\subsection*{Geometric Intuition}

\begin{exposition}
Cauchy makes the simplest possible height choice: always sample the left edge
of each slab. The result is a staircase of left-anchored rectangles. As the
slabs thin out, the question is whether those staircase areas settle.
\end{exposition}

\begin{remark*}[Interpretation]
Cauchy's construction is the seed. It answers the geometric problem for the
case where the curve is tame enough that left-edge staircases forget their
initial sampling bias as the mesh shrinks. The construction is deliberately
primitive: it fixes a height rule first and asks for convergence second.
\end{remark*}

\subsection*{Definition}

\begin{definitionbox}{Definition (Cauchy Sum)}
\begin{definition}[Left-Endpoint Cauchy Sum]
\label{def:cauchy-sum}
For \(f:[a,b]\to\mathbb R\) and \(P=\{a=x_0<\cdots<x_n=b\}\), define
\[
  C(f,P)=\sum_{i=1}^{n} f(x_{i-1})\,\Delta x_i.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
C(f,P)=\sum_{i=1}^{n} f(x_{i-1})\,\Delta x_i.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The Cauchy sum assigns each subinterval the height at its left endpoint and
adds the resulting rectangle areas.
\end{remark*}

\begin{definitionbox}{Definition (Cauchy Integral)}
\begin{definition}[Cauchy Integral]
\label{def:cauchy-integral}
The function \(f\) is \emph{Cauchy-integrable} with value \(L\) if
\[
  \forall\varepsilon>0\ \exists\delta>0\ \forall P\,
  \bigl(\|P\|<\delta\Rightarrow |C(f,P)-L|<\varepsilon\bigr).
\]
The value \(L\) is denoted \(\int_a^b f(x)\,dx\).
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{HasDefiniteIntegral}(f,[a,b],\operatorname{CauchyCriterion},L)
\Longleftrightarrow
\forall\varepsilon>0\ \exists\delta>0\ \forall P\,
\bigl(\|P\|<\delta\Rightarrow |C(f,P)-L|<\varepsilon\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Cauchy integrability means that left-endpoint sums converge to a single value
as the partition mesh tends to zero.
\end{remark*}

\subsection*{Worked Positive Example}

\begin{propositionbox}{Proposition (Integral of a Constant)}
\begin{proposition}[Cauchy Integral of a Constant]
\label{prop:cauchy-integral-constant}
If \(f(x)=c\) on \([a,b]\), then \(\int_a^b c\,dx=c(b-a)\).
\hyperref[prf:cauchy-integral-constant]{\textit{Go to proof.}}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
f(x)=c\text{ on }[a,b]
\Longrightarrow
\int_a^b c\,dx=c(b-a).
\]
\end{remark*}

\begin{remark*}[Interpretation]
A constant integrand gives rectangles with the same height on every
subinterval, so the total area is height times interval length.
\end{remark*}

\subsection*{Algebraic and Order Properties}

\begin{theorembox}{Theorem (Linearity)}
\begin{theorem}[Linearity of the Cauchy Integral]
\label{thm:cauchy-integral-linearity}
If \(f\) and \(g\) are Cauchy-integrable on \([a,b]\), then
\[
  \int_a^b(\alpha f+\beta g)\,dx
  =
  \alpha\int_a^b f\,dx+\beta\int_a^b g\,dx.
\]
\hyperref[prf:cauchy-integral-linearity]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{IsIntegrable}(f,[a,b],\operatorname{CauchyCriterion})\land
\operatorname{IsIntegrable}(g,[a,b],\operatorname{CauchyCriterion})
\Longrightarrow
\int_a^b(\alpha f+\beta g)\,dx
=
\alpha\int_a^b f\,dx+\beta\int_a^b g\,dx.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Linearity says that Cauchy integration distributes over finite linear
combinations of integrable functions.
\end{remark*}

\begin{theorembox}{Theorem (Monotonicity)}
\begin{theorem}[Monotonicity of the Cauchy Integral]
\label{thm:cauchy-integral-monotonicity}
If \(f\) and \(g\) are Cauchy-integrable on \([a,b]\) and \(f(x)\le g(x)\)
for every \(x\in[a,b]\), then
\[
  \int_a^b f\,dx\le \int_a^b g\,dx .
\]
\hyperref[prf:cauchy-integral-monotonicity]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{IsIntegrable}(f,[a,b],\operatorname{CauchyCriterion})\land
\operatorname{IsIntegrable}(g,[a,b],\operatorname{CauchyCriterion})\land
(\forall x\in[a,b])\bigl(f(x)\le g(x)\bigr)
\Longrightarrow
\int_a^b f\,dx\le \int_a^b g\,dx.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Pointwise order between integrands is preserved by the Cauchy integral.
\end{remark*}

\begin{corollarybox}{Corollary (Bounds)}
\begin{corollary}[Bounds for the Cauchy Integral]
\label{cor:cauchy-integral-bounds}
If \(f\) is continuous on \([a,b]\), \(m=\min_{[a,b]}f\), and
\(M=\max_{[a,b]}f\), then
\[
  m(b-a)\le \int_a^b f\,dx\le M(b-a).
\]
\hyperref[prf:cauchy-integral-bounds]{\textit{Go to proof.}}
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
m=\min_{[a,b]}f\land M=\max_{[a,b]}f
\Longrightarrow
m(b-a)\le \int_a^b f\,dx\le M(b-a).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The integral of a continuous function is trapped between the areas of the
constant lower and upper bounding rectangles.
\end{remark*}

\begin{theorembox}{Theorem (Triangle Inequality)}
\begin{theorem}[Triangle Inequality for the Cauchy Integral]
\label{thm:cauchy-integral-triangle-inequality}
If \(f\) and \(|f|\) are Cauchy-integrable on \([a,b]\), then
\[
  \left|\int_a^b f\,dx\right|\le \int_a^b |f|\,dx .
\]
\hyperref[prf:cauchy-integral-triangle-inequality]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{IsIntegrable}(f,[a,b],\operatorname{CauchyCriterion})\land
\operatorname{IsIntegrable}(|f|,[a,b],\operatorname{CauchyCriterion})
\Longrightarrow
\left|\int_a^b f\,dx\right|\le \int_a^b |f|\,dx.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The absolute value of the signed accumulated area is bounded by the accumulated
absolute area.
\end{remark*}

\begin{theorembox}{Theorem (Interval Additivity)}
\begin{theorem}[Interval Additivity for the Cauchy Integral]
\label{thm:cauchy-integral-interval-additivity}
If \(c\in[a,b]\) and \(f\) is Cauchy-integrable on the relevant intervals,
then
\[
  \int_a^b f\,dx=\int_a^c f\,dx+\int_c^b f\,dx .
\]
\hyperref[prf:cauchy-integral-interval-additivity]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
c\in[a,b]\land
\operatorname{IsIntegrable}(f,[a,b],\operatorname{CauchyCriterion})
\Longrightarrow
\int_a^b f\,dx=\int_a^c f\,dx+\int_c^b f\,dx.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The Cauchy integral over an interval decomposes into the sum of the integrals
over adjacent subintervals.
\end{remark*}

\begin{definitionbox}{Definition (Oscillation on an Interval)}
\begin{definition}[Oscillation on an Interval]
\label{def:interval-oscillation-integration}
For bounded \(f\) on an interval \(I\), define
\[
  \omega_f(I)=\sup_I f-\inf_I f.
\]
Oscillation is the error controller: if \(\omega_f(I)\) is small, every sample
height on \(I\) is close to every other sample height.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\omega_f(I)=\sup_I f-\inf_I f.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Oscillation measures the vertical spread of a bounded function on an interval.
\end{remark*}

\subsection*{Existence and Applicability Criterion}

\begin{theorembox}{Theorem (Continuous Implies Cauchy-Integrable)}
\begin{theorem}[Continuous Functions Are Cauchy-Integrable]
\label{thm:continuous-cauchy-integrable}
Every continuous function \(f:[a,b]\to\mathbb R\) is Cauchy-integrable.
\hyperref[prf:continuous-cauchy-integrable]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{IsContinuous}(f,[a,b])
\Longrightarrow
\operatorname{IsIntegrable}(f,[a,b],\operatorname{CauchyCriterion}).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Continuity on the compact interval gives uniform continuity. Hence short slabs
have small oscillation. Given two fine partitions, pass to their common
refinement and compare sums slab by slab; each re-sampling error is bounded by
\(\omega_f(I)\) times the slab width. The sums form a Cauchy net, and
completene
ss of \(\mathbb R\) supplies the limit.
\end{remark*}

\begin{remark*}[Interpretation]
Uniform continuity is the load-bearing ingredient. It converts pointwise
control into one global scale: if the slabs are all shorter than that scale,
then every slab has small oscillation. Without this global scale, Cauchy's
left-endpoint definition has no mechanism for proving that unrelated
partitions settle to the same number.
\end{remark*}

\subsection*{Worked Failure Mode}

\begin{remark*}[Interpretation]
A single jump should still have an area. If
\[
  f(x)=
  \begin{cases}
    0,&x<c,\\
    1,&x\ge c,
  \end{cases}
\]
on \([a,b]\), the only dangerous slab is the one crossing \(c\). Its total
possible error is bounded by its width, and that width tends to zero as the
mesh tends to zero. This example is not hard; the point is that Cauchy's
definition gives no internal test explaining why it is harmless. The proof
above proves continuity is sufficient, not that controlled discontinuities are
allowed.
\end{remark*}

\subsection*{What the Failure Motivates Next}

\begin{theorembox}{Theorem (Tag Independence)}
\begin{theorem}[Tag Independence for Continuous Functions]
\label{thm:cauchy-tag-independence}

If \(f\) is continuous on \([a,b]\), then every choice of tags gives the same
limit as the left-endpoint sums as \(\|P\|\to 0\).
\hyperref[prf:cauchy-tag-independence]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{IsContinuous}(f,[a,b])
\Longrightarrow
\text{all tagged sums have the same limit as the left-endpoint sums as }\|P\|\to0.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Cauchy has two defects. First, the left endpoint is privileged by convention;
tag independence is a theorem only after continuity has already been assumed.
Second, the existence proof gives no usable criterion for discontinuous
functions. The patch request is to stop privileging the left endpoint and make
tag independence part of the definition.
\end{remark*}

\begin{remark*}[Interpretation]
The next definition should not say ``sample on the left.'' It should say:
sample anywhere, even adversarially, and require all fine enough choices to
agree. Riemann is exactly Cauchy's tag-independence theorem promoted into the
definition of integrability.
\end{remark*}
