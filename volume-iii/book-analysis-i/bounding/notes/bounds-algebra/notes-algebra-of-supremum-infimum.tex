\begin{remark*}[Section boundary]

\begin{definitionbox}{Definition (Set Arithmetic Images)}
\begin{definition}[Set Arithmetic Images]
\label{def:set-arithmetic-images}
Let $A,B\subseteq\mathbb{R}$.  Define
\[
  A+B:=\{a+b:a\in A,\ b\in B\},\qquad
  A-B:=\{a-b:a\in A,\ b\in B\},
\]
\[
  AB:=\{ab:a\in A,\ b\in B\},\qquad
  A/B:=\{a/b:a\in A,\ b\in B,\ b\neq0\}.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
x\in A+B\Longleftrightarrow(\exists a\in A)(\exists b\in B)(x=a+b),
\]
and similarly for difference, product, and quotient with the quotient condition
$b\neq0$.
\end{remark*}

\begin{remark*}[Predicate reading]
Let
\[
D=A\times B,\qquad D_{\div}=\{(a,b)\in A\times B:b\ne0\},
\]
and define functional rules on these domains by
\[
\varphi_{+}\colon D\to\mathbb{R},\quad (a,b)\mapsto a+b,\qquad
\varphi_{-}\colon D\to\mathbb{R},\quad (a,b)\mapsto a-b,
\]
\[
\varphi_{\cdot}\colon D\to\mathbb{R},\quad (a,b)\mapsto ab,\qquad
\varphi_{\div}\colon D_{\div}\to\mathbb{R},\quad (a,b)\mapsto a/b.
\]
Then
\[
\operatorname{ReplacementImage}(A+B,D,\varphi_{+}),
\]
\[
\operatorname{ReplacementImage}(A-B,D,\varphi_{-}),
\quad
\operatorname{ReplacementImage}(AB,D,\varphi_{\cdot}),
\]
\[
\operatorname{ReplacementImage}(A/B,D_{\div},\varphi_{\div}).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
x\notin A+B\Longleftrightarrow(\forall a\in A)(\forall b\in B)(x\neq a+b),
\]
with the analogous readings for the other arithmetic images.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
C\ne AB
\quad\text{means}\quad
(\exists x)\bigl[x\in C\triangle AB\bigr],
\]
and similarly for sum, difference, and quotient.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
An arithmetic-image definition fails only if it omits a value produced by the
rule or includes a value not produced by the rule.


  \item[\exists-condition.]
  \[
x\notin A+B\Longleftrightarrow(\forall a\in A)(\forall b\in B)(x\neq a+b),
\]
\[
C\ne AB
\quad\text{means}\quad
(\exists x)\bigl[x\in C\triangle AB\bigr],
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
Every membership claim decomposes into witnesses from the input sets and the
arithmetic operation that produced the image point.


  \item[\exists-condition.]
  \[
x\notin A+B\Longleftrightarrow(\forall a\in A)(\forall b\in B)(x\neq a+b),
\]
\[
C\ne AB
\quad\text{means}\quad
(\exists x)\bigl[x\in C\triangle AB\bigr],
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Set arithmetic turns pointwise arithmetic into images of sets. Membership in
one of these images is always witnessed by input elements from the original
sets and the corresponding arithmetic operation.
\end{remark*}

\begin{definitionbox}{Definition (Reciprocal Image)}
\begin{definition}[Reciprocal Image]
\label{def:reciprocal-image}
Let $A\subseteq\mathbb{R}$ and suppose $0\notin A$.  The reciprocal image of
$A$ is
\[
  A^{-1}:=\{1/a:a\in A\}.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
x\in A^{-1}\Longleftrightarrow(\exists a\in A)(x=1/a).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&A\subseteq\mathbb{R},\qquad 0\notin A,\\
&\mathsf{Reciprocal}(A)=A^{-1}.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
x\notin A^{-1}\Longleftrightarrow(\forall a\in A)(x\neq1/a).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&A\subseteq\mathbb{R},\qquad 0\notin A,\qquad B\subseteq\mathbb{R},\\
&B\ne\mathsf{Reciprocal}(A)
\quad\text{means}\quad
(\exists x)\bigl[x\in B\triangle A^{-1}\bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
The reciprocal image is undefined if zero is allowed as an input value.  Once
zero is excluded, the only definition failures are omitted or spurious
reciprocal values.


  \item[\mathsf{Reciprocal}.]
  \[
x\notin A^{-1}\Longleftrightarrow(\forall a\in A)(x\neq1/a).
\]
\[
\begin{aligned}
&A\subseteq\mathbb{R},\qquad 0\notin A,\qquad B\subseteq\mathbb{R},\\
&B\ne\mathsf{Reciprocal}(A)
\quad\text{means}\quad
(\exists x)\bigl[x\in B\triangle A^{-1}\bigr].
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
Every element of $A^{-1}$ has the form $1/a$ for a nonzero $a\in A$.


  \item[\mathsf{Reciprocal}.]
  \[
x\notin A^{-1}\Longleftrightarrow(\forall a\in A)(x\neq1/a).
\]
\[
\begin{aligned}
&A\subseteq\mathbb{R},\qquad 0\notin A,\qquad B\subseteq\mathbb{R},\\
&B\ne\mathsf{Reciprocal}(A)
\quad\text{means}\quad
(\exists x)\bigl[x\in B\triangle A^{-1}\bigr].
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
The reciprocal image applies inversion to every element of $A$. The exclusion
of zero is structural, since reciprocal values are undefined at zero.
\end{remark*}

\begin{theorembox}{Theorem (Translation Invariance of the Supremum)}
\begin{theorem}[Translation Invariance of the Supremum]
\label{thm:translation-invariance-supremum}
Let $A\subseteq\mathbb{R}$ be nonempty and bounded above, and let $c\in\mathbb{R}$. Then
\[
  \sup(A+c)=\sup A+c,
  \qquad
  A+c=\{a+c:a\in A\}.
\]
\hyperref[prf:translation-invariance-supremum]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let }A\subseteq\mathbb{R}\text{ be nonempty and bounded above, and let }c\in\mathbb{R}.\\
&\sup\{a+c:a\in A\}=\sup A+c.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad A+c=\{a+c:a\in A\},\\
&\operatorname{LeastUpperBound}(s,A,P)
\Longrightarrow
\operatorname{LeastUpperBound}(s+c,A+c,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\sup(A+c)\ne \sup A+c
\]
is impossible under the stated hypotheses.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad A+c=\{a+c:a\in A\},\\
&\operatorname{LeastUpperBound}(s,A,P)\land
\neg\operatorname{LeastUpperBound}(s+c,A+c,P)
\end{aligned}
\]
is a contradictory configuration.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
The formula could fail only if translation either failed to preserve upper bounds or failed to preserve leastness. Adding the same real number to every comparison preserves the order, so neither failure mode can occur.


  \item[\operatorname{LeastUpperBound}.]
  \[
\sup(A+c)\ne \sup A+c
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad A+c=\{a+c:a\in A\},\\
&\operatorname{LeastUpperBound}(s,A,P)\land
\neg\operatorname{LeastUpperBound}(s+c,A+c,P)
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
\begin{aligned}
\forall a\in A\;(a\le s)
&\Longleftrightarrow
\forall a\in A\;(a+c\le s+c),\\
\forall u\,[\operatorname{UpperBound}(u,A)\Longrightarrow s\le u]
&\Longleftrightarrow
\forall v\,[\operatorname{UpperBound}(v,A+c)\Longrightarrow s+c\le v].
\end{aligned}
\]


  \item[\operatorname{LeastUpperBound}.]
  \[
\sup(A+c)\ne \sup A+c
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad A+c=\{a+c:a\in A\},\\
&\operatorname{LeastUpperBound}(s,A,P)\land
\neg\operatorname{LeastUpperBound}(s+c,A+c,P)
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Translation shifts the entire set and all of its ceilings by the same amount. The least ceiling therefore shifts by exactly that amount.
\end{remark*}

\begin{theorembox}{Theorem (Scalar Multiplication by a Positive Scalar)}
\begin{theorem}[Scalar Multiplication by a Positive Scalar]
\label{thm:scalar-mult-positive}
Let $A\subseteq\mathbb{R}$ be nonempty and bounded above, and let $k>0$. Then
\[
  \sup(kA)=k\sup A,
  \qquad
  kA=\{ka:a\in A\}.
\]
\hyperref[prf:scalar-mult-positive]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let }A\subseteq\mathbb{R}\text{ be nonempty and bounded above, and let }k>0.\\
&\sup\{ka:a\in A\}=k\sup A.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad kA=\{ka:a\in A\},\\
&\operatorname{LeastUpperBound}(s,A,P)\land k>0
\Longrightarrow
\operatorname{LeastUpperBound}(ks,kA,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
k>0,\quad \sup(kA)\ne k\sup A
\]
is impossible under the stated hypotheses.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad kA=\{ka:a\in A\},\\
&\operatorname{LeastUpperBound}(s,A,P)\land k>0\land
\neg\operatorname{LeastUpperBound}(ks,kA,P)
\end{aligned}
\]
is a contradictory configuration.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
The formula could fail only if multiplying by a positive scalar failed to preserve upper bounds or failed to preserve leastness. Positive multiplication preserves the order, so both properties scale exactly.


  \item[\operatorname{LeastUpperBound}.]
  \[
k>0,\quad \sup(kA)\ne k\sup A
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad kA=\{ka:a\in A\},\\
&\operatorname{LeastUpperBound}(s,A,P)\land k>0\land
\neg\operatorname{LeastUpperBound}(ks,kA,P)
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
\begin{aligned}
\forall a\in A\;(a\le s)
&\Longleftrightarrow
\forall a\in A\;(ka\le ks),\\
\operatorname{UpperBound}(u,A)
&\Longleftrightarrow
\operatorname{UpperBound}(ku,kA).
\end{aligned}
\]


  \item[\operatorname{LeastUpperBound}.]
  \[
k>0,\quad \sup(kA)\ne k\sup A
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad kA=\{ka:a\in A\},\\
&\operatorname{LeastUpperBound}(s,A,P)\land k>0\land
\neg\operatorname{LeastUpperBound}(ks,kA,P)
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Positive scaling stretches the set without reversing order. The least ceiling stretches by the same positive factor.
\end{remark*}

\begin{theorembox}{Theorem (Scalar Multiplication by a Negative Scalar)}

\label{thm:scalar-mult-negative}
Let $A\subseteq\mathbb{R}$ be nonempty and bounded below, and let $k<0$. Then
\[
  \sup(kA)=k\inf A,
  \qquad

\hyperref[prf:scalar-mult-negative]{Go to proof.}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let }A\subseteq\mathbb{R}\text{ be nonempty and bounded below, and let }k<0.\\
&\sup\{ka:a\in A\}=k\inf A.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad kA=\{ka:a\in A\},\\
&\operatorname{GreatestLowerBound}(t,A,P)\land k<0
\Longrightarrow
\operatorname{LeastUpperBound}(kt,kA,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
k<0,\quad \sup(kA)\ne k\inf A
\]
is impossible under the stated hypotheses.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad kA=\{ka:a\in A\},\\
&\operatorname{GreatestLowerBound}(t,A,P)\land k<0\land
\neg\operatorname{LeastUpperBound}(kt,kA,P)
\end{aligned}
\]
is a contradictory configuration.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
The formula could fail only if negative multiplication did not reverse lower bounds into upper bounds, or if the extremal value lost its leastness after reversal. Multiplication by a negative scalar reverses order exactly, so the infimum of $A$ becomes the supremum of $kA$.


  \item[\operatorname{GreatestLowerBound} / \operatorname{LeastUpperBound}.]
  \[
k<0,\quad \sup(kA)\ne k\inf A
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad kA=\{ka:a\in A\},\\
&\operatorname{GreatestLowerBound}(t,A,P)\land k<0\land
\neg\operatorname{LeastUpperBound}(kt,kA,P)
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
\begin{aligned}
\forall a\in A\;(t\le a)
&\Longleftrightarrow
\forall a\in A\;(ka\le kt),\\
\operatorname{LowerBound}(v,A)
&\Longleftrightarrow

\begin{theorembox}{Theorem (Supremum of a Dilation)}
\begin{theorem}[Supremum of a Dilation]
\label{thm:supremum-dilation}
Let $A\subseteq\mathbb{R}$ be nonempty and bounded, and let $\lambda\in\mathbb{R}$. Then
\[
  \sup(\lambda A)=\max\{\lambda\sup A,\lambda\inf A\},
  \qquad

\]

\end{theorem}

\[

&\text{Let }A\subseteq\mathbb{R}\text{ be nonempty and bounded, and let }\lambda\in\mathbb{R}.\\

\end{aligned}
\]
\end{remark*}

\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad \lambda A=\{\lambda a:a\in A\},\\
&M=\max\{\lambda s,\lambda t\},\\
&\operatorname{LeastUpperBound}(s,A,P)\land \operatorname{GreatestLowerBound}(t,A,P)
\Longrightarrow
\operatorname{LeastUpperBound}(M,\lambda A,P).
\end{aligned}
\]
\end{remark*}

\[
\sup(\lambda A)\ne \max\{\lambda\sup A,\lambda\inf A\}
\]
is impossible under the stated hypotheses.
\end{remark*}

&\operatorname{LeastUpperBound}(s,A,P)\land \operatorname{GreatestLowerBound}(t,A,P)
\land
\neg\operatorname{LeastUpperBound}(M,\lambda A,P)
\end{aligned}

is a contradictory configuration.
\end{remark*}

\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
\sup(\lambda A)=
\begin{cases}
\lambda\sup A, & \lambda>0,\\
0, & \lambda=0,\\
\lambda\inf A, & \lambda<0.
\end{cases}
\]

  \[
\sup(\lambda A)\ne \max\{\lambda\sup A,\lambda\inf A\}
\]
\[
\begin{aligned}

\end{remark*}

\begin{remark*}[Interpretation]
A dilation either preserves the order, collapses all points to zero, or reverses the order. The displayed maximum packages those three cases into one formula.
\end{remark*}

The absolute-value image measures distance from the origin. For a bounded subset of the real line, the largest possible distance is controlled by whichever extremal endpoint lies farther from zero.

\begin{theorembox}{Theorem (Supremum of a Reciprocal Set)}
\begin{theorem}[Supremum of a Reciprocal Set]
\label{thm:supremum-reciprocal-set}
Let $A\subseteq\mathbb{R}$ be nonempty and bounded, and suppose that
$0<\inf A$ or $\sup A<0$.  Define
\[
  A^{-1}:=\{1/a:a\in A\}.
\]
Then
\[
  \sup(A^{-1})=\frac{1}{\inf A}.
\]
\hyperref[prf:supremum-reciprocal-set]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
A\neq\varnothing,\quad A\text{ bounded},\quad
0\notin[\inf A,\sup A]
\quad\Longrightarrow\quad
\sup\{1/a:a\in A\}=(\inf A)^{-1}.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad A^{-1}=\{1/a:a\in A\},\\
&u=m^{-1},\\
&\operatorname{GreatestLowerBound}(m,A,P)\land\operatorname{LeastUpperBound}(M,A,P)\land
0\notin[m,M]
\Longrightarrow
\operatorname{LeastUpperBound}(u,A^{-1},P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
A\neq\varnothing,\quad A\text{ bounded},\quad 0\notin[\inf A,\sup A],
\quad
\sup(A^{-1})\neq(\inf A)^{-1}.
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad A^{-1}=\{1/a:a\in A\},\\
&\operatorname{LeastUpperBound}(M,A,P),\qquad u=m^{-1},\\
&\operatorname{GreatestLowerBound}(m,A,P)\land 0\notin[m,M]\land
\neg\operatorname{LeastUpperBound}(u,A^{-1},P)
\end{aligned}
\]
is contradictory under the stated hypotheses.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
The formula fails without a nonzero gap from zero: reciprocal images may be
unbounded or undefined.  With the gap, reciprocal is order-reversing on the
whole interval containing $A$, so the floor of $A$ becomes the ceiling of
$A^{-1}$.


  \item[\operatorname{LeastUpperBound} / \operatorname{GreatestLowerBound}.]
  \[
A\neq\varnothing,\quad A\text{ bounded},\quad 0\notin[\inf A,\sup A],
\quad
\sup(A^{-1})\neq(\inf A)^{-1}.
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad A^{-1}=\{1/a:a\in A\},\\
&\operatorname{LeastUpperBound}(M,A,P),\qquad u=m^{-1},\\
&\operatorname{GreatestLowerBound}(m,A,P)\land 0\notin[m,M]\land
\neg\operatorname{LeastUpperBound}(u,A^{-1},P)
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
Every $a\in A$ lies between $\inf A$ and $\sup A$, and the interval does not
cross zero.  The reciprocal map reverses comparisons there, so
$1/a\leq 1/\inf A$.  Leastness follows by applying the same reversal to values
of $A$ approaching $\inf A$.


  \item[\operatorname{LeastUpperBound} / \operatorname{GreatestLowerBound}.]
  \[
A\neq\varnothing,\quad A\text{ bounded},\quad 0\notin[\inf A,\sup A],
\quad
\sup(A^{-1})\neq(\inf A)^{-1}.
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad A^{-1}=\{1/a:a\in A\},\\
&\operatorname{LeastUpperBound}(M,A,P),\qquad u=m^{-1},\\
&\operatorname{GreatestLowerBound}(m,A,P)\land 0\notin[m,M]\land
\neg\operatorname{LeastUpperBound}(u,A^{-1},P)
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Taking reciprocals turns small nonzero magnitudes into large magnitudes.  The
least value threshold of the original set therefore controls the largest value
threshold of the reciprocal set.
\end{remark*}

\begin{theorembox}{Theorem (Supremum of a Product Set)}
\begin{theorem}[Supremum of a Product Set]
\label{thm:supremum-product-set}
Let $A,B\subseteq\mathbb{R}$ be nonempty and bounded. Then
\[
  \sup(AB)=
  \max\{(\inf A)(\inf B),(\inf A)(\sup B),(\sup A)(\inf B),(\sup A)(\sup B)\}.
\]
\hyperref[prf:supremum-product-set]{Go to proof.}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
A,B\neq\varnothing,\quad A,B\text{ bounded}
\quad\Longrightarrow\quad
\sup\{ab:a\in A,\ b\in B\}
\]
equals the maximum of the four endpoint products.
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad AB=\{ab:a\in A,\ b\in B\},\\
&M=\max\{\alpha\gamma,\alpha\delta,\beta\gamma,\beta\delta\},\\
&\operatorname{GreatestLowerBound}(\alpha,A,P)\land\operatorname{LeastUpperBound}(\beta,A,P)\land\\
&\operatorname{GreatestLowerBound}(\gamma,B,P)\land\operatorname{LeastUpperBound}(\delta,B,P)
\Longrightarrow
\operatorname{LeastUpperBound}(M,AB,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\sup(AB)\neq
\max\{(\inf A)(\inf B),(\inf A)(\sup B),(\sup A)(\inf B),(\sup A)(\sup B)\}
\]
is impossible under the stated hypotheses.

\begin{description}
  \item[Exposition.]
The lower inequality could fail only if a sum fell below the sum of the two lower thresholds. The upper inequality could fail only if a sum rose above the sum of the two upper thresholds.


  \item[\operatorname{GreatestLowerBound} / \operatorname{LeastUpperBound}.]
  \[
\inf(A+B)<\inf A+\inf B
\quad\text{or}\quad
\sup A+\sup B<\sup(A+B)
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
A,B\subseteq\mathbb{R},\qquad C=A+B=\{a+b:a\in A,\ b\in B\},\\
&\bigl[\operatorname{GreatestLowerBound}(t_A,A,P)\land
\operatorname{GreatestLowerBound}(t_B,B,P)\land \inf C<t_A+t_B\bigr]\\
\lor
&\bigl[\operatorname{LeastUpperBound}(s_A,A,P)\land
\operatorname{LeastUpperBound}(s_B,B,P)\land s_A+s_B<\sup C\bigr]
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
\inf A\le a\le \sup A,\quad \inf B\le b\le \sup B
\Longrightarrow
\inf A+\inf B\le a+b\le \sup A+\sup B.
\]


  \item[\operatorname{GreatestLowerBound} / \operatorname{LeastUpperBound}.]
  \[
\inf(A+B)<\inf A+\inf B
\quad\text{or}\quad
\sup A+\sup B<\sup(A+B)
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
A,B\subseteq\mathbb{R},\qquad C=A+B=\{a+b:a\in A,\ b\in B\},\\
&\bigl[\operatorname{GreatestLowerBound}(t_A,A,P)\land
\operatorname{GreatestLowerBound}(t_B,B,P)\land \inf C<t_A+t_B\bigr]\\
\lor
&\bigl[\operatorname{LeastUpperBound}(s_A,A,P)\land
\operatorname{LeastUpperBound}(s_B,B,P)\land s_A+s_B<\sup C\bigr]
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Every element of the sum set is built from one element of $A$ and one element of $B$. Adding the individual lower and upper controls gives immediate lower and upper controls for every such sum.
\end{remark*}

\begin{propositionbox}{Proposition (Upper Bounds Depend on the Ambient Order)}
\begin{proposition}[Upper Bounds Depend on the Ambient Order]
\label{prop:upper-bounds-ambient-order}
Let $(S,\le_S)$ and $(T,\le_T)$ be ordered sets, and let $A\subseteq S\cap T$. Whether $u$ is an upper bound of $A$ must be interpreted relative to the chosen ambient order.
\hyperref[prf:upper-bounds-ambient-order]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&u\text{ is an upper bound of }A\text{ in }S
\Longleftrightarrow
u\in S\text{ and }\forall x\in A\;(x\le_S u),\\
&u\text{ is an upper bound of }A\text{ in }T
\Longleftrightarrow
u\in T\text{ and }\forall x\in A\;(x\le_T u).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P_S=\mathsf{OrderedSet}(S,\le_S),\qquad
P_T=\mathsf{OrderedSet}(T,\le_T),\qquad A\subseteq S\cap T,\\
&\operatorname{UpperBound}(u,A,P_S)
\Longleftrightarrow
u\in S\land \forall x\in A\;(x\le_S u),\\
&\operatorname{UpperBound}(u,A,P_T)
\Longleftrightarrow
u\in T\land \forall x\in A\;(x\le_T u).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\text{The phrase ``}u\text{ is an upper bound of }A\text{''}
\]
is not well-determined until the ambient ordered set and its order relation are fixed.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P_S=\mathsf{OrderedSet}(S,\le_S),\qquad
P_T=\mathsf{OrderedSet}(T,\le_T),\\
&X=\text{``$u$ is an upper bound of $A$''},\\
&\neg\operatorname{WellDefined}(X,\text{ambient omitted})
\end{aligned}
\]
when the ambient order is omitted.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
Ambiguity occurs when the same set $A$ is considered inside different ordered ambient sets or under different order relations. The candidate bound may belong to one ambient set and not another, or the relevant comparison relation may change.


  \item[\operatorname{WellDefined}.]
  \[
\text{The phrase ``}u\text{ is an upper bound of }A\text{''}
\]
\[
\begin{aligned}
&P_S=\mathsf{OrderedSet}(S,\le_S),\qquad
P_T=\mathsf{OrderedSet}(T,\le_T),\\
&X=\text{``$u$ is an upper bound of $A$''},\\
&\neg\operatorname{WellDefined}(X,\text{ambient omitted})
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
\operatorname{UpperBound}_S(u,A)
\text{ and }
\operatorname{UpperBound}_T(u,A)
\]
are separate assertions unless $S=T$ and $\le_S=\le_T$ on the elements being compared.


  \item[\operatorname{WellDefined}.]
  \[
\text{The phrase ``}u\text{ is an upper bound of }A\text{''}
\]
\[
\begin{aligned}
&P_S=\mathsf{OrderedSet}(S,\le_S),\qquad
P_T=\mathsf{OrderedSet}(T,\le_T),\\
&X=\text{``$u$ is an upper bound of $A$''},\\
&\neg\operatorname{WellDefined}(X,\text{ambient omitted})
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
An upper bound is not only about the subset; it is also about where the subset lives. Changing the ambient order changes the universe of possible bounds and the comparison used to test them.
\end{remark*}

\begin{propositionbox}{Proposition (Suprema Depend on the Ambient Set)}
\begin{proposition}[Suprema Depend on the Ambient Set]
\label{prop:suprema-ambient-set}
Let $A\subseteq S\subseteq S'$ be subsets of an ordered set. The supremum of $A$ relative to $S$, if it exists, need not agree with the supremum of $A$ relative to $S'$.
\hyperref[prf:suprema-ambient-set]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
\sup_S A\text{ and }\sup_{S'} A
\]
are ambient-relative objects and may differ in existence or value.
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&T=S',\qquad A\subseteq S,\qquad S\subseteq T,\\
&P_S=\mathsf{OrderedSet}(S,\leq),\qquad
P_T=\mathsf{OrderedSet}(T,\leq),\\
&\operatorname{LeastUpperBound}(s,A,P_S)
\quad\text{and}\quad
\operatorname{LeastUpperBound}(s,A,P_T)
\end{aligned}
\]
are different assertions unless the ambient upper-bound sets agree.
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\sup_S A=\sup_{S'} A
\]
is not guaranteed by $A\subseteq S\subseteq S'$ alone.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&T=S',\qquad A\subseteq S,\qquad S\subseteq T,\\
&P_S=\mathsf{OrderedSet}(S,\leq),\qquad
P_T=\mathsf{OrderedSet}(T,\leq),\\
&\operatorname{LeastUpperBound}(s,A,P_T)\land
\neg\operatorname{LeastUpperBound}(s,A,P_S)
\end{aligned}
\]
is possible when the larger ambient set contains the least upper bound but the smaller one does not.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
The ambient-relative supremum can fail to transfer because the candidate least upper bound may not belong to the smaller ambient set, or because the smaller ambient set may have a different collection of upper bounds.


  \item[\operatorname{LeastUpperBound}.]
  \[
\sup_S A=\sup_{S'} A
\]
\[
\begin{aligned}
&T=S',\qquad A\subseteq S,\qquad S\subseteq T,\\
&P_S=\mathsf{OrderedSet}(S,\leq),\qquad
P_T=\mathsf{OrderedSet}(T,\leq),\\
&\operatorname{LeastUpperBound}(s,A,P_T)\land
\neg\operatorname{LeastUpperBound}(s,A,P_S)
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
s=\sup_{S'} A,\quad s\notin S
\Longrightarrow
s\text{ cannot be the supremum of }A\text{ relative to }S.
\]


  \item[\operatorname{LeastUpperBound}.]
  \[
\sup_S A=\sup_{S'} A
\]
\[
\begin{aligned}
&T=S',\qquad A\subseteq S,\qquad S\subseteq T,\\
&P_S=\mathsf{OrderedSet}(S,\leq),\qquad
P_T=\mathsf{OrderedSet}(T,\leq),\\
&\operatorname{LeastUpperBound}(s,A,P_T)\land
\neg\operatorname{LeastUpperBound}(s,A,P_S)
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Suprema are least upper bounds inside a chosen universe. Enlarging or shrinking that universe can create or remove the point that should serve as the least ceiling.
\end{remark*}

\begin{theorembox}{Theorem (Ambient Existence of Supremum)}
\begin{theorem}[Ambient Existence of Supremum]
\label{thm:ambient-existence-supremum}
There are ordered sets $S\subseteq S'$ and a subset $A\subseteq S$ such that $A$ has a supremum relative to $S'$ but has no supremum relative to $S$.
\hyperref[prf:ambient-existence-supremum]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\exists S\subseteq S'\;\exists A\subseteq S:
\quad
\sup_{S'} A\text{ exists}
\quad\text{and}\quad
\sup_S A\text{ does not exist}.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&T=S',\qquad A\subseteq S,\qquad S\subseteq T,\\
&P_S=\mathsf{OrderedSet}(S,\leq),\qquad
P_T=\mathsf{OrderedSet}(T,\leq),\\
&\exists s\in T:\operatorname{LeastUpperBound}(s,A,P_T)
\land
\neg\exists t\in S\;\operatorname{LeastUpperBound}(t,A,P_S).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\text{Every ambient enlargement preserves and reflects existence of suprema.}
\]
This assertion is false.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&T=S',\qquad A\subseteq S,\qquad S\subseteq T,\\
&P_S=\mathsf{OrderedSet}(S,\leq),\qquad
P_T=\mathsf{OrderedSet}(T,\leq),\\
&\bigl[\exists s\in T\,\operatorname{LeastUpperBound}(s,A,P_T)
\Longrightarrow
\exists t\in S\,\operatorname{LeastUpperBound}(t,A,P_S)\bigr]
\end{aligned}
\]
is false.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
The transfer from $S'$ to $S$ fails when the least upper bound exists in the larger ambient set but is missing from the smaller one. The smaller ambient set may contain upper bounds without containing a least upper bound.


  \item[\operatorname{LeastUpperBound}.]
  \[
\text{Every ambient enlargement preserves and reflects existence of suprema.}
\]
\[
\begin{aligned}
&T=S',\qquad A\subseteq S,\qquad S\subseteq T,\\
&P_S=\mathsf{OrderedSet}(S,\leq),\qquad
P_T=\mathsf{OrderedSet}(T,\leq),\\
&\bigl[\exists s\in T\,\operatorname{LeastUpperBound}(s,A,P_T)
\Longrightarrow
\exists t\in S\,\operatorname{LeastUpperBound}(t,A,P_S)\bigr]
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
s=\sup_{S'} A,\quad s\notin S,\quad
\text{and no }t\in S\text{ is least among the upper bounds of }A\text{ in }S.
\]


  \item[\operatorname{LeastUpperBound}.]
  \[
\text{Every ambient enlargement preserves and reflects existence of suprema.}
\]
\[
\begin{aligned}
&T=S',\qquad A\subseteq S,\qquad S\subseteq T,\\
&P_S=\mathsf{OrderedSet}(S,\leq),\qquad
P_T=\mathsf{OrderedSet}(T,\leq),\\
&\bigl[\exists s\in T\,\operatorname{LeastUpperBound}(s,A,P_T)
\Longrightarrow
\exists t\in S\,\operatorname{LeastUpperBound}(t,A,P_S)\bigr]
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Completeness is an ambient property. A larger ordered universe can fill a gap that remains absent in a smaller ordered universe.
\end{remark*}

\begin{lemmabox}{Lemma (Rational Gap Example for Suprema)}
\begin{lemma}[Rational Gap Example for Suprema]
\label{lem:rational-gap-suprema}
Let
\[
  A=\{q\in\mathbb{Q}:q^2<2\}.
\]
Then $A$ is nonempty and bounded above in $\mathbb{Q}$, but $A$ has no supremum relative to $\mathbb{Q}$.
\hyperref[prf:rational-gap-suprema]{Go to proof.}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
A=\{q\in\mathbb{Q}:q^2<2\}
\quad\Longrightarrow\quad
\sup_{\mathbb{Q}} A\text{ does not exist}.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P_{\mathbb{Q}}=\mathsf{OrderedSet}(\mathbb{Q},\leq),\qquad
A=\{q\in\mathbb{Q}:q^2<2\},\\
&\neg\exists s\in\mathbb{Q}\;\operatorname{LeastUpperBound}(s,A,P_{\mathbb{Q}}).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists s\in\mathbb{Q}\;(s=\sup_{\mathbb{Q}}A)
\]
is false for this set.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P_{\mathbb{Q}}=\mathsf{OrderedSet}(\mathbb{Q},\leq),\qquad
A=\{q\in\mathbb{Q}:q^2<2\},\\
&\exists s\in\mathbb{Q}\;\operatorname{LeastUpperBound}(s,A,P_{\mathbb{Q}})
\end{aligned}
\]
is false.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
A rational candidate below $\sqrt{2}$ fails to be an upper bound. A rational candidate above $\sqrt{2}$ is an upper bound but is not least, because another rational upper bound lies between it and $\sqrt{2}$.


  \item[\operatorname{LeastUpperBound}.]
  \[
\exists s\in\mathbb{Q}\;(s=\sup_{\mathbb{Q}}A)
\]
\[
\begin{aligned}
&P_{\mathbb{Q}}=\mathsf{OrderedSet}(\mathbb{Q},\leq),\qquad
A=\{q\in\mathbb{Q}:q^2<2\},\\
&\exists s\in\mathbb{Q}\;\operatorname{LeastUpperBound}(s,A,P_{\mathbb{Q}})
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
\begin{cases}
s^2<2 &\Longrightarrow s\text{ is not an upper bound of }A,\\
s^2>2 &\Longrightarrow s\text{ is not the least upper bound of }A.
\end{cases}
\qquad(s\in\mathbb{Q})
\]


  \item[\operatorname{LeastUpperBound}.]
  \[
\exists s\in\mathbb{Q}\;(s=\sup_{\mathbb{Q}}A)
\]
\[
\begin{aligned}
&P_{\mathbb{Q}}=\mathsf{OrderedSet}(\mathbb{Q},\leq),\qquad
A=\{q\in\mathbb{Q}:q^2<2\},\\
&\exists s\in\mathbb{Q}\;\operatorname{LeastUpperBound}(s,A,P_{\mathbb{Q}})
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
The set wants $\sqrt{2}$ as its least ceiling, but that number is not rational. The rational ambient set therefore has a visible gap where the supremum should be.
\end{remark*}
