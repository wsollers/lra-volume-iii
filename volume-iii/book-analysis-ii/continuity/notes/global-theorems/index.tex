\section{Global Theorems}

\begin{toolkitbox}{Global Theorems --- Quick Reference}
\small
\begin{tabular}{l l}
\toprule
\textbf{Name} & \textbf{Meaning}\\
\midrule
\hyperref[def:bounded-on-a-set]{Bounded on a set} & The output set is bounded in $\mathbb{R}$.\\
\hyperref[thm:boundedness-theorem]{Boundedness theorem} & Continuous functions on $[a,b]$ are bounded.\\
\hyperref[def:absolute-extrema]{Absolute extrema} & Global maximum and minimum values on a set.\\
\hyperref[thm:extreme-value-theorem]{Extreme value theorem} & Continuous functions on $[a,b]$ attain extrema.\\
\hyperref[thm:bolzano-intermediate-value]{Intermediate value theorem} & Continuous functions do not skip values.\\
\hyperref[thm:image-of-closed-bounded-interval]{Image interval theorem} & $f([a,b])$ is a closed bounded interval.\\
\hyperref[thm:darboux-property]{Darboux property} & Interval values are preserved without gaps.\\
