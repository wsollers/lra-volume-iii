\begin{theorem}[A Rational Between Any Two Irrationals]
\label{thm:rational-between-any-two-irrationals}
For every $x,y\in\mathbb{R}\setminus\mathbb{Q}$, if $x<y$, then there exists
$q\in\mathbb{Q}$ such that
\[
  x<q<y.
\]
\hyperref[prf:rational-between-any-two-irrationals]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall x,y\in\mathbb{R}\setminus\mathbb{Q}\;
\bigl(x<y\Longrightarrow \exists q\in\mathbb{Q}\;(x<q<y)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
x,y\in\mathbb{R}\setminus\mathbb{Q}\land x<y
\Longrightarrow \exists q\in\mathbb{Q}\;(x<q<y).
\]
\end{remark*}

\begin{remark*}[Interpretation]
This is the density of the rationals restricted to intervals whose endpoints
are irrational.
\end{remark*}
