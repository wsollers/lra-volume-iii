\begin{theorem*}[Maximum Implies Supremum]
If a set $A \subseteq \mathbb{R}$ has a maximum $m$, then $m$ is also the supremum of $A$.
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
Let $m = \max A$. By definition, $m \in A$ and $m$ is an upper bound of $A$. Let $u$ be any upper bound of $A$. Since $m \in A$ and $u$ is an upper bound, it follows by the definition of an upper bound that $m \le u$. Thus, $m$ is the least upper bound of $A$, so $m = \sup A$.
\end{proof}

\begin{proof}
\textbf{Detailed Learning Proof.}~\\

\textbf{Step 1. Establish the upper bound property.}
Assume $\operatorname{Maximum}(m, A)$. By the definition of a maximum:
\[ m \in A \quad \text{and} \quad \forall x \in A, x \le m \]
The second part of this statement satisfies the first requirement of being a supremum: $m$ is an upper bound of $A$.

\textbf{Step 2. Set up the "leastness" requirement.}
To show $m$ is the \textit{least} upper bound, we must show that any other upper bound $u$ must be greater than or equal to $m$. Let $u$ be an arbitrary upper bound of $A$.

\textbf{Step 3. Use set membership as the pivot.}
By the definition of an upper bound:
\[ \forall x \in A, x \le u \]
Because $m \in A$ (from Step 1), we can substitute $m$ for $x$:
\[ m \le u \]

\textbf{Step 4. Conclude by definition.}
We have shown that $m$ is an upper bound and that $m \le u$ for any upper bound $u$. By the definition of the supremum, $m = \sup A$.

\textbf{Step 5. Alternative via $\varepsilon$-characterization.}
Alternatively, let $\varepsilon > 0$ be arbitrary. We need to find $a \in A$ such that $m - \varepsilon < a$. Since $m \in A$ and $m - \varepsilon < m$, we can simply choose $a = m$. Thus, $m$ satisfies the $\varepsilon$-characterization of the supremum.
\end{proof}

\begin{remark*}[Proof structure]
This is a direct proof utilizing the definition of an upper bound. It leverages the fact that a maximum is an element of the set to force any other upper bound to be at least as large as that maximum.
\end{remark*}
