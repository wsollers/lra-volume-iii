\begin{lemmabox}{Lemma (Minimum of Positive Numbers Is Positive)}
\begin{lemma}[Minimum of Positive Numbers Is Positive]
\label{lem:minimum-of-positive-numbers-is-positive}
For every $a,b\in\mathbb{R}$,
\[
  (a>0\land b>0)
  \Longrightarrow
  \min\{a,b\}>0.
\]
\hyperref[prf:minimum-of-positive-numbers-is-positive]{Go to proof.}
\end{lemma}
\end{lemmabox}
\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{R}\;\bigl((a>0\land b>0)\Longrightarrow \min\{a,b\}>0\bigr).
\]
\end{remark*}
\begin{remark*}[Predicate reading]
\[
m=\min\{a,b\},\qquad
\operatorname{Positive}(a,\mathbb{R})\land\operatorname{Positive}(b,\mathbb{R})
\Longrightarrow \operatorname{Positive}(m,\mathbb{R}).
\]
\end{remark*}
\begin{remark*}[Negated quantified statement]
\[
\exists a,b\in\mathbb{R}\;\bigl(a>0\land b>0\land \min\{a,b\}\leq 0\bigr).
\]
\end{remark*}
\begin{remark*}[Negation predicate reading]
\[
m=\min\{a,b\},\qquad
\operatorname{Positive}(a,\mathbb{R})\land\operatorname{Positive}(b,\mathbb{R})\land
\neg\operatorname{Positive}(m,\mathbb{R}).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
  Failure is witnessed by the displayed negated or contrapositive condition.

  \item[\operatorname{Positive}.]
  \[
\exists a,b\in\mathbb{R}\;\bigl(a>0\land b>0\land \min\{a,b\}\leq 0\bigr).
\]
\[
m=\min\{a,b\},\qquad
\operatorname{Positive}(a,\mathbb{R})\land\operatorname{Positive}(b,\mathbb{R})\land
\neg\operatorname{Positive}(m,\mathbb{R}).
\]
\end{description}
\end{remark*}
\begin{remark*}[Interpretation]
Choosing the smaller of two positive bounds still leaves a positive bound.
\end{remark*}
