\begin{theorem*}[Least Upper Bound Property Implies Existence of Suprema]
Let $S$ be a partially ordered set. If $S$ has the Least Upper Bound Property, then every nonempty subset of $S$ that is bounded above in $S$ has a supremum in $S$.
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
Assume that $S$ has the Least Upper Bound Property. Let $A \subseteq S$ be nonempty and bounded above in $S$. By the Least Upper Bound Property, there exists an element $s \in S$ that is a least upper bound of $A$ in $S$. By definition, this means that $s$ is a supremum of $A$ in $S$. Hence
\[
\exists s \in S \; \operatorname{Supremum}(s,A).
\]
Therefore every nonempty subset of $S$ that is bounded above in $S$ has a supremum in $S$.
\end{proof}

\begin{proof}
\textbf{Detailed Learning Proof.}~\\

\textbf{Step 1.} Fix a subset to which the property applies.

Assume that $S$ has the Least Upper Bound Property. Let $A \subseteq S$ be nonempty and bounded above in $S$.

\textbf{Step 2.} Match the hypotheses on $A$ with the definition of the Least Upper Bound Property.

The Least Upper Bound Property says that every nonempty subset of $S$ that is bounded above in $S$ has a least upper bound in $S$. The subset $A$ satisfies these hypotheses.

\textbf{Step 3.} Apply the property.

By the Least Upper Bound Property, there exists $s \in S$ such that $s$ is a least upper bound of $A$.

\textbf{Step 4.} Translate to supremum.

By definition, this means $\operatorname{Supremum}(s,A)$.

\textbf{Step 5.} Conclude.

\[
\exists s \in S \; \operatorname{Supremum}(s,A).
\]
\end{proof}

\begin{remark*}[Proof structure]
Direct application of the Least Upper Bound Property followed by unpacking the definition of supremum.
\end{remark*}
