\begin{theorem*}[Supremum of a Sum Set]
Let $A, B \subseteq \mathbb{R}$ be non-empty sets bounded above. Define $A + B = \{a + b : a \in A, b \in B\}$. Then $\sup(A + B) = \sup A + \sup B$.
\end{theorem*}

\begin{proof}[Professional Standard Proof]
\LRAProofBodyStart
Let
\[
s_A=\sup A
\qquad\text{and}\qquad
s_B=\sup B.
\]
For every $a\in A$ and $b\in B$, the defining upper-bound property of the
supremum gives
\[
a\le s_A
\qquad\text{and}\qquad
b\le s_B.
\]
Hence
\[
a+b\le s_A+s_B.
\]
Thus $s_A+s_B$ is an upper bound for $A+B$.

It remains to show that no smaller number is an upper bound. Let
$\varepsilon>0$. By the epsilon characterization of the supremum, there exist
$a_\varepsilon\in A$ and $b_\varepsilon\in B$ such that
\[
s_A-\frac{\varepsilon}{2}<a_\varepsilon\le s_A
\qquad\text{and}\qquad
s_B-\frac{\varepsilon}{2}<b_\varepsilon\le s_B.
\]
Adding these inequalities gives
\[
s_A+s_B-\varepsilon
<
a_\varepsilon+b_\varepsilon
\le
s_A+s_B.
\]
Since $a_\varepsilon+b_\varepsilon\in A+B$, every positive epsilon places an
element of $A+B$ above $s_A+s_B-\varepsilon$. Therefore, by the epsilon
characterization of the supremum,
\[
\sup(A+B)=s_A+s_B=\sup A+\sup B.
\]
\end{proof}

\begin{proof}[Detailed Learning Proof]
\LRAProofBodyStart
Let
\[
s_A=\sup A
\qquad\text{and}\qquad
s_B=\sup B.
\]
The proof has two tasks: first show that $s_A+s_B$ is an upper bound for the
sum set, and then show that it is the least upper bound.

For the upper-bound part, take an arbitrary element of $A+B$. By definition of
the sum set, it has the form $a+b$ with $a\in A$ and $b\in B$. Since $s_A$ is
an upper bound for $A$ and $s_B$ is an upper bound for $B$,
\[
a\le s_A
\qquad\text{and}\qquad
b\le s_B.
\]
Adding the two inequalities gives
\[
a+b\le s_A+s_B.
\]
Because this holds for every $a+b\in A+B$, the number $s_A+s_B$ is an upper
bound for $A+B$.

For the least-upper-bound part, let $\varepsilon>0$. Since $s_A=\sup A$, the
epsilon characterization of the supremum gives an element
$a_\varepsilon\in A$ satisfying
\[
s_A-\frac{\varepsilon}{2}<a_\varepsilon\le s_A.
\]
Likewise, since $s_B=\sup B$, there is an element $b_\varepsilon\in B$ such
that
\[
s_B-\frac{\varepsilon}{2}<b_\varepsilon\le s_B.
\]
Adding these two estimates yields
\[
s_A+s_B-\varepsilon
<
a_\varepsilon+b_\varepsilon
\le
s_A+s_B.
\]
The element $a_\varepsilon+b_\varepsilon$ belongs to $A+B$. Thus for every
$\varepsilon>0$, the sum set contains an element larger than
$s_A+s_B-\varepsilon$ while $s_A+s_B$ remains an upper bound. This is exactly
the epsilon characterization that $s_A+s_B$ is the supremum of $A+B$.
Therefore
\[
\sup(A+B)=\sup A+\sup B.
\]
\end{proof}

\begin{remark*}[Proof structure]
The upper-bound half comes from adding $a\le\sup A$ and $b\le\sup B$. The
leastness half splits an arbitrary $\varepsilon$ into two halves, chooses
$a_\varepsilon$ and $b_\varepsilon$ within those errors of the separate
suprema, and then adds them to produce an element of $A+B$ within
$\varepsilon$ of $\sup A+\sup B$.
\end{remark*}
