\begin{theorem}[Decreasing Image Sends Infimum to Supremum]
\label{thm:decreasing-image-infimum-to-supremum}
Let $I\subseteq\mathbb{R}$ be an interval, let $A\subseteq I$ be nonempty and
bounded below, and suppose $i:=\inf A\in I$.  If $f\colon I\to\mathbb{R}$ is
decreasing and continuous at $i$ relative to $I$, then
$f[A]:=\{f(a):a\in A\}$ is bounded above, and
\[
  \sup f[A]=f(\inf A).
\]
\hyperref[prf:decreasing-image-infimum-to-supremum]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
A\neq\varnothing,\quad t=\inf A\in I,\quad f\text{ decreasing and continuous at }t
\Longrightarrow
\sup\{f(a):a\in A\}=f(t).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&A\subseteq I,\qquad f\colon I\to\mathbb{R},\qquad
B=f(A)=\{f(a):a\in A\},\\
&P_I=\mathsf{OrderedSet}(I,\leq),\qquad
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\\
&\operatorname{GreatestLowerBound}(t,A,P_I)\land
\operatorname{OrderReversing}(f,P_I,P_{\mathbb{R}})\land
\operatorname{IsContinuous}(f,t,I,\mathbb{R})
\Longrightarrow
\operatorname{LeastUpperBound}(f(t),B,P_{\mathbb{R}}).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\sup f(A)\neq f(\inf A)
\]
is impossible under the stated hypotheses.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&A\subseteq I,\qquad f\colon I\to\mathbb{R},\qquad
B=f(A)=\{f(a):a\in A\},\\
&P_I=\mathsf{OrderedSet}(I,\leq),\qquad
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\\
&\operatorname{GreatestLowerBound}(t,A,P_I)\land
\operatorname{OrderReversing}(f,P_I,P_{\mathbb{R}})\land
\operatorname{IsContinuous}(f,t,I,\mathbb{R})\land
\neg\operatorname{LeastUpperBound}(f(t),B,P_{\mathbb{R}}).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
The equality can fail if $f$ is not decreasing, if $\inf A$ is outside the
domain, or if $f$ is not continuous at the lower endpoint.


  \item[\operatorname{GreatestLowerBound} / \operatorname{OrderReversing}.]
  \[
\sup f(A)\neq f(\inf A)
\]
\[
\begin{aligned}
&A\subseteq I,\qquad f\colon I\to\mathbb{R},\qquad
B=f(A)=\{f(a):a\in A\},\\
&P_I=\mathsf{OrderedSet}(I,\leq),\qquad
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\\
&\operatorname{GreatestLowerBound}(t,A,P_I)\land
\operatorname{OrderReversing}(f,P_I,P_{\mathbb{R}})\land
\operatorname{IsContinuous}(f,t,I,\mathbb{R})\land
\neg\operatorname{LeastUpperBound}(f(t),B,P_{\mathbb{R}}).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Decreasingness gives $f(a)\leq f(t)$ for every $a\in A$.  Approximation of
$t$ from above and continuity at $t$ make this upper bound sharp.


  \item[\operatorname{GreatestLowerBound} / \operatorname{OrderReversing}.]
  \[
\sup f(A)\neq f(\inf A)
\]
\[
\begin{aligned}
&A\subseteq I,\qquad f\colon I\to\mathbb{R},\qquad
B=f(A)=\{f(a):a\in A\},\\
&P_I=\mathsf{OrderedSet}(I,\leq),\qquad
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\\
&\operatorname{GreatestLowerBound}(t,A,P_I)\land
\operatorname{OrderReversing}(f,P_I,P_{\mathbb{R}})\land
\operatorname{IsContinuous}(f,t,I,\mathbb{R})\land
\neg\operatorname{LeastUpperBound}(f(t),B,P_{\mathbb{R}}).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
A decreasing continuous function reverses endpoint roles: the lower endpoint
of the original set becomes the upper endpoint of the image.
\end{remark*}
