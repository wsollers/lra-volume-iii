% Section: Derivative Definition

\begin{remark*}[Section guide]
The derivative is the number the secant slopes approach, but the word
``limit'' has three useful faces. The epsilon--delta form is the contract;
the topological form strips it to neighbourhoods; the sequential form is the
disproving tool, since one bad approach sequence is enough.
\end{remark*}

\begin{definitionbox}{Definition (Derivative at a Point)}
\begin{definition}[Derivative at a Point]
\label{def:derivative-at-a-point}
Let $f: D \to \mathbb{R}$ be a real-valued function and $c \in \operatorname{int}(D)$. We say $f$ is \emph{differentiable at $c$} with \emph{derivative} $L = f'(c)$ if and only if:
\[
\forall \varepsilon > 0 \;\exists \delta > 0 \;\forall x \in D
\Bigl(0 < |x - c| < \delta \implies \left| \frac{f(x) - f(c)}{x - c} - L \right| < \varepsilon\Bigr).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\forall \varepsilon > 0 \;\exists \delta > 0 \;\forall x \in D
\Bigl(0 < |x - c| < \delta \implies \left|\frac{f(x)-f(c)}{x-c} - L\right| < \varepsilon\Bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{DerivativeAt}(f,c,L)
\;\Longleftrightarrow\;
\operatorname{FunctionLimit}\!\left(\lambda x.\,\frac{f(x)-f(c)}{x-c},\,D,\,c,\,L\right).
\]
The predicate $\operatorname{DerivativeAt}(f,c,L)$ asserts that $L$ is the value of the
limit of the difference quotient of $f$ at $c$.
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\forall L \in \mathbb{R}\;\exists \varepsilon > 0\;\forall \delta > 0\;\exists x \in D
\Bigl(0 < |x-c| < \delta \;\land\; \left|\frac{f(x)-f(c)}{x-c} - L\right| \ge \varepsilon\Bigr).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{DerivativeAt}(f,c,L)
\;\Longleftrightarrow\;
\neg\operatorname{FunctionLimit}\!\left(\lambda x.\,\frac{f(x)-f(c)}{x-c},\,D,\,c,\,L\right).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{itemize}
  \item The difference quotient has no finite limit at $c$ (e.g.\ $|f(x)-f(c)|/|x-c| \to \infty$).
  \item The difference quotient oscillates without settling near any single value.
  \item The proposed value $L$ is the wrong candidate.
\end{itemize}
\end{remark*}

\begin{remark*}[Interpretation]
The derivative is a limit of slopes of secant lines. The $\varepsilon$-$\delta$ form demands that the difference quotient can be made arbitrarily close to $L$ by taking $x$ sufficiently close to $c$. The input condition $0 < |x-c|$ is punctured: $x = c$ is excluded because $(f(c)-f(c))/(c-c)$ is $0/0$.
\end{remark*}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:limit-function]{Limit of a Function}
\end{itemize}
\end{dependencies}

\begin{example*}[Derivative of \(x^2\) from the Definition]
\phantomsection\label{ex:x2-derivative}
Let \(f(x)=x^2\) and fix \(c\in\mathbb{R}\). For \(x\neq c\),
\[
  \frac{f(x)-f(c)}{x-c}
  =
  \frac{x^2-c^2}{x-c}
  =
  x+c.
\]
The singularity has been removed by exact algebra, so the limit as
\(x\to c\) is \(2c\). Hence \(f'(c)=2c\). Read through
Carath\'{e}odory, this is the factor \(\varphi(x)=x+c\), continuous at
\(c\) with \(\varphi(c)=2c\).
\end{example*}

\begin{definitionbox}{Definition (Topological Definition of Derivative at a Point)}
\begin{definition}[Topological Definition of Derivative at a Point]
\label{def:topological-definition-of-derivative-at-a-point}
Let $f: D \to \mathbb{R}$ and $c \in \operatorname{int}(D)$. $f$ is differentiable at $c$ with derivative $L$ if for every neighbourhood $V$ of $L$, there exists a neighbourhood $U$ of $c$ such that for all $x \in (U \setminus \{c\}) \cap D$:
\[
\frac{f(x) - f(c)}{x - c} \in V.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\forall V \text{ neighbourhood of } L\;\exists U \text{ neighbourhood of } c\;
\forall x \in (U \setminus \{c\}) \cap D:\;
\frac{f(x)-f(c)}{x-c} \in V.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{DerivativeAt}_{\mathrm{top}}(f,c,L)
\;\Longleftrightarrow\;
\forall V \ni L\;\exists U \ni c\;\forall x \in (U\setminus\{c\})\cap D:\;
\frac{f(x)-f(c)}{x-c}\in V.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists V \text{ neighbourhood of } L\;\forall U \text{ neighbourhood of } c\;
\exists x \in (U \setminus \{c\}) \cap D:\;
\frac{f(x)-f(c)}{x-c} \notin V.
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{DerivativeAt}_{\mathrm{top}}(f,c,L)
\;\Longleftrightarrow\;
\exists V \ni L\;\forall U \ni c\;\exists x \in (U\setminus\{c\})\cap D:\;
\frac{f(x)-f(c)}{x-c}\notin V.
\]
\end{remark*}

\begin{remark*}[Failure modes]
The topological derivative fails when a fixed output neighbourhood of $L$ cannot be entered by the difference quotient no matter how tightly the input is restricted around $c$. Geometric reading: no punctured neighbourhood around $c$ maps all secant slopes into any prescribed window around $L$.
\end{remark*}

\begin{remark*}[Interpretation]
The topological form replaces $\varepsilon$-balls with abstract neighbourhoods, making the definition coordinate-free and suited for generalisation to metric spaces and manifolds.
\end{remark*}

\begin{definitionbox}{Definition (Sequential Definition of Derivative at a Point)}
\begin{definition}[Sequential Definition of Derivative at a Point]
\label{def:sequential-definition-of-derivative-at-a-point}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let $c \in A$ be a limit point of $A$. We say that $f$ is differentiable at $c$ with derivative $L$ if
\[
\forall (x_n) \subseteq A \setminus \{c\}\;
\bigl(x_n \to c \Rightarrow \tfrac{f(x_n)-f(c)}{x_n-c} \to L\bigr).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\forall (x_n) \subseteq A \setminus \{c\}:\;
x_n \to c \;\Rightarrow\; \frac{f(x_n)-f(c)}{x_n-c} \to L.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{DerivativeAt}_{\mathrm{seq}}(f,c,L)
\;\Longleftrightarrow\;
\forall (x_n)\subseteq A\setminus\{c\}:\;
x_n\to c \;\Rightarrow\; \frac{f(x_n)-f(c)}{x_n-c}\to L.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists (x_n) \subseteq A \setminus \{c\}:\;
x_n \to c \;\land\; \frac{f(x_n)-f(c)}{x_n-c} \not\to L.
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{DerivativeAt}_{\mathrm{seq}}(f,c,L)
\;\Longleftrightarrow\;
\exists (x_n)\subseteq A\setminus\{c\}:\;
x_n\to c \;\land\; \frac{f(x_n)-f(c)}{x_n-c}\not\to L.
\]
\end{remark*}

\begin{remark*}[Failure modes]
Failure occurs when a single approach sequence $(x_n) \to c$ exists along which the difference quotients either: (i) diverge to infinity (vertical tangent); (ii) oscillate without settling; (iii) converge but to a value other than $L$ (wrong slope candidate). A single such sequence certifies non-differentiability.
\end{remark*}

\begin{remark*}[Failure modes]
\[
\underbrace{x_n \to c}_{\text{approach condition}} \;\land\; \underbrace{\frac{f(x_n)-f(c)}{x_n-c} \not\to L}_{\text{quotient failure}}.
\]
The two canonical witness types are: two subsequences with different limiting difference quotients (e.g., $f(x) = |x|$ at $c = 0$), or quotients growing without bound.
\end{remark*}

\begin{remark*}[Interpretation]
The sequential form is the primary tool for disproving differentiability: exhibit a single sequence $x_n \to c$ along which the difference quotient fails to converge to any single value.
\end{remark*}

\begin{theorembox}{Theorem (Equivalence of Definitions of the Derivative at a Point)}
\begin{theorem}[Equivalence of Definitions of the Derivative at a Point]
\label{thm:derivative-equivalence}
Let $f : A \to \mathbb{R}$ and $c \in A$ be a limit point of $A$. The following are equivalent:
\begin{enumerate}
  \item $f$ is differentiable at $c$ in the $\varepsilon$-$\delta$ sense with derivative $L$.
  \item $f$ is differentiable at $c$ in the topological sense with derivative $L$.
  \item $f$ is differentiable at $c$ in the sequential sense with derivative $L$.
\end{enumerate}

\smallskip\noindent
\hyperref[prf:derivative-equivalence]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\text{(i) } \Longleftrightarrow \text{ (ii) } \Longleftrightarrow \text{ (iii).}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{DerivativeAt}(f,c,L)
\;\Longleftrightarrow\;
\operatorname{DerivativeAt}_{\mathrm{top}}(f,c,L)
\;\Longleftrightarrow\;
\operatorname{DerivativeAt}_{\mathrm{seq}}(f,c,L).
\]
All three predicates name the same underlying condition on the difference quotient of $f$ at $c$.
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
\neg\text{(i)} &\;\Longleftrightarrow\; \neg\text{(ii)} \;\Longleftrightarrow\; \neg\text{(iii)}\\
&\;\Longleftrightarrow\; \exists (x_n) \subseteq A \setminus \{c\}:\;
x_n \to c \;\land\; \frac{f(x_n)-f(c)}{x_n-c} \not\to L.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
The equivalence cannot produce contradictory verdicts: if $c$ is a limit point of $A$, all three forms agree. Degenerate case: if $c$ is not a limit point of $A$, the punctured-ball and sequence conditions are vacuously satisfied by every candidate $L$ simultaneously, making the equivalence trivially true for all $L$ rather than characterizing a unique derivative.
\end{remark*}

\begin{remark*}[Interpretation]
The three forms are notational restatements of the same underlying limit condition on the difference quotient. Analysts choose whichever form is most convenient for the proof at hand.
\end{remark*}

\begin{propositionbox}{Proposition (Derivative: $h$-Form Equivalence)}
\begin{proposition}[Derivative: $h$-Form Equivalence]
\label{prop:derivative-h-form-equivalence}
Let $f : A \to \mathbb{R}$ and $c \in \operatorname{int}(A)$. Then $f$ is differentiable at $c$ with derivative $L$ if and only if
\[
\lim_{h \to 0} \frac{f(c+h) - f(c)}{h} = L.
\]

\smallskip\noindent
\hyperref[prf:derivative-h-form-equivalence]{\textit{Go to proof.}}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{DerivativeAt}(f,c,L)
\;\Longleftrightarrow\;
\lim_{h\to 0}\frac{f(c+h)-f(c)}{h} = L.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\neg\operatorname{DerivativeAt}(f,c,L)
\;\Longleftrightarrow\;
\lim_{h\to 0}\frac{f(c+h)-f(c)}{h} \neq L.
\]
\end{remark*}

\begin{remark*}[Failure modes]
The $h$-form fails in two branches: (i) The incremental ratio $(f(c+h)-f(c))/h$ has no finite limit as $h \to 0$ (corner, cusp, or vertical tangent at $c$). (ii) The ratio converges to a finite value different from $L$ (proposed $L$ is the wrong slope).
\end{remark*}

\begin{remark*}[Interpretation]
The $h$-form substitutes $x = c + h$ in the difference quotient. It is the standard notation in calculus and makes the incremental structure of the derivative explicit.
\end{remark*}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:derivative-at-a-point]{Derivative at a Point}
\end{itemize}
\end{dependencies}

\begin{remark*}[Discrete and continuous change]
The forward difference operator
\[
  \Delta f(n)=f(n+1)-f(n)
\]
is the discrete analogue of the derivative. The \(h\)-form above is its
continuous descendant: replace the fixed step \(1\) by a shrinking step
\(h\), divide by \(h\), and pass to the limit.

The kinship matters, but so does the divergence. The discrete product rule
has an unavoidable shift:
\[
  \Delta(fg)(n)
  =
  (\Delta f)(n)g(n)+f(n+1)(\Delta g)(n).
\]
The continuous product rule loses this shift because \(f(x+h)\to f(x)\) as
\(h\to0\). Likewise, ordinary powers are not the clean basis for \(\Delta\);
falling factorials are:
\[
  \Delta\,x^{\underline{k}}=k\,x^{\underline{k-1}}.
\]
Passing to the limit is therefore not just a sharpening of a quotient. It
restores symmetries that the discrete operator must carry as shifts.
\end{remark*}

\subsection{First Consequences}

\begin{theorembox}{Theorem (Differentiable Implies Continuous)}
\begin{theorem}[Differentiable Implies Continuous]
\label{thm:differentiable-implies-continuous}
Let $f : A \to \mathbb{R}$ and $c \in A$. If $f$ is differentiable at $c$, then $f$ is continuous at $c$.

\smallskip\noindent
\hyperref[prf:differentiable-implies-continuous]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{DerivableAt}(f,c) \;\Rightarrow\; \operatorname{ContinuousAt}(f,c).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{DifferentiableAt}(f,c)
\;\Longrightarrow\;
\operatorname{ContinuousAt}(f,c).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The implication can only fail in the reverse direction: continuity at $c$
does not force differentiability at $c$.
\end{remark*}

\begin{remark*}[Failure modes]
The converse fails: $f(x) = |x|$ is continuous at $0$ but not differentiable there.
\end{remark*}

\begin{remark*}[Contrapositive quantified statement]
\[
\neg \text{ContinuousAt}(f,c) \;\Rightarrow\; \neg \text{DifferentiableAt}(f,c).
\]
\end{remark*}

\begin{remark*}[Contrapositive predicate reading]
\[
\neg\operatorname{ContinuousAt}(f,c)
\;\Longrightarrow\;
\neg\operatorname{DifferentiableAt}(f,c).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Differentiability is strictly stronger than continuity: the existence of a well-defined tangent slope forces the function values to coalesce at $f(c)$ as $x \to c$.
\end{remark*}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:derivative-at-a-point]{Derivative at a Point}
  \item \hyperref[def:continuous-at-point]{Continuity at a Point}
\end{itemize}
\end{dependencies}

\begin{remark*}[Converse failure]
The theorem runs one way only. Differentiability forces continuity, but
continuity does not force differentiability. The negation of the converse is
\[
  \exists f\;\exists c:\;
  \operatorname{ContinuousAt}(f,c)\land
  \neg\operatorname{DifferentiableAt}(f,c).
\]
The examples below escalate the failure: one corner, infinitely many isolated
corners, and no differentiability anywhere.
\end{remark*}

\begin{example*}[One Corner: \(|x|\)]
\phantomsection\label{ex:cont-not-diff-corner}
The function \(f(x)=|x|\) is continuous on \(\mathbb{R}\), but it is not
differentiable at \(0\). Its one-sided derivatives are \(-1\) and \(+1\), so
the two one-sided slopes do not agree. Thus a single corner is enough to
separate continuity from differentiability.
\end{example*}

\begin{example*}[Finitely Many Corners: A Sawtooth]
\phantomsection\label{ex:cont-not-diff-sawtooth}
On a compact interval, a triangular sawtooth made from finitely many line
segments can be continuous while failing to be differentiable at each corner.
For example,
\[
  s(x)=
  \begin{cases}
    x, & 0\leq x\leq 1,\\
    2-x, & 1\leq x\leq 2,
  \end{cases}
\]
is continuous on \([0,2]\), but the left derivative at \(1\) is \(1\) and the
right derivative is \(-1\). Adding more teeth gives any finite number of
corner failures without breaking continuity.
\end{example*}

\begin{example*}[The Weierstrass Function]
\phantomsection\label{ex:weierstrass}
For \(0<a<1\) and an odd integer \(b\) satisfying
\(ab>1+\frac{3}{2}\pi\), the Weierstrass function
\[
  W(x)=\sum_{n=0}^{\infty}a^n\cos(b^n\pi x)
\]
is continuous on \(\mathbb{R}\), because the series converges uniformly by
the Weierstrass \(M\)-test. Classically, under the displayed condition, it is
differentiable at no point. Thus continuity may hold everywhere while
differentiability holds nowhere.
\end{example*}

\begin{remark*}[Proof status]
The continuity of \(W\) follows from uniform convergence once the
function-series machinery is available. The nowhere-differentiability proof
is a classical result of Weierstrass and is deferred to the function-series
development; it is used here only as a counterexample to the converse of
Differentiable Implies Continuous.
\end{remark*}

\begin{remark*}[Forward flag]
Brownian motion has sample paths that are almost surely continuous
everywhere and differentiable nowhere. This is the stochastic analogue of the
Weierstrass phenomenon and one reason ordinary calculus must be replaced by
the It\^o integral in stochastic analysis.
\end{remark*}

\begin{theorembox}{Theorem (Uniqueness of the Derivative)}
\begin{theorem}[Uniqueness of the Derivative]
\label{thm:uniqueness-of-the-derivative}
Let $f : A \to \mathbb{R}$ and $c \in A$. If $f$ is differentiable at $c$, its derivative is unique.

\smallskip\noindent
\hyperref[prf:uniqueness-of-the-derivative]{\textit{Go to proof.}}
\hyperref[prf:uniqueness-of-the-derivative]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{DerivativeAt}(f,c,L) \;\land\; \operatorname{DerivativeAt}(f,c,L')
\;\Rightarrow\; L = L'.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{DerivativeAt}(f,c,L)
\;\land\;
\operatorname{DerivativeAt}(f,c,L')
\;\Longrightarrow\; L = L'.
\]
\end{remark*}

\begin{remark*}[Failure modes]
The uniqueness proof assumes $L \neq L'$ and takes $\varepsilon = |L - L'|/2 > 0$. The definition supplies $\delta$ making the difference quotient within $\varepsilon$ of both $L$ and $L'$ simultaneously, forcing $|L - L'| < 2\varepsilon = |L - L'|$, a contradiction. The cluster-point hypothesis on $c$ is essential: without it, the punctured-ball condition is vacuous and the limit is trivially satisfied by every value.
\end{remark*}

\begin{remark*}[Contrapositive quantified statement]
\[
L \neq L' \;\Rightarrow\; \neg\bigl(\operatorname{DerivativeAt}(f,c,L) \;\land\; \operatorname{DerivativeAt}(f,c,L')\bigr).
\]
Two distinct candidates cannot both be the derivative of $f$ at $c$.
\end{remark*}

\begin{remark*}[Contrapositive predicate reading]
\[
L \neq L'
\;\Longrightarrow\;
\neg\operatorname{DerivativeAt}(f,c,L)
\;\lor\;
\neg\operatorname{DerivativeAt}(f,c,L').
\]
\end{remark*}

\begin{remark*}[Interpretation]
Uniqueness licenses writing $f'(c)$ as a definite value rather than a set of candidates.
\end{remark*}

\begin{propositionbox}{Proposition (Derivative of the Identity)}
\begin{proposition}[Derivative of the Identity]
\label{prop:derivative-of-the-identity}
Let $f(x) = x$ for all $x \in \mathbb{R}$. Then $f'(c) = 1$ for every $c \in \mathbb{R}$.

\smallskip\noindent
\hyperref[prf:derivative-of-the-identity]{\textit{Go to proof.}}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
\forall c \in \mathbb{R}:\; \frac{d}{dx}[x]\big|_{x=c} = 1.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The identity function has slope $1$ everywhere. It is the base case for computing derivatives of polynomials via the algebra of derivatives.
\end{remark*}

\begin{propositionbox}{Proposition (Derivative of a Constant)}
\begin{proposition}[Derivative of a Constant]
\label{prop:derivative-of-a-constant}
Let $k\in\mathbb{R}$ and let $f(x)=k$ for all $x\in\mathbb{R}$. Then $f'(c)=0$ for every $c\in\mathbb{R}$.

\smallskip\noindent
\hyperref[prf:derivative-of-a-constant]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
  \forall k\in\mathbb{R}\,\forall c\in\mathbb{R}:\;
  \frac{d}{dx}[k]\big|_{x=c}=0.
\]
\end{remark*}

\begin{remark*}[Interpretation]
A constant function has zero change across every secant line, so its instantaneous rate of change is zero everywhere. Together with the derivative of the identity, this is one of the base cases for the algebraic computation rules.
\end{remark*}
