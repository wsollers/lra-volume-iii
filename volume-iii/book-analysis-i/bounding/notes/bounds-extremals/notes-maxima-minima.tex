\subsection{Maxima and Minima}

\begin{remark*}[Exposition]
Maxima and minima are attained sharp bounds.  They are stronger than
suprema and infima because the extremal element must belong to the set
itself.
\end{remark*}

  \caption{Maxima and minima are suprema and infima that are attained by the set.}

\begin{definitionbox}{Maximum}
\begin{definition}[Maximum]
\label{def:maximum}
Let $A\subseteq\mathbb{R}$.  An element $m\in A$ is the \emph{maximum}
of $A$, written $m=\max A$, if every element of $A$ is less than or equal
to $m$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  m\in A
  \quad\text{and}\quad
  (\forall a\in A)(a\leq m).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  A\subseteq\mathbb{R},\qquad P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
  \operatorname{GreatestElement}(m,A,P).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  m\notin A
  \quad\text{or}\quad
  (\exists a\in A)(m<a).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  A\subseteq\mathbb{R},\qquad P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
  \neg\operatorname{GreatestElement}(m,A,P).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
  Maximum failure means the candidate is not an attained upper bound of
  the set.

  \item[\operatorname{GreatestElement}.]
  The candidate is not an element of the set.
  \[
    m\notin A.
  \]
  \[
    P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
    \neg\operatorname{GreatestElement}(m,A,P).
  \]

  \item[\operatorname{UpperBound}.]
  Some element of the set lies strictly above the candidate.
  \[
    (\exists a\in A)(m<a).
  \]
  \[
    P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
    \neg\operatorname{UpperBound}(m,A,P).
  \]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
A maximum is an upper bound that is attained by the set itself.  The
attainment condition separates maximum from supremum.
\end{remark*}

\begin{remark*}[Empty-set convention]
The empty set has no maximum, since there is no element $m\in\varnothing$
that can attain the upper bound.
\end{remark*}

\begin{propositionbox}{Uniqueness of the Maximum}
\begin{proposition}[Uniqueness of the Maximum]
\label{prop:uniqueness-of-the-maximum}
Let $A\subseteq\mathbb{R}$.  If $m_{1},m_{2}\in A$ are both maxima of
$A$, then $m_{1}=m_{2}$.

\hyperref[prf:uniqueness-of-the-maximum]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{GreatestElement}(m_{1},A,P)
  \quad\text{and}\quad
  \operatorname{GreatestElement}(m_{2},A,P)
  \quad\Longrightarrow\quad
  m_{1}=m_{2}.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  A\subseteq\mathbb{R},\qquad P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
  \operatorname{GreatestElement}(m_{1},A,P)\land
  \operatorname{GreatestElement}(m_{2},A,P)
  \Longrightarrow m_{1}=m_{2}.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \operatorname{GreatestElement}(m_{1},A,P)
  \quad\text{and}\quad
  \operatorname{GreatestElement}(m_{2},A,P)
  \quad\text{and}\quad
  m_{1}\neq m_{2}.
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  A\subseteq\mathbb{R},\qquad P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
  \operatorname{GreatestElement}(m_{1},A,P)\land
  \operatorname{GreatestElement}(m_{2},A,P)\land m_{1}\neq m_{2}.
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
  Failure would require two distinct elements of $A$ each to dominate the
  entire set.

  \item[\operatorname{GreatestElement}.]
  Two different candidates both satisfy the maximum predicate.
  \[
    \operatorname{GreatestElement}(m_{1},A,P)\land
    \operatorname{GreatestElement}(m_{2},A,P)\land m_{1}\neq m_{2}.
  \]
  \[
    P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
    \operatorname{GreatestElement}(m_{1},A,P)\land
    \operatorname{GreatestElement}(m_{2},A,P)\land m_{1}\neq m_{2}.
  \]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Uniqueness says that attainment at the top of a real set cannot occur at
two distinct real elements.
\end{remark*}

\begin{propositionbox}{Maximum Implies Supremum}
\begin{proposition}[Maximum Implies Supremum]
\label{prop:maximum-implies-supremum}
Let $A\subseteq\mathbb{R}$.  If $m=\max A$, then $m=\sup A$.

\hyperref[prf:maximum-implies-supremum]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{GreatestElement}(m,A,P)
  \quad\Longrightarrow\quad
  \operatorname{LeastUpperBound}(m,A,P).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  A\subseteq\mathbb{R},\qquad P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
  \operatorname{GreatestElement}(m,A,P)\Longrightarrow
  \operatorname{LeastUpperBound}(m,A,P).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \operatorname{GreatestElement}(m,A,P)
  \quad\text{and}\quad
  \neg\operatorname{LeastUpperBound}(m,A,P).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  \operatorname{GreatestElement}(m,A,P)
  \quad\text{and}\quad
  \neg\operatorname{LeastUpperBound}(m,A,P).
\]

The negation predicate reading is the predicate-level form of the displayed negated statement.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
  Failure would require an attained upper bound that is not least among
  upper bounds.

  \item[\operatorname{GreatestElement} / \operatorname{LeastUpperBound}.]
  The candidate is a maximum but not the least upper bound.
  \[
    \operatorname{GreatestElement}(m,A,P)\land
    \neg\operatorname{LeastUpperBound}(m,A,P).
  \]
  \[
    P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
    \operatorname{GreatestElement}(m,A,P)\land
    \neg\operatorname{LeastUpperBound}(m,A,P).
  \]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
An attained upper bound is automatically least among upper bounds because
any upper bound must lie above the attained element.
\end{remark*}

\begin{propositionbox}{Supremum in the Set is the Maximum}
\begin{proposition}[Supremum in the Set is the Maximum]
\label{prop:supremum-in-the-set-is-the-maximum}
Let $A\subseteq\mathbb{R}$ and suppose $s=\sup A$.  If $s\in A$, then
$s=\max A$.

\hyperref[prf:supremum-in-the-set-is-the-maximum]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{LeastUpperBound}(s,A,P)
  \quad\text{and}\quad
  s\in A
  \quad\Longrightarrow\quad
  \operatorname{GreatestElement}(s,A,P).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  A\subseteq\mathbb{R},\qquad P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
  \operatorname{LeastUpperBound}(s,A,P)\land s\in A
  \Longrightarrow \operatorname{GreatestElement}(s,A,P).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \operatorname{LeastUpperBound}(s,A,P)
  \quad\text{and}\quad
  s\in A
  \quad\text{and}\quad
  \neg\operatorname{GreatestElement}(s,A,P).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  \operatorname{LeastUpperBound}(s,A,P)
  \quad\text{and}\quad
  s\in A
  \quad\text{and}\quad
  \neg\operatorname{GreatestElement}(s,A,P).
\]

The negation predicate reading is the predicate-level form of the displayed negated statement.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
  Failure would require a supremum that belongs to the set but is not its
  maximum.

  \item[\operatorname{LeastUpperBound} / \operatorname{GreatestElement}.]
  The candidate is a least upper bound, belongs to the set, and yet is not
  the greatest element.
  \[
    \operatorname{LeastUpperBound}(s,A,P)\land s\in A\land
    \neg\operatorname{GreatestElement}(s,A,P).
  \]
  \[
    P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
    \operatorname{LeastUpperBound}(s,A,P)\land s\in A\land
    \neg\operatorname{GreatestElement}(s,A,P).
  \]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
A supremum that belongs to the set is an attained upper bound, so it is the
maximum.
\end{remark*}
