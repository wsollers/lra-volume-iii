\begin{corollary*}[Nested Interval Intersection Contains the Endpoint Infimum]
Let $\{[a_n,b_n]\}_{n\in\mathbb N}$ be a nested sequence of nonempty closed
bounded intervals. If $\beta=\inf\{b_n:n\in\mathbb N\}$, then
$\beta\in \bigcap_{n=1}^{\infty}[a_n,b_n]$.
\end{corollary*}
