\section{Algebra of Summations}
\label{sec:summation-algebra}

\begin{toolkitbox}{Summation Algebra --- Quick Reference}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{@{}ll@{}}
\textbf{Linearity} & $\sum(af+bg) = a\sum f + b\sum g$ \\
\textbf{Splitting} & $\sum_a^c = \sum_a^b + \sum_{b+1}^c$ \\
\textbf{Index shift} & $\sum_{k=a}^{b} f(k) = \sum_{k=a+1}^{b+1} f(k-1)$ \\
\textbf{Constant sum} & $\sum_{k=a}^{b} c = (b-a+1)c$ \\
\textbf{Empty sum} & $\sum_{k=a}^{b} f(k) = 0$ when $a > b$ \\
\textbf{Nested sums} & $\sum_i\sum_j a_{ij} = \sum_j\sum_i a_{ij}$ (independent limits) \\
\end{tabular}
\end{toolkitbox}
