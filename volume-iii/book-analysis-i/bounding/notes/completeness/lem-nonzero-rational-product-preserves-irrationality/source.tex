\begin{lemma}[Nonzero Rational Product Preserves Irrationality]
\label{lem:nonzero-rational-product-preserves-irrationality}
For every $q\in\mathbb{Q}\setminus\{0\}$ and every
$x\in\mathbb{R}\setminus\mathbb{Q}$,
\[
  qx\in\mathbb{R}\setminus\mathbb{Q}.
\]
\hyperref[prf:nonzero-rational-product-preserves-irrationality]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\forall q\in\mathbb{Q}\setminus\{0\}\;\forall x\in\mathbb{R}\setminus\mathbb{Q}\;
\bigl(qx\in\mathbb{R}\setminus\mathbb{Q}\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
q\in\mathbb{Q}\setminus\{0\}\land x\in\mathbb{R}\setminus\mathbb{Q}
\Longrightarrow qx\in\mathbb{R}\setminus\mathbb{Q}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Multiplying an irrational by a nonzero rational cannot make it rational,
because division by the same nonzero rational would recover the original
irrational as a rational number.
\end{remark*}
