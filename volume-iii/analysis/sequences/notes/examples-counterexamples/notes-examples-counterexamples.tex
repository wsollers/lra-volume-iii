\subsection*{Examples and Counterexamples}

\begin{tcolorbox}[colback=gray!8, colframe=gray!40, title=\textbf{Sequence Examples Toolkit}]

\medskip
\textbf{Atomic items in this section:}
\begin{itemize}
  \item Example: Constant sequence
  \item Example: Reciprocal sequence
  \item Example: Alternating null sequence
  \item Example: Oscillating sequence
  \item Example: Geometric sequence
  \item Counterexample: Bounded does not imply convergent
  \item Counterexample: Successive differences tending to zero do not imply Cauchy
\end{itemize}

\begin{exposition}
Examples provide test objects for the definitions. A good sequence example
usually isolates one behavior: settling to a limit, oscillating between
several values, escaping to infinity, or satisfying a tempting condition that
is still too weak to force convergence.
\end{exposition}

\begin{example}[Constant Sequence]
\label{ex:sequence-example-constant}
Let $c\in\mathbb{R}$ and define $x_n=c$ for every $n\in\mathbb{N}$. Then
$(x_n)$ converges to $c$.
\end{example}

\begin{remark*}[Interpretation]
A constant sequence is the baseline example of convergence: every term is
already equal to the proposed limit.
\end{remark*}

\begin{example}[Reciprocal Sequence]
\label{ex:sequence-example-reciprocal}
Define $x_n=1/n$ for every $n\in\mathbb{N}$. Then $(x_n)$ is decreasing,
bounded below by $0$, and converges to $0$.
\end{example}

\begin{remark*}[Interpretation]
The reciprocal sequence is the standard positive null sequence. It is also
the model for estimates of the form ``eventually smaller than any fixed
positive tolerance.''
\end{remark*}

\begin{example}[Alternating Null Sequence]
\label{ex:sequence-example-alternating-null}
Define
\[
  x_n=\frac{(-1)^n}{n}
  \qquad
  \text{for every } n\in\mathbb{N}.
\]
Then $(x_n)$ is not monotone, but it converges to $0$.
\end{example}

\begin{remark*}[Interpretation]
Oscillation in sign does not by itself prevent convergence. The amplitudes
must be watched: here the oscillation collapses to zero.
\end{remark*}

\begin{example}[Oscillating Sequence]
\label{ex:sequence-example-oscillating}
Define
\[
  x_n=(-1)^n
  \qquad
  \text{for every } n\in\mathbb{N}.
\]
Then $(x_n)$ is bounded and divergent. Its even-indexed subsequence converges
to $1$, and its odd-indexed subsequence converges to $-1$.
\end{example}

\begin{remark*}[Interpretation]
This is the canonical oscillator. The sequence never settles because two
subsequences settle to different limits.
\end{remark*}

\begin{example}[Geometric Sequence]
\label{ex:sequence-example-geometric}
Let $r\in\mathbb{R}$ and define $x_n=r^n$ for every $n\in\mathbb{N}$.
Then:
\begin{itemize}
  \item if $|r|<1$, then $x_n\to 0$;
  \item if $r=1$, then $x_n\to 1$;
  \item if $r=-1$, then $(x_n)$ oscillates and diverges;
  \item if $|r|>1$, then $(x_n)$ is unbounded.
\end{itemize}
\end{example}

\begin{remark*}[Interpretation]
The geometric sequence is the standard laboratory for rate, sign, and
magnitude. The threshold $|r|=1$ separates decay from growth.
\end{remark*}

\begin{example}[Bounded Does Not Imply Convergent]
\label{ex:sequence-counterexample-bounded-not-convergent}
The sequence $x_n=(-1)^n$ is bounded, since $|x_n|\le 1$ for every
$n\in\mathbb{N}$, but it does not converge.
\end{example}

\begin{remark*}[Interpretation]
Boundedness prevents escape to infinity, but it does not prevent persistent
oscillation.
\end{remark*}

\begin{example}[Vanishing Successive Differences Do Not Imply Cauchy]
\label{ex:sequence-counterexample-successive-differences-not-cauchy}
Let
\[
  x_n=1+\frac{1}{2}+\cdots+\frac{1}{n}
  \qquad
  \text{for every } n\in\mathbb{N}.
\]
Then
\[
  x_{n+1}-x_n=\frac{1}{n+1}\to 0,
\]
but $(x_n)$ is not Cauchy.
\end{example}

\begin{remark*}[Interpretation]
Cauchy control requires all sufficiently late terms to be mutually close.
Controlling only adjacent jumps is weaker.
\end{remark*}
