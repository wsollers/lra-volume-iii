\subsection{Higher Differences}
\label{subsec:higher-differences}

\begin{definition}[Higher Differences]
\label{def:higher-differences}
The \emph{second difference} of \(f\) is defined by
\[
\Delta^2 f(n) = \Delta(\Delta f(n)).
\]
More generally, the \(k\)-th difference \(\Delta^k f(n)\) denotes the
\(k\)-fold application of the difference operator, defined recursively by
\[
\Delta^0 f = f, \qquad \Delta^k f = \Delta(\Delta^{k-1} f).
\]
\end{definition}

\begin{remark*}[Fully quantified form]
\(\forall k \in \mathbb{N},\; \forall f : \mathbb{Z}\to\mathbb{R},\;
\forall n \in \mathbb{Z}:\;
\Delta^k f(n) := (\underbrace{\Delta \circ \cdots \circ \Delta}_{k})\,f(n)\).
\end{remark*}

\begin{remark*}[Interpretation]
Higher differences measure repeated one-step change, just as higher
derivatives measure repeated infinitesimal change.
\end{remark*}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:forward-difference]{Forward Difference Operator}.
\end{itemize}
\end{dependencies}

\subsection*{Example}

Let \(f(n) = n^2\). Then
\[
\begin{aligned}
\Delta f(n) &= 2n + 1,\\
\Delta^2 f(n) &= \Delta(2n+1) = 2(n+1)+1 - (2n+1) = 2,\\
\Delta^3 f(n) &= \Delta(2) = 0.
\end{aligned}
\]

\begin{remark*}[Analogy with higher derivatives]
Higher differences behave analogously to higher derivatives.
For polynomial functions of degree \(d\), the \(d\)-th difference is
constant and all higher differences are zero --- just as the \(d\)-th
derivative of a degree-\(d\) polynomial is constant and the next
derivative vanishes.
\end{remark*}
