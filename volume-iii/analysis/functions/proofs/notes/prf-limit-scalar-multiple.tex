\begin{theorem*}[Scalar Multiple Rule for Function Limits]
Let $f : A \to \mathbb{R}$, let $c$ be a cluster point of $A$, and let
$\alpha \in \mathbb{R}$. If $\lim_{x\to c}f(x)=L$, then
$\lim_{x\to c}(\alpha f)(x) = \alpha L$.
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
Let $(x_n) \subseteq A \setminus \{c\}$ with $x_n \to c$.
By the Sequential Criterion, $f(x_n) \to L$.
By the scalar multiple rule for sequence limits, $(\alpha f)(x_n) = \alpha f(x_n) \to \alpha L$.
Since $(x_n)$ was arbitrary, the Sequential Criterion gives
$\lim_{x\to c}(\alpha f)(x) = \alpha L$.
\end{proof}

\begin{proof}
\textbf{Detailed Learning Proof.}~\\
\textbf{Step 1. Set up a test sequence.}
Let $(x_n) \subseteq A \setminus \{c\}$ be any sequence with $x_n \to c$.

\textbf{Step 2. Apply the Sequential Criterion.}
Since $\lim_{x\to c}f(x)=L$, the Sequential Criterion gives $f(x_n) \to L$.

\textbf{Step 3. Apply the scalar multiple rule for sequences.}
The algebra of sequence limits gives
\[
  (\alpha f)(x_n) = \alpha f(x_n) \to \alpha L.
\]

\textbf{Step 4. Lift back by the Sequential Criterion.}
Since every such $(x_n)$ satisfies $(\alpha f)(x_n) \to \alpha L$, the
Sequential Criterion yields $\lim_{x\to c}(\alpha f)(x) = \alpha L$.
\end{proof}

\begin{remark*}[Proof structure]
Bridge argument. The case $\alpha = 0$ is trivial (constant zero function);
the case $\alpha \neq 0$ follows from the scalar multiple rule for sequences.
\end{remark*}
