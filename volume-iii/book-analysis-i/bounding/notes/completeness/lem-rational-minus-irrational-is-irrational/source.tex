\begin{lemma}[Rational Minus Irrational Is Irrational]
\label{lem:rational-minus-irrational-is-irrational}
For every $q\in\mathbb{Q}$ and every $x\in\mathbb{R}\setminus\mathbb{Q}$,
\[
  q-x\in\mathbb{R}\setminus\mathbb{Q}.
\]
\hyperref[prf:rational-minus-irrational-is-irrational]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\forall q\in\mathbb{Q}\;\forall x\in\mathbb{R}\setminus\mathbb{Q}\;
\bigl(q-x\in\mathbb{R}\setminus\mathbb{Q}\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
q\in\mathbb{Q}\land x\in\mathbb{R}\setminus\mathbb{Q}
\Longrightarrow q-x\in\mathbb{R}\setminus\mathbb{Q}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
If $q-x$ were rational, then $x=q-(q-x)$ would be rational, contradicting the
hypothesis on $x$.
\end{remark*}
