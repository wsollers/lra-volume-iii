\begin{proposition*}[Integer Interval as a Finite Set]
Let \(n \le m\) be integers, and let
\[
  X:=\{i\in\mathbb{Z}:n\le i\le m\}.
\]
If \(a_i\) is a real number assigned to each integer \(i\in X\), then
\[
  \sum_{i=n}^{m} a_i
  =
  \sum_{i\in X} a_i.
\]
\end{proposition*}
