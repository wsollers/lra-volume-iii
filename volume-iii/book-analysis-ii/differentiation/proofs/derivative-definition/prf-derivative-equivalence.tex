\begin{theorem*}[Equivalence of Definitions of the Derivative at a Point]
Let $f : A \to \mathbb{R}$ and $c \in A$ be a limit point of $A$. The following are equivalent:
\begin{enumerate}
  \item $f$ is differentiable at $c$ in the $\varepsilon$-$\delta$ sense with derivative $L$.
  \item $f$ is differentiable at $c$ in the topological sense with derivative $L$.
  \item $f$ is differentiable at $c$ in the sequential sense with derivative $L$.
\end{enumerate}
\end{theorem*}
