\begin{lemma}[Irrational Minus Rational Is Irrational]
\label{lem:irrational-minus-rational-is-irrational}
For every $x\in\mathbb{R}\setminus\mathbb{Q}$ and every $q\in\mathbb{Q}$,
\[
  x-q\in\mathbb{R}\setminus\mathbb{Q}.
\]
\hyperref[prf:irrational-minus-rational-is-irrational]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\forall x\in\mathbb{R}\setminus\mathbb{Q}\;\forall q\in\mathbb{Q}\;
\bigl(x-q\in\mathbb{R}\setminus\mathbb{Q}\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
x\in\mathbb{R}\setminus\mathbb{Q}\land q\in\mathbb{Q}
\Longrightarrow x-q\in\mathbb{R}\setminus\mathbb{Q}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Subtracting a rational is the same as adding its rational additive inverse, so
an irrational remains irrational.
\end{remark*}
