\begin{theorem*}[Convergence iff Liminf Equals Limsup]
Let $(x_n)$ be a bounded real sequence, and let $L\in\mathbb{R}$. Then
$x_n\to L$ if and only if
\[
  \liminf_{n\to\infty}x_n
  =
  \limsup_{n\to\infty}x_n
  =
  L.
\]

\smallskip
\noindent
\end{theorem*}
