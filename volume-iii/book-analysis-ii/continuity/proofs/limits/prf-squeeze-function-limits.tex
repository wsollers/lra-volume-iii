\begin{theorem*}[Squeeze Theorem for Function Limits]
Let $f, g, h : A \to \mathbb{R}$ and let $c$ be a cluster point of
$A$. Suppose $\lim_{x\to c}f(x)=L =
\lim_{x\to c}g(x)=L$, and suppose there exists
$\delta_0 > 0$ such that
\[
  \forall x \in A,\;
  0 < |x - c| < \delta_0
  \;\Rightarrow\;
  f(x) \leq h(x) \leq g(x).
\]
Then $\lim_{x\to c}h(x)=L$.

\smallskip\noindent
\end{theorem*}
