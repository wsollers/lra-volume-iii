\begin{lemma}[Irrational Plus Rational Is Irrational]
\label{lem:irrational-plus-rational-is-irrational}
For every $x\in\mathbb{R}\setminus\mathbb{Q}$ and every $q\in\mathbb{Q}$,
\[
  x+q\in\mathbb{R}\setminus\mathbb{Q}.
\]
\hyperref[prf:irrational-plus-rational-is-irrational]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\forall x\in\mathbb{R}\setminus\mathbb{Q}\;\forall q\in\mathbb{Q}\;
\bigl(x+q\in\mathbb{R}\setminus\mathbb{Q}\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
x\in\mathbb{R}\setminus\mathbb{Q}\land q\in\mathbb{Q}
\Longrightarrow x+q\in\mathbb{R}\setminus\mathbb{Q}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
This is the commuted form of rational translation preserving irrationality.
\end{remark*}
