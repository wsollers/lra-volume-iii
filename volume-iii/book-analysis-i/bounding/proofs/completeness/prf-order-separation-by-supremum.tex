\begin{theorem*}[Order Separation by Supremum]
Let $A,B\subseteq\mathbb{R}$ be nonempty sets such that $a\le b$ for every
$a\in A$ and every $b\in B$. Then there exists $c\in\mathbb{R}$ such that
$a\le c\le b$ for every $a\in A$ and every $b\in B$.
\end{theorem*}
