\section{Monotone Functions}

\begin{toolkitbox}{Monotone Functions --- Quick Reference}
\small
\begin{tabular}{l l}
\toprule
\textbf{Concept/Statement} & \textbf{Meaning/Role}\\
\midrule
\hyperref[thm:monotone-one-sided-limits]{One-sided limits} & Monotone functions have finite or infinite side limits.\\
\hyperref[cor:monotone-continuity-criterion]{Continuity criterion} & Continuity means both side limits equal the value.\\
\hyperref[def:jump-of-function]{Jump} & The size of the monotone discontinuity.\\
\hyperref[prop:monotone-discontinuities-first-kind]{First-kind gaps} & Monotone discontinuities are jumps.\\
\hyperref[thm:monotone-discontinuities-countable]{Countability} & Only countably many jumps can occur.\\
\hyperref[thm:continuous-inverse-theorem]{Continuous inverse} & Strict monotonicity gives a continuous inverse.\\
