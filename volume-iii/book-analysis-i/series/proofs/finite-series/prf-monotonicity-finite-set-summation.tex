\begin{proposition*}[Monotonicity for Finite Set Summation]
Let \(X\) be a finite set, and let \(f:X\to\mathbb{R}\) and
\(g:X\to\mathbb{R}\) be functions such that \(f(x)\le g(x)\) for all
\(x\in X\). Then
\[
  \sum_{x\in X} f(x)
  \le
  \sum_{x\in X} g(x).
\]
\end{proposition*}
