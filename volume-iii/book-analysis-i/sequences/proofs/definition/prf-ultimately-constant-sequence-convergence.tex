\begin{theorem*}[Ultimately Constant Sequence Convergence]
Let $(x_n)$ be a real sequence. If there exist $N_0 \in \mathbb{N}$ and
$c \in \mathbb{R}$ such that $x_n = c$ for every $n \ge N_0$, then
$(x_n)$ converges to $c$.

\noindent
\end{theorem*}
