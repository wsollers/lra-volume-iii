\begin{theorem}[Increasing Inverse Preserves Infima]
\label{thm:increasing-inverse-preserves-infima}
Let $I,J\subseteq\mathbb{R}$ be intervals, and let $f\colon I\to J$ be a continuous
strictly increasing bijection.  Let $B\subseteq J$ be nonempty and bounded
below, and suppose $\inf B\in J$.  Then
$f^{-1}[B]:=\{f^{-1}(b):b\in B\}$ is bounded below, and
\[
  \inf f^{-1}[B]=f^{-1}(\inf B).
\]
\hyperref[prf:increasing-inverse-preserves-infima]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
B\neq\varnothing,\quad \inf B\in J,\quad f^{-1}\text{ increasing and continuous at }\inf B
\Longrightarrow
\inf f^{-1}(B)=f^{-1}(\inf B).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&B\subseteq J,\qquad g=f^{-1},\qquad g\colon J\to I,\qquad
C=g(B)=\{g(b):b\in B\},\\
&P_J=\mathsf{OrderedSet}(J,\leq),\qquad
P_I=\mathsf{OrderedSet}(I,\leq),\\
&\operatorname{GreatestLowerBound}(v,B,P_J)\land
\operatorname{OrderPreserving}(g,P_J,P_I)\land
\operatorname{IsContinuous}(g,v,J,I)
\Longrightarrow
\operatorname{GreatestLowerBound}(g(v),C,P_I).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\inf f^{-1}(B)\neq f^{-1}(\inf B)
\]
is impossible under the stated hypotheses.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&B\subseteq J,\qquad g=f^{-1},\qquad g\colon J\to I,\qquad
C=g(B)=\{g(b):b\in B\},\\
&P_J=\mathsf{OrderedSet}(J,\leq),\qquad
P_I=\mathsf{OrderedSet}(I,\leq),\\
&\operatorname{GreatestLowerBound}(v,B,P_J)\land
\operatorname{OrderPreserving}(g,P_J,P_I)\land
\operatorname{IsContinuous}(g,v,J,I)\land
\neg\operatorname{GreatestLowerBound}(g(v),C,P_I).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
The formula can fail if the inverse is not increasing, if $\inf B$ is outside
the codomain set $J$, or if the inverse is not continuous at $\inf B$.


  \item[\operatorname{GreatestLowerBound} / \operatorname{OrderPreserving}.]
  \[
\inf f^{-1}(B)\neq f^{-1}(\inf B)
\]
\[
\begin{aligned}
&B\subseteq J,\qquad g=f^{-1},\qquad g\colon J\to I,\qquad
C=g(B)=\{g(b):b\in B\},\\
&P_J=\mathsf{OrderedSet}(J,\leq),\qquad
P_I=\mathsf{OrderedSet}(I,\leq),\\
&\operatorname{GreatestLowerBound}(v,B,P_J)\land
\operatorname{OrderPreserving}(g,P_J,P_I)\land
\operatorname{IsContinuous}(g,v,J,I)\land
\neg\operatorname{GreatestLowerBound}(g(v),C,P_I).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Apply the increasing image infimum theorem to $g=f^{-1}$ and the set $B$.


  \item[\operatorname{GreatestLowerBound} / \operatorname{OrderPreserving}.]
  \[
\inf f^{-1}(B)\neq f^{-1}(\inf B)
\]
\[
\begin{aligned}
&B\subseteq J,\qquad g=f^{-1},\qquad g\colon J\to I,\qquad
C=g(B)=\{g(b):b\in B\},\\
&P_J=\mathsf{OrderedSet}(J,\leq),\qquad
P_I=\mathsf{OrderedSet}(I,\leq),\\
&\operatorname{GreatestLowerBound}(v,B,P_J)\land
\operatorname{OrderPreserving}(g,P_J,P_I)\land
\operatorname{IsContinuous}(g,v,J,I)\land
\neg\operatorname{GreatestLowerBound}(g(v),C,P_I).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
The inverse preserves lower endpoints under the same increasing and continuity
hypotheses required for the direct image theorem.
\end{remark*}
