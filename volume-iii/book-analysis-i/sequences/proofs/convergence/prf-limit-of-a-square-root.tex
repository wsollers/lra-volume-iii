\begin{theorem*}[Limit of a Square Root]
Let $(x_n)$ be a real sequence such that $0\le x_n$ for every
$n\in\mathbb{N}$, and let $L\in\mathbb{R}$. If $x_n\to L$, then
$0\le L$ and $\sqrt{x_n}\to\sqrt{L}$.

\smallskip
\noindent
\end{theorem*}
