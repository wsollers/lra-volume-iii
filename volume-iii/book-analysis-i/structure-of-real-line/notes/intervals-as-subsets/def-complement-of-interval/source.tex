\begin{definitionbox}{Definition (Complement of an Interval)}
\begin{definition}[Complement of an Interval]
\label{def:complement-of-interval}
Let $I\subseteq\mathbb{R}$ be an interval. Its \emph{complement} in
$\mathbb{R}$ is the set
\[
  I^{c}:=\mathbb{R}\setminus I
  =
  \{x\in\mathbb{R}:x\notin I\}.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
I^c=\{x\in\mathbb{R}:x\notin I\}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The complement records every real point excluded by the interval. Endpoint
membership reverses: an endpoint included in $I$ is excluded from $I^c$, and an
endpoint excluded from $I$ is included in $I^c$.
\end{remark*}
