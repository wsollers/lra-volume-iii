\begin{definitionbox}{Definition (Epsilon Characterization of Supremum)}
\begin{definition}[Epsilon Characterization of Supremum]
\label{def:epsilon-characterization-of-supremum}
Let $A\subseteq\mathbb{R}$ be nonempty and bounded above, and let
$s\in\mathbb{R}$.  Then $s=\sup A$ if and only if $s$ is an upper bound of
$A$ and for every $\varepsilon>0$ there exists $a\in A$ such that
$s-\varepsilon<a$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } A\subseteq\mathbb{R} \text{ be nonempty and bounded above, and let } s\in\mathbb{R}.\\
&s=\sup A\\
&\Longleftrightarrow
\Bigl[\forall a\in A\;(a\le s)\Bigr]
\land
\Bigl[\forall \varepsilon>0\;\exists a\in A\;(s-\varepsilon<a)\Bigr].
\end{aligned}
\]
\[
s=\sup A
\Longleftrightarrow
\left[
(\forall a\in A)(a\leq s)
\land
(\forall\varepsilon>0)(\exists a\in A)(s-\varepsilon<a\leq s)
\right].
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\\
&\operatorname{EpsilonCharacterizationOfSupremum}(s,A,P)
\Longleftrightarrow
\operatorname{LeastUpperBound}(s,A,P)
\Longleftrightarrow
\operatorname{UpperBound}(s,A,P)
\land
(\forall \varepsilon>0)(\exists a\in A)(s-\varepsilon<a\leq s).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let } A\subseteq\mathbb{R} \text{ and } s\in\mathbb{R}.\\
&s \text{ does not satisfy the epsilon characterization of } \sup A\\
&\Longleftrightarrow
\Bigl[\exists x\in A\;(s<x)\Bigr]
\lor
\Bigl[\exists \varepsilon_0>0\;\forall a\in A\;(a\le s-\varepsilon_0)\Bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\\
&\neg\operatorname{EpsilonCharacterizationOfSupremum}(s,A,P)\Longleftrightarrow
\Bigl[\exists x\in A\;(s<x)\Bigr]
\lor
\Bigl[\exists \varepsilon_0>0\;\forall a\in A\;(a\le s-\varepsilon_0)\Bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
The characterization fails either because $s$ is not an upper bound of $A$, or because $A$ remains at least some positive distance below $s$.


  \item[\operatorname{EpsilonCharacterizationOfSupremum}.]
  \[
\begin{aligned}
&\text{Let } A\subseteq\mathbb{R} \text{ and } s\in\mathbb{R}.\\
&s \text{ does not satisfy the epsilon characterization of } \sup A\\
&\Longleftrightarrow
\Bigl[\exists x\in A\;(s<x)\Bigr]
\lor
\Bigl[\exists \varepsilon_0>0\;\forall a\in A\;(a\le s-\varepsilon_0)\Bigr].
\end{aligned}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\\
&\neg\operatorname{EpsilonCharacterizationOfSupremum}(s,A,P)\Longleftrightarrow
\Bigl[\exists x\in A\;(s<x)\Bigr]
\lor
\Bigl[\exists \varepsilon_0>0\;\forall a\in A\;(a\le s-\varepsilon_0)\Bigr].
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
\neg\operatorname{EpsilonCharacterizationOfSupremum}(s,A,\mathbb{R})
=
\underbrace{\exists x\in A\;(s<x)}_{\text{upper-bound failure}}
\;\lor\;
\underbrace{\exists \varepsilon_0>0\;\forall a\in A\;(a\le s-\varepsilon_0)}_{\text{positive gap below }s}.
\]


  \item[\operatorname{EpsilonCharacterizationOfSupremum}.]
  \[
\begin{aligned}
&\text{Let } A\subseteq\mathbb{R} \text{ and } s\in\mathbb{R}.\\
&s \text{ does not satisfy the epsilon characterization of } \sup A\\
&\Longleftrightarrow
\Bigl[\exists x\in A\;(s<x)\Bigr]
\lor
\Bigl[\exists \varepsilon_0>0\;\forall a\in A\;(a\le s-\varepsilon_0)\Bigr].
\end{aligned}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\\
&\neg\operatorname{EpsilonCharacterizationOfSupremum}(s,A,P)\Longleftrightarrow
\Bigl[\exists x\in A\;(s<x)\Bigr]
\lor
\Bigl[\exists \varepsilon_0>0\;\forall a\in A\;(a\le s-\varepsilon_0)\Bigr].
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
This condition says that $s$ caps the set from above while points of $A$ approach $s$ from below at every positive scale. The upper-bound clause prevents $A$ from crossing above $s$, and the epsilon clause prevents $s$ from floating strictly above the set. Geometrically, $s$ is a ceiling that the set presses against arbitrarily closely.
\end{remark*}

\begin{definitionbox}{Definition (Epsilon Characterization of Infimum)}
\begin{definition}[Epsilon Characterization of Infimum]
\label{def:epsilon-characterization-of-infimum}
Let $A\subseteq\mathbb{R}$ be nonempty and bounded below, and let
$i\in\mathbb{R}$.  Then $i=\inf A$ if and only if $i$ is a lower bound of
$A$ and for every $\varepsilon>0$ there exists $a\in A$ such that
$a<i+\varepsilon$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } A\subseteq\mathbb{R} \text{ be nonempty and bounded below, and let } i\in\mathbb{R}.\\
&i=\inf A\\
&\Longleftrightarrow
\Bigl[\forall a\in A\;(i\le a)\Bigr]
\land
\Bigl[\forall \varepsilon>0\;\exists a\in A\;(a<i+\varepsilon)\Bigr].
\end{aligned}
\]
\[
i=\inf A
\Longleftrightarrow
\left[
(\forall a\in A)(i\leq a)
\land

\begin{remark*}[Interpretation]
This condition says that $i$ sits below the set while points of $A$ approach $i$ from above at every positive scale. The lower-bound clause prevents $A$ from crossing below $i$, and the epsilon clause prevents $i$ from sitting strictly below the set with a positive gap. Geometrically, $i$ is a floor that the set presses against arbitrarily closely from above.


The epsilon characterizations are one way to prove the sharpness part of a
supremum or infimum claim.  After proving the candidate is an upper or lower
bound, one can either compare it with arbitrary bounds directly or build
near-extremal points using an arbitrary $\varepsilon>0$.  The proof-techniques
chapter gives a fuller guide to choosing between these approaches:
\hyperref[subsec:eps-char]{The $\varepsilon$-Characterisation of Suprema and
Infima}.
\end{remark*}

\begin{lemmabox}{Lemma (Supremum Strict Upper Approximation)}
\begin{lemma}[Supremum Strict Upper Approximation]
\label{lem:supremum-strict-upper-approximation}
Let $A\subseteq\mathbb{R}$ be nonempty and bounded above, and let
$s=\sup A$.  For every $x\in\mathbb{R}$ with $x<s$, there exists
$a\in A$ such that
\[
x<a\le s.
\]
\hyperref[prf:supremum-strict-upper-approximation]{Go to proof.}
\end{lemma}
\end{lemmabox}
\begin{remark*}[Standard quantified statement]
\[
\forall x\in\mathbb{R},\qquad
x<\sup A
\Rightarrow
(\exists a\in A)(x<a\leq\sup A).
\]
\end{remark*}
\begin{remark*}[Predicate reading]
\[
x<\sup A\Rightarrow A\cap(x,\sup A]\ne\varnothing.
\]
\end{remark*}
\begin{remark*}[Negated quantified statement]
\[
\exists A\subseteq\mathbb{R}\;\exists x,s\in\mathbb{R}\;
\bigl(A\ne\varnothing\land s=\sup A\land x<s\land
\forall a\in A\;(a\le x\lor s<a)\bigr).
\]
\end{remark*}
\begin{remark*}[Negation predicate reading]
\[
x<\sup A\land A\cap(x,\sup A]=\varnothing.
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
  Failure is witnessed by the displayed negated or contrapositive condition.

  \item[\varnothing.]
  \[
\exists A\subseteq\mathbb{R}\;\exists x,s\in\mathbb{R}\;
\bigl(A\ne\varnothing\land s=\sup A\land x<s\land
\forall a\in A\;(a\le x\lor s<a)\bigr).
\]
\[
x<\sup A\land A\cap(x,\sup A]=\varnothing.
\]
\end{description}
\end{remark*}
\begin{remark*}[Interpretation]
Every number strictly below a supremum is crossed by some element of the set.
\end{remark*}

\begin{lemmabox}{Lemma (Infimum Strict Lower Approximation)}
\begin{lemma}[Infimum Strict Lower Approximation]
\label{lem:infimum-strict-lower-approximation}
Let $A\subseteq\mathbb{R}$ be nonempty and bounded below, and let
$i=\inf A$.  For every $x\in\mathbb{R}$ with $i<x$, there exists
$a\in A$ such that
\[
i\le a<x.
\]
\hyperref[prf:infimum-strict-lower-approximation]{Go to proof.}
\end{lemma}
\end{lemmabox}
\begin{remark*}[Standard quantified statement]
\[
\forall x\in\mathbb{R},\qquad
\inf A<x
\Rightarrow
(\exists a\in A)(\inf A\leq a<x).
\]
\end{remark*}
\begin{remark*}[Predicate reading]
\[
\inf A<x\Rightarrow A\cap[\inf A,x)\ne\varnothing.
\]
\end{remark*}
\begin{remark*}[Negated quantified statement]
\[
\exists A\subseteq\mathbb{R}\;\exists i,x\in\mathbb{R}\;
\bigl(A\ne\varnothing\land i=\inf A\land i<x\land
\forall a\in A\;(a<i\lor x\le a)\bigr).
\]
\end{remark*}
\begin{remark*}[Negation predicate reading]
\[
\inf A<x\land A\cap[\inf A,x)=\varnothing.
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
  Failure is witnessed by the displayed negated or contrapositive condition.

  \item[\varnothing.]
  \[
\exists A\subseteq\mathbb{R}\;\exists i,x\in\mathbb{R}\;

\]
\[
\inf A<x\land A\cap[\inf A,x)=\varnothing.

\end{description}
\end{remark*}
\begin{remark*}[Interpretation]
Every number strictly above an infimum is reached from below by some element
of the set.
\end{remark*}
