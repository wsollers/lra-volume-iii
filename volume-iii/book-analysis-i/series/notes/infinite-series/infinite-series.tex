\begin{definition}[Formal Infinite Series]
\label{def:formal-infinite-series}

A formal infinite series is an expression of the form
\[
  \sum_{n=m}^{\infty} a_n,
\]
where \(m\) is an integer, and \(a_n\) is a real number for every
integer \(n\ge m\). This series is sometimes written as
\[
  a_m+a_{m+1}+a_{m+2}+\cdots.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  \sum_{n=m}^{\infty} a_n
  \quad\text{denotes a formal infinite series with } m\in\mathbb{Z}
  \text{ and } a_n\in\mathbb{R}\text{ for every } n\ge m.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The notation records an infinite list of real terms beginning at the
integer index \(m\). At this stage it is a formal expression rather than a
claim that a value has already been assigned to the series.
\end{remark*}
