\subsection*{Characterization of Discontinuity}
\label{sec:discontinuity}
% ---------------------------------------------------------
% Definition --- Point of Discontinuity
% ---------------------------------------------------------

\begin{definitionbox}{Definition (Point of Discontinuity)}
\begin{definition}[Point of Discontinuity]
\label{def:point-of-discontinuity}
Let $A \subseteq \mathbb{R}$, let $f \colon A \to \mathbb{R}$, and let $a \in A$. The function $f$ is discontinuous at $a$ if and only if $f$ is not continuous at $a$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } A \subseteq \mathbb{R},\; f \colon A \to \mathbb{R},\; a \in A.\\
&f \text{ is discontinuous at } a
\Longleftrightarrow
\neg\Bigl(\forall \varepsilon > 0\; \exists \delta > 0\; \forall x \in A\;
\bigl(|x-a|<\delta \Rightarrow |f(x)-f(a)| < \varepsilon\bigr)\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{DiscontinuousAt}(f,a) \coloneqq \neg \operatorname{ContinuousAt}(f,a).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let } A \subseteq \mathbb{R},\; f \colon A \to \mathbb{R},\; a \in A.\\
&f \text{ is discontinuous at } a
\Longleftrightarrow
\exists \varepsilon > 0\; \forall \delta > 0\; \exists x \in A\;
\bigl(|x-a|<\delta \land |f(x)-f(a)| \ge \varepsilon\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg \operatorname{DiscontinuousAt}(f,a)
\Longleftrightarrow
\operatorname{ContinuousAt}(f,a)
\Longleftrightarrow
\forall \varepsilon > 0\; \exists \delta > 0\; \forall x \in A\;
\bigl(|x-a|<\delta \Rightarrow |f(x)-f(a)| < \varepsilon\bigr).
\]
\end{remark*}

\begin{remark*}[Failure modes]
Discontinuity arises either because the nearby values of $f$ do not approach any single real number as $x \to a$, or because a well-defined limit exists but disagrees with the actual value $f(a)$.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\underbrace{\neg \exists L \in \mathbb{R}\; \operatorname{FunctionLimit}(f,A,a,L)}_{\text{no limit exists}}
\;\lor\;
\underbrace{\bigl(\exists L \in \mathbb{R}\; \operatorname{FunctionLimit}(f,A,a,L) \land L \ne f(a)\bigr)}_{\text{limit-value mismatch}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
A point of discontinuity is detected by the breakdown of the usual epsilon-delta control. Understanding removable, jump, and essential cases guides how to remedy or classify singular behavior in proofs.
\end{remark*}

% Definition --- Sequential Characterization
% ---------------------------------------------------------

\begin{definitionbox}{Definition (Sequential Characterization of Discontinuity)}
\begin{definition}[Sequential Characterization of Discontinuity]
\label{def:sequential-discontinuity-at-a-point}
Let $A \subseteq \mathbb{R}$, let $f:A \to \mathbb{R}$, and let $c \in A$. The function $f$ is discontinuous at $c$ if and only if there exists a sequence $(x_n)_{n\in\mathbb{N}} \subseteq A$ such that $x_n \to c$ while $f(x_n) \not\to f(c)$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
f \text{ is discontinuous at } c \Longleftrightarrow
\exists (x_n)\subseteq A \;\bigl[x_n \to c \;\land\; f(x_n) \not\to f(c)\bigr].
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{DiscontinuousAt}(f,c) \coloneqq \exists (x_n)\subseteq A:\; x_n\to c \;\land\; \neg\bigl(f(x_n)\to f(c)\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
f \text{ is continuous at } c \Longleftrightarrow
\forall (x_n)\subseteq A:\; x_n\to c \;\Rightarrow\; f(x_n)\to f(c).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg \operatorname{DiscontinuousAt}(f,c)
\Longleftrightarrow
\operatorname{ContinuousAt}(f,c)
\Longleftrightarrow
\forall (x_n)\subseteq A:\; x_n\to c \;\Rightarrow\; f(x_n)\to f(c).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The characterization fails --- meaning $f$ is NOT sequentially discontinuous --- when every sequence $(x_n)$ in $A$ converging to $c$ has $f(x_n) \to f(c)$. No sequential witness exists.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\neg \operatorname{DiscontinuousAt}(f,c)
\Longleftrightarrow
\underbrace{\forall (x_n)\subseteq A:\;
  x_n\to c \;\Rightarrow\; f(x_n)\to f(c)
}_{\text{every approach preserves the limit}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
A discontinuity at $c$ can be certified by a single sequence approaching $c$ whose images refuse to settle at $f(c)$.
\end{remark*}

\begin{definitionbox}{Definition (Neighbourhood Characterization of Discontinuity)}
\begin{definition}[Neighbourhood Characterization of Discontinuity]
\label{def:neighborhood-discontinuity-at-a-point}
Let $A \subseteq \mathbb{R}$, let $f \colon A \to \mathbb{R}$, and let $c \in A$. The function $f$ is discontinuous at $c$ if and only if there exists a neighborhood $V$ of $f(c)$ such that for every neighborhood $U$ of $c$ relative to $A$ there exists $x \in U \cap A$ with $f(x) \notin V$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
f \text{ is discontinuous at } c
\Longleftrightarrow
\exists V \text{ nbhd of } f(c)\;
\forall U \text{ nbhd of } c\;
\exists x \in U \cap A\; \bigl(f(x) \notin V\bigr).
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{DiscontinuousAt}(f,c) \coloneqq
\exists V \text{ nbhd of } f(c)\;
\forall U \text{ nbhd of } c\;
\exists x \in U \cap A:\; f(x) \notin V.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
f \text{ is continuous at } c
\Longleftrightarrow
\forall V \text{ nbhd of } f(c)\;
\exists U \text{ nbhd of } c\;
\forall x \in U \cap A\; \bigl(f(x) \in V\bigr).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg \operatorname{DiscontinuousAt}(f,c)
\Longleftrightarrow
\operatorname{ContinuousAt}(f,c)
\Longleftrightarrow
\forall V \text{ nbhd of } f(c)\;
\exists U \text{ nbhd of } c\;
\forall x \in U \cap A\; \bigl(f(x) \in V\bigr).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The neighbourhood discontinuity characterization fails --- meaning $f$ is actually continuous --- when every output neighbourhood $V$ of $f(c)$ can be matched by some input neighbourhood $U$ such that $f$ maps all of $U \cap A$ into $V$. No perpetually-missed output neighbourhood exists.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\neg \operatorname{DiscontinuousAt}(f,c)
\Longleftrightarrow
\forall V \text{ nbhd of } f(c)\;
\underbrace{\exists U \text{ nbhd of } c}_{\text{some input window}}
:\;
\underbrace{\forall x \in U \cap A:\; f(x) \in V}_{\text{all outputs inside tolerance}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Discontinuity means a fixed output neighborhood $V$ is perpetually missed no matter how tightly the input is restricted around $c$.
\end{remark*}

% Lemma --- Equivalence
% ---------------------------------------------------------

\begin{lemmabox}{Lemma (Equivalent Formulations of Discontinuity at a Point)}
\begin{lemma}[Equivalent Formulations of Discontinuity at a Point]
\label{lem:equivalent-discontinuity-at-a-point}
Let $A\subseteq \mathbb{R}$, let $f:A\to\mathbb{R}$, and let $c\in A$. The following are equivalent:
\begin{enumerate}
\item $c$ is a point of discontinuity in the $\varepsilon$-$\delta$ sense.
\item $c$ is a sequential point of discontinuity.
\item $c$ is a neighbourhood point of discontinuity.
\end{enumerate}

\smallskip\noindent
\hyperref[prf:equivalent-discontinuity-at-a-point]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
(1)\; &\exists \varepsilon_0>0\;\forall \delta>0\;\exists x\in A:\;0<|x-c|<\delta \land |f(x)-f(c)|\ge \varepsilon_0\\
\Longleftrightarrow\;
(2)\; &\exists (x_n)\subseteq A:\; x_n \to c \land \neg\bigl(f(x_n)\to f(c)\bigr)\\
\Longleftrightarrow\;
(3)\; &\exists V \text{ nbhd of } f(c)\;\forall U \text{ nbhd of } c\;\exists x\in U\cap A:\;f(x)\notin V.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
\operatorname{DiscontinuousAt}(f,c)
&\Longleftrightarrow
\exists \varepsilon_0>0\;\forall \delta>0\;\exists x\in A:\;0<|x-c|<\delta \land |f(x)-f(c)|\ge \varepsilon_0\\
&\Longleftrightarrow
\exists (x_n)\subseteq A:\; x_n \to c \land \neg\bigl(f(x_n)\to f(c)\bigr)\\
&\Longleftrightarrow
\exists V \text{ nbhd of } f(c)\;\forall U \text{ nbhd of } c\;\exists x\in U\cap A:\;f(x)\notin V.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
The equivalence cannot fail for functions at limit points of $A$: the proof that all three characterizations are equivalent is constructive in both directions. In the degenerate case where $c$ is not a limit point of $A$, the punctured-neighbourhood conditions are vacuously satisfied by every $L$, and the equivalence becomes trivial rather than informative.
\end{remark*}

\begin{remark*}[Interpretation]
The three characterizations of discontinuity are mutually convertible. Analysts select whichever viewpoint simplifies a particular argument.
\end{remark*}

% Definition --- Types of Discontinuity

\begin{definitionbox}{Definition (Types of Discontinuity)}
\begin{definition}[Types of Discontinuity]
\label{def:types-of-discontinuity-at-a-point}
Let $A \subseteq \mathbb{R}$, let $f:A \to \mathbb{R}$, and let $x_0 \in A$ with $f$ discontinuous at $x_0$.
\begin{enumerate}[label=(\roman*)]
  \item \emph{Removable} iff $\exists L \in \mathbb{R}:\;\lim_{x \to x_0} f(x) = L$ and $L \ne f(x_0)$.
  \item \emph{Jump} iff $\exists L_{-}, L_{+} \in \mathbb{R}:\;\lim_{x \to x_0^-} f(x) = L_{-}$, $\lim_{x \to x_0^+} f(x) = L_{+}$, and $L_{-} \ne L_{+}$.
  \item \emph{Essential} iff it is not removable.
\end{enumerate}
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Removable } \Longleftrightarrow \exists L \in \mathbb{R}\; \bigl(\lim_{x \to x_0} f(x) = L \land L \ne f(x_0)\bigr).\\
&\text{Jump } \Longleftrightarrow \exists L_{-}, L_{+} \in \mathbb{R}\; \bigl(\lim_{x \to x_0^-} f(x) = L_{-} \land \lim_{x \to x_0^+} f(x) = L_{+} \land L_{-} \ne L_{+}\bigr).\\
&\text{Essential } \Longleftrightarrow \neg \exists L \in \mathbb{R}\; \bigl(\lim_{x \to x_0} f(x) = L \land L \ne f(x_0)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\begin{aligned}
\operatorname{RemovableDiscontinuity}(f,x_0) &\coloneqq \exists L \in \mathbb{R}\; \bigl(\operatorname{FunctionLimit}(f,A,x_0,L) \land L \ne f(x_0)\bigr).\\
\operatorname{JumpDiscontinuity}(f,x_0) &\coloneqq \exists L_{-}, L_{+} \in \mathbb{R}\; \bigl(\operatorname{LeftLimit}(f,A,x_0,L_{-}) \land \operatorname{RightLimit}(f,A,x_0,L_{+}) \land L_{-} \ne L_{+}\bigr).\\
\operatorname{EssentialDiscontinuity}(f,x_0) &\coloneqq \neg \operatorname{RemovableDiscontinuity}(f,x_0).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
\neg\text{Removable}
&\Longleftrightarrow
\forall L \in \mathbb{R}\;
\bigl(\lim_{x \to x_0} f(x) \ne L \lor L = f(x_0)\bigr),\\
\neg\text{Jump}
&\Longleftrightarrow
\forall L_-,L_+ \in \mathbb{R}\;
\bigl(\lim_{x \to x_0^-} f(x)\ne L_-
\lor \lim_{x \to x_0^+} f(x)\ne L_+
\lor L_-=L_+\bigr),\\
\neg\text{Essential}
&\Longleftrightarrow
\text{Removable}.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
\neg \operatorname{RemovableDiscontinuity}(f,x_0) &\Longleftrightarrow \forall L \in \mathbb{R}\; \bigl(\neg\operatorname{FunctionLimit}(f,A,x_0,L) \lor L = f(x_0)\bigr).\\
\neg \operatorname{JumpDiscontinuity}(f,x_0) &\Longleftrightarrow \forall L_{-}, L_{+} \in \mathbb{R}\; \bigl(\neg\operatorname{LeftLimit}(f,A,x_0,L_{-}) \lor \neg\operatorname{RightLimit}(f,A,x_0,L_{+}) \lor L_{-} = L_{+}\bigr).\\
\neg \operatorname{EssentialDiscontinuity}(f,x_0) &\Longleftrightarrow \operatorname{RemovableDiscontinuity}(f,x_0).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{itemize}
\item Removable discontinuity fails when no bilateral limit exists at $x_0$, or when the bilateral limit exists but equals $f(x_0)$ (making the function actually continuous).
\item Jump discontinuity fails when at least one of the one-sided limits does not exist, or when both one-sided limits exist but are equal (making it potentially removable or continuous).
\item Essential discontinuity fails when the discontinuity is removable (the bilateral limit exists and differs from $f(x_0)$).
\end{itemize}
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\begin{aligned}
\neg \operatorname{RemovableDiscontinuity}(f,x_0) &\Longleftrightarrow
\underbrace{\neg \exists L \in \mathbb{R}\; \lim_{x \to x_0} f(x) = L}_{\text{no bilateral limit exists}}
\;\lor\;
\underbrace{\lim_{x \to x_0} f(x) = f(x_0)}_{\text{limit equals function value}}.\\
\neg \operatorname{JumpDiscontinuity}(f,x_0) &\Longleftrightarrow
\underbrace{\neg \exists L_{-} \in \mathbb{R}\; \lim_{x \to x_0^-} f(x) = L_{-}}_{\text{no left limit}}
\;\lor\;
\underbrace{\neg \exists L_{+} \in \mathbb{R}\; \lim_{x \to x_0^+} f(x) = L_{+}}_{\text{no right limit}}\\
&\qquad\lor\;
\underbrace{\lim_{x \to x_0^-} f(x) = \lim_{x \to x_0^+} f(x)}_{\text{one-sided limits agree}}.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Removable discontinuities can be healed by redefining $f(x_0)$. Jump discontinuities provide well-defined one-sided limits useful in integration theory. Essential discontinuities are pathological and require more global tools.
\end{remark*}
