\subsection*{Density}




\begin{definitionbox}{Definition (Dense Subset)}
\begin{definition}[Dense Subset]
\label{def:dense-subset}
Let $(X,\tau)$ be a topological space and let $D\subseteq X$.  The set
$D$ is \emph{dense in} $X$ if $\overline{D}=X$.  Equivalently, every
nonempty open subset of $X$ intersects $D$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\overline{D}=X
\Longleftrightarrow
\forall U\in\tau\;
\bigl(U\neq\varnothing\Longrightarrow \exists d\in D\;(d\in U)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
X\in\mathbf{Set},\qquad D\subseteq X,\qquad
\operatorname{DenseSubset}(D,X,\tau)\coloneqq
\overline{D}=X
\Longleftrightarrow
\forall U\in\tau\;
\bigl(U\neq\varnothing\Longrightarrow \exists d\in D\;(d\in U)\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let }(X,\tau)\text{ be a topological space and }D\subseteq X.\\
&D\text{ is not dense in }X
\Longleftrightarrow
\exists U\in\tau\;\bigl(U\neq\varnothing\land U\cap D=\varnothing\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
X\in\mathbf{Set},\qquad D\subseteq X,\qquad
\neg\operatorname{DenseSubset}(D,X,\tau)\Longleftrightarrow
\exists U\in\tau\;\bigl(U\neq\varnothing\land U\cap D=\varnothing\bigr).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Density fails when some nonempty open subset of $X$ contains no point of
$D$.


  \item[\operatorname{DenseSubset}.]
  \[
\begin{aligned}
&\text{Let }(X,\tau)\text{ be a topological space and }D\subseteq X.\\
&D\text{ is not dense in }X
\Longleftrightarrow
\exists U\in\tau\;\bigl(U\neq\varnothing\land U\cap D=\varnothing\bigr).
\end{aligned}
\]
\[
X\in\mathbf{Set},\qquad D\subseteq X,\qquad
\neg\operatorname{DenseSubset}(D,X,\tau)\Longleftrightarrow
\exists U\in\tau\;\bigl(U\neq\varnothing\land U\cap D=\varnothing\bigr).
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
\[
\neg\operatorname{DenseSubset}(D,X,\tau)
=
\underbrace{\exists U\in\tau\;(U\neq\varnothing)}_{\text{nonempty open set}}
\;\land\;
\underbrace{U\cap D=\varnothing}_{\text{no point of }D\text{ lies in }U}.
\]


  \item[\operatorname{DenseSubset}.]
  \[
\begin{aligned}
&\text{Let }(X,\tau)\text{ be a topological space and }D\subseteq X.\\
&D\text{ is not dense in }X
\Longleftrightarrow
\exists U\in\tau\;\bigl(U\neq\varnothing\land U\cap D=\varnothing\bigr).
\end{aligned}
\]
\[
X\in\mathbf{Set},\qquad D\subseteq X,\qquad
\neg\operatorname{DenseSubset}(D,X,\tau)\Longleftrightarrow
\exists U\in\tau\;\bigl(U\neq\varnothing\land U\cap D=\varnothing\bigr).
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Density says that no nonempty open part of $X$ misses $D$.  It does not
say that every point of $X$ belongs to $D$; it says that $D$ is
topologically unavoidable inside $X$.
\end{remark*}

\begin{definitionbox}{Definition (Order Dense Subset)}
\begin{definition}[Order Dense Subset]
\label{def:order-dense-subset}
Let $(S,\leq)$ be a linearly ordered set and let $D\subseteq S$.  The set
$D$ is \emph{order dense in} $S$ if between every two elements $x<y$ of
$S$ there lies an element of $D$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\operatorname{OrderDenseSubset}(D,P)
\Longleftrightarrow
\forall x,y\in S\;\bigl(x<y\Longrightarrow \exists d\in D\;(x<d<y)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
S\in\mathbf{Set},\qquad D\subseteq S,\qquad P=\mathsf{TotallyOrderedSet}(S,\leq),\qquad
\operatorname{OrderDenseSubset}(D,P)\coloneqq
\forall x,y\in S\;\bigl(x<y\Longrightarrow \exists d\in D\;(x<d<y)\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let }(S,\leq)\text{ be a linearly ordered set and }D\subseteq S.\\
&D\text{ is not order dense in }S
\Longleftrightarrow
\exists x,y\in S\;\bigl(x<y\land \forall d\in D\;(d\leq x\lor y\leq d)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
S\in\mathbf{Set},\qquad D\subseteq S,\qquad P=\mathsf{TotallyOrderedSet}(S,\leq),\qquad
\neg\operatorname{OrderDenseSubset}(D,P)\Longleftrightarrow
\exists x,y\in S\;\bigl(x<y\land \forall d\in D\;(d\leq x\lor y\leq d)\bigr).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Order density fails when there is a nonempty open interval of $S$ that
contains no point of $D$.


  \item[\operatorname{OrderDenseSubset}.]
  \[
\begin{aligned}
&\text{Let }(S,\leq)\text{ be a linearly ordered set and }D\subseteq S.\\
&D\text{ is not order dense in }S
\Longleftrightarrow
\exists x,y\in S\;\bigl(x<y\land \forall d\in D\;(d\leq x\lor y\leq d)\bigr).
\end{aligned}
\]
\[
S\in\mathbf{Set},\qquad D\subseteq S,\qquad P=\mathsf{TotallyOrderedSet}(S,\leq),\qquad
\neg\operatorname{OrderDenseSubset}(D,P)\Longleftrightarrow
\exists x,y\in S\;\bigl(x<y\land \forall d\in D\;(d\leq x\lor y\leq d)\bigr).
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
\[
\neg\operatorname{OrderDenseSubset}(D,P)
=
\underbrace{\exists x,y\in S\;(x<y)}_{\text{nonempty interval}}
\;\land\;
\underbrace{\forall d\in D\;(d\leq x\lor y\leq d)}_{\text{no point of }D\text{ lies between }x\text{ and }y}.
\]


  \item[\operatorname{OrderDenseSubset}.]
  \[
\begin{aligned}
&\text{Let }(S,\leq)\text{ be a linearly ordered set and }D\subseteq S.\\
&D\text{ is not order dense in }S
\Longleftrightarrow
\exists x,y\in S\;\bigl(x<y\land \forall d\in D\;(d\leq x\lor y\leq d)\bigr).
\end{aligned}
\]
\[
S\in\mathbf{Set},\qquad D\subseteq S,\qquad P=\mathsf{TotallyOrderedSet}(S,\leq),\qquad
\neg\operatorname{OrderDenseSubset}(D,P)\Longleftrightarrow
\exists x,y\in S\;\bigl(x<y\land \forall d\in D\;(d\leq x\lor y\leq d)\bigr).
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Order density is an interval-penetration property. It ignores metric distance
and asks only whether every ordered gap contains a point of the subset. A
subset can therefore be order dense in $S$ precisely when no nonempty open
interval of $S$ is missed entirely by that subset.
\end{remark*}

\begin{theorembox}{Theorem (Density of the Rationals)}
\begin{theorem}[Density of the Rationals]
\label{thm:density-of-rationals-in-reals}
For every $a,b\in\mathbb{R}$ with $a<b$, there exists $q\in\mathbb{Q}$ such
that
\[
  a<q<b.
\]
Equivalently,
\[
  (\forall a,b\in\mathbb{R})
  \bigl[a<b\Longrightarrow(\exists q\in\mathbb{Q})(a<q<b)\bigr].
\]
\hyperref[prf:density-of-rationals-in-reals]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{R}\;\bigl(a<b\Longrightarrow \exists q\in\mathbb{Q}\;(a<q<b)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad \mathbb{Q}\subseteq\mathbb{R},\qquad
D_{\mathbb{Q}}=\mathsf{OrderDenseSet}(\mathbb{Q},P_{\mathbb{R}}),\qquad
\operatorname{IsOrderDenseSubset}(\mathbb{Q},P_{\mathbb{R}})\Longleftrightarrow
\forall a,b\in\mathbb{R}\;\bigl(a<b\Longrightarrow \exists q\in\mathbb{Q}\;(a<q<b)\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists a,b\in\mathbb{R}\;\bigl(a<b\land \forall q\in\mathbb{Q}\;(q\le a\lor b\le q)\bigr).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad \mathbb{Q}\subseteq\mathbb{R},\qquad
D_{\mathbb{Q}}=\mathsf{OrderDenseSet}(\mathbb{Q},P_{\mathbb{R}}),\qquad
\neg\operatorname{IsOrderDenseSubset}(\mathbb{Q},P_{\mathbb{R}})\Longleftrightarrow
\exists a,b\in\mathbb{R}\;\bigl(a<b\land \forall q\in\mathbb{Q}\;(q\le a\lor b\le q)\bigr).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
The theorem would fail precisely if some nonempty real interval contained no rational number.


  \item[\operatorname{IsOrderDenseSubset}.]
  \[
\exists a,b\in\mathbb{R}\;\bigl(a<b\land \forall q\in\mathbb{Q}\;(q\le a\lor b\le q)\bigr).
\]
\[
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad \mathbb{Q}\subseteq\mathbb{R},\qquad
D_{\mathbb{Q}}=\mathsf{OrderDenseSet}(\mathbb{Q},P_{\mathbb{R}}),\qquad
\neg\operatorname{IsOrderDenseSubset}(\mathbb{Q},P_{\mathbb{R}})\Longleftrightarrow
\exists a,b\in\mathbb{R}\;\bigl(a<b\land \forall q\in\mathbb{Q}\;(q\le a\lor b\le q)\bigr).
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
\[
\neg\operatorname{IsOrderDenseSubset}(\mathbb{Q},\mathbb{R})
=
\underbrace{\exists a,b\in\mathbb{R}\;(a<b)}_{\text{nonempty real interval}}
\;\land\;
\underbrace{\forall q\in\mathbb{Q}\;(q\le a\lor b\le q)}_{\text{no rational lies inside}}.
\]


  \item[\operatorname{IsOrderDenseSubset}.]
  \[
\exists a,b\in\mathbb{R}\;\bigl(a<b\land \forall q\in\mathbb{Q}\;(q\le a\lor b\le q)\bigr).
\]
\[
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad \mathbb{Q}\subseteq\mathbb{R},\qquad
D_{\mathbb{Q}}=\mathsf{OrderDenseSet}(\mathbb{Q},P_{\mathbb{R}}),\qquad
\neg\operatorname{IsOrderDenseSubset}(\mathbb{Q},P_{\mathbb{R}})\Longleftrightarrow
\exists a,b\in\mathbb{R}\;\bigl(a<b\land \forall q\in\mathbb{Q}\;(q\le a\lor b\le q)\bigr).
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Rational numbers penetrate every real interval. The theorem does not say that every real number is rational; it says that rational numbers are never separated from a real interval by a positive order gap. This is the interval form of rational density in the real line.
\end{remark*}

\label{lem:rational-translation-preserves-rationality}

  r+q\in\mathbb{Q}.

The rational numbers are closed under addition, so translating a rational by a
rational keeps it rational.

\begin{theorembox}{Theorem (Density of the Irrationals)}
\begin{theorem}[Density of the Irrationals]
\label{thm:density-of-irrationals-in-reals}
For every $a,b\in\mathbb{R}$ with $a<b$, there exists
$s\in\mathbb{R}\setminus\mathbb{Q}$ such that
\[
  a<s<b.
\]
Equivalently,
\[
  (\forall a,b\in\mathbb{R})
  \bigl[a<b\Longrightarrow
  (\exists s\in\mathbb{R}\setminus\mathbb{Q})(a<s<b)\bigr].
\]
\hyperref[prf:density-of-irrationals-in-reals]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{R}\;\bigl(a<b\Longrightarrow \exists s\in\mathbb{R}\setminus\mathbb{Q}\;(a<s<b)\bigr).

\]
\end{remark*}

\begin{exposition}
Density in $\mathbb{R}$ has a scale-invariant consequence: no matter how small a nonempty
open interval becomes, it still contains both rational and irrational points. The interleaving
of $\mathbb{Q}$ and $\mathbb{R}\setminus\mathbb{Q}$ is therefore not a large-scale phenomenon.
It persists at every interval scale.

This has three important consequences. First, the continuum does not become locally discrete
under magnification. For every nonempty open interval $(a,b)$, both
\[
\mathbb{Q}\cap(a,b)
\qquad\text{and}\qquad
(\mathbb{R}\setminus\mathbb{Q})\cap(a,b)
\]
remain nonempty, and in fact infinite. Density is therefore stable under repeated restriction
to smaller intervals.

Second, this behavior sharply distinguishes the real line from discrete numerical models.
Floating-point systems form a discrete set of representable values, so repeated magnification
eventually reaches a scale at which adjacent representable numbers leave no further point
strictly between them. At that stage, interval penetration fails in the discrete model even
though it continues to hold in $\mathbb{R}$.

Third, the Archimedean Property supplies the scale mechanism behind this behavior. Given any
positive scale $\varepsilon$, there exists $n\in\mathbb{N}$ with
\[
\frac{1}{n}<\varepsilon.
\]
This makes it possible to construct rational approximations at arbitrarily small interval
scales, which is one of the basic engines behind density arguments.

The resulting contrast is structural. Smooth mathematical models are built in a setting where
interval penetration persists at every scale, whereas finite computational models eventually
encounter a smallest representable separation. The transition from continuum behavior to
discrete behavior occurs exactly where nonempty intervals in the numerical model cease to
contain further \\representable points.
\end{exposition}
