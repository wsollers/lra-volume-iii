\subsection*{Monotone Functions and Continuity}
\label{sec:monotone-functions}
\begin{theorembox}{Theorem (One-Sided Limits of Monotone Functions)}
\begin{theorem}[One-Sided Limits of Monotone Functions]
\label{thm:monotone-one-sided-limits}
Let $I \subseteq \mathbb{R}$ be an interval, let $c$ be interior to $I$, and let $f:I\to\mathbb{R}$ be increasing. Define
\[
L_- \coloneqq \sup\{f(x): x\in I,\; x<c\},
\qquad
L_+ \coloneqq \inf\{f(x): x\in I,\; c<x\}.
\]
Then both one-sided limits exist and satisfy
\[
\lim_{x\to c^-} f(x) = L_-,
\qquad
\lim_{x\to c^+} f(x) = L_+.
\]
\hyperref[prf:monotone-one-sided-limits]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&I\subseteq\mathbb{R}\land
\bigl(\forall x,y\in I\;\forall z\in\mathbb{R}\;(x<z<y\Rightarrow z\in I)\bigr)
\land f:I\to\mathbb{R}\\
&\land\bigl(\forall x,y\in I\;(x<y\Rightarrow f(x)\le f(y))\bigr)
\land c\in I
\land \exists a,b\in I\;(a<c<b)\\
&\land L_- = \sup\{f(x): x\in I,\; x<c\}
\land L_+ = \inf\{f(x): x\in I,\; c<x\}\\
&\Longrightarrow\\
&\forall \varepsilon>0\; \exists \delta_- > 0\; \forall x\in I\; \bigl((0<c-x<\delta_- \land x<c) \Rightarrow |f(x)-L_-|<\varepsilon \bigr)\\
&\land \forall \varepsilon>0\; \exists \delta_+ > 0\; \forall x\in I\; \bigl((0<x-c<\delta_+ \land c<x) \Rightarrow |f(x)-L_+|<\varepsilon \bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&\text{Let } I\subseteq \mathbb{R} \text{ be an interval, } f:I\to\mathbb{R} \text{ increasing, and } c \text{ interior to } I.\\
&\operatorname{LeftLimit}\!\Bigl(f,c,\sup\{f(x): x\in I,\; x<c\}\Bigr)
\land
\operatorname{RightLimit}\!\Bigl(f,c,\inf\{f(x): x\in I,\; x>c\}\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
For an increasing function, the values just to the left of $c$ form a monotone increasing set bounded above by any right neighborhood, so their supremum matches the left-hand limit. Symmetrically, the values just to the right form a monotone increasing tail bounded below, making the infimum coincide with the right-hand limit. The theorem shows that monotonicity alone guarantees both one-sided limits exist and equals these extremal values. Failure occurs when $c$ is not an interior point or $f$ is not monotone, because the comparison sets may be empty or oscillatory, preventing the suprema or infima from governing the boundary behavior.
\end{remark*}

\begin{corollarybox}{Corollary (Continuity Criterion for Monotone Functions)}
\begin{corollary}[Continuity Criterion for Monotone Functions]
\label{cor:monotone-continuity-criterion}
Let $I \subseteq \mathbb{R}$ be an interval, let $f:I \to \mathbb{R}$ be monotone increasing, and let $c$ be an interior point of $I$. Then $f$ is continuous at $c$ if and only if its one-sided limits satisfy $f(c^-)=f(c)=f(c^+)$, where $f(c^-)=\lim_{x\to c^-} f(x)$ and $f(c^+)=\lim_{x\to c^+} f(x)$.
\hyperref[prf:monotone-continuity-criterion]{Go to proof.}
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } I \subseteq \mathbb{R} \text{ be an interval, } f:I\to\mathbb{R} \text{ monotone increasing, and } c \text{ an interior point of } I.\\
&\bigl(f \text{ is continuous at } c\bigr)
\Longleftrightarrow
\left(\lim_{x\to c^-} f(x) = f(c) = \lim_{x\to c^+} f(x)\right).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{IsContinuous}(f,c)
\Longleftrightarrow
\bigl(\operatorname{LeftLimit}(f,c,f(c)) \land \operatorname{RightLimit}(f,c,f(c))\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let } I \subseteq \mathbb{R} \text{ be an interval, } f:I\to\mathbb{R} \text{ monotone increasing, and } c \text{ an interior point of } I.\\
&\neg\bigl(f \text{ is continuous at } c\bigr)
\Longleftrightarrow
\left(\lim_{x\to c^-} f(x) < f(c)\right)
\lor
\left(f(c) < \lim_{x\to c^+} f(x)\right).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{IsContinuous}(f,c)
\Longleftrightarrow
\left(\exists L^{-}\,[\operatorname{LeftLimit}(f,c,L^{-}) \land L^{-} < f(c)]\right)
\lor
\left(\exists L^{+}\,[\operatorname{RightLimit}(f,c,L^{+}) \land f(c) < L^{+}]\right).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
Continuity can fail only through a jump: either the graph arrives at $c$ from the left strictly below $f(c)$, or it departs to the right strictly above $f(c)$. Both jumps reflect the inability of the one-sided limits to match the function value at the interior point.
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
\neg\operatorname{IsContinuous}(f,c)
=
\underbrace{\left(\exists L^{-}\,[\operatorname{LeftLimit}(f,c,L^{-}) \land L^{-} < f(c)]\right)}_{\text{left jump at } c}
\lor
\underbrace{\left(\exists L^{+}\,[\operatorname{RightLimit}(f,c,L^{+}) \land f(c) < L^{+}]\right)}_{\text{right jump at } c}.
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
A monotone graph possesses finite one-sided limits at every interior point. Continuity at $c$ demands that these approach values exactly meet the function value, so the graph neither overshoots nor undershoots when passing through $c$. Any failure manifests as a jump discontinuity: the graph either leaps upward at $c$ or remains below the right-hand approach. This criterion is indispensable when classifying discontinuities of monotone functions, since it converts continuity into a simple equality check between the function value and the adjacent plateau levels.
\end{remark*}

\begin{definitionbox}{Definition (Jump of a Function)}
\begin{definition}[Jump of a Function]
\label{def:jump-of-function}
Let $I \subseteq \mathbb{R}$ be an interval, let $f:I\to\mathbb{R}$ be
increasing, and let $c$ be an interior point of $I$. If the one-sided
limits $f(c^-)$ and $f(c^+)$ exist, the jump of $f$ at $c$ is
\[
j_f(c)\coloneqq f(c^+)-f(c^-).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\forall I\subseteq\mathbb{R}\;\forall f:I\to\mathbb{R}\;\forall c\in I\;\forall j\in\mathbb{R}\\
&\Bigl[
\bigl(\forall x,y\in I\;\forall z\in\mathbb{R}\;(x<z<y\Rightarrow z\in I)\bigr)
\land \exists a,b\in I\;(a<c<b)\\
&\qquad\land \bigl(\forall x,y\in I\;(x<y\Rightarrow f(x)\le f(y))\bigr)
\land \exists L_-,L_+\in\mathbb{R}\;
\bigl(\lim_{x\to c^-}f(x)=L_- \land \lim_{x\to c^+}f(x)=L_+\bigr)
\Bigr]\\
&\qquad\Rightarrow
\bigl(j=j_f(c)\Longleftrightarrow j=f(c^+)-f(c^-)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{IsInterval}(I,\mathbb{R})
\land \operatorname{OrderPreserving}(f,I)
\land \operatorname{IsInterior}(c,I)
\land \exists L_-,L_+\in\mathbb{R}\;
\bigl(\operatorname{LeftLimit}(f,c,L_-)
\land \operatorname{RightLimit}(f,c,L_+)
\land j=L_+-L_-\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\forall I\subseteq\mathbb{R}\;\forall f:I\to\mathbb{R}\;\forall c\in I\;\forall j\in\mathbb{R}:\\
&\neg\bigl(j=j_f(c)\bigr)
\Longleftrightarrow
\neg\bigl(\forall x,y\in I\;\forall z\in\mathbb{R}\;(x<z<y\Rightarrow z\in I)\bigr)\\
&\qquad\lor
\neg\bigl(\forall x,y\in I\;(x<y\Rightarrow f(x)\le f(y))\bigr)
\lor \neg\bigl(\exists a,b\in I\;(a<c<b)\bigr)\\
&\qquad\lor
\forall L_-,L_+\in\mathbb{R}\;
\bigl(\lim_{x\to c^-}f(x)\ne L_-
\lor \lim_{x\to c^+}f(x)\ne L_+
\lor j\ne L_+-L_-\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{IsInterval}(I,\mathbb{R})
\lor \neg\operatorname{OrderPreserving}(f,I)
\lor \neg\operatorname{IsInterior}(c,I)
\lor
\forall L_-,L_+\in\mathbb{R}\;
\bigl(\neg\operatorname{LeftLimit}(f,c,L_-)
\lor \neg\operatorname{RightLimit}(f,c,L_+)
\lor j\ne L_+-L_-\bigr).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
A proposed jump value fails when a required one-sided limit does not
exist, or when both one-sided limits exist but the proposed value is not
their difference.
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
\underbrace{\neg\bigl(f(c^-)\text{ exists}\bigr)}_{\text{missing left limit}}
\lor
\underbrace{\neg\bigl(f(c^+)\text{ exists}\bigr)}_{\text{missing right limit}}
\lor
\underbrace{j\ne f(c^+)-f(c^-)}_{\text{incorrect jump magnitude}}.
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
For a monotone function the one-sided limits exist at every interior
point, and their difference measures the vertical gap in the graph at
that point. A zero jump means the two limiting heights agree; a positive
jump records a discontinuity of first kind.
\end{remark*}

\begin{propositionbox}{Proposition (Discontinuities of a Monotone Function Are of First Kind)}
\begin{proposition}[Discontinuities of a Monotone Function Are of First Kind]
\label{prop:monotone-discontinuities-first-kind}
Let $I \subseteq \mathbb{R}$ be an interval, let $f:I \to \mathbb{R}$ be monotone, and let $a \in I$ be such that there exist points of $I$ strictly less and strictly greater than $a$. If $f$ is discontinuous at $a$, then both one-sided limits $\lim_{x\uparrow a} f(x)$ and $\lim_{x\downarrow a} f(x)$ exist in $\mathbb{R}$.
\hyperref[prf:monotone-discontinuities-first-kind]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } I \subseteq \mathbb{R} \text{ be an interval, } f:I \to \mathbb{R} \text{ monotone, and } a \in I \text{ interior to } I.\\
&\bigl[f \text{ is discontinuous at } a\bigr]
\Longrightarrow
\exists L^{-},L^{+} \in \mathbb{R} :
\begin{cases}
\displaystyle \lim_{x \uparrow a} f(x) = L^{-},\\[4pt]
\displaystyle \lim_{x \downarrow a} f(x) = L^{+}.
\end{cases}
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{Monotone}\bigl(f,\mathsf{OrderedSet}(I,\leq)\bigr) \land \operatorname{IsDiscontinuous}(f,a) \Longrightarrow \exists L^{-},L^{+} \in \mathbb{R} \; \bigl(\operatorname{LeftLimit}(f,a,L^{-}) \land \operatorname{RightLimit}(f,a,L^{+})\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
A monotone graph can jump only by settling on distinct finite one-sided limits; it cannot oscillate wildly or blow up to infinity at a point of discontinuity. Thus every discontinuity of a monotone function is a jump discontinuity of the first kind, which explains why such functions are easy to integrate and why their exceptional sets are well behaved. Geometrically, the graph approaches a finite height from the left and another finite height from the right before making the jump.
\end{remark*}

\begin{corollarybox}{Corollary (Jump Intervals Are Disjoint)}
\begin{corollary}[Jump Intervals Are Disjoint]
\label{cor:jump-intervals-for-monotone-discontinuities}
Let $I \subseteq \mathbb{R}$ be an interval and let $f:I \to \mathbb{R}$ be nondecreasing. If $a,b \in I$ are distinct points of discontinuity of $f$, then the open intervals $(f(a^-),f(a^+))$ and $(f(b^-),f(b^+))$ are disjoint.%
\hyperref[prf:jump-intervals-for-monotone-discontinuities]{Go to proof.}
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in I\;
\Bigl[
a\ne b\land
\bigl(\exists \varepsilon_a>0\;\forall \delta>0\;\exists x\in I\;
(|x-a|<\delta\land |f(x)-f(a)|\ge\varepsilon_a)\bigr)\land
\bigl(\exists \varepsilon_b>0\;\forall \delta>0\;\exists y\in I\;
(|y-b|<\delta\land |f(y)-f(b)|\ge\varepsilon_b)\bigr)
\Rightarrow
(f(a^-),f(a^+))\cap(f(b^-),f(b^+))=\varnothing
\Bigr].
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{Monotone}\bigl(f,\mathsf{OrderedSet}(I,\leq)\bigr)\land
\begin{cases}
a,b\in I,\\
a\neq b,\\
\operatorname{IsDiscontinuous}(f,a),\\
\operatorname{IsDiscontinuous}(f,b)
\end{cases}
\Longrightarrow
(f(a^-),f(a^+))\cap(f(b^-),f(b^+))=\varnothing.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The corollary states that a nondecreasing function assigns disjoint open value intervals to distinct jump discontinuities. Because the function never reverses direction, the lower one-sided limit at the later point cannot drop below the upper one-sided limit at the earlier point, so the corresponding ranges cannot overlap. This distinction of jump intervals ensures that the cumulative vertical jumps of a monotone function stay separated, preventing the graph from revisiting the same band of values via different discontinuities.
\end{remark*}

\begin{theorembox}{Theorem (Discontinuities of a Monotone Function Are Countable)}
\begin{theorem}[Discontinuities of a Monotone Function Are Countable]
\label{thm:monotone-discontinuities-countable}
Let $I \subseteq \mathbb{R}$ be an interval and let $f:I\to\mathbb{R}$ be monotone. Then the set $D_f \coloneqq \{c \in I : f \text{ is discontinuous at } c\}$ is at most countable. \hyperref[prf:monotone-discontinuities-countable]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } I \subseteq \mathbb{R} \text{ be an interval and } f:I\to\mathbb{R} \text{ be monotone}.\\
&\#\{c \in I : f \text{ fails to be continuous at } c\} \le \aleph_0.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{Monotone}\bigl(f,\mathsf{OrderedSet}(I,\leq)\bigr) \Longrightarrow \#\{c \in I : \operatorname{IsDiscontinuous}(f,c)\} \le \aleph_0.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Every discontinuity of a monotone function is a jump discontinuity with a nondegenerate vertical gap $(f(c^-),f(c^+))$. Distinct discontinuities yield pairwise disjoint gaps, and each such interval contains a rational number. Associating to every discontinuity the rational number inside its gap produces an injection of the discontinuity set into $\mathbb{Q}$, forcing the set to be countable. The only way the conclusion could fail would be the existence of uncountably many disjoint jump intervals, which is impossible because the rationals are countable.
\end{remark*}

\begin{corollarybox}{Corollary (Monotone Function Is Continuous on a Dense Set)}
\begin{corollary}[Monotone Function Is Continuous on a Dense Set]
\label{cor:monotone-continuous-on-dense-set}
Let $f:[a,b]\to\mathbb{R}$ be monotone. Then the set $\{x\in[a,b]:f \text{ is continuous at } x\}$ is dense in $[a,b]$.
\hyperref[prf:monotone-continuous-on-dense-set]{Go to proof.}
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } f:[a,b]\to\mathbb{R} \text{ be monotone.}\\
&\forall u,v\in[a,b] \; (u<v \Rightarrow \exists c\in(u,v)\text{ such that }f \text{ is continuous at } c).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{Monotone}\bigl(f,\mathsf{OrderedSet}([a,b],\leq)\bigr)
\Longrightarrow
\operatorname{IsDense}\bigl(\{x\in[a,b]:\operatorname{IsContinuous}(f,x)\},[a,b]\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists u,v\in[a,b]\;
\Bigl(u<v \land
\forall c\in(u,v)\;\exists \varepsilon_c>0\;\forall \delta>0\;\exists x\in[a,b]\;
\bigl(|x-c|<\delta\land |f(x)-f(c)|\ge\varepsilon_c\bigr)\Bigr).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\exists u,v\in[a,b]\;
\Bigl(u<v \land
\forall c\in(u,v)\;\neg\operatorname{IsContinuous}(f,c)\Bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
A monotone function on a closed interval can only jump at most countably many times, so the points of discontinuity leave room inside every subinterval for a continuity point. Thus, although the function may fail to be continuous at isolated or countably many locations, continuity still occurs throughout any interval one inspects. This property ensures that monotone functions retain many features of continuous functions when arguments depend only on dense sets of continuity points.
\end{remark*}

\begin{propositionbox}{Proposition (Continuous Injective Is Strictly Monotone)}
\begin{proposition}[Continuous Injective Is Strictly Monotone]
\label{prop:continuous-injective-iff-strictly-monotone}
Let $f:[a,b]\to\mathbb{R}$ be continuous on $[a,b]$. Then $f$ is injective if and only if $f$ is strictly monotone on $[a,b]$.
\hyperref[prf:continuous-injective-iff-strictly-monotone]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } f:[a,b]\to\mathbb{R} \text{ be continuous on } [a,b].\\
&\bigl[\forall x,y\in[a,b]\;(f(x)=f(y)\Rightarrow x=y)\bigr] \Longleftrightarrow
\bigl[\forall x,y\in[a,b]\;(x<y\Rightarrow f(x)<f(y))\bigr]\\
&\qquad\qquad\qquad\quad\lor
\bigl[\forall x,y\in[a,b]\,(x<y\Rightarrow f(y)<f(x))\bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{IsContinuous}(f,[a,b]) \Longrightarrow
\bigl(\operatorname{Injective}(f) \Longleftrightarrow
\operatorname{StrictlyMonotone}(f,[a,b])\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
A continuous function on a closed interval cannot reverse direction without repeating a value, so injectivity enforces a single increasing or decreasing trend across the entire domain. Conversely any strictly monotone map is automatically injective. The result emphasizes that topological constraints supplied by continuity interact with order-theoretic injectivity to force global monotonicity; failure occurs precisely when the graph turns back on itself, producing repeated values. Geometrically the continuous injective graph must cross each horizontal line at most once, which is only possible when the graph rises everywhere or falls everywhere.
\end{remark*}

\begin{theorembox}{Theorem (Continuous Inverse Theorem)}
\begin{theorem}[Continuous Inverse Theorem]
\label{thm:continuous-inverse-theorem}
Let $I\subseteq\mathbb{R}$ be an interval and let $f:I\to\mathbb{R}$ be strictly monotone and continuous on $I$. Then the inverse function $f^{-1}:f(I)\to I$ exists, is strictly monotone, and is continuous on $f(I)$.
\hyperref[prf:continuous-inverse-theorem]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } I\subseteq\mathbb{R} \text{ be an interval and } f:I\to\mathbb{R}.\\
&\Bigl[(\forall x,y\in I)(x<y \Rightarrow f(x)<f(y)) \land \forall c\in I\;\forall \varepsilon>0\;\exists \delta>0\;\forall x\in I\bigl(|x-c|<\delta \Rightarrow |f(x)-f(c)|<\varepsilon\bigr)\Bigr]\\
&\quad\Longrightarrow \Bigl[\exists g:f(I)\to I \text{ such that } (\forall x\in I)\, g(f(x))=x \land (\forall y\in f(I))\, f(g(y))=y\\
&\qquad\land (\forall u,v\in f(I))(u<v \Rightarrow g(u)<g(v)) \land \forall y\in f(I)\;\forall \varepsilon>0\;\exists \eta>0\;\forall u\in f(I)\bigl(|u-y|<\eta \Rightarrow |g(u)-g(y)|<\varepsilon\bigr)\Bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
The displayed quantified statement is the predicate-level form of $label: its hypotheses name the input conditions, and its conclusion names the asserted structure or relation.
\end{remark*}

\begin{remark*}[Interpretation]
A continuous strictly monotone function on an interval never reverses the order of points, so it is automatically bijective onto its image. This theorem records that the inverse map retains the same order orientation and inherits continuity, reflecting the fact that no sudden jumps can occur when the original graph crosses each horizontal line exactly once. Failure would require either loss of monotonicity or a discontinuity, either of which would create a flat spot or a jump preventing the existence of a continuous inverse. Geometrically, the graph of $f^{-1}$ is the reflection of the graph of $f$ across the diagonal line $y=x$, so the smoothness and strict increase or decrease are visibly preserved by symmetry.
\end{remark*}
