\subsection*{Real-Valued Functions}
\label{sec:real-valued-functions}

\begin{remark*}[Background from Volume I]
Volume I develops the set-theoretic theory of functions, including function
equality, composition, inverse functions, and one-sided inverse behavior.  This
section does not repeat that material.  Instead, it studies the extra algebraic
and order structure available when the codomain is \(\mathbb{R}\).
\end{remark*}

\begin{definitionbox}{Definition (At-Point Operation)}
\begin{definition}[At-Point Operation]
\label{def:at-point-operation}
Let \(A\subseteq\mathbb{R}\), let \(x\in A\), let
\(f_1,\ldots,f_n:A\to\mathbb{R}\), and let
\(\Phi:\mathbb{R}^n\to\mathbb{R}\) be an \(n\)-variable real operation.
The at-point application of \(\Phi\) to \(f_1,\ldots,f_n\) at \(x\) is the
real number
\[
  \Phi\bigl(f_1(x),\ldots,f_n(x)\bigr).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  x\in A
  \quad\Longrightarrow\quad
  \Phi\bigl(f_1(x),\ldots,f_n(x)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{AtPointOperation}(\Phi,(f_1,\ldots,f_n),x).
\]
\end{remark*}

\begin{remark*}[Interpretation]
At a single input, no new function has been built yet.  The operation is just
ordinary real arithmetic after the listed functions have been evaluated at the
same point.
\end{remark*}

\begin{definitionbox}{Definition (Pointwise Operation on a Set)}
\begin{definition}[Pointwise Operation on a Set]
\label{def:pointwise-operation-on-set}
Let \(A\subseteq\mathbb{R}\), let \(f_1,\ldots,f_n:A\to\mathbb{R}\),
and let \(\Phi:\mathbb{R}^n\to\mathbb{R}\) be an \(n\)-variable real
operation.  The pointwise operation determined by \(\Phi\) on \(A\) is the
function \(F:A\to\mathbb{R}\) defined by
\[
  F(x)=\Phi\bigl(f_1(x),\ldots,f_n(x)\bigr)
  \qquad (x\in A).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  \forall x\in A\;
  \bigl(F(x)=\Phi(f_1(x),\ldots,f_n(x))\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{PointwiseOperation}(\Phi,(f_1,\ldots,f_n),A).
\]
\end{remark*}

\begin{remark*}[Interpretation]
A pointwise operation applies the same real operation at every input in the
domain.
\end{remark*}

\begin{definitionbox}{Definition (At-Point Relation)}
\begin{definition}[At-Point Relation]
\label{def:at-point-relation}
Let \(A\subseteq\mathbb{R}\), let \(x\in A\), let
\(f_1,\ldots,f_n:A\to\mathbb{R}\), and let \(R\) be an \(n\)-variable
relation on \(\mathbb{R}\).  The relation \(R\) holds at \(x\) for
\(f_1,\ldots,f_n\) if
\[
  R\bigl(f_1(x),\ldots,f_n(x)\bigr)
\]
holds.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  R\text{ holds at }x
  \;\Longleftrightarrow\;
  R\bigl(f_1(x),\ldots,f_n(x)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  R\colon \mathbb{R}^n\to\{\mathrm{true},\mathrm{false}\},\qquad
  \operatorname{AtPointRelation}(R,(f_1,\ldots,f_n),x).
\]
\end{remark*}

\begin{remark*}[Interpretation]
For example, \(f\leq g\) at \(x\) means exactly \(f(x)\leq g(x)\).
\end{remark*}

\begin{definitionbox}{Definition (Pointwise Relation on a Set)}
\begin{definition}[Pointwise Relation on a Set]
\label{def:pointwise-relation-on-set}
Let \(A\subseteq\mathbb{R}\), let \(f_1,\ldots,f_n:A\to\mathbb{R}\),
and let \(R\) be an \(n\)-variable relation on \(\mathbb{R}\).  The
relation \(R\) holds pointwise on \(A\) for \(f_1,\ldots,f_n\) if
\[
  R\bigl(f_1(x),\ldots,f_n(x)\bigr)
  \qquad (x\in A).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  \forall x\in A\;
  R\bigl(f_1(x),\ldots,f_n(x)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  R\colon \mathbb{R}^n\to\{\mathrm{true},\mathrm{false}\},\qquad
  \operatorname{PointwiseRelation}(R,(f_1,\ldots,f_n),A)
  \;\Longleftrightarrow\;
  \forall x\in A\;R(f_1(x),\ldots,f_n(x)).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Thus \(f\leq g\) on \(A\) means \(f(x)\leq g(x)\) for every \(x\in A\).
This compares two functions at the same input, not one function at two ordered
inputs.
\end{remark*}

\begin{definitionbox}{Definition (Pointwise Relation Near a Point)}
\begin{definition}[Pointwise Relation Near a Point]
\label{def:pointwise-relation-near-point}
Let \(A\subseteq\mathbb{R}\), let \(c\in\mathbb{R}\), let
\(f_1,\ldots,f_n:A\to\mathbb{R}\), and let \(R\) be an \(n\)-variable
relation on \(\mathbb{R}\).  The relation \(R\) holds pointwise near \(c\)
in \(A\) for \(f_1,\ldots,f_n\) if there exists \(\delta>0\) such that
\[
  x\in A,\quad 0<|x-c|<\delta
  \quad\Longrightarrow\quad
  R\bigl(f_1(x),\ldots,f_n(x)\bigr).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  \exists\delta>0\;\forall x\in A\;
  \bigl(0<|x-c|<\delta
  \Rightarrow R(f_1(x),\ldots,f_n(x))\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  R\colon \mathbb{R}^n\to\{\mathrm{true},\mathrm{false}\},\qquad
  \operatorname{PointwiseRelationNear}(R,(f_1,\ldots,f_n),c,A).
\]
\end{remark*}

\begin{remark*}[Interpretation]
This is the precise form of phrases such as \(f\leq g\) near \(c\).  The
base point itself is omitted because limit statements are punctured.
\end{remark*}

\begin{propositionbox}{Proposition (Pointwise Operation Evaluation)}
\begin{proposition}[Pointwise Operation Evaluation]
\label{prop:pointwise-operation-evaluation}
Let \(A\subseteq\mathbb{R}\), let \(f_1,\ldots,f_n:A\to\mathbb{R}\),
and let \(\Phi:\mathbb{R}^n\to\mathbb{R}\) be an \(n\)-variable real
operation.  If \(F\) is the pointwise operation determined by
\(\Phi\) and \(f_1,\ldots,f_n\) on \(A\), then for every \(x\in A\),
\[
  F(x)
  =
  \Phi\bigl(f_1(x),\ldots,f_n(x)\bigr).
\]
\hyperref[prf:pointwise-operation-evaluation]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
  \forall x\in A\;
  \bigl(F(x)=\Phi(f_1(x),\ldots,f_n(x))\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Evaluating a pointwise operation at a point means first evaluating all input
functions at that point and then applying the underlying real operation.
\end{remark*}

\begin{propositionbox}{Proposition (Pointwise Relation Evaluation)}
\begin{proposition}[Pointwise Relation Evaluation]
\label{prop:pointwise-relation-evaluation}
Let \(A\subseteq\mathbb{R}\), let \(f_1,\ldots,f_n:A\to\mathbb{R}\),
and let \(R\) be an \(n\)-variable relation on \(\mathbb{R}\).  Then
\(R\) holds pointwise on \(A\) for \(f_1,\ldots,f_n\) if and only if
\(R\) holds at every \(x\in A\) for \(f_1,\ldots,f_n\).
\hyperref[prf:pointwise-relation-evaluation]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{PointwiseRelation}(R,(f_1,\ldots,f_n),A)
  \;\Longleftrightarrow\;
  \forall x\in A\;\operatorname{AtPointRelation}(R,(f_1,\ldots,f_n),x).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Pointwise relation on a set is exactly the collection of all its at-point
relation statements.
\end{remark*}

\begin{propositionbox}{Proposition (Pointwise Relation Near a Point Unpacking)}
\begin{proposition}[Pointwise Relation Near a Point Unpacking]
\label{prop:pointwise-relation-near-unpacking}
Let \(A\subseteq\mathbb{R}\), let \(c\in\mathbb{R}\), let
\(f_1,\ldots,f_n:A\to\mathbb{R}\), and let \(R\) be an \(n\)-variable
relation on \(\mathbb{R}\).  Then \(R\) holds pointwise near \(c\) in

\begin{definitionbox}{Definition (Pointwise Sum of Functions)}
\begin{definition}[Pointwise Sum of Functions]
\label{def:pointwise-sum-of-functions}
Let $X\subseteq\mathbb{R}$ and let $f,g:X\to\mathbb{R}$.  The
pointwise sum $f+g:X\to\mathbb{R}$ is the function defined by
\[
  (f+g)(x)=f(x)+g(x)
  \qquad (x\in X).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  \forall x\in X\;\bigl((f+g)(x)=f(x)+g(x)\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The sum is computed output by output; at each input, add the two function
values at that same input.
\end{remark*}

\begin{definitionbox}{Definition (Pointwise Quotient of Functions)}
\begin{definition}[Pointwise Quotient of Functions]
\label{def:pointwise-quotient-of-functions}
Let $X\subseteq\mathbb{R}$ and let $f,g:X\to\mathbb{R}$.  Suppose
$g(x)\neq 0$ for every $x\in X$.  The pointwise quotient $f/g:X\to\mathbb{R}$
is the function defined by
\[
  (f/g)(x)=\frac{f(x)}{g(x)}
  \qquad (x\in X).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  \bigl(\forall x\in X\;(g(x)\neq 0)\bigr)
  \;\Longrightarrow\;
  \forall x\in X\;\bigl((f/g)(x)=f(x)/g(x)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  f/g
  =
  \bigl(x\mapsto f(x)/g(x)\bigr)\big|_{X}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The quotient is defined pointwise only when the denominator function never
vanishes on the domain.
\end{remark*}

\begin{definitionbox}{Definition (Pointwise Reciprocal of a Function)}
\begin{definition}[Pointwise Reciprocal of a Function]

Let \(X\subseteq\mathbb{R}\) and let \(f:X\to\mathbb{R}\).  Suppose
\(f(x)\neq0\) for every \(x\in X\).  The pointwise reciprocal

\end{remark*}

\begin{remark*}[Interpretation]
The reciprocal is the quotient of the constant function \(1\) by \(f\), and it
is defined on the whole domain only when \(f\) is nowhere zero there.

\begin{theorembox}{Theorem (Function Algebra Closure)}
\begin{theorem}[Function Algebra Closure]
\label{thm:function-algebra-closure}
Let \(A\subseteq\mathbb{R}\), let \(f,g:A\to\mathbb{R}\), and let
\(\lambda\in\mathbb{R}\).  Then \(f+g\), \(f-g\), \(fg\), \(\lambda f\),
\(|f|\), \(\max(f,g)\), and \(\min(f,g)\) are functions \(A\to\mathbb{R}\).
\hyperref[prf:function-algebra-closure]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
Each listed function assigns to every \(x\in A\) a real number by applying the
corresponding real operation to \(f(x)\) and \(g(x)\).
\end{remark*}

\begin{remark*}[Interpretation]
Real-valued functions are closed under the basic pointwise algebra operations
whose real output is always defined.
\end{remark*}

\begin{theorembox}{Theorem (Function Quotient Closure)}
\begin{theorem}[Function Quotient Closure]
\label{thm:function-quotient-closure}
Let \(A\subseteq\mathbb{R}\) and let \(f,g:A\to\mathbb{R}\).  If
\(g(x)\neq0\) for every \(x\in A\), then \(f/g:A\to\mathbb{R}\) is a
well-defined function.
\hyperref[prf:function-quotient-closure]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
  \bigl(\forall x\in A\;(g(x)\neq0)\bigr)
  \;\Longrightarrow\;
  \forall x\in A\;\bigl((f/g)(x)\in\mathbb{R}\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  f/g
  =
  \bigl(x\mapsto f(x)/g(x)\bigr)\big|_{A}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The only extra closure condition for a quotient is that the denominator never
vanishes on the domain.
\end{remark*}

\begin{definitionbox}{Definition (Pointwise Equality on a Set)}
\begin{definition}[Pointwise Equality on a Set]
\label{def:pointwise-equality-on-set}
Let \(A\subseteq\mathbb{R}\) and let \(f,g:A\to\mathbb{R}\).  The

\[
  f(x)=g(x)
  \qquad (x\in A).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  f=g\text{ on }A
  \;\Longleftrightarrow\;
  \forall x\in A\;\bigl(f(x)=g(x)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{PointwiseRelation}(=,(f,g),A).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \neg(f=g\text{ on }A)
  \;\Longleftrightarrow\;
  \exists x\in A\;\bigl(f(x)\neq g(x)\bigr).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  \neg(f=g\text{ on }A)
  \;\Longleftrightarrow\;
  \exists x\in A\;\bigl(f(x)\neq g(x)\bigr).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
  The negated statement isolates the obstruction to the asserted condition.

  \item[\exists-condition.]
  The displayed obstruction is the corresponding failure mode.
  \[
  \neg(f=g\text{ on }A)
  \;\Longleftrightarrow\;
  \exists x\in A\;\bigl(f(x)\neq g(x)\bigr).
\]
  \[
  \neg(f=g\text{ on }A)
  \;\Longleftrightarrow\;
  \exists x\in A\;\bigl(f(x)\neq g(x)\bigr).
\]
\end{description}
\end{remark*}

two real-valued functions are equal exactly when every value agrees.  The
generic predicate \(\operatorname{PointwiseRelation}(R,(f,g),A)\)
specializes to \(f\leq g\), \(f<g\), \(f=g\), \(f\geq g\), and \(f>g\) on

\begin{theorembox}{Theorem (Pointwise Order Laws)}
\begin{theorem}[Pointwise Order Laws]
\label{thm:pointwise-order-laws}
Let \(A\subseteq\mathbb{R}\).  The relation \(\leq\) on real-valued
functions \(A\to\mathbb{R}\), defined pointwise, is a partial order:
\[
  f\leq f,\qquad
  (f\leq g\ \text{and}\ g\leq f)\Rightarrow f=g,\qquad
  (f\leq g\ \text{and}\ g\leq h)\Rightarrow f\leq h.
\]
\hyperref[prf:pointwise-order-laws]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
Each order law is obtained by fixing \(x\in A\) and applying the
corresponding order law in \(\mathbb{R}\) to the values at \(x\).
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \leq\colon \mathbb{R}^2\to\{\mathrm{true},\mathrm{false}\},\qquad
  \operatorname{PointwiseRelation}(\leq,(f,g),A)
  \;\Longleftrightarrow\;
  \forall x\in A\;\bigl(f(x)\leq g(x)\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Pointwise order compares functions vertically at the same input.  It is only a
partial order because many functions cross and are therefore incomparable.
\end{remark*}

\begin{theorembox}{Theorem (Pointwise Order Arithmetic)}
\begin{theorem}[Pointwise Order Arithmetic]
\label{thm:pointwise-order-arithmetic}
Let \(A\subseteq\mathbb{R}\) and let \(f,g,h:A\to\mathbb{R}\).  If
\(f\leq g\) on \(A\), then \(f+h\leq g+h\) on \(A\),
\(-g\leq -f\) on \(A\), \(\lambda f\leq\lambda g\) on \(A\) for
\(\lambda\geq0\), and \(\lambda g\leq\lambda f\) on \(A\) for
\(\lambda\leq0\).  If also \(f_1\leq g_1\) and \(f_2\leq g_2\) on \(A\),
then \(f_1+f_2\leq g_1+g_2\) on \(A\).
\hyperref[prf:pointwise-order-arithmetic]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
  \forall x\in A\;(f(x)\leq g(x))
  \;\Longrightarrow\;
  \forall x\in A\;(f(x)+h(x)\leq g(x)+h(x)).
\]
The scalar and negation laws are obtained by the corresponding real order laws
at each \(x\in A\).
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \leq\colon \mathbb{R}^2\to\{\mathrm{true},\mathrm{false}\},\qquad
  \operatorname{PointwiseRelation}(\leq,(f,g),A)
  \;\Longrightarrow\;
  \operatorname{PointwiseRelation}(\leq,(f+h,g+h),A).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Every permitted manipulation is a real inequality manipulation performed at a
fixed input and then requantified over \(A\).
\end{remark*}

\begin{propositionbox}{Proposition (Reciprocal Order Reversal)}
\begin{proposition}[Reciprocal Order Reversal]
\label{prop:reciprocal-order-reversal}
Let \(A\subseteq\mathbb{R}\) and let \(f,g:A\to\mathbb{R}\).  If
\(0<f\leq g\) on \(A\), then
\[
  \frac{1}{g}\leq\frac{1}{f}

\]

\[
  \forall x\in A\;\bigl(0<f(x)\leq g(x)\bigr)

\]
\end{remark*}

\[

  \operatorname{PointwiseRelation}(\leq,(1/g,1/f),A),
\]
under the pointwise positivity hypothesis \(0<f\) on \(A\).

Taking reciprocals reverses inequalities on positive real values, so the
pointwise reciprocal reverses the pointwise order.

finite linear-combination rules for functions.

\begin{definitionbox}{Definition (Real-Linear Rule)}
\begin{definition}[Real-Linear Rule]
\label{def:real-linear-rule}
Let $\mathcal{C}$ be a class of real-valued functions closed under real
linear combinations.  A rule $T$ on $\mathcal{C}$ is real-linear if,
whenever $f,g\in\mathcal{C}$ and $\alpha,\beta\in\mathbb{R}$,
\[
  T(\alpha f+\beta g)=\alpha T(f)+\beta T(g)
\]
whenever both sides are defined.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  f,g\in\mathcal{C}
  \;\Longrightarrow\;
  T(\alpha f+\beta g)=\alpha T(f)+\beta T(g).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Linearity records exactly the behavior of preserving addition and scalar
multiplication. Nonlinear operations such as products, quotients, absolute
values, and order comparisons require their own dependency information.
\end{remark*}

\begin{propositionbox}{Proposition}
\begin{proposition}[Real-Linearity Test]
\label{prop:real-linear-rule-test}
Let \(\mathcal{C}\) be closed under real linear combinations.  A rule
\(T\) on \(\mathcal{C}\) is real-linear if and only if
\[
  T(f+g)=T(f)+T(g)
  \quad\text{and}\quad
  T(\alpha f)=\alpha T(f)
\]
whenever the displayed expressions are defined.

\smallskip\noindent
\hyperref[prf:real-linear-rule-test]{\textit{Go to proof.}}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
  T\text{ is real-linear}
  \;\Longleftrightarrow\;
  T\text{ is additive and homogeneous over }\mathbb{R}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The two-rule form is often easier to verify; the linear-combination form is
often easier to use.
\end{remark*}

\begin{definition}[Bounded Above Function]
\label{def:function-bounded-above}

  f(x)\leq M
  \qquad (x\in A).

\end{definition}

\[
  \exists M\in\mathbb{R}\;\forall x\in A\;\bigl(f(x)\leq M\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{BoundedAbove}(f,A)
  \;\Longleftrightarrow\;
  \exists M\in\mathbb{R}\;\forall x\in A\;\bigl(f(x)\leq M\bigr).
\]
\end{remark*}

$f$ is bounded on $A$ if there exists $B>0$ such that
\[
  |f(x)|\leq B
  \qquad (x\in A).
\]
\end{definition}
\end{definitionbox}

\[
  \exists B>0\;\forall x\in A\;\bigl(|f(x)|\leq B\bigr).
\]
\end{remark*}

  \;\coloneqq\;
  \exists B>0\;\forall x\in A\;\bigl(|f(x)|\leq B\bigr).

  \forall B>0\;\exists x\in A\;\bigl(|f(x)|>B\bigr).

\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}

  \;\Longleftrightarrow\;
  \forall B>0\;\exists x\in A\;\bigl(|f(x)|>B\bigr).

\end{description}
\end{remark*}

  \item[\operatorname{Bounded}.]

  \forall B>0\;\exists x\in A\;\bigl(|f(x)|>B\bigr).

  \;\Longleftrightarrow\;

\end{theorem}

\begin{definitionbox}{Definition (Supremum of a Function on a Set)}
\begin{definition}[Supremum of a Function on a Set]
\label{def:function-supremum-on-set}
Let $A\subseteq\mathbb{R}$ and let $f:A\to\mathbb{R}$.  The supremum of
$f$ on $A$ is
\[
  \sup_{x\in A} f(x) = \sup f(A),
\]
provided the supremum of $f(A)$ exists.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  s=\sup_{x\in A}f(x)
  \;\Longleftrightarrow\;
  \bigl(\forall x\in A\;(f(x)\leq s)\bigr)
  \;\wedge\;
  \bigl(\forall u\in\mathbb{R}\;((\forall x\in A\;(f(x)\leq u))\Rightarrow s\leq u)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  \operatorname{LeastUpperBound}(s,f(A),\mathsf{OrderedSet}(\mathbb{R},\leq))
  \;\Longleftrightarrow\;
  s=\sup_{x\in A}f(x).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The supremum is the least real ceiling for the output set $f(A)$; it need
not be an output value of $f$.
\end{remark*}

\begin{propositionbox}{Proposition (Supremum Monotonicity Converse Fails)}
\begin{proposition}[Supremum Monotonicity Converse Fails]
\label{prop:supremum-monotonicity-converse-fails}
There exist \(f,g:A\to\mathbb{R}\) such that \(\sup_A f\leq\sup_A g\) but
\(f\leq g\) does not hold on \(A\).
\hyperref[prf:supremum-monotonicity-converse-fails]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
  \sup_A f\leq\sup_A g
  \;\not\Longrightarrow\;
  \operatorname{PointwiseRelation}(\leq,(f,g),A).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
A scalar comparison of two least upper bounds loses pointwise information.
\[
  \sup_A f\leq\sup_A g
  \quad\text{and}\quad
  \exists x\in A\;\bigl(f(x)>g(x)\bigr)
\]
can hold simultaneously.


  \item[\exists\text{-condition}.]
  The displayed obstruction is the corresponding failure mode.
  \[
  \sup_A f\leq\sup_A g
  \quad\text{and}\quad
  \exists x\in A\;\bigl(f(x)>g(x)\bigr)
\]
  \[
  \sup_A f\leq\sup_A g
  \quad\text{and}\quad
  \exists x\in A\;\bigl(f(x)>g(x)\bigr)
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Supremum monotonicity is one-directional.
\end{remark*}

If \(f\) and \(g\) are bounded above on \(A\), then
\[
  \sup_A(f+g)\leq \sup_A f+\sup_A g.
\]
\hyperref[prf:supremum-subadditivity]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{BoundedAbove}(f,A)\wedge\operatorname{BoundedAbove}(g,A)
  \Rightarrow
  \sup_A(f+g)\leq \sup_A f+\sup_A g.
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
Equality need not hold because \(f\) and \(g\) may attain their largest values
at different inputs.
\[
  \sup_A(f+g)<\sup_A f+\sup_A g
\]
can occur.


  \item[\exists\text{-condition}.]
  The displayed obstruction is the corresponding failure mode.
  \[
  \sup_A(f+g)<\sup_A f+\sup_A g
\]
  \[
  \sup_A(f+g)<\sup_A f+\sup_A g
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
The sum of separate ceilings is a ceiling for the pointwise sum.
\end{remark*}

  (f+g)(x)\leq (f+g)(x_0).
\]
\end{remark*}

\begin{propositionbox}{Proposition (Common Extremum Converse Fails)}
\begin{proposition}[Common Extremum Converse Fails]
\label{prop:common-extremum-converse-fails}
The converse of the common maximum theorem is false: \(f+g\) may attain a
maximum at a point where neither \(f\) nor \(g\) attains its maximum.
\hyperref[prf:common-extremum-converse-fails]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
  x_0\text{ maximizes }f+g
  \;\not\Longrightarrow\;
  x_0\text{ maximizes }f\text{ and }x_0\text{ maximizes }g.
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
The summands can trade off: one decreases while the other increases, so their
sum is maximized at a balancing point.
\[
  x_0\in\operatorname*{arg\,max}_A(f+g)
  \quad\text{but}\quad
  x_0\notin\operatorname*{arg\,max}_A f.
\]


  \item[\exists\text{-condition}.]

  \[
  x_0\in\operatorname*{arg\,max}_A(f+g)
  \quad\text{but}\quad
  x_0\notin\operatorname*{arg\,max}_A f.
\]
  \[
  x_0\in\operatorname*{arg\,max}_A(f+g)
  \quad\text{but}\quad
  x_0\notin\operatorname*{arg\,max}_A f.
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
The common-extremum theorem is one-directional.

\begin{definitionbox}{Definition (Increasing Function)}
\begin{definition}[Increasing Function]
\label{def:function-increasing}
Let $A\subseteq\mathbb{R}$ and let $f:A\to\mathbb{R}$.  The function
$f$ is increasing on $A$ if
\[
  x_1<x_2\;\Longrightarrow\;f(x_1)\leq f(x_2)
  \qquad (x_1,x_2\in A).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  \forall x_1,x_2\in A\;\bigl(x_1<x_2\Rightarrow f(x_1)\leq f(x_2)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  A\subseteq\mathbb{R},\qquad f\colon A\to\mathbb{R},\qquad
  \operatorname{OrderPreserving}(f,\mathsf{OrderedSet}(A,\leq),\mathsf{OrderedSet}(\mathbb{R},\leq))
  \;\Longleftrightarrow\;
  \forall x_1,x_2\in A\;\bigl(x_1<x_2\Rightarrow f(x_1)\leq f(x_2)\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \exists x_1,x_2\in A\;\bigl(x_1<x_2\;\wedge\;f(x_1)>f(x_2)\bigr).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  \exists x_1,x_2\in A\;\bigl(x_1<x_2\;\wedge\;f(x_1)>f(x_2)\bigr).
\]
\end{remark*}



\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
The function fails to be increasing if some later input has a smaller
output than an earlier input.


  \item[\exists-condition.]

  \exists x_1,x_2\in A\;\bigl(x_1<x_2\;\wedge\;f(x_1)>f(x_2)\bigr).

\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]

\[
  \exists x_1,x_2\in A\;\bigl(x_1<x_2\;\wedge\;f(x_1)>f(x_2)\bigr).

  \item[\exists-condition.]

  \exists x_1,x_2\in A\;\bigl(x_1<x_2\;\wedge\;f(x_1)>f(x_2)\bigr).

  \exists x_1,x_2\in A\;\bigl(x_1<x_2\;\wedge\;f(x_1)>f(x_2)\bigr).

\label{def:function-strictly-increasing}

\begin{definitionbox}{Definition (Monotone Function)}
\begin{definition}[Monotone Function]
\label{def:function-monotone}
Let $A\subseteq\mathbb{R}$ and let $f:A\to\mathbb{R}$.  The function
$f$ is monotone on $A$ if $f$ is increasing on $A$ or decreasing on $A$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  \bigl(\forall x_1,x_2\in A\;(x_1<x_2\Rightarrow f(x_1)\leq f(x_2))\bigr)
  \;\vee\;
  \bigl(\forall x_1,x_2\in A\;(x_1<x_2\Rightarrow f(x_1)\geq f(x_2))\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  A\subseteq\mathbb{R},\qquad f\colon A\to\mathbb{R},\qquad P=\mathsf{OrderedSet}(A,\leq),\quad Q=\mathsf{OrderedSet}(\mathbb{R},\leq),
  \qquad
  \operatorname{Monotone}(f,P)
  \;\coloneqq\;
  \operatorname{OrderPreserving}(f,P,Q)
  \;\vee\;
  \operatorname{OrderReversing}(f,P,Q).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \bigl(\exists x_1,x_2\in A\;(x_1<x_2\wedge f(x_1)>f(x_2))\bigr)
  \;\wedge\;
  \bigl(\exists y_1,y_2\in A\;(y_1<y_2\wedge f(y_1)<f(y_2))\bigr).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  A\subseteq\mathbb{R},\qquad f\colon A\to\mathbb{R},\qquad P=\mathsf{OrderedSet}(A,\leq),\quad
  \neg\operatorname{Monotone}(f,P)
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
A nonmonotone function has evidence of both kinds of reversal: one pair
where the outputs go down as inputs increase, and another pair where the
outputs go up as inputs increase.


  \item[\operatorname{Monotone}.]
  The displayed obstruction is the corresponding failure mode.
  \[
  \bigl(\exists x_1,x_2\in A\;(x_1<x_2\wedge f(x_1)>f(x_2))\bigr)
  \;\wedge\;
  \bigl(\exists y_1,y_2\in A\;(y_1<y_2\wedge f(y_1)<f(y_2))\bigr).
\]
  \[
  A\subseteq\mathbb{R},\qquad f\colon A\to\mathbb{R},\qquad P=\mathsf{OrderedSet}(A,\leq),\quad
  \neg\operatorname{Monotone}(f,P)
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}

\[
  \begin{aligned}
  &\exists x_1,x_2\in A\;(x_1<x_2\wedge f(x_1)>f(x_2))\\
  &\qquad\wedge\;
  \exists y_1,y_2\in A\;(y_1<y_2\wedge f(y_1)<f(y_2)).
