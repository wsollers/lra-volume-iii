\begin{theorem*}[Convergence by Domination]
Let $(x_n) \subseteq \mathbb{R}$ and let $L \in \mathbb{R}$. If
$|x_n-L| \le c a_n$ on a tail, where $c>0$, $a_n>0$, and $a_n\to 0$,
then $x_n\to L$.
\end{theorem*}
