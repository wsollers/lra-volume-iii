\begin{theorem*}[Supremum Is Monotone Under Inclusion]
Let $A$ and $B$ be non-empty subsets of $\mathbb{R}$ that are bounded above. If $A \subseteq B$, then $\sup A \le \sup B$.
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
Let $s_B = \sup B$. Then $s_B$ is an upper bound of $B$, so
\[
\forall b \in B,\quad b \le s_B.
\]
Since $A \subseteq B$, it follows that
\[
\forall a \in A,\quad a \le s_B,
\]
so $s_B$ is an upper bound of $A$. Let $s_A = \sup A$. Since $s_A$ is the least upper bound of $A$, and $s_B$ is an upper bound of $A$, it follows that
\[
s_A \le s_B.
\]
\end{proof}

\begin{proof}
\textbf{Detailed Learning Proof.}~\\

\textbf{Step 1.} Let $s_B = \sup B$. By definition of supremum, $s_B$ is an upper bound of $B$, so $b \le s_B$ for all $b \in B$.

\textbf{Step 2.} Since $A \subseteq B$, every element of $A$ is an element of $B$. Therefore $a \le s_B$ for all $a \in A$, so $s_B$ is an upper bound of $A$.

\textbf{Step 3.} Let $s_A = \sup A$. By definition, $s_A$ is the least upper bound of $A$.

\textbf{Step 4.} Since $s_B$ is an upper bound of $A$, the defining property of the least upper bound gives $s_A \le s_B$.
\end{proof}

\begin{remark*}[Proof structure]
Direct argument using the definition of upper bound and the minimality property of the supremum.
\end{remark*}
