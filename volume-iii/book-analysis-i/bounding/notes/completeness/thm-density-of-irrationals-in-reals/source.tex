\begin{theorem}[Density of the Irrationals]
\label{thm:density-of-irrationals-in-reals}
For every $a,b\in\mathbb{R}$ with $a<b$, there exists
$s\in\mathbb{R}\setminus\mathbb{Q}$ such that
\[
  a<s<b.
\]
Equivalently,
\[
  (\forall a,b\in\mathbb{R})
  \bigl[a<b\Longrightarrow
  (\exists s\in\mathbb{R}\setminus\mathbb{Q})(a<s<b)\bigr].
\]
\hyperref[prf:density-of-irrationals-in-reals]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{R}\;\bigl(a<b\Longrightarrow \exists s\in\mathbb{R}\setminus\mathbb{Q}\;(a<s<b)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad I=\mathbb{R}\setminus\mathbb{Q},\qquad
D_I=\mathsf{OrderDenseSet}(I,P_{\mathbb{R}}),\qquad
\operatorname{IsOrderDenseSubset}(I,P_{\mathbb{R}})\Longleftrightarrow
\forall a,b\in\mathbb{R}\;\bigl(a<b\Longrightarrow \exists s\in\mathbb{R}\setminus\mathbb{Q}\;(a<s<b)\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists a,b\in\mathbb{R}\;\bigl(a<b\land \forall s\in\mathbb{R}\setminus\mathbb{Q}\;(s\le a\lor b\le s)\bigr).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad I=\mathbb{R}\setminus\mathbb{Q},\qquad
D_I=\mathsf{OrderDenseSet}(I,P_{\mathbb{R}}),\qquad
\neg\operatorname{IsOrderDenseSubset}(I,P_{\mathbb{R}})\Longleftrightarrow
\exists a,b\in\mathbb{R}\;\bigl(a<b\land \forall s\in\mathbb{R}\setminus\mathbb{Q}\;(s\le a\lor b\le s)\bigr).
\]
\end{remark*}

\begin{description}
\item[Exposition.]
The theorem would fail precisely if some nonempty real interval contained no irrational number.

\]
\[
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad I=\mathbb{R}\setminus\mathbb{Q},\qquad
D_I=\mathsf{OrderDenseSet}(I,P_{\mathbb{R}}),\qquad
\neg\operatorname{IsOrderDenseSubset}(I,P_{\mathbb{R}})\Longleftrightarrow
\exists a,b\in\mathbb{R}\;\bigl(a<b\land \forall s\in\mathbb{R}\setminus\mathbb{Q}\;(s\le a\lor b\le s)\bigr).
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}

\[
\neg\operatorname{IsOrderDenseSubset}(\mathbb{R}\setminus\mathbb{Q},\mathbb{R})
=
\underbrace{\exists a,b\in\mathbb{R}\;(a<b)}_{\text{nonempty real interval}}
\;\land\;
\underbrace{\forall s\in\mathbb{R}\setminus\mathbb{Q}\;(s\le a\lor b\le s)}_{\text{no irrational lies inside}}.
\]


  \item[\operatorname{IsOrderDenseSubset}.]

\]

\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Irrational numbers also penetrate every real interval. One way to see the result is to insert a rational point into a translated and rescaled interval, then shift it by an irrational amount. The outcome is that the irrational numbers, like the rational numbers, leave no interval gap in $\mathbb{R}$.
\end{remark*}
