% Riemann bars with a sample dot.  #1=color  #2=mode (1 L,2 R,3 mid,4 inf,5 sup)
\newcommand{\riemannbars}[2]{%
  \foreach \i in {0,...,5}{%
    \pgfmathsetmacro\xa{\i*(4/6)}%
    \pgfmathsetmacro\xb{\xa+4/6}%
    \pgfmathsetmacro\xs{(#2==1)*\xa+(#2==2)*\xb+(#2==3)*((\xa+\xb)/2)%
       +(#2==4)*((\xb<=2.0001)?\xa:\xb)+(#2==5)*((\xb<=2.0001)?\xb:\xa)}%
    \pgfmathsetmacro\hh{2.2-0.3*(\xs-2)*(\xs-2)}%
    \fill[#1!15] (\xa,0) rectangle (\xb,\hh);
    \draw[#1!55,line width=0.5pt] (\xa,0) rectangle (\xb,\hh);
    \node[atlas dot,fill=#1!80!atlasink,minimum size=3pt] at (\xs,\hh){};
  }%
}
% Fill-only bars for overlays (no dot, no edge).  #1=color!alpha  #2=mode
\newcommand{\riemannfill}[2]{%
  \foreach \i in {0,...,5}{%
    \pgfmathsetmacro\xa{\i*(4/6)}%
    \pgfmathsetmacro\xb{\xa+4/6}%
    \pgfmathsetmacro\xs{(#2==4)*((\xb<=2.0001)?\xa:\xb)+(#2==5)*((\xb<=2.0001)?\xb:\xa)}%
    \pgfmathsetmacro\hh{2.2-0.3*(\xs-2)*(\xs-2)}%
    \fill[#1] (\xa,0) rectangle (\xb,\hh);
  }%
}
