\begin{theorem*}[Absolute Bound Gives Upper and Lower Bounds]
Let $(x_n)$ be a real sequence, and let $K > 0$. If $|x_n| \le K$ for every
$n \in \mathbb{N}$, then $-K \le x_n \le K$ for every
$n \in \mathbb{N}$.

\smallskip
\noindent
\end{theorem*}
