\begin{lemma}[Rational Translation Preserves Rationality]
\label{lem:rational-translation-preserves-rationality}
For every $r,q\in\mathbb{Q}$,
\[
  r+q\in\mathbb{Q}.
\]
\hyperref[prf:rational-translation-preserves-rationality]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\forall r,q\in\mathbb{Q}\;(r+q\in\mathbb{Q}).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
r\in\mathbb{Q}\land q\in\mathbb{Q}\Longrightarrow r+q\in\mathbb{Q}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The rational numbers are closed under addition, so translating a rational by a
rational keeps it rational.
\end{remark*}
