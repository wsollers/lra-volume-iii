\begin{remark*}[Orientation]
In this section, \emph{lattice} refers to the max--min structure already
present in the usual order on $\mathbb{R}$.  For two real numbers, their
join is their maximum and their meet is their minimum:
\[
  x\vee y=\max\{x,y\},
  \qquad
  x\wedge y=\min\{x,y\}.
\]
Thus lattice operations are not a new order relation; they are the operations
created by the order.  The set-level constructions below apply those pairwise
operations to all pairs of points from two sets.
\end{remark*}

\begin{definitionbox}{Definition (Pairwise Maximum and Minimum Sets)}
\begin{definition}[Pairwise Maximum and Minimum Sets]
\label{def:pairwise-maximum-minimum-sets}
Let $A,B\subseteq\mathbb{R}$.  The \emph{pairwise maximum set} and
\emph{pairwise minimum set} of $A$ and $B$ are defined, respectively, by
\[
  A\vee B:=\{\max\{a,b\}:a\in A,\ b\in B\},
\]
and
\[
  A\wedge B:=\{\min\{a,b\}:a\in A,\ b\in B\}.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
x\in A\vee B
\Longleftrightarrow
(\exists a\in A)(\exists b\in B)\bigl(x=\max\{a,b\}\bigr),
\]
and
\[
x\in A\wedge B
\Longleftrightarrow
(\exists a\in A)(\exists b\in B)\bigl(x=\min\{a,b\}\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&A,B\subseteq\mathbb{R},\qquad D=A\times B,\\
&M=A\vee B=\{\max\{a,b\}:a\in A,\ b\in B\},\\
&N=A\wedge B=\{\min\{a,b\}:a\in A,\ b\in B\},\\
&\varphi_{\max}\colon D\to M,\qquad \varphi_{\min}\colon D\to N,\\
&\varphi_{\max}(a,b)=\max\{a,b\},\qquad
\varphi_{\min}(a,b)=\min\{a,b\},\\
&\operatorname{ReplacementImage}(M,D,\varphi_{\max}),\qquad
\operatorname{ReplacementImage}(N,D,\varphi_{\min}).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
x\notin A\vee B
\Longleftrightarrow
(\forall a\in A)(\forall b\in B)\bigl(x\neq\max\{a,b\}\bigr),
\]
and
\[
x\notin A\wedge B
\Longleftrightarrow
(\forall a\in A)(\forall b\in B)\bigl(x\neq\min\{a,b\}\bigr).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&A,B\subseteq\mathbb{R},\qquad
M=A\vee B=\{\max\{a,b\}:a\in A,\ b\in B\},\\
&C\ne M
\end{aligned}
\quad\text{means}\quad
(\exists x)\bigl[x\in C\triangle M\bigr],
\]
and similarly for $A\wedge B$.
\end{remark*}

\begin{theorembox}{Theorem (Supremum of the Pairwise Maximum Set)}
\begin{theorem}[Supremum of the Pairwise Maximum Set]
\label{thm:supremum-pairwise-maximum-set}
Let $A,B\subseteq\mathbb{R}$ be nonempty and bounded above. Then $A\vee B$
is bounded above and
\[
\sup(A\vee B)=\max\{\sup A,\sup B\}.
\]
\hyperref[prf:supremum-pairwise-maximum-set]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
A,B\neq\varnothing,\quad A,B\text{ bounded above}
\quad\Longrightarrow\quad
\sup(A\vee B)=\max\{\sup A,\sup B\}.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&M=A\vee B,\qquad U=\max\{s_A,s_B\},\\
&\operatorname{LeastUpperBound}(s_A,A,P)\land\operatorname{LeastUpperBound}(s_B,B,P)
\Longrightarrow
\operatorname{LeastUpperBound}(U,M,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\sup(A\vee B)\neq\max\{\sup A,\sup B\}
\]
is impossible under the stated hypotheses.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&M=A\vee B,\qquad U=\max\{s_A,s_B\},\\
&\operatorname{LeastUpperBound}(s_A,A,P)\land\operatorname{LeastUpperBound}(s_B,B,P)\land
\neg\operatorname{LeastUpperBound}(U,M,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
The formula could fail only if the maximum of the two suprema failed to bound
all pairwise maxima, or if it were not the least such bound.


  \item[\operatorname{LeastUpperBound}.]
  \[
\sup(A\vee B)\neq\max\{\sup A,\sup B\}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&M=A\vee B,\qquad U=\max\{s_A,s_B\},\\
&\operatorname{LeastUpperBound}(s_A,A,P)\land\operatorname{LeastUpperBound}(s_B,B,P)\land
\neg\operatorname{LeastUpperBound}(U,M,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Every pairwise maximum is bounded by $\max\{\sup A,\sup B\}$.  Sharpness comes
from holding one input fixed in the nonempty other set and letting the relevant
set approach its own supremum.


  \item[\operatorname{LeastUpperBound}.]
  \[
\sup(A\vee B)\neq\max\{\sup A,\sup B\}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&M=A\vee B,\qquad U=\max\{s_A,s_B\},\\
&\operatorname{LeastUpperBound}(s_A,A,P)\land\operatorname{LeastUpperBound}(s_B,B,P)\land
\neg\operatorname{LeastUpperBound}(U,M,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
The largest possible pairwise maximum is controlled by the larger of the two
individual upper endpoints.
\end{remark*}
