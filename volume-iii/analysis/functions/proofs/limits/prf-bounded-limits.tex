\begin{theorem*}[Bounded Limits]
Let $f : A \to \mathbb{R}$, let $c$ be a cluster point of $A$, and
let $a, b \in \mathbb{R}$. If $a \leq f(x) \leq b$ for all $x \in A$
with $x \neq c$, and $\lim_{x\to c}f(x)=L$, then
$a \leq L \leq b$.

\smallskip\noindent
\end{theorem*}
