\begin{remark*}[Interpretation]
This chapter is the narrative version of the reference list. The reference
orders the definitions correctly, but the real subject is the sequence of
defects and patches:
\[
\begin{aligned}
  \text{geometric problem}&\to \text{Cauchy}\to \text{Riemann}\\
  &\to \text{Darboux}\to \text{Stieltjes}\\
  &\to \text{Henstock--Kurzweil and McShane}\\
  &\to \text{Lebesgue}.
\end{aligned}
\]
Each integral answers a concrete complaint about the one before it. The point
is not to memorize several integrals, but to understand why each one had to be
invented.
\end{remark*}

\subsection*{The Geometric Problem}

\begin{exposition}
Forget integrals for a moment. The physical question is always the same:

\begin{quote}
Given a pointwise rate \(f\), how much total quantity accumulates from
\(a\) to \(b\)?
\end{quote}

A speedometer reads \(f(t)\); the accumulated quantity is distance. A wire has
linear density \(f(x)\); the accumulated quantity is mass. A curve \(y=f(x)\)
over \([a,b]\) asks for signed area. The only available strategy is:

\begin{enumerate}
  \item Chop \([a,b]\) into thin slabs.
  \item Pretend \(f\) is constant on each slab.
  \item Add the resulting rectangles.
  \item Refine the chopping and ask whether the sums settle.
\end{enumerate}

Every construction below makes two choices: which height to use in a slab, and
what it means for all these rectangle sums to settle to one number.
\end{exposition}

\begin{remark*}[Interpretation]
The constructions differ by the height rule, the ruler, or the fineness rule:
\[
\begin{array}{c|c|c}
\text{Construction} & \text{Height rule} & \text{Ruler / fineness}\\
\hline
\text{Cauchy} & f(x_{i-1})\text{, left endpoint} & \Delta x_i\\
\text{Riemann} & f(\xi_i)\text{, arbitrary tag} & \Delta x_i\\
\text{Darboux} & \inf_I f,\ \sup_I f\text{ on each slab} & \Delta x_i\\
\text{Riemann--Stieltjes} & f(\xi_i)\text{, tagged height} & \Delta\alpha_i\\
\text{Henstock--Kurzweil} & f(\xi_i),\ \xi_i\in I_i &
\Delta x_i,\ \delta(\xi_i)\text{-fine, tag tethered}\\
\text{McShane} & f(t_i),\ t_i\text{ may float} &
\Delta x_i,\ \delta(t_i)\text{-fine, tag untethered}
\end{array}
\]
Henstock--Kurzweil does not change the sampled height or the ordinary ruler;
it changes the admissible chopping by replacing a global mesh bound with a
local gauge condition. McShane keeps the same gauge idea but frees the tag
from the slab, which pulls the theory back toward Lebesgue's absolute notion
of size.
\end{remark*}

\begin{definitionbox}{Definition (Partition)}
\begin{definition}[Partition of a Compact Interval]
\label{def:integration-partition}
A \emph{partition} of \([a,b]\) is a finite increasing list
\[
  P=\{a=x_0<x_1<\cdots<x_n=b\}.
\]
Equivalently,
\[
\begin{aligned}
\operatorname{Partition}(P,[a,b])\Longleftrightarrow{}
&\exists n\in\mathbb N\,\exists x_0,\ldots,x_n\in\mathbb R\\
&\bigl(P=\{x_0,\ldots,x_n\}\wedge x_0=a\wedge x_n=b\bigr)\\
&{}\wedge \forall i\,\bigl(1\le i\le n\Rightarrow x_{i-1}<x_i\bigr).
\end{aligned}
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
A partition \(P\) of \([a,b]\) is a finite increasing list
\[
  P=\{a=x_0<x_1<\cdots<x_n=b\}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
A partition records the cut-points that decompose the interval into finitely
many adjacent slabs.
\end{remark*}

\begin{definitionbox}{Definition (Subinterval Width)}
\begin{definition}[Subinterval Width]
\label{def:subinterval-width-integration}
The \(i\)-th slab has width
\[
  \Delta x_i=x_i-x_{i-1}.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
For a partition \(P=\{x_0,\ldots,x_n\}\), the \(i\)-th subinterval width is
\[
  \Delta x_i=x_i-x_{i-1}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The width measures the length of a single slab in the partition.
\end{remark*}

\begin{definitionbox}{Definition (Mesh)}
\begin{definition}[Mesh of a Partition]
\label{def:partition-mesh}
The \emph{mesh} or \emph{norm} of \(P\) is the width of its fattest slab:
\[
  \|P\|=\max_{1\le i\le n}\Delta x_i.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
For a partition \(P=\{x_0,\ldots,x_n\}\),
\[
  \|P\|=\max_{1\le i\le n}(x_i-x_{i-1}).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The mesh is the resolution dial. The phrase ``as the partition is refined''
will usually mean that \(\|P\|\to 0\): no slab is allowed to remain fat.
\end{remark*}

\begin{definitionbox}{Definition (Tagged Partition)}
\begin{definition}[Tagged Partition]
\label{def:tagged-partition-integration}
A \emph{tagged partition} is a partition \(P=\{x_0,\ldots,x_n\}\) together
with points \(\xi_i\in[x_{i-1},x_i]\). The tag is the point where the height
\(f(\xi_i)\) is sampled.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
A tagged partition of \([a,b]\) is a partition
\(P=\{x_0,\ldots,x_n\}\) together with points
\[
  \xi_i\in[x_{i-1},x_i]\qquad(1\le i\le n).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The partition chooses the slabs, while the tags choose where each slab samples
the function height.
\end{remark*}

\begin{definitionbox}{Definition (Refinement)}
\begin{definition}[Refinement of a Partition]
\label{def:partition-refinement}
A partition \(P'\) \emph{refines} \(P\) when \(P\subseteq P'\). Refinement adds
cut-points; it does not move or delete existing cut-points.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
A partition \(P'\) refines a partition \(P\) if
\[
  P\subseteq P'.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Refinement only adds cut-points. It makes the slab decomposition finer while
preserving the old cuts.
\end{remark*}

\begin{lemmabox}{Lemma (Common Refinement)}
\begin{lemma}[Common Refinement of Two Partitions]
\label{lem:common-refinement-partitions}
Any two partitions \(P\) and \(P'\) of \([a,b]\) have a common refinement:
order the finite set \(P\cup P'\) increasingly.
\hyperref[prf:common-refinement-partitions]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
For any two partitions \(P\) and \(P'\) of \([a,b]\), there exists a
partition \(R\) of \([a,b]\) such that
\[
  P\subseteq R
  \qquad\text{and}\qquad
  P'\subseteq R.
\]
\end{remark*}

\begin{remark*}[Interpretation]
This small fact is the reason independently chosen partitions can be compared.
Whenever a proof says ``refine them together,'' it is using this lemma. It is
the structural engine behind Cauchy's existence proof and Darboux's squeeze.
\end{remark*}
