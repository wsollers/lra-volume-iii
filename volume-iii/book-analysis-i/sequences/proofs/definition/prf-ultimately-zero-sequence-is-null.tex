\begin{theorem*}[Ultimately Zero Sequence is Null]
Let $(x_n)$ be a real sequence. If there exists $N_0 \in \mathbb{N}$ such
that $x_n = 0$ for every $n \ge N_0$, then $(x_n)$ is a null sequence.

\smallskip
\noindent
\end{theorem*}
