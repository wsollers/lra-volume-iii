\subsection*{Archimedean Property}

\begin{remark*}[Why this requires proof]
The Archimedean Property states that the natural numbers are unbounded in
$\mathbb{R}$. That conclusion is not a formal consequence of the ordered-field
axioms alone. There exist ordered fields satisfying all ordered-field axioms in
which the natural numbers are bounded above, so the statement must be justified
from stronger structure.

The needed structure is completeness. If $\mathbb{N}$ were bounded above in
$\mathbb{R}$, then the Least Upper Bound Property would produce a supremum
$\alpha \in \mathbb{R}$ for $\mathbb{N}$. But then $\alpha - 1$ cannot be an
upper bound, so there exists $n \in \mathbb{N}$ with $n > \alpha - 1$. Hence
$n+1 > \alpha$, contradicting the fact that $\alpha$ is an upper bound of
$\mathbb{N}$.

\smallskip
\noindent
\hyperref[prf:archimedean-property]{\textit{Go to proof.}}
\end{remark*}




\begin{theorembox}{Theorem (Archimedean Property)}
\begin{theorem}[Archimedean Property]
\label{thm:archimedean-property}
For every real number $x$, there exists a natural number $n$ such that $n>x$.

\smallskip
\noindent
\hyperref[prf:archimedean-property]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall x\in\mathbb{R}\;\exists n\in\mathbb{N}\;(n>x).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{ArchimedeanProperty}(\mathbb{R})\Longleftrightarrow
\forall x\in\mathbb{R}\;\exists n\in\mathbb{N}\;(n>x).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists x\in\mathbb{R}\;\forall n\in\mathbb{N}\;(n\le x).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{ArchimedeanProperty}(\mathbb{R})\Longleftrightarrow
\exists x\in\mathbb{R}\;\forall n\in\mathbb{N}\;(n\le x).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The theorem fails exactly when the natural numbers are bounded above by some real number.
\end{remark*}

\begin{remark*}[Failure modes]
\[
\neg\operatorname{ArchimedeanProperty}(\mathbb{R})
=
\underbrace{\exists x\in\mathbb{R}}_{\text{real upper bound}}
\;\underbrace{\forall n\in\mathbb{N}\;(n\le x)}_{\text{all natural numbers remain below it}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The Archimedean Property says that the natural numbers do not stop below any real ceiling. No real number is so large that every natural number remains below it. In this chapter, the result is a consequence of completeness: a bounded copy of $\mathbb{N}$ would have a supremum, and that supremum is contradicted by shifting one natural number upward.
\end{remark*}

\begin{corollarybox}{Corollary (Archimedean Reciprocal Form)}
\begin{corollary}[Archimedean Reciprocal Form]
\label{cor:archimedean-reciprocal-form}
For every real number $\varepsilon>0$, there exists a natural number
$n\in\mathbb{N}$ such that
\[
0<\frac{1}{n}<\varepsilon.
\]

\smallskip
\noindent
\hyperref[prf:archimedean-reciprocal-form]{\textit{Go to proof.}}
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
\forall \varepsilon\in\mathbb{R}\;
\bigl(\varepsilon>0\rightarrow
\exists n\in\mathbb{N}\;(0<1/n<\varepsilon)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{ArchimedeanReciprocalForm}(\mathbb{R})\Longleftrightarrow
\forall \varepsilon>0\;\exists n\in\mathbb{N}\;(0<1/n<\varepsilon).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists \varepsilon>0\;\forall n\in\mathbb{N}\;\left(\frac{1}{n}\ge \varepsilon\right).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{ArchimedeanReciprocalForm}(\mathbb{R})
\Longleftrightarrow
\exists \varepsilon>0\;\forall n\in\mathbb{N}\;(1/n\ge \varepsilon).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The corollary fails only if some positive scale remains below every reciprocal
natural number.
\end{remark*}

\begin{remark*}[Failure modes]
\[
\underbrace{\varepsilon>0}_{\text{positive scale}}
\qquad\text{and}\qquad
\underbrace{\forall n\in\mathbb{N}\;(1/n\ge\varepsilon)}_{\text{reciprocals never pass below it}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The reciprocals of the natural numbers descend below every positive real
tolerance.
\end{remark*}

\begin{corollarybox}{Corollary (Archimedean Scaling Form)}
\begin{corollary}[Archimedean Scaling Form]
\label{cor:archimedean-reciprocal}
For every $x>0$ and every $y\in\mathbb{R}$, there exists $n\in\mathbb{N}$ such that $nx>y$.

\smallskip
\noindent
\hyperref[prf:archimedean-reciprocal]{\textit{Go to proof.}}
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
\forall x,y\in\mathbb{R}\;\bigl(x>0\rightarrow \exists n\in\mathbb{N}\;(nx>y)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{ArchimedeanScalingForm}(\mathbb{R})\Longleftrightarrow
\forall x,y\in\mathbb{R}\;\bigl(x>0\rightarrow \exists n\in\mathbb{N}\;(nx>y)\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists x,y\in\mathbb{R}\;\bigl(x>0\land \forall n\in\mathbb{N}\;(nx\le y)\bigr).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{ArchimedeanScalingForm}(\mathbb{R})\Longleftrightarrow
\exists x,y\in\mathbb{R}\;\bigl(x>0\land \forall n\in\mathbb{N}\;(nx\le y)\bigr).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The corollary fails when a positive real step size has all of its natural multiples bounded above by some real threshold.
\end{remark*}

\begin{remark*}[Failure modes]
\[
\neg\operatorname{ArchimedeanScalingForm}(\mathbb{R})
=
\underbrace{\exists x\in\mathbb{R}\;(x>0)}_{\text{positive step size}}
\;\land\;
\underbrace{\exists y\in\mathbb{R}\;\forall n\in\mathbb{N}\;(nx\le y)}_{\text{all natural multiples stay bounded}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Any fixed positive real step eventually passes any real threshold when repeated enough times. The statement is the Archimedean Property applied after rescaling by the positive number $x$: growth by units becomes growth by steps of size $x$.
\end{remark*}

\end{dependencies}

\begin{lemmabox}{Lemma (Integer Part Lemma)}
\begin{lemma}[Integer Part Lemma]
\label{lem:integer-part-lemma}
For every real number $x\in\mathbb{R}$, there exists an integer
$m\in\mathbb{Z}$ such that
\[
m\le x<m+1.
\]

\smallskip
\noindent
\hyperref[prf:integer-part-lemma]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\forall x\in\mathbb{R}\;\exists m\in\mathbb{Z}\;(m\le x<m+1).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{IntegerPart}(m,x)\Longleftrightarrow
m\in\mathbb{Z}\land m\le x<m+1.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists x\in\mathbb{R}\;\forall m\in\mathbb{Z}\;(x<m\lor m+1\le x).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\exists m\in\mathbb{Z}\;\operatorname{IntegerPart}(m,x)
\Longleftrightarrow
\forall m\in\mathbb{Z}\;(x<m\lor m+1\le x).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The lemma fails only if some real number lies in no half-open unit interval
$[m,m+1)$ with integer left endpoint.
\end{remark*}

\begin{remark*}[Failure modes]
\[
\underbrace{x<m}_{\text{integer starts too high}}
\qquad\text{or}\qquad
\underbrace{m+1\le x}_{\text{integer interval ends too low}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Every real number has an integer part: the real line is tiled by consecutive
unit intervals whose left endpoints are integers.
\end{remark*}

\hyperref[thm:archimedean-property]{Theorem~\ref*{thm:archimedean-property}}.
\end{dependencies}



\begin{lemmabox}{Lemma (Archimedean Integer Part Lemma)}
\begin{lemma}[Archimedean Integer Part Lemma]
\label{lem:archimedean-integer-part-lemma}
For every $x\in\mathbb{R}$, there exists $m\in\mathbb{Z}$ such that
\[
m-1\le x<m.
\]

\smallskip
\noindent
\hyperref[prf:archimedean-integer-part-lemma]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\forall x\in\mathbb{R}\;\exists m\in\mathbb{Z}\;(m-1\le x<m).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{IntegerCeilingAbove}(m,x)\Longleftrightarrow
m\in\mathbb{Z}\land m-1\le x<m.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists x\in\mathbb{R}\;\forall m\in\mathbb{Z}\;(x<m-1\lor m\le x).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\exists m\in\mathbb{Z}\;\operatorname{IntegerCeilingAbove}(m,x)
\Longleftrightarrow
\forall m\in\mathbb{Z}\;(x<m-1\lor m\le x).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The lemma fails only if some real number lies in no half-open unit interval
$[m-1,m)$ with integer right endpoint.
\end{remark*}

\begin{remark*}[Failure modes]
\[
\underbrace{x<m-1}_{\text{right endpoint too high}}
\qquad\text{or}\qquad
\underbrace{m\le x}_{\text{right endpoint not above }x}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
This is the integer-part lemma with the endpoint shifted by one.  It produces
the first integer strictly above $x$ when the unit interval is written with its
integer endpoint on the right.
\end{remark*}

\begin{lemmabox}{Lemma (Integer Ceiling Lemma)}
\begin{lemma}[Integer Ceiling Lemma]
\label{lem:integer-ceiling-lemma}
For every $x\in\mathbb{R}$, there exists $m\in\mathbb{Z}$ such that
\[
x<m\le x+1.
\]

\smallskip
\noindent
\hyperref[prf:integer-ceiling-lemma]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\forall x\in\mathbb{R}\;\exists m\in\mathbb{Z}\;(x<m\le x+1).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{StrictIntegerCeiling}(m,x)\Longleftrightarrow
m\in\mathbb{Z}\land x<m\le x+1.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists x\in\mathbb{R}\;\forall m\in\mathbb{Z}\;(m\le x\lor x+1<m).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\exists m\in\mathbb{Z}\;\operatorname{StrictIntegerCeiling}(m,x)
\Longleftrightarrow
\forall m\in\mathbb{Z}\;(m\le x\lor x+1<m).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The lemma fails only if every integer is either at or below $x$, or lies
strictly above the next unit endpoint $x+1$.
\end{remark*}

\begin{remark*}[Failure modes]
\[
\underbrace{m\le x}_{\text{not strictly above }x}
\qquad\text{or}\qquad
\underbrace{x+1<m}_{\text{overshoots one unit}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
There is always an integer just above a real number, and it can be chosen no
farther than one unit above that number.
\end{remark*}

\begin{lemmabox}{Lemma (Integer Above Lemma)}
\begin{lemma}[Integer Above Lemma]
\label{lem:integer-above-lemma}
For every $x\in\mathbb{R}$, there exists an integer $m\in\mathbb{Z}$ such
that
\[
x<m.
\]

\smallskip
\noindent
\hyperref[prf:integer-above-lemma]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\forall x\in\mathbb{R}\;\exists m\in\mathbb{Z}\;(x<m).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{IntegerAbove}(m,x)\Longleftrightarrow
m\in\mathbb{Z}\land x<m.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists x\in\mathbb{R}\;\forall m\in\mathbb{Z}\;(m\le x).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\exists m\in\mathbb{Z}\;\operatorname{IntegerAbove}(m,x)
\Longleftrightarrow
\forall m\in\mathbb{Z}\;(m\le x).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The lemma fails only if some real number is an upper bound for all integers.
\end{remark*}

\begin{remark*}[Failure modes]
\[
\underbrace{\exists x\in\mathbb{R}}_{\text{candidate upper bound}}
\;\underbrace{\forall m\in\mathbb{Z}\;(m\le x)}_{\text{no integer lies above it}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Integers are unbounded above in the real line. The Integer Ceiling Lemma gives
the sharper version with $m$ no farther than one unit above $x$.
\end{remark*}

\begin{lemmabox}{Lemma (Unit-Length Interval Contains an Integer)}
\begin{lemma}[Unit-Length Interval Contains an Integer]
\label{lem:unit-length-interval-contains-integer}
If $\alpha,\beta\in\mathbb{R}$ and $\beta-\alpha>1$, then there exists
$m\in\mathbb{Z}$ such that
\[
\alpha<m<\beta.
\]

\smallskip
\noindent
\hyperref[prf:unit-length-interval-contains-integer]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\forall \alpha,\beta\in\mathbb{R}\;
\bigl(\beta-\alpha>1\rightarrow
\exists m\in\mathbb{Z}\;(\alpha<m<\beta)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{IntegerInOpenInterval}(m,\alpha,\beta)
\Longleftrightarrow
m\in\mathbb{Z}\land \alpha<m<\beta.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists \alpha,\beta\in\mathbb{R}\;
\bigl(\beta-\alpha>1\land
\forall m\in\mathbb{Z}\;(m\le \alpha\lor \beta\le m)\bigr).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\exists m\in\mathbb{Z}\;\operatorname{IntegerInOpenInterval}(m,\alpha,\beta)
\Longleftrightarrow
\forall m\in\mathbb{Z}\;(m\le \alpha\lor \beta\le m).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The lemma fails only if an open interval of length greater than one still
misses every integer.
\end{remark*}

\begin{remark*}[Failure modes]
\[
\underbrace{m\le \alpha}_{\text{integer at or left of the interval}}
\qquad\text{or}\qquad
\underbrace{\beta\le m}_{\text{integer at or right of the interval}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Open intervals longer than one unit cannot fit between consecutive integers.
At least one integer must lie strictly inside.
\end{remark*}
