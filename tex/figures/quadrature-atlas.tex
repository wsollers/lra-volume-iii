\def\Gexpr{1.7+0.5*sin(75*\x)}

% trapezoid cell on [#1,#2]
\newcommand{\trapcell}[2]{%
  \pgfmathsetmacro\fL{1.7+0.5*sin(75*#1)}\pgfmathsetmacro\fR{1.7+0.5*sin(75*#2)}%
  \fill[atlasblue!14] (#1,0)--(#1,\fL)--(#2,\fR)--(#2,0)--cycle;
  \draw[atlasblue!60,line width=0.5pt] (#1,0)--(#1,\fL)--(#2,\fR)--(#2,0)--cycle;
  \node[atlas dot,minimum size=2.6pt] at (#1,\fL){};
  \node[atlas dot,minimum size=2.6pt] at (#2,\fR){};
}
% Simpson parabola over a pair [#1,#3] with interior node #2 (exact interpolant via quad Bezier->cubic)
\newcommand{\simpsonpair}[3]{%
  \pgfmathsetmacro\fL{1.7+0.5*sin(75*#1)}%
  \pgfmathsetmacro\fM{1.7+0.5*sin(75*#2)}%
  \pgfmathsetmacro\fR{1.7+0.5*sin(75*#3)}%
  \pgfmathsetmacro\Qy{2*\fM-(\fL+\fR)/2}%
  \pgfmathsetmacro\CLx{(#1+2*#2)/3}\pgfmathsetmacro\CLy{(\fL+2*\Qy)/3}%
  \pgfmathsetmacro\CRx{(#3+2*#2)/3}\pgfmathsetmacro\CRy{(\fR+2*\Qy)/3}%
  \fill[atlasmagenta!13]
    (#1,0)--(#1,\fL) .. controls (\CLx,\CLy) and (\CRx,\CRy) .. (#3,\fR) --(#3,0)--cycle;
  \draw[atlasmagenta,line width=1.3pt]
    (#1,\fL) .. controls (\CLx,\CLy) and (\CRx,\CRy) .. (#3,\fR);
  \node[atlas dot,minimum size=2.6pt] at (#1,\fL){};
  \node[atlas dot,minimum size=2.6pt] at (#2,\fM){};
  \node[atlas dot,minimum size=2.6pt] at (#3,\fR){};
}
