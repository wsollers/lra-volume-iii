\begin{theorem}[Supremum of the Pairwise Maximum Set]
\label{thm:supremum-pairwise-maximum-set}
Let $A,B\subseteq\mathbb{R}$ be nonempty and bounded above. Then $A\vee B$
is bounded above and
\[
\sup(A\vee B)=\max\{\sup A,\sup B\}.
\]
\hyperref[prf:supremum-pairwise-maximum-set]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
A,B\neq\varnothing,\quad A,B\text{ bounded above}
\quad\Longrightarrow\quad
\sup(A\vee B)=\max\{\sup A,\sup B\}.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&M=A\vee B,\qquad U=\max\{s_A,s_B\},\\
&\operatorname{LeastUpperBound}(s_A,A,P)\land\operatorname{LeastUpperBound}(s_B,B,P)
\Longrightarrow
\operatorname{LeastUpperBound}(U,M,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\sup(A\vee B)\neq\max\{\sup A,\sup B\}
\]
is impossible under the stated hypotheses.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&M=A\vee B,\qquad U=\max\{s_A,s_B\},\\
&\operatorname{LeastUpperBound}(s_A,A,P)\land\operatorname{LeastUpperBound}(s_B,B,P)\land
\neg\operatorname{LeastUpperBound}(U,M,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
The formula could fail only if the maximum of the two suprema failed to bound
all pairwise maxima, or if it were not the least such bound.


  \item[\operatorname{LeastUpperBound}.]
  \[
\sup(A\vee B)\neq\max\{\sup A,\sup B\}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&M=A\vee B,\qquad U=\max\{s_A,s_B\},\\
&\operatorname{LeastUpperBound}(s_A,A,P)\land\operatorname{LeastUpperBound}(s_B,B,P)\land
\neg\operatorname{LeastUpperBound}(U,M,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Every pairwise maximum is bounded by $\max\{\sup A,\sup B\}$.  Sharpness comes
from holding one input fixed in the nonempty other set and letting the relevant
set approach its own supremum.


  \item[\operatorname{LeastUpperBound}.]
  \[
\sup(A\vee B)\neq\max\{\sup A,\sup B\}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&M=A\vee B,\qquad U=\max\{s_A,s_B\},\\
&\operatorname{LeastUpperBound}(s_A,A,P)\land\operatorname{LeastUpperBound}(s_B,B,P)\land
\neg\operatorname{LeastUpperBound}(U,M,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
The largest possible pairwise maximum is controlled by the larger of the two
individual upper endpoints.
\end{remark*}
