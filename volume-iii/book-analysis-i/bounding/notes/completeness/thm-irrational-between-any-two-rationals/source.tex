\begin{theorem}[An Irrational Between Any Two Rationals]
\label{thm:irrational-between-any-two-rationals}
For every $r,s\in\mathbb{Q}$, if $r<s$, then there exists
$x\in\mathbb{R}\setminus\mathbb{Q}$ such that
\[
  r<x<s.
\]
\hyperref[prf:irrational-between-any-two-rationals]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall r,s\in\mathbb{Q}\;
\bigl(r<s\Longrightarrow
\exists x\in\mathbb{R}\setminus\mathbb{Q}\;(r<x<s)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
r,s\in\mathbb{Q}\land r<s
\Longrightarrow \exists x\in\mathbb{R}\setminus\mathbb{Q}\;(r<x<s).
\]
\end{remark*}

\begin{remark*}[Interpretation]
This is the density of the irrationals restricted to intervals whose endpoints
are rational.
\end{remark*}
