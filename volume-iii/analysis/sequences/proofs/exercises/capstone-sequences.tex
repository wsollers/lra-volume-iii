\subsection*{Capstone --- Sequences}

\begin{remark*}[Capstone Problem]
This exercise synthesizes the basic sequence tools: epsilon tails,
neighborhoods, uniqueness of limits, convergence of tails, domination,
and residue-class subsequences.

\medskip
\noindent
Let $(x_n)\subseteq\mathbb{R}$ and let $L\in\mathbb{R}$.

\begin{enumerate}[label=(\alph*)]

\item \textbf{Tail control from domination.}
Suppose there exist $m\in\mathbb{N}$, $c>0$, and a sequence
$(a_n)\subseteq(0,\infty)$ such that $a_n\to 0$ and
\[
|x_n-L|\leq c\,a_n
\qquad\text{for all } n\geq m.
\]
Prove directly from the epsilon definition that $x_n\to L$.

\item \textbf{Neighborhood translation.}
Rewrite the conclusion of part (a) in neighborhood language: prove that
for every $\varepsilon>0$ there exists $N\in\mathbb{N}$ such that
\[
n\geq N \quad\Longrightarrow\quad x_n\in V_\varepsilon(L).
\]
Then explain why this is the same statement as $|x_n-L|<\varepsilon$.

\item \textbf{Tail invariance.}
Fix $q\in\mathbb{N}$ and define the $q$-tail
\[
y_n:=x_{n+q}.
\]
Prove that $(x_n)$ converges to $L$ if and only if $(y_n)$ converges to
$L$. Your proof must explicitly account for the shift in indices.

\item \textbf{Uniqueness test.}
Suppose $x_n\to L$ and $x_n\to K$ for some $K\in\mathbb{R}$. Prove
$L=K$ without citing the Uniqueness of Limits theorem by name. Your
argument should choose a single tolerance from $|L-K|$ and force a
contradiction if $L\neq K$.

\item \textbf{Residue-class reconstruction.}
Define the two residue subsequences
\[
u_n:=x_{2n},
\qquad
v_n:=x_{2n-1}.
\]
Prove that if $u_n\to L$ and $v_n\to L$, then $x_n\to L$.

\item \textbf{Residue-class obstruction.}
Prove that if $u_n\to L$, $v_n\to K$, and $L\neq K$, then $(x_n)$ does
not converge. Your proof should use part (d), not a direct epsilon
negation.

\item \textbf{Construct and classify.}
Construct an explicit real sequence $(x_n)$ such that:
\[
x_{2n}\to L,\qquad x_{2n-1}\to L,\qquad\text{and}\qquad x_n\neq L
\text{ for every } n\in\mathbb{N}.
\]
Then prove that your example converges to $L$ and identify which earlier
parts certify the conclusion.

\end{enumerate}

\medskip
\textit{Concepts synthesized:}
sequence as a function,
epsilon definition of convergence,
neighborhood formulation,
uniqueness of limits,
tails,
convergence by domination,
and residue-class subsequences.

\medskip
\textit{Connection to later chapters.}
Continuity and limits of functions repeatedly convert local questions into
sequence questions. The residue-class parts train the habit of decomposing
one sequence into structured subsequences, while the domination part trains
the habit of replacing a complicated error by a simpler null sequence. These
two moves reappear in proofs of sequential continuity, compactness, uniform
continuity, and numerical convergence.
\end{remark*}
