\begin{proposition*}[Pointwise Max/Min Absolute Value Formulas]
Let \(A\subseteq\mathbb{R}\) and let \(f,g:A\to\mathbb{R}\).  For every
\(x\in A\),
\[
  \max(f,g)(x)=\frac{f(x)+g(x)+|f(x)-g(x)|}{2}
\]
and
\[
  \min(f,g)(x)=\frac{f(x)+g(x)-|f(x)-g(x)|}{2}.
\]
\end{proposition*}
