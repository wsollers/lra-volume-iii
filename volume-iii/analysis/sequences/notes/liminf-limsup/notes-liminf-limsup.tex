\subsection*{Liminf and Limsup}
\begin{exposition}
The limit superior and limit inferior measure the eventual upper and lower
envelopes of a bounded sequence. Instead of asking whether all late terms
settle to one number, they ask how high and how low the tails continue to
reach. This makes them natural tools for detecting convergence, oscillation,
and subsequential limits.
\end{exposition}

% ---------------------------------------------------------
% Definition --- Tail Supremum Sequence
% ---------------------------------------------------------

\begin{definitionbox}{Definition (Tail Supremum Sequence)}
\begin{definition}[Tail Supremum Sequence]
\label{def:tail-supremum-sequence}
Let $(x_n)$ be a bounded real sequence. The \textbf{tail supremum
sequence} of $(x_n)$ is the real sequence $(s_n)$ defined by
\[
  s_n:=\sup\{x_k:k\ge n\}
  \qquad\text{for every } n\in\mathbb{N}.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\forall n\in\mathbb{N}\;
\bigl(s_n=\sup\{x_k:k\ge n\}\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The term $s_n$ records the least ceiling of the $n$-th tail. As $n$ grows,
earlier terms are discarded, so the tail has fewer opportunities to reach
large values.
\end{remark*}

% Definition --- Limit Superior of a Sequence

\begin{definitionbox}{Definition (Limit Superior of a Sequence)}
\begin{definition}[Limit Superior of a Sequence]
\label{def:limsup-sequence}
Let $(x_n)$ be a bounded real sequence, and let $(s_n)$ be its tail
supremum sequence. The \textbf{limit superior} of $(x_n)$ is
\[
  \limsup_{n\to\infty}x_n:=\lim_{n\to\infty}s_n.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\limsup_{n\to\infty}x_n=L
\Longleftrightarrow
\forall \varepsilon>0\;\exists N\in\mathbb{N}\;\forall n\ge N
\;\bigl(|s_n-L|<\varepsilon\bigr).
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{LimsupSequence}(x_n,L)
\coloneqq
\operatorname{ConvergentSequence}(s_n,L).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The limit superior is the eventual upper envelope of the sequence. It is the
number to which the ceilings of the tails descend.
\end{remark*}

% Theorem --- Tail Suprema Are Decreasing

\begin{theorembox}{Theorem (Tail Suprema Are Decreasing)}
\begin{theorem}[Tail Suprema Are Decreasing]
\label{thm:tail-suprema-are-decreasing}
Let $(x_n)$ be a bounded real sequence, and let $(s_n)$ be its tail
supremum sequence. Then $(s_n)$ is decreasing.

\smallskip
\noindent
\hyperref[prf:tail-suprema-are-decreasing]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall n\in\mathbb{N}\;\bigl(s_{n+1}\le s_n\bigr).
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{MonotoneDecreasingSequence}(s_n).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The $(n+1)$-st tail is contained in the $n$-th tail. Passing from a set to a
smaller subset cannot raise its supremum.
\end{remark*}

\begin{theorembox}{Theorem (Liminf Is Below Limsup)}
\begin{theorem}[Liminf Is Below Limsup]
\label{thm:liminf-below-limsup}

\[
  \liminf_{n\to\infty}x_n\le \limsup_{n\to\infty}x_n.
\]

\smallskip

\hyperref[prf:liminf-below-limsup]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\liminf_{n\to\infty}x_n\le \limsup_{n\to\infty}x_n.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Every tail infimum lies below every corresponding tail supremum. Passing to
the limits preserves this order.
\end{remark*}

\begin{theorembox}{Theorem (Convergence iff Liminf Equals Limsup)}
\begin{theorem}[Convergence iff Liminf Equals Limsup]
\label{thm:convergence-iff-liminf-equals-limsup}
Let $(x_n)$ be a bounded real sequence, and let $L\in\mathbb{R}$. Then
$x_n\to L$ if and only if
\[
  \liminf_{n\to\infty}x_n
  =
  \limsup_{n\to\infty}x_n
  =
  L.
\]

\smallskip
\noindent
\hyperref[prf:convergence-iff-liminf-equals-limsup]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
x_n\to L
\quad\Longleftrightarrow\quad
\liminf_{n\to\infty}x_n
=
\limsup_{n\to\infty}x_n
=L.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{ConvergentSequence}(x_n,L)
\Longleftrightarrow
\bigl(
  \operatorname{LiminfSequence}(x_n,L)
  \land
  \operatorname{LimsupSequence}(x_n,L)
\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Convergence occurs exactly when the eventual lower envelope and eventual
upper envelope collapse to the same number.
\end{remark*}

% Theorem --- Limsup Is the Largest Subsequential Limit
% ---------------------------------------------------------

\begin{theorembox}{Theorem (Limsup Is the Largest Subsequential Limit)}
\begin{theorem}[Limsup Is the Largest Subsequential Limit]
\label{thm:limsup-largest-subsequential-limit}
Let $(x_n)$ be a bounded real sequence, and let
$S=\limsup_{n\to\infty}x_n$. Then $S$ is a subsequential limit of
$(x_n)$, and every subsequential limit of $(x_n)$ is less than or equal
to $S$.

\smallskip
\noindent
\hyperref[prf:limsup-largest-subsequential-limit]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&S=\limsup_{n\to\infty}x_n\\
&\quad\Longrightarrow\quad
\Bigl[
  S\text{ is a subsequential limit of }(x_n)
  \land
  \forall L\in\mathbb{R}\;
  \bigl(L\text{ is a subsequential limit of }(x_n)\Longrightarrow L\le S\bigr)
\Bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{LimsupSequence}(x_n,S)
\Longrightarrow
\bigl(
  \operatorname{SubsequentialLimit}(x_n,S)
  \land
  \forall L\in\mathbb{R}\;
  (\operatorname{SubsequentialLimit}(x_n,L)\Longrightarrow L\le S)
\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The limit superior is not merely an upper bound on subsequential behavior. It
is actually attained as a subsequential limit, and no subsequential limit lies
above it.
\end{remark*}

% Theorem --- Oscillation Criterion via Liminf and Limsup
% ---------------------------------------------------------

\begin{theorembox}{Theorem (Oscillation Criterion via Liminf and Limsup)}
\begin{theorem}[Oscillation Criterion via Liminf and Limsup]
\label{thm:oscillation-criterion-via-liminf-limsup}
Let $(x_n)$ be a bounded real sequence. Then
\[
  \liminf_{n\to\infty}x_n
  <
  \limsup_{n\to\infty}x_n
\]
if and only if $(x_n)$ has two distinct subsequential limits.

\smallskip
\noindent
\hyperref[prf:oscillation-criterion-via-liminf-limsup]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\liminf_{n\to\infty}x_n<\limsup_{n\to\infty}x_n\\
&\quad\Longleftrightarrow\quad
\exists L,K\in\mathbb{R}\;
\bigl(
  L\ne K
  \land L\text{ is a subsequential limit of }(x_n)
  \land K\text{ is a subsequential limit of }(x_n)
\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\begin{aligned}
&\exists I,S\in\mathbb{R}\;
\bigl(
  \operatorname{LiminfSequence}(x_n,I)
  \land
  \operatorname{LimsupSequence}(x_n,S)
  \land I<S
\bigr)\\
&\quad\Longleftrightarrow\quad
\exists L,K\in\mathbb{R}\;
\bigl(
  L\ne K
  \land \operatorname{SubsequentialLimit}(x_n,L)
  \land \operatorname{SubsequentialLimit}(x_n,K)
\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
A strict gap between the lower and upper envelopes is exactly persistent
oscillation in bounded sequences. The sequence keeps returning near at least
two incompatible limiting levels.
\end{remark*}

\begin{theorem}[Limsup Comparison Under Eventual Order]
\label{thm:limsup-comparison-under-eventual-order}
Let $(x_n)$ and $(y_n)$ be bounded real sequences. If there exists
$N\in\mathbb{N}$ such that $x_n\le y_n$ for every $n\ge N$, then
\[
  \limsup_{n\to\infty}x_n
  \le
  \limsup_{n\to\infty}y_n.
\]

\smallskip
\noindent
\hyperref[prf:limsup-comparison-under-eventual-order]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\exists N\in\mathbb{N}\;\forall n\ge N\;(x_n\le y_n)\\
&\quad\Longrightarrow\quad
\limsup_{n\to\infty}x_n
\le
\limsup_{n\to\infty}y_n.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Eventual order of sequences passes to eventual order of their upper tail
envelopes. Taking the limiting ceiling preserves that comparison.
\end{remark*}

\begin{theorembox}{Theorem (Limsup Squeeze Under Eventual Order)}
\begin{theorem}[Limsup Squeeze Under Eventual Order]
\label{thm:limsup-squeeze-under-eventual-order}
Let $(a_n)$, $(x_n)$, and $(b_n)$ be bounded real sequences. If there
exists $N\in\mathbb{N}$ such that
\[
  a_n\le x_n\le b_n
  \qquad\text{for every }n\ge N,
\]
then
\[
  \limsup_{n\to\infty}a_n
  \le
  \limsup_{n\to\infty}x_n
  \le
  \limsup_{n\to\infty}b_n.
\]

\smallskip
\noindent
\hyperref[prf:limsup-squeeze-under-eventual-order]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\exists N\in\mathbb{N}\;\forall n\ge N\;(a_n\le x_n\le b_n)\\
&\quad\Longrightarrow\quad
\limsup_{n\to\infty}a_n
\le
\limsup_{n\to\infty}x_n
\le
\limsup_{n\to\infty}b_n.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
If one sequence is eventually trapped between two others, then its eventual
upper envelope is trapped between their eventual upper envelopes.
\end{remark*}

\begin{remark*}[Illustrative cases]
For the alternating sequence $x_n=(-1)^n$, the tail suprema are constantly
$1$ and the tail infima are constantly $-1$, so
\[
  \limsup_{n\to\infty}(-1)^n=1,
  \qquad
  \liminf_{n\to\infty}(-1)^n=-1.
\]
For any convergent sequence $x_n\to L$, both envelopes collapse to $L$.
\end{remark*}
