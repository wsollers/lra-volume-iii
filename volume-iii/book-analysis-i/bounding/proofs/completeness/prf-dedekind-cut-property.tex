\begin{theorem*}[Dedekind Cut Property]
Let $A,B\subseteq\mathbb{R}$ be nonempty sets such that
$A\cup B=\mathbb{R}$, $A\cap B=\varnothing$, and $a<b$ for every $a\in A$
and every $b\in B$. Then there exists $c\in\mathbb{R}$ such that
$a\le c\le b$ for every $a\in A$ and every $b\in B$.
\end{theorem*}
