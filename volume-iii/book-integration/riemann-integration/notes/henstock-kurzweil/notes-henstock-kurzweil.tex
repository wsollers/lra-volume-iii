\subsection*{Where This Sits}

\begin{exposition}
Darboux exposed the Riemann disease before measure theory names it precisely:
one uniform fineness parameter must serve the whole interval. There are two
ways forward. Lebesgue will change what is measured. The gauge integrals keep
the tagged-sum picture and change exactly one symbol: the constant fineness
\(\delta>0\) becomes a positive function \(\delta(x)\).
\end{exposition}

\begin{remark*}[Interpretation]
Henstock--Kurzweil belongs after the measure-zero diagnosis because it is a
Riemann-style response to the same obstruction: keep slabs, tags, and sums,
but replace one uniform mesh tolerance by local fineness. McShane will modify
the tag rule one more time, producing a gauge-sum version of Lebesgue
integration.
\end{remark*}

\subsection*{Geometric Intuition}

\begin{exposition}
A gauge is adaptive mesh refinement written as analysis. It is tiny where the
function needs tight local control and generous where the function is calm. A
tagged slab is allowed only if it fits inside the local radius prescribed by
its own tag.
\end{exposition}

\begin{remark*}[Interpretation]
Riemann uses a positive number:
\[
  \exists\delta>0.
\]
Henstock--Kurzweil uses a positive function:
\[
  \exists\delta:[a,b]\to(0,\infty).
\]
The mesh condition \(\|P\|<\delta\) becomes the local condition that each slab
is \(\delta\)-fine at its tag.
\end{remark*}

\subsection*{Definition}

\begin{definitionbox}{Definition (Gauge)}
\begin{definition}[Gauge on an Interval]
\label{def:gauge-on-interval}
A \emph{gauge} on \([a,b]\) is a function
\[
  \delta:[a,b]\to(0,\infty).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{Gauge}(\delta,[a,b])
\Longleftrightarrow
\delta:[a,b]\to(0,\infty).
\]
\end{remark*}

\begin{remark*}[Interpretation]
A gauge assigns a positive local radius to each point of the interval.
\end{remark*}

\begin{itemize}
  \item \hyperref[def:interval-partition]{Interval Partition}
\end{itemize}
\end{dependencies}

\begin{definitionbox}{Definition (Delta-Fine Tagged Partition)}
\begin{definition}[\(\delta\)-Fine Tagged Partition]
\label{def:delta-fine-tagged-partition}
A tagged partition \((P,\xi)\) is \emph{\(\delta\)-fine} if
\[
  [x_{i-1},x_i]\subset
  \bigl(\xi_i-\delta(\xi_i),\,\xi_i+\delta(\xi_i)\bigr)
  \qquad\text{for every }i.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
(P,\xi)\text{ is }\delta\text{-fine}
\Longleftrightarrow
\forall i\;
[x_{i-1},x_i]\subset
\bigl(\xi_i-\delta(\xi_i),\,\xi_i+\delta(\xi_i)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{IsDeltaFine}((P,\xi),\delta,[a,b])
\Longleftrightarrow
\forall i\;
[x_{i-1},x_i]\subset
\bigl(\xi_i-\delta(\xi_i),\,\xi_i+\delta(\xi_i)\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
A tagged partition is gauge-fine when every subinterval fits inside the local
neighborhood prescribed by its own tag.
\end{remark*}

\begin{definitionbox}{Definition (Henstock--Kurzweil Integral)}
\begin{definition}[Henstock--Kurzweil Integral]
\label{def:henstock-kurzweil-integral}
A function \(f:[a,b]\to\mathbb R\) is \emph{Henstock--Kurzweil integrable}
with value \(A\) if
\[
  \forall\varepsilon>0\ \exists\delta:[a,b]\to(0,\infty)\ \forall(P,\xi)\,
  \bigl((P,\xi)\text{ is }\delta\text{-fine}
  \Rightarrow |S(f,P,\xi)-A|<\varepsilon\bigr).
\]
The value is denoted \((\mathrm{HK})\int_a^b f\).
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{HasDefiniteIntegral}(f,[a,b],\operatorname{HenstockKurzweilCriterion},A)
\Longleftrightarrow
\forall\varepsilon>0\ \exists\delta:[a,b]\to(0,\infty)\ \forall(P,\xi)\,
\bigl((P,\xi)\text{ is }\delta\text{-fine}
\Rightarrow |S(f,P,\xi)-A|<\varepsilon\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{HasDefiniteIntegral}(f,[a,b],\operatorname{HenstockKurzweilCriterion},A)
\Longleftrightarrow
\forall\varepsilon>0\ \exists\delta:[a,b]\to(0,\infty)\ \forall(P,\xi)\,
\bigl(\operatorname{IsDeltaFine}((P,\xi),\delta,[a,b])
\Rightarrow |S(f,P,\xi)-A|<\varepsilon\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Henstock--Kurzweil integrability requires all sufficiently local tagged sums
to converge to the same value under a suitable gauge.
\end{remark*}

\begin{lemmabox}{Lemma (Cousin)}
\begin{lemma}[Cousin's Lemma]
\label{lem:cousin}
For every gauge \(\delta\) on \([a,b]\), there exists a \(\delta\)-fine tagged
partition of \([a,b]\).
\hyperref[prf:cousin]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{Gauge}(\delta,[a,b])
\Longrightarrow
\exists(P,\xi)\;\bigl((P,\xi)\text{ is }\delta\text{-fine}\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{Gauge}(\delta,[a,b])
\Longrightarrow
\exists(P,\xi)\;\operatorname{IsDeltaFine}((P,\xi),\delta,[a,b]).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Cousin's Lemma prevents the definition from being empty. No matter how wildly
the gauge varies, compactness of \([a,b]\) guarantees that at least one tagged
partition respects all the local step-size demands.
\end{remark*}

\subsection*{Existence and Applicability Criterion}

\begin{theorembox}{Theorem (Riemann Sits Inside Henstock--Kurzweil)}
\begin{theorem}[Riemann Integrable Implies Henstock--Kurzweil Integrable]
\label{thm:riemann-integrable-implies-hk-integrable}
If \(f\) is Riemann-integrable on \([a,b]\), then \(f\) is
Henstock--Kurzweil integrable on \([a,b]\), and the two integrals have the
same value.
\hyperref[prf:riemann-integrable-implies-hk-integrable]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{IsIntegrable}(f,[a,b],\operatorname{RiemannCriterion})
\Longrightarrow
\operatorname{IsIntegrable}(f,[a,b],\operatorname{HenstockKurzweilCriterion}),
\]
with the same integral value.
\end{remark*}

\begin{remark*}[Interpretation]
Riemann is the special case where the gauge is constant. If a Riemann proof
hands over one mesh tolerance \(\delta_0\), the gauge proof uses
\(\delta(x)=\delta_0\) for every \(x\).
\end{remark*}

\subsection*{Worked Positive Example}

\begin{lemmabox}{Lemma (Straddle)}
\begin{lemma}[Straddle Lemma]
\label{lem:hk-straddle}
If \(F\) is differentiable at \(\xi\) with \(F'(\xi)=f(\xi)\), then for every
\(\varepsilon>0\) there is \(\delta(\xi)>0\) such that whenever
\(u\le \xi\le v\), \(u,v\in(\xi-\delta(\xi),\xi+\delta(\xi))\),
\[
  |F(v)-F(u)-f(\xi)(v-u)|\le \varepsilon(v-u).
\]
\hyperref[prf:hk-straddle]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
F'(\xi)=f(\xi)
\Longrightarrow
\forall\varepsilon>0\ \exists\delta(\xi)>0\ \forall u,v\,
\bigl(u\le \xi\le v\land u,v\in(\xi-\delta(\xi),\xi+\delta(\xi))
\Rightarrow |F(v)-F(u)-f(\xi)(v-u)|\le \varepsilon(v-u)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{Derivative}(F,\xi)=f(\xi)
\Longrightarrow
\forall\varepsilon>0\ \exists\delta(\xi)>0\ \forall u,v\,
\bigl(u\le \xi\le v\land u,v\in(\xi-\delta(\xi),\xi+\delta(\xi))
\Rightarrow |F(v)-F(u)-f(\xi)(v-u)|\le \varepsilon(v-u)\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The straddle estimate is the derivative approximation written for intervals
that contain the tag.
\end{remark*}

\begin{theorembox}{Theorem (FTC for the Gauge Integral)}
\begin{theorem}[Fundamental Theorem for the Henstock--Kurzweil Integral]
\label{thm:hk-fundamental-theorem}
If \(F:[a,b]\to\mathbb R\) is differentiable at every point and \(F'=f\), then
\(f\) is Henstock--Kurzweil integrable and
\[
  (\mathrm{HK})\int_a^b f=F(b)-F(a).
\]
\hyperref[prf:hk-fundamental-theorem]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
(\forall x\in[a,b])\bigl(F'(x)=f(x)\bigr)
\Longrightarrow
\operatorname{IsIntegrable}(f,[a,b],\operatorname{HenstockKurzweilCriterion})
\land
(\mathrm{HK})\int_a^b f=F(b)-F(a).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The gauge integral restores the fundamental theorem for every derivative on a
compact interval.
\end{remark*}

\subsection*{Worked Failure Mode}

\begin{remark*}[Interpretation]
On a compact interval,
\[
  \mathcal R[a,b]\subsetneq \mathcal M[a,b]=\mathcal L[a,b]
  \subsetneq \mathcal{HK}[a,b].
\]
Figure~\ref{fig:integration-lattice} records this inclusion lattice and the
standard witness at each frontier.
The first strict inclusion is witnessed by functions like
\(\mathbf 1_{\mathbb Q}\), which are Lebesgue-integrable but not Riemann
integrable. The second is witnessed by derivatives such as that of
\(F(x)=x^2\sin(1/x
^2)\) with \(F(0)=0\): the derivative is HK-integrable by the
gauge FTC even when its absolute value is not Lebesgue-integrable.

This inclusion statement anticipates the next volume's Lebesgue theory. The
present chapter has not built Lebesgue integration yet; it has only isolated
the Riemann-side machinery, the gauge machinery, and the reason a
measure-theoretic successor is needed.
\end{remark*}

\begin{remark*}[Interpretation]
Let \(F(x)=x^2\sin(1/x^2)\) for \(x\ne0\), and set \(F(0)=0\). Then \(F\) is
differentiable at \(0\), with \(F'(0)=0\), while for \(x\ne0\),
\[
  F'(x)=2x\sin(1/x^2)-\frac{2}{x}\cos(1/x^2).
\]
The oscillatory term has nonintegrable absolute size near \(0\), so this
derivative is the standard witness that HK reaches beyond Lebesgue while the
FTC remains exact.
\end{remark*}

\subsection*{What the Failure Motivates Next}

\begin{remark*}[Interpretation]
Henstock--Kurzweil is not simply ``the best integral.'' Lebesgue has stronger
limit theorems, a complete \(L^1\) normed-space home, and a clean abstract
measure-space setting. HK is the jewel for the one-dimensional FTC and
conditional accumulation; Lebesgue and Ito remain the engines for probability
and stochastic integration.
\end{remark*}

\begin{remark*}[Interpretation]
A gauge \(\delta(x)\) is a spatially varying step-size field. Lipschitz
control is the case where a constant gauge is enough; singular or non-Lipschitz
regions demand a genuinely varying gauge. Cousin's Lemma is the compactness
guarantee that a valid adaptive partition exists.
\end{remark*}
