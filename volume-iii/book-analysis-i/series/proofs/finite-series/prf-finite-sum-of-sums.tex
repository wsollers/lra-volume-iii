\begin{lemma*}[Finite Sum of Sums]
Let \(m \le n\) be integers, and let \(a_i\) and \(b_i\) be real
numbers assigned to each integer \(i\) satisfying \(m \le i \le n\).
Then
\[
  \sum_{i=m}^{n} (a_i+b_i)
  =
  \left(\sum_{i=m}^{n} a_i\right)
  +
  \left(\sum_{i=m}^{n} b_i\right).
\]
\end{lemma*}
