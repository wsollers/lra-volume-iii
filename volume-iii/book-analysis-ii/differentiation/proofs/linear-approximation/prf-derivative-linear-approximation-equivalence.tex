\begin{theorem*}[Derivative and Linear Approximation]
Let $f:A\to\mathbb{R}$ and let $a\in A$ be an interior point of $A$. Then
$f$ is differentiable at $a$ with derivative $L$ if and only if
\[
  f(a+h)=f(a)+Lh+o(h)
  \qquad (h\to 0).
\]
\end{theorem*}
