\begin{proposition*}[Singleton Finite Set Summation]
If \(X\) consists of a single element, \(X=\{x_0\}\), and
\(f:X\to\mathbb{R}\) is a function, then
\[
  \sum_{x\in X} f(x)=f(x_0).
\]
\end{proposition*}
