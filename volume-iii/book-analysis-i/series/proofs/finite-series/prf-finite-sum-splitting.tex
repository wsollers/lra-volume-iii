\begin{lemma*}[Splitting a Finite Sum]
Let \(m \le n < p\) be integers, and let \(a_i\) be a real number
assigned to each integer \(i\) satisfying \(m \le i \le p\). Then
\[
  \sum_{i=m}^{n} a_i
  +
  \sum_{i=n+1}^{p} a_i
  =
  \sum_{i=m}^{p} a_i.
\]
\end{lemma*}
