\section{Bounds and Extremal Values}

\begin{remark*}[Predicate vs.\ Existence: Structural Template]
Each bound-type concept (maximum, minimum, upper bound, supremum, etc.) is
defined by a predicate relative to an ambient ordered set $S$. Given
$A \subseteq S$, let $P_A(x)$ denote such a predicate.

\medskip
\noindent\textbf{Predicate form.}
The statement ``$x$ is a $P$-object of $A$'' means that $P_A(x)$ holds for a
specific identified element $x \in S$. To prove this, one exhibits the element
and verifies each condition in $P_A(x)$.

\medskip
\noindent\textbf{Existence form.}
The statement ``$A$ has a $P$-object'' means that
\[
\exists x \in S \; P_A(x).
\]
Once established, one may fix such an $x$ and reason with the fact that
$P_A(x)$ holds.

\medskip
\noindent\textbf{Notational glosses.}
\begin{itemize}
  \item ``$x$ is a $P$-object of $A$'' abbreviates $P_A(x)$.
  \item ``$A$ has a $P$-object'' abbreviates $\exists x \in S \; P_A(x)$.
\end{itemize}

\medskip
\noindent\textbf{Proof strategy summary.}

\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{@{} p{4cm} p{3cm} p{6cm} @{}}
\hline
\textbf{Goal} & \textbf{Form} & \textbf{Strategy} \\
\hline
$P_A(x)$ & Predicate & Exhibit $x$ and verify $P_A(x)$ directly. \\
\noalign{\vskip 0.3em}
$\exists x \in S \; P_A(x)$ & Existence & Construct or identify a candidate and verify $P_A(x)$. \\
\noalign{\vskip 0.3em}
\textit{Hypothesis:} $\exists x \in S \; P_A(x)$ & --- & Fix such an $x$ and use the fact that $P_A(x)$ holds. \\
\hline
\end{tabular}
\end{center}

\noindent\emph{Note.}
The hypothesis row is not a proof goal. It records the standard move of fixing
an existential witness once existence has been established.
\end{remark*}

\subsection*{Bounds and Extremal Elements}

\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Upper Bound)}},
  fonttitle=\small\bfseries]
\begin{definition}[Upper Bound]
\label{def:upper-bound}
Let $(S,\le)$ be an ordered set, let $A\subseteq S$, and let $u\in S$. The element $u$ is an \emph{upper bound} of $A$ if and only if $x\le u$ for every $x\in A$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let }(S,\le)\text{ be an ordered set, }A\subseteq S,\text{ and }u\in S.\\
&u \text{ is an upper bound of } A
\Longleftrightarrow
\forall x\in A\;(x\le u).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{UpperBound}(u,A)\coloneqq \forall x\in A\;(x\le u).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let }(S,\le)\text{ be an ordered set, }A\subseteq S,\text{ and }u\in S.\\
&u \text{ is not an upper bound of } A
\Longleftrightarrow
\exists x\in A\;(u<x).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{UpperBound}(u,A)\Longleftrightarrow \exists x\in A\;(u<x).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The attempt to declare $u$ an upper bound collapses precisely when some element of $A$ lies strictly above $u$, providing a concrete counterexample to the domination condition.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\neg\operatorname{UpperBound}(u,A)
=
\underbrace{\exists x\in A\;(u<x)}_{\text{witness in }A\text{ exceeding }u}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
An upper bound caps a subset of an ordered structure from above, so every element of the subset fits beneath the chosen guard $u$. The negation highlights the standard obstruction: locating even one element of the set beyond $u$ demolishes the claim. Geometrically, $u$ places a horizontal ceiling over $A$ on the number line, and the only failure mode consists of punching through that ceiling with a higher point of $A$.
\end{remark*}

\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,

  title={\small\textbf{Definition (Lower Bound)}},
  fonttitle=\small\bfseries]
\begin{definition}[Lower Bound]
\label{def:lower-bound}
Let $(S,\le)$ be an ordered set, $A\subseteq S$, and $l\in S$. The element $l$ is a \emph{lower bound} of $A$ if and only if every $x\in A$ satisfies $l\le x$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let }(S,\le)\text{ be an ordered set, }A\subseteq S,\text{ and }l\in S.\\
&l\text{ is a lower bound of }A
\Longleftrightarrow
\forall x\in A\;(l\le x).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{LowerBound}(l,A) \coloneqq \forall x\in A\;(l\le x).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let }(S,\le)\text{ be an ordered set, }A\subseteq S,\text{ and }l\in S.\\
&\neg\bigl[l\text{ is a lower bound of }A\bigr]
\Longleftrightarrow
\exists x\in A\;(x<l).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{LowerBound}(l,A) \Longleftrightarrow \exists x\in A\;(x<l).
\]
\end{remark*}

\begin{remark*}[Failure modes]
Failure occurs precisely when some element of $A$ lies strictly below $l$, so $l$ cannot bound $A$ from below.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\neg\operatorname{LowerBound}(l,A)
=
\underbrace{\exists x\in A\text{ with }x<l}_{\text{witness below }l}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
A lower bound clamps the set $A$ from below: starting at $l$ and moving upward along the order one encounters all of $A$, but no point of $A$ sits beneath $l$. Failure means that at least one element of $A$ pokes under $l$, revealing that $l$ was chosen too high to constrain the set from below.
\end{remark*}

\NoLocalDependencies
\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Minimum)}},
  fonttitle=\small\bfseries]
\begin{definition}[Minimum]
\label{def:minimum}
Let $(S,\le)$ be an ordered set, $A\subseteq S$, and $m\in A$. The element $m$ is the \emph{minimum} of $A$ if and only if for every $x\in A$ one has $m\le x$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let }(S,\le)\text{ be an ordered set, }A\subseteq S,\text{ and }m\in S.\\
&m \text{ is the minimum of }A
\Longleftrightarrow
\bigl(m\in A \land \forall x\in A\;(m\le x)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{Minimum}(m,A)\coloneqq (m\in A)\land \forall x\in A\;(m\le x).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let }(S,\le)\text{ be an ordered set, }A\subseteq S,\text{ and }m\in S.\\
&\neg\bigl(m \text{ is the minimum of }A\bigr)
\Longleftrightarrow
\bigl(m\notin A \lor \exists x\in A\;(m\nleq x)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{Minimum}(m,A)\Longleftrightarrow \bigl(m\notin A \lor \exists x\in A\;(m\nleq x)\bigr).
\]
\end{remark*}

\begin{remark*}[Failure modes]
Failure occurs either because $m$ is not a member of $A$, or because some element of $A$ lies strictly below $m$, violating the global lower-bound condition.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\neg\operatorname{Minimum}(m,A)=
\underbrace{m\notin A}_{\text{not even a candidate}}
\;\lor\;
\underbrace{\exists x\in A\;(m\nleq x)}_{\text{candidate but not below all elements}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
A minimum point both belongs to the set and sits at its absolute bottom: every other element rests above it in the ambient order. The definition splits into the membership requirement, ensuring the bound is realized, and the universal inequality that promotes the element from a mere lower bound to a genuine extremal. Failure arises if the candidate lies outside the set or if another element dips below it; geometrically, the set either has no realized floor at that point or its supposed floor is undercut elsewhere.
\end{remark*}

\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Maximum)}},
  fonttitle=\small\bfseries]
\begin{definition}[Maximum]
\label{def:maximum}
Let $(S,\le)$ be an ordered set and let $A \subseteq S$. An element $M \in S$ is the maximum of $A$ if and only if $M \in A$ and every $x \in A$ satisfies $x \le M$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (S,\le) \text{ be an ordered set and } A \subseteq S.\\
&M \text{ is the maximum of } A
\Longleftrightarrow
(M \in A) \land \forall x \in A\;(x \le M).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{Maximum}(M,A) \coloneqq (M \in A) \land \forall x \in A\;(x \le M).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let } (S,\le) \text{ be an ordered set and } A \subseteq S.\\
&\neg\bigl(M \text{ is the maximum of } A\bigr)
\Longleftrightarrow
(M \notin A) \lor \exists x \in A\;(M < x).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg \operatorname{Maximum}(M,A) \Longleftrightarrow (M \notin A) \lor \exists x \in A\;(M < x).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The property fails either because the candidate value is missing from $A$ or because some element of $A$ sits strictly above it, violating the domination requirement.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\neg \operatorname{Maximum}(M,A)
=
\underbrace{(M \notin A)}_{\text{membership failure}}
\;\lor\;
\underbrace{\bigl(\exists x \in A\;(M < x)\bigr)}_{\text{domination failure}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
A maximum is an element of the set that simultaneously belongs to the set and dominates it under the ambient order. The domination clause guarantees that no point of $A$ lies above the maximum, so the element captures the extreme right edge of the subset. Failure occurs either when the supposed extremal is not actually part of the set or when another member protrudes farther to the right, showing the candidate was not truly maximal. In geometric terms, the maximum is the endpoint where the subset comes to rest when viewed along the ordering axis.
\end{remark*}

\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Supremum)}},
  fonttitle=\small\bfseries]
\begin{definition}[Supremum]
\label{def:supremum}
Let $(S,\le)$ be an ordered set, let $A \subseteq S$, and let $s \in S$. The element $s$ is the supremum of $A$ if and only if $s$ is an upper bound of $A$ and every upper bound $u$ of $A$ satisfies $s \le u$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (S,\le) \text{ be an ordered set, } A \subseteq S, \text{ and } s \in S.\\
&s \text{ is the supremum of } A \Longleftrightarrow \Bigl[(\forall x \in A)(x \le s) \land \forall u \in S\,\bigl((\forall x \in A)(x \le u) \rightarrow s \le u\bigr)\Bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{Supremum}(s,A) \coloneqq \operatorname{UpperBound}(s,A) \land \forall u \in S\,\bigl(\operatorname{UpperBound}(u,A) \rightarrow s \le u\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let } (S,\le) \text{ be an ordered set, } A \subseteq S, \text{ and } s \in S.\\
&\neg\bigl(s \text{ is the supremum of } A\bigr) \Longleftrightarrow \Bigl[\exists x \in A\, (s < x) \lor \exists u \in S\,\bigl((\forall x \in A)(x \le u) \land u < s\bigr)\Bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{Supremum}(s,A) \Longleftrightarrow \bigl(\exists x \in A\, (s < x)\bigr) \lor \bigl(\exists u \in S\,(\operatorname{UpperBound}(u,A) \land u < s)\bigr).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The definition fails in exactly two ways: either $s$ is not an upper bound of $A$, or there exists a strictly smaller upper bound $u$ of $A$, showing that $s$ is not the least among all upper bounds.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\begin{aligned}
&\text{Let } (S,\le) \text{ be an ordered set, } A \subseteq S, \text{ and } s \in S.\\[2pt]
&\neg\operatorname{Supremum}(s,A) \Longleftrightarrow
\underbrace{\exists x \in A\,(s < x)}_{\text{fails to be an upper bound}}
\;\lor\;
\underbrace{\exists u \in S\,\bigl(\operatorname{UpperBound}(u,A) \land u < s\bigr)}_{\text{upper bound lies below } s}.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The supremum of $A$ is the smallest ambient element that dominates the whole set. It exists precisely when $s$ bounds $A$ from above and no stricter bound sits below it. Failure occurs either because $A$ pokes above $s$ or because another upper bound nestles strictly beneath $s$, contradicting minimality. Geometrically, the supremum is the leftmost point in the order that caps the set, so every other cap must lie to its right.
\end{remark*}

\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Infimum)}},
  fonttitle=\small\bfseries]
\begin{definition}[Infimum]
\label{def:infimum}
Let $(S,\le)$ be an ordered set, let $A\subseteq S$, and let $i\in S$. The element $i$ is the infimum of $A$ if and only if $i$ is a lower bound of $A$ and every lower bound $l$ of $A$ satisfies $l\le i$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let }(S,\le)\text{ be an ordered set, }A\subseteq S,\text{ and }i\in S.\\
&i\text{ is the infimum of }A\\
&\Longleftrightarrow\Bigl[\forall x\in A\;(i\le x)\Bigr]\land\forall l\in S\;\Bigl(\bigl[\forall x\in A\;(l\le x)\bigr]\Rightarrow l\le i\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{Infimum}(i,A)\coloneqq \operatorname{LowerBound}(i,A)\land \forall l\in S\;\bigl(\operatorname{LowerBound}(l,A)\Rightarrow l\le i\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let }(S,\le)\text{ be an ordered set, }A\subseteq S,\text{ and }i\in S.\\
&i\text{ is not the infimum of }A\\
&\Longleftrightarrow \Bigl[\exists x\in A\;(i\nleq x)\Bigr]\lor \exists l\in S\;\Bigl(\bigl[\forall x\in A\;(l\le x)\bigr]\land l\nleq i\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{Infimum}(i,A)\Longleftrightarrow \neg\operatorname{LowerBound}(i,A)\lor \exists l\in S\;(\operatorname{LowerBound}(l,A)\land l\nleq i).
\]
\end{remark*}

\begin{remark*}[Failure modes]
The definition fails either because $i$ is not a lower bound of $A$, or because some other lower bound $l$ is not beneath $i$, preventing $i$ from being the greatest among all lower bounds.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\neg\operatorname{Infimum}(i,A)=
\underbrace{\neg\operatorname{LowerBound}(i,A)}_{\text{does not bound }A\text{ below}}
\;\lor\;
\underbrace{\exists l\in S\;(\operatorname{LowerBound}(l,A)\land l\nleq i)}_{\text{competing lower bound not below }i}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The infimum of $A$ is the greatest lower bound: it still lies below every element of $A$, yet no other lower bound stands above it. If $i$ fails to be an infimum, either it already misses some element of $A$ or there exists a better lower bound lying strictly higher in the order. Geometrically, in a number line picture, the infimum is the highest point lying to the left of $A$ that is still a universal left barrier for the set.
\end{remark*}

\noindent\textbf{Contrast: Maximum vs.\ Supremum.}

\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{@{} p{3.2cm} p{5.2cm} p{5.2cm} @{}}
\hline
\textbf{Property} & \textbf{Maximum $m=\max A$} & \textbf{Supremum $s=\sup A$} \\
\hline
Must belong to $A$?
&
Yes. $m \in A$ is required.
&
No. $s$ may or may not belong to $A$.
\\
\noalign{\vskip 0.3em}

Must be an upper bound?
&
Yes. $\forall x \in A \; (x \le m)$.
&
Yes. $\forall x \in A \; (x \le s)$.
\\
\noalign{\vskip 0.3em}

Must be the least upper bound?
&
Not stated separately, but membership together with the upper-bound condition forces it.
&
Yes. This is required explicitly.
\\
\noalign{\vskip 0.3em}

Existence guaranteed?
&
Only when the set attains its supremum.
&
For nonempty sets bounded above in $\mathbb{R}$, yes, by the Completeness Axiom.
\\
\hline
\end{tabular}
\end{center}

\noindent\emph{Note.}
A maximum is an element of the set, whereas a supremum is an extremal upper bound that may lie
outside the set. The two coincide exactly when the supremum belongs to the set.







\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Least Upper Bound Property Implies Existence of Suprema)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Least Upper Bound Property Implies Existence of Suprema]
\label{thm:lub-property-implies-existence-of-suprema}
Let $S$ be an ordered set with the least upper bound property. If $A\subseteq S$ is nonempty and bounded above in $S$, then $A$ has a supremum in $S$.

\smallskip
\noindent
\hyperref[prf:lub-property-implies-existence-of-suprema]{\textit{Go to proof.}}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let }S\text{ be an ordered set with the least upper bound property.}\\
&\forall A\subseteq S\;
\bigl[
A\ne\varnothing
\land
\exists u\in S\;\forall x\in A\;(x\le u)
\Longrightarrow
\exists s\in S\;(s=\sup_S A)
\bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{LeastUpperBoundProperty}(S)
\Longrightarrow
\forall A\subseteq S\;
\bigl[
A\ne\varnothing
\land
\operatorname{BoundedAbove}_S(A)
\Longrightarrow
\exists s\in S\;\operatorname{Supremum}_S(s,A)
\bigr].
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\operatorname{LeastUpperBoundProperty}(S)
\quad\text{and}\quad
\exists A\subseteq S:
A\ne\varnothing,\ A\text{ bounded above in }S,\ \text{and }\sup_S A\text{ does not exist}
\]
is impossible.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\operatorname{LeastUpperBoundProperty}(S)
\land
\exists A\subseteq S\;
\bigl[A\ne\varnothing
\land \operatorname{BoundedAbove}_S(A)
\land
\neg\exists s\in S\;\operatorname{Supremum}_S(s,A)\bigr]
\]
is a contradictory configuration.
\end{remark*}

\begin{remark*}[Failure modes]
The theorem could fail only if an admissible subset were nonempty and bounded above while still lacking a least upper bound in $S$. That is precisely the situation ruled out by the least upper bound property.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\operatorname{LeastUpperBoundProperty}(S)
=
\underbrace{
\forall A\subseteq S\;
\bigl[A\ne\varnothing\land \operatorname{BoundedAbove}_S(A)
\Longrightarrow
\exists s\in S\;\operatorname{Supremum}_S(s,A)\bigr]
}_{\text{every admissible subset has a least upper bound in }S}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
This theorem is the operational use of the least upper bound property. Once the ambient ordered set has that property, every nonempty bounded-above subset receives the least ceiling that the definition promises.
\end{remark*}

\begin{tcolorbox}[colback=propbox, colframe=propborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Proposition (Upper Bound Property of Supremum)}},
  fonttitle=\small\bfseries]
\begin{proposition}[Upper Bound Property of Supremum]
\label{prop:upper-bound-property-of-supremum}
Let $A\subseteq\mathbb{R}$ and let $s=\sup A$. Then $x\le s$ for every $x\in A$.

\smallskip
\noindent
\hyperref[prf:upper-bound-property-of-supremum]{\textit{Go to proof.}}
\end{proposition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let }A\subseteq\mathbb{R}\text{ and }s=\sup A.\\
&\forall x\in A\;(x\le s).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{Supremum}(s,A)
\Longrightarrow
\operatorname{UpperBound}(s,A).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
s=\sup A
\quad\text{and}\quad
\exists x\in A\;(s<x)
\]
is impossible.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\operatorname{Supremum}(s,A)\land \neg\operatorname{UpperBound}(s,A)
\]
is a contradictory configuration.
\end{remark*}

\begin{remark*}[Failure modes]
The proposition could fail only if a point of $A$ sat strictly above its claimed supremum. Such a point would show that the claimed supremum is not even an upper bound.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\neg\operatorname{UpperBound}(s,A)
\Longleftrightarrow
\underbrace{\exists x\in A\;(s<x)}_{\text{element above the claimed supremum}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The supremum has two jobs: it must bound the set from above, and it must be least among all such bounds. This proposition isolates the first job.
\end{remark*}

\begin{remark*}[Dependencies]
\hyperref[def:upper-bound]{Upper Bound};
\hyperref[def:supremum]{Supremum}.
\end{remark*}

\subsection{Consequences}

\begin{remark*}[Logical structure]
Bounds, extrema, and least/greatest bounds form a hierarchy of increasing
logical complexity.
Upper and lower bounds are defined by a universal quantifier alone.
Maximum and minimum add a membership condition: the extremum must belong
to the set.
Supremum and infimum relax membership but add a minimality (or maximality)
condition: the bound must be the \emph{least} (or \emph{greatest}) among
all bounds.
Maximum implies supremum --- when $\max A$ exists, it equals $\sup A$
(Proposition~\ref{prop:maximum-implies-supremum} and the proof at
\hyperref[prf:maximum-implies-supremum]{maximum-implies-supremum}) --- but the converse fails:
$(0,1)$ has $\sup = 1$ but no maximum.
\end{remark*}

\begin{remark*}[Connections forward]
The $\varepsilon$-characterization (Proposition~\ref{prop:eps-char}) is the
workhorse tool of the bounding chapter.
Every proof in the Bound Arithmetic section
(\S\ref{sec:bound-arithmetic}) invokes it to verify that a candidate value
is the least upper bound, not merely an upper bound.
The Monotone Convergence Theorem (convergence chapter) uses the same
mechanism: the limit of a bounded monotone sequence is identified as
$\sup\{a_n\}$, and the $\varepsilon$-characterization produces the required
$N$.
\end{remark*}
