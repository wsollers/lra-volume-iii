\begin{proposition*}[Little-o Quotient Characterisation]
Let $a\in\mathbb{R}$, and let $f$ and $g$ be real-valued functions defined on a
punctured neighbourhood of $a$, with $g(x)\neq 0$ for $x$ sufficiently close to
$a$ and $x\neq a$. Then
\[
  f(x)=o(g(x))\ (x\to a)
  \quad\Longleftrightarrow\quad
  \frac{f(x)}{g(x)}\to 0\ (x\to a).
\]
\end{proposition*}

\begin{proof}[Professional Standard Proof]
\LRAProofBodyStart
By definition, $f=o(g)$ as $x\to a$ means that for every $\varepsilon>0$,
eventually $|f(x)|\leq\varepsilon |g(x)|$. Since $g(x)\neq 0$ eventually, this
is equivalent to
\[
  \left|\frac{f(x)}{g(x)}\right|\leq\varepsilon
\]
eventually. That is precisely the assertion that $f(x)/g(x)\to 0$ as $x\to a$.
\end{proof}

\begin{proof}[Detailed Learning Proof]
\LRAProofBodyStart
Assume first that $f=o(g)$. Given $\varepsilon>0$, choose $\delta>0$ so that
$0<|x-a|<\delta$ implies $|f(x)|\leq\varepsilon |g(x)|$. On a smaller punctured
neighbourhood, if necessary, $g(x)\neq 0$. Dividing by $|g(x)|$ gives
$|f(x)/g(x)|\leq\varepsilon$, so $f(x)/g(x)\to 0$.

Conversely, assume $f(x)/g(x)\to 0$. Given $\varepsilon>0$, choose $\delta>0$
so that $0<|x-a|<\delta$ implies $|f(x)/g(x)|\leq\varepsilon$, again within a
punctured neighbourhood where $g$ is nonzero. Multiplying by $|g(x)|$ gives
$|f(x)|\leq\varepsilon |g(x)|$, hence $f=o(g)$.
\end{proof}
