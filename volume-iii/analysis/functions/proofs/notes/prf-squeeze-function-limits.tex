\begin{theorem*}[Squeeze Theorem for Function Limits]
If $\lim_{x\to c}f(x)=L=\lim_{x\to c}g(x)$ and $f(x) \leq h(x) \leq g(x)$ near $c$, then $\lim_{x\to c}h(x)=L$.
\end{theorem*}
