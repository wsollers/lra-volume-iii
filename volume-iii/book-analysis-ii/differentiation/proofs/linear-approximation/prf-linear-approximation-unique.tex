\begin{proposition*}[Uniqueness of the Linear Coefficient]
Let $f:A\to\mathbb{R}$ and let $a\in A$ be an interior point of $A$. If
\[
  f(a+h)=f(a)+Lh+o(h)
  \qquad\text{and}\qquad
  f(a+h)=f(a)+Mh+o(h),
\]
then $L=M$.
\end{proposition*}
