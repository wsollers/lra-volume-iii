\begin{theorem*}[Holder's Inequality]
Let $p,q>1$ satisfy $1/p+1/q=1$. For real numbers $a_1,\dots,a_n$ and
$b_1,\dots,b_n$,
\[
\sum_{j=1}^{n}|a_jb_j|
\le
\left(\sum_{j=1}^{n}|a_j|^p\right)^{1/p}
\left(\sum_{j=1}^{n}|b_j|^q\right)^{1/q}.
\]
\end{theorem*}
