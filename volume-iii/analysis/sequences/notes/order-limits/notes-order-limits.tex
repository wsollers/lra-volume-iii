\subsection*{Order-Theoretic Limit Constructions}
\begin{exposition}
Completeness lets order data produce limits. An increasing sequence bounded
above converges to the least ceiling of its range; a decreasing sequence
bounded below converges to the greatest floor of its range. Applying these
same ideas to tail suprema and tail infima produces limsup and liminf for
every bounded sequence.
\end{exposition}

% ---------------------------------------------------------
% Theorem --- Increasing Sequence Limit as Supremum
% ---------------------------------------------------------

\begin{theorembox}{Theorem (Increasing Sequence Limit as Supremum)}
\begin{theorem}[Increasing Sequence Limit as Supremum]
\label{thm:increasing-sequence-limit-as-supremum}
Let $(x_n)$ be an increasing real sequence that is bounded above. Then
\[
  \lim_{n\to\infty}x_n=\sup\{x_n:n\in\mathbb{N}\}.
\]

\smallskip
\noindent
\hyperref[prf:increasing-sequence-limit-as-supremum]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\forall (x_n)\subseteq\mathbb{R}\;
\Bigl(
  \bigl[\forall n\in\mathbb{N}\;(x_n\le x_{n+1})\bigr]
  \land
  \bigl[\exists M\in\mathbb{R}\;\forall n\in\mathbb{N}\;(x_n\le M)\bigr]\\
&\qquad\Longrightarrow\quad
  \lim_{n\to\infty}x_n=\sup\{x_n:n\in\mathbb{N}\}
\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Theorem predicate reading]
\[
\operatorname{MonotoneIncreasingSequence}(x_n)
\land
\operatorname{BoundedAboveSequence}(x_n)
\Longrightarrow
\operatorname{ConvergentSequence}\bigl(x_n,\sup\{x_n:n\in\mathbb{N}\}\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The supremum is the only possible destination: all terms lie below it, and
the defining approximation property of the supremum forces the sequence
eventually above every lower tolerance.
\end{remark*}

\end{remark*}

% Theorem --- Decreasing Sequence Limit as Infimum
% ---------------------------------------------------------

\begin{theorembox}{Theorem (Decreasing Sequence Limit as Infimum)}
\begin{theorem}[Decreasing Sequence Limit as Infimum]
\label{thm:decreasing-sequence-limit-as-infimum}
Let $(x_n)$ be a decreasing real sequence that is bounded below. Then
\[
  \lim_{n\to\infty}x_n=\inf\{x_n:n\in\mathbb{N}\}.
\]

\smallskip
\noindent
\hyperref[prf:decreasing-sequence-limit-as-infimum]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\forall (x_n)\subseteq\mathbb{R}\;
\Bigl(
  \bigl[\forall n\in\mathbb{N}\;(x_{n+1}\le x_n)\bigr]
  \land
  \bigl[\exists m\in\mathbb{R}\;\forall n\in\mathbb{N}\;(m\le x_n)\bigr]\\
&\qquad\Longrightarrow\quad
  \lim_{n\to\infty}x_n=\inf\{x_n:n\in\mathbb{N}\}
\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Theorem predicate reading]
\[
\operatorname{MonotoneDecreasingSequence}(x_n)
\land
\operatorname{BoundedBelowSequence}(x_n)
\Longrightarrow
\operatorname{ConvergentSequence}\bigl(x_n,\inf\{x_n:n\in\mathbb{N}\}\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The infimum is the decreasing analogue of the supremum target: the sequence
falls toward its greatest lower barrier.
\end{remark*}

% Theorem --- Tail Suprema and Infima Converge
% ---------------------------------------------------------

\begin{theorembox}{Theorem (Tail Suprema and Infima Converge)}
\begin{theorem}[Tail Suprema and Infima Converge]
\label{thm:tail-suprema-and-infima-converge}
Let $(x_n)$ be a bounded real sequence. Define
\[
  s_n=\sup\{x_k:k\ge n\},
  \qquad
  i_n=\inf\{x_k:k\ge n\}.
\]
Then $(s_n)$ and $(i_n)$ converge.

\smallskip
\noindent
\hyperref[prf:tail-suprema-and-infima-converge]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\forall (x_n)\subseteq\mathbb{R}\;
\Bigl(
  (x_n)\text{ is bounded}
  \Longrightarrow\\
&\quad
  \exists S,I\in\mathbb{R}\;
  \Bigl[
    \lim_{n\to\infty}\sup\{x_k:k\ge n\}=S
    \land
    \lim_{n\to\infty}\inf\{x_k:k\ge n\}=I
  \Bigr]
\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Theorem predicate reading]
\[
\operatorname{BoundedSequence}(x_n)
\Longrightarrow
\exists S,I\in\mathbb{R}\;
\bigl(
  \operatorname{ConvergentSequence}(\sup\{x_k:k\ge n\},S)
  \land
  \operatorname{ConvergentSequence}(\inf\{x_k:k\ge n\},I)
\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Tail suprema decrease because later tails contain fewer terms; tail infima
increase for the same reason. Boundedness supplies the opposite bound needed
for monotone convergence.
\end{remark*}

\end{remark*}

% Theorem --- Bounded Sequence Has Limsup and Liminf
% ---------------------------------------------------------

\begin{theorembox}{Theorem (Bounded Sequence Has Limsup and Liminf)}
\begin{theorem}[Bounded Sequence Has Limsup and Liminf]
\label{thm:bounded-sequence-has-limsup-and-liminf}
Every bounded real sequence has a limit superior and a limit inferior.

\smallskip
\noindent
\hyperref[prf:bounded-sequence-has-limsup-and-liminf]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall (x_n)\subseteq\mathbb{R}\;
\Bigl(
  (x_n)\text{ is bounded}
  \Longrightarrow
  \exists S,I\in\mathbb{R}\;
  \bigl(S=\limsup_{n\to\infty}x_n
  \land I=\liminf_{n\to\infty}x_n\bigr)
\Bigr).
\]
\end{remark*}

\begin{remark*}[Theorem predicate reading]
\[
\operatorname{BoundedSequence}(x_n)
\Longrightarrow
\exists S,I\in\mathbb{R}\;
\bigl(
  \operatorname{LimsupSequence}(x_n,S)
  \land
  \operatorname{LiminfSequence}(x_n,I)
\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
For bounded sequences, limsup and liminf are not extra assumptions. They are
guaranteed order-theoretic limits of the tail envelopes.
\end{remark*}
