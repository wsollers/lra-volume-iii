\begin{theorem}[Infimum of a Product Set]
\label{thm:infimum-product-set}
Let $A,B\subseteq\mathbb{R}$ be nonempty and bounded. Then
\[
  \inf(AB)=
  \min\{(\inf A)(\inf B),(\inf A)(\sup B),(\sup A)(\inf B),(\sup A)(\sup B)\}.
\]
\hyperref[prf:infimum-product-set]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
A,B\neq\varnothing,\quad A,B\text{ bounded}
\quad\Longrightarrow\quad
\inf\{ab:a\in A,\ b\in B\}
\]
equals the minimum of the four endpoint products.
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
A,B\subseteq\mathbb{R},\qquad AB=\{ab:a\in A,\ b\in B\},\\
&L=\min\{\alpha\gamma,\alpha\delta,\beta\gamma,\beta\delta\},\\
&\operatorname{GreatestLowerBound}(\alpha,A,P)\land
\operatorname{LeastUpperBound}(\beta,A,P)\land\\
&\operatorname{GreatestLowerBound}(\gamma,B,P)\land
\operatorname{LeastUpperBound}(\delta,B,P)
\Longrightarrow
\operatorname{GreatestLowerBound}(L,AB,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\inf(AB)\neq
\min\{(\inf A)(\inf B),(\inf A)(\sup B),(\sup A)(\inf B),(\sup A)(\sup B)\}
\]
is impossible under the stated hypotheses.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
A,B\subseteq\mathbb{R},\qquad AB=\{ab:a\in A,\ b\in B\},\\
&L=\min\{\alpha\gamma,\alpha\delta,\beta\gamma,\beta\delta\},\\
&\operatorname{GreatestLowerBound}(\alpha,A,P)\land
\operatorname{LeastUpperBound}(\beta,A,P)\land\\
&\operatorname{GreatestLowerBound}(\gamma,B,P)\land
\operatorname{LeastUpperBound}(\delta,B,P)\land
\neg\operatorname{GreatestLowerBound}(L,AB,P)
\end{aligned}
\]
is a contradictory configuration.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
A failure would require a product below all four endpoint products or would
require the endpoint minimum not to be greatest.


  \item[\operatorname{GreatestLowerBound} / \operatorname{LeastUpperBound}.]
  \[
\inf(AB)\neq
\min\{(\inf A)(\inf B),(\inf A)(\sup B),(\sup A)(\inf B),(\sup A)(\sup B)\}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
A,B\subseteq\mathbb{R},\qquad AB=\{ab:a\in A,\ b\in B\},\\
&L=\min\{\alpha\gamma,\alpha\delta,\beta\gamma,\beta\delta\},\\
&\operatorname{GreatestLowerBound}(\alpha,A,P)\land
\operatorname{LeastUpperBound}(\beta,A,P)\land\\
&\operatorname{GreatestLowerBound}(\gamma,B,P)\land
\operatorname{LeastUpperBound}(\delta,B,P)\land
\neg\operatorname{GreatestLowerBound}(L,AB,P)
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
Reduce to the same endpoint analysis as the supremum product formula, replacing
the corner maximum by the corner minimum.


  \item[\operatorname{GreatestLowerBound} / \operatorname{LeastUpperBound}.]
  \[
\inf(AB)\neq
\min\{(\inf A)(\inf B),(\inf A)(\sup B),(\sup A)(\inf B),(\sup A)(\sup B)\}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
A,B\subseteq\mathbb{R},\qquad AB=\{ab:a\in A,\ b\in B\},\\
&L=\min\{\alpha\gamma,\alpha\delta,\beta\gamma,\beta\delta\},\\
&\operatorname{GreatestLowerBound}(\alpha,A,P)\land

&\operatorname{GreatestLowerBound}(\gamma,B,P)\land

\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
The infimum of a bounded product set is the smallest of the same four endpoint
products that control the enclosing rectangle.
\end{remark*}
