\begin{lemma}[Rational Difference Preserves Rationality]
\label{lem:rational-difference-preserves-rationality}
For every $r,q\in\mathbb{Q}$,
\[
  r-q\in\mathbb{Q}.
\]
\hyperref[prf:rational-difference-preserves-rationality]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\forall r,q\in\mathbb{Q}\;(r-q\in\mathbb{Q}).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
r\in\mathbb{Q}\land q\in\mathbb{Q}\Longrightarrow r-q\in\mathbb{Q}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Subtraction is addition of the additive inverse, and rational numbers are
closed under both operations.
\end{remark*}
