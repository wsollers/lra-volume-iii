\begin{proposition*}[Triangle Inequality for Finite Set Summation]
Let \(X\) be a finite set, and let \(f:X\to\mathbb{R}\) be a function.
Then
\[
  \left\lvert \sum_{x\in X} f(x) \right\rvert
  \le
  \sum_{x\in X} \lvert f(x)\rvert.
\]
\end{proposition*}
