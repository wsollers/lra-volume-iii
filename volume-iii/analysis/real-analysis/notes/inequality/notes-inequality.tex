\section{Inequality}
\label{sec:inequality}

\begin{tcolorbox}[
  colback=gray!6,
  colframe=gray!40,
  arc=2pt,
  left=8pt, right=8pt, top=6pt, bottom=6pt,
  title={\small\textbf{Toolkit: Inequality}},
  fonttitle=\small\bfseries
]
The algebra of inequalities governs how strict and nonstrict order
relations behave under addition, multiplication, reciprocation, and
composition.  The atomic notions formalised below are:
\begin{itemize}
  \item \textbf{addition on both sides} --- adding the same term preserves
        strict and nonstrict inequalities;
  \item \textbf{addition of inequalities} --- strict, nonstrict, and mixed
        pairwise addition;
  \item \textbf{multiplication by a positive scalar} --- preserves direction;
  \item \textbf{multiplication by a negative scalar} --- reverses direction;
  \item \textbf{nonstrict multiplication by a nonneg scalar} --- the zero
        edge case;
  \item \textbf{squeeze theorem} --- equality forced by double bounding;
  \item \textbf{transitivity} --- strict and mixed chaining;
  \item \textbf{reciprocal inequalities} --- order reversal under
        reciprocation of positive quantities;
  \item \textbf{square monotonicity} --- ordering of squares for
        nonneg inputs;
  \item \textbf{square-root monotonicity} --- ordering of square roots
        (requires \texttt{def:square-root});
  \item \textbf{AM-GM for two terms} --- arithmetic--geometric mean
        inequality;
  \item \textbf{Bernoulli's inequality} --- $(1+x)^n \ge 1 + nx$.
\end{itemize}
\end{tcolorbox}

\begin{theorem}[Addition Preserves Strict Inequality]
\label{thm:ineq-add-both-sides}
For all $a, b, c \in \mathbb{R}$,
\[
  a < b \;\Rightarrow\; a + c < b + c.
\]
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  \forall a, b, c \in \mathbb{R},\quad
  a < b \;\Rightarrow\; a + c < b + c.
\]
\end{remark*}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Squeeze)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Squeeze]
\label{thm:ineq-squeeze}
For all $a, b, c \in \mathbb{R}$,
\[
  a \le b \le c \;\land\; a = c \;\Rightarrow\; b = a.
\]
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  \forall a, b, c \in \mathbb{R},\quad
  a \le b \le c \;\land\; a = c \;\Rightarrow\; b = a.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The squeeze principle forces equality when a value is simultaneously
bounded above and below by equal quantities.  The continuous analogue
(if $f(x) \le g(x) \le h(x)$ and $f(x) \to L$ and $h(x) \to L$ then
$g(x) \to L$) is the central tool for computing limits.  The algebraic
version stated here is its foundation.
\end{remark*}

\hyperref[def:ordered-field]{Ordered field},
\hyperref[thm:additive-cancellation]{additive cancellation}.
\end{remark*}

\begin{theorem}[AM-GM Inequality for Two Terms]
\label{thm:ineq-am-gm-two}
For all $a, b \ge 0$,
\[
  \sqrt{ab} \le \frac{a + b}{2}.
\]
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  \forall a, b \in \mathbb{R},\;
  a \ge 0 \;\land\; b \ge 0 \;\Rightarrow\;
  \sqrt{ab} \le \frac{a + b}{2}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The arithmetic--geometric mean inequality for two terms is the simplest
case of the AM-GM family.  The standard proof proceeds by squaring both
sides (using
\hyperref[thm:ineq-square-monotone]{square monotonicity}) and reducing
to $(\sqrt{a} - \sqrt{b})^2 \ge 0$, which follows from
\hyperref[thm:squares-nonnegative]{squares being nonnegative}.
\end{remark*}

\begin{remark*}[Dependencies]
\hyperref[thm:squares-nonnegative]{squares are nonnegative},
\hyperref[thm:ineq-square-monotone]{square is strictly monotone on nonneg reals},
\texttt{def:square-root} (not yet registered --- see audit note in
\hyperref[thm:ineq-square-root-monotone]{square root monotonicity}).
\end{remark*}

\begin{remark*}[Proof]
\hyperref[prf:ineq-am-gm-two]{Go to proof.}
\end{remark*}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Bernoulli's Inequality)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Bernoulli's Inequality]
\label{thm:ineq-bernoulli}
For all $x \in \mathbb{R}$ with $x \ge -1$ and all $n \in \mathbb{N}$,
\[
  (1 + x)^n \ge 1 + nx.
\]
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  \forall x \in \mathbb{R},\;
  \forall n \in \mathbb{N},\;
  x \ge -1 \;\Rightarrow\;
  (1 + x)^n \ge 1 + nx.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Bernoulli's inequality is proved by induction and is a key estimate in
analysis: it provides a linear lower bound for exponential growth and
is used to establish the divergence of the harmonic series, bounds on
compound interest, and properties of the exponential function.
\end{remark*}
