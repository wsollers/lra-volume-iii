\begin{corollary*}[Fubini's Theorem for Finite Series]
Let \(X\) and \(Y\) be finite sets, and let
\(f:X\times Y\to\mathbb{R}\) be a function. Then
\[
  \sum_{x\in X}
  \left(
    \sum_{y\in Y} f(x,y)
  \right)
  =
  \sum_{(x,y)\in X\times Y} f(x,y)
  =
  \sum_{(y,x)\in Y\times X} f(x,y)
  =
  \sum_{y\in Y}
  \left(
    \sum_{x\in X} f(x,y)
  \right).
\]
\end{corollary*}
