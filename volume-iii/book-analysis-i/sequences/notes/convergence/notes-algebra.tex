\subsection*{Algebra of Convergent Sequences}

\begin{exposition}
The algebra of convergent sequences records that limits are compatible with the
ordinary algebraic operations on real numbers. The pointwise operations below
turn real sequences into new real sequences, and the theorem blocks that follow
state how convergence passes through those operations. The reciprocal and
quotient laws require nonzero denominator hypotheses, while the square-root
law requires nonnegative terms.
\end{exposition}

% ---------------------------------------------------------
% Definition --- Pointwise Sum of Sequences
% ---------------------------------------------------------

\begin{definitionbox}{Definition (Pointwise Sum of Sequences)}
\begin{definition}[Pointwise Sum of Sequences]
\label{def:pointwise-sum-sequence}
Let $(x_n)$ and $(y_n)$ be real sequences. The \textbf{pointwise sum}
of $(x_n)$ and $(y_n)$ is the real sequence $(x_n+y_n)$ defined by
\[
(x+y)_n := x_n+y_n
\qquad
\text{for every } n\in\mathbb{N}.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (x_n) \text{ and } (y_n) \text{ be real sequences.}\\
&\forall n\in\mathbb{N}\;\bigl((x+y)_n=x_n+y_n\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The sum sequence is formed term by term. No limiting process is involved in
the definition; convergence enters only when the behavior of the resulting
sequence is studied.
\end{remark*}

\begin{definitionbox}{Definition (Linear Combination of Real Sequences)}
\begin{definition}[Linear Combination of Real Sequences]
\label{def:linear-combination-of-real-sequences}
Let $(x_n)$ and $(y_n)$ be real sequences, and let
$\alpha,\beta\in\mathbb{R}$.  The real linear combination
$\alpha(x_n)+\beta(y_n)$ is the real sequence $(\alpha x_n+\beta y_n)$

\[
(\alpha x+\beta y)_n := \alpha x_n+\beta y_n
\qquad
\text{for every } n\in\mathbb{N}.
\]
\end{definition}
\end{definitionbox}

\[
\begin{aligned}
&\text{Let } (x_n),(y_n) \text{ be real sequences and }

  =\alpha x_n+\beta y_n\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
A linear combination of sequences is built term by term from scalar
multiples and sums. It is the common algebraic shape behind sequence limit
rules and Cauchy closure under addition and scalar multiplication.
\end{remark*}

\begin{definitionbox}{Definition (Reciprocal Sequence)}
\begin{definition}[Reciprocal Sequence]
\label{def:reciprocal-sequence}
Let $(x_n)$ be a real sequence such that $x_n\neq 0$ for every
$n\in\mathbb{N}$. The \textbf{reciprocal sequence} of $(x_n)$ is the real
sequence $(1/x_n)$ defined by
\[
\left(\frac{1}{x}\right)_n := \frac{1}{x_n}
\qquad
\text{for every } n\in\mathbb{N}.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (x_n) \text{ be a real sequence with }
  \forall n\in\mathbb{N}\;(x_n\neq 0).\\
&\forall n\in\mathbb{N}\;
\left(\left(\frac{1}{x}\right)_n=\frac{1}{x_n}\right).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&\text{Let } (x_n) \text{ be a real sequence with }
  \forall n\in\mathbb{N}\;(x_n\neq 0).\\
&\forall n\in\mathbb{N}\;
\left(\left(\frac{1}{x}\right)_n=\frac{1}{x_n}\right).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The reciprocal sequence is defined only when division by each term is legal.
For limit theorems, a nonzero limit guarantees eventual nonzero behavior, but
the sequence-level construction itself requires nonzero terms wherever it is
defined.
\end{remark*}

\begin{theorembox}{Theorem (Limit of a Scalar Multiple)}
\begin{theorem}[Limit of a Scalar Multiple]
\label{thm:limit-of-scalar-multiple}
Let $(x_n)$ be a real sequence, let $L\in\mathbb{R}$, and let
$\alpha\in\mathbb{R}$. If $x_n\to L$, then $\alpha x_n\to \alpha L$.

\smallskip
\noindent
\hyperref[prf:limit-of-scalar-multiple]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (x_n) \text{ be a real sequence and let }
  L,\alpha\in\mathbb{R}.\\
&\bigl[
\forall \varepsilon>0\;\exists N\in\mathbb{N}\;\forall n\ge N\;
(|x_n-L|<\varepsilon)
\bigr]\\
&\qquad\Longrightarrow\qquad
\bigl[
\forall \varepsilon>0\;\exists N\in\mathbb{N}\;\forall n\ge N\;
(|\alpha x_n-\alpha L|<\varepsilon)
\bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{ConvergesTo}(x_n,L,\mathbb{R})
\Longrightarrow
\operatorname{ConvergesTo}(\alpha x_n,\alpha L,\mathbb{R}).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Multiplying every term by a fixed scalar multiplies the limiting value by the
same scalar. The case $\alpha=0$ collapses the sequence to the constant zero
sequence.
\end{remark*}

\begin{theorembox}{Theorem (Limit of a Sum)}
\begin{theorem}[Limit of a Sum]
\label{thm:limit-of-a-sum}
Let $(x_n)$ and $(y_n)$ be real sequences, and let $L,M\in\mathbb{R}$.
If $x_n\to L$ and $y_n\to M$, then $x_n+y_n\to L+M$.

\smallskip
\noindent
\hyperref[prf:limit-of-a-sum]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (x_n),(y_n) \text{ be real sequences and let } L,M\in\mathbb{R}.\\
&\bigl[
x_n\to L \;\land\; y_n\to M
\bigr]
\Longrightarrow
\bigl[
x_n+y_n\to L+M
\bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\bigl(
\operatorname{ConvergesTo}(x_n,L,\mathbb{R})
\land
\operatorname{ConvergesTo}(y_n,M,\mathbb{R})
\bigr)
\Longrightarrow
\operatorname{ConvergesTo}(x_n+y_n,L+M,\mathbb{R}).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The sum law says that convergent sequences may be added before or after
taking limits. The proof splits a target tolerance into two smaller tolerances,
one for each summand.
\end{remark*}

\begin{theorembox}{Theorem (Nonzero Limit is Eventually Nonzero)}
\begin{theorem}[Nonzero Limit is Eventually Nonzero]
\label{thm:nonzero-limit-eventually-nonzero}
Let $(x_n)$ be a real sequence, and let $L\in\mathbb{R}$ with $L\neq 0$.
If $x_n\to L$, then there exists $N\in\mathbb{N}$ such that $x_n\neq 0$
for every $n\ge N$.

\smallskip
\noindent
\hyperref[prf:nonzero-limit-eventually-nonzero]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (x_n) \text{ be a real sequence and let } L\in\mathbb{R}.\\
&\bigl[
L\neq 0 \;\land\; x_n\to L
\bigr]
\Longrightarrow
\exists N\in\mathbb{N}\;\forall n\ge N\;(x_n\neq 0).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&\text{Let } (x_n) \text{ be a real sequence and let } L\in\mathbb{R}.\\
&\bigl[
L\neq 0 \;\land\; x_n\to L
\bigr]
\Longrightarrow
\exists N\in\mathbb{N}\;\forall n\ge N\;(x_n\neq 0).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
A sequence converging to a nonzero number eventually stays away from zero.
This is the denominator-control result needed for reciprocal and quotient
rules.
\end{remark*}

Let $(x_n)$ be a real sequence such that $x_n\neq 0$ for every
$n\in\mathbb{N}$, and let $L\in\mathbb{R}$ with $L\neq 0$. If $x_n\to L$,
then $1/x_n\to 1/L$.

\smallskip
\noindent
\hyperref[prf:limit-of-a-reciprocal]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (x_n) \text{ be a real sequence with }
  \forall n\in\mathbb{N}\;(x_n\neq 0), \text{ and let } L\neq 0.\\
&\bigl[x_n\to L\bigr]
\Longrightarrow
\bigl[\frac{1}{x_n}\to \frac{1}{L}\bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{ConvergesTo}(x_n,L,\mathbb{R})
\Longrightarrow
\operatorname{ConvergesTo}\left(\frac{1}{x_n},\frac{1}{L},\mathbb{R}\right).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Reciprocation is continuous away from zero. The nonzero limit supplies a tail
on which the denominators stay uniformly separated from zero.
\end{remark*}

\begin{theorembox}{Theorem (Limit of a Quotient)}
\begin{theorem}[Limit of a Quotient]
\label{thm:limit-of-a-quotient}
Let $(x_n)$ and $(y_n)$ be real sequences such that $y_n\neq 0$ for every
$n\in\mathbb{N}$, and let $L,M\in\mathbb{R}$ with $M\neq 0$. If
$x_n\to L$ and $y_n\to M$, then $x_n/y_n\to L/M$.

\smallskip
\noindent
\hyperref[prf:limit-of-a-quotient]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (x_n),(y_n) \text{ be real sequences with }
  \forall n\in\mathbb{N}\;(y_n\neq 0), \text{ and let } M\neq 0.\\
&\bigl[
x_n\to L \;\land\; y_n\to M
\bigr]
\Longrightarrow
\bigl[
\frac{x_n}{y_n}\to \frac{L}{M}
\bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\bigl(
\operatorname{ConvergesTo}(x_n,L,\mathbb{R})
\land
\operatorname{ConvergesTo}(y_n,M,\mathbb{R})
\bigr)
\Longrightarrow
\operatorname{ConvergesTo}\left(\frac{x_n}{y_n},\frac{L}{M},\mathbb{R}\right).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The quotient law is the product law applied to $(x_n)$ and the reciprocal of
$(y_n)$. The condition $M\neq 0$ prevents the limiting denominator from
collapsing.
\end{remark*}

\subsection*{Error-Track Method for Limit Algebra}
\label{subsec:error-track-method-limit-algebra}

\begin{exposition}
The algebra of limits can be read as error propagation with vanishing errors.
If
\[
  x_n=A+\alpha_n,\qquad y_n=B+\beta_n,
  \qquad \alpha_n\to 0,\quad \beta_n\to 0,
\]
then $\alpha_n$ and $\beta_n$ are the moving errors. Sum, product, and quotient
limits follow by expanding the target expression into a main term plus an
error expression that tends to zero.
\end{exposition}

\begin{example*}[Sum]
If $x_n=A+\alpha_n$ and $y_n=B+\beta_n$, then
\[
  x_n+y_n
  =
  A+B+(\alpha_n+\beta_n).
\]
The new error is $\alpha_n+\beta_n$, which tends to zero by the sum rule for
null sequences.
\end{example*}

\begin{example*}[Product]
The same decomposition gives
\[
  x_ny_n
  =
  (A+\alpha_n)(B+\beta_n)
  =
  AB+A\beta_n+B\alpha_n+\alpha_n\beta_n.
\]
The first two error terms are linear in the original errors. The last term is
second-order and vanishes because a product of null sequences is null.
\end{example*}

\begin{example*}[Quotient]
If $B\neq 0$ and $y_n\neq 0$ eventually, then
\[
  \frac{x_n}{y_n}-\frac{A}{B}
  =
  \frac{B\alpha_n-A\beta_n}{B(B+\beta_n)}.
\]
The numerator tends to zero, while the denominator tends to $B^2\neq 0$.
This is the quotient law in error coordinates.
\end{example*}

\begin{remark*}[Interpretation]
The finite error formulas for approximate numbers have the same algebraic
shape. The passage to limits replaces fixed error bounds by null sequences
and replaces informal smallness by the assertion that a remainder tends to
zero.
\end{remark*}

\begin{theorembox}{Theorem (Positive Limit is Eventually Positive)}
\begin{theorem}[Positive Limit is Eventually Positive]
\label{thm:positive-limit-eventually-positive}
Let $(x_n)$ be a real sequence, and let $L>0$. If $x_n\to L$, then there
exists $N\in\mathbb{N}$ such that $0<x_n$ for every $n\ge N$.

\smallskip
\noindent
\hyperref[prf:positive-limit-eventually-positive]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (x_n) \text{ be a real sequence and let } L>0.\\
&\bigl[x_n\to L\bigr]
\Longrightarrow
\exists N\in\mathbb{N}\;\forall n\ge N\;(0<x_n).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&\text{Let } (x_n) \text{ be a real sequence and let } L>0.\\
&\bigl[x_n\to L\bigr]
\Longrightarrow
\exists N\in\mathbb{N}\;\forall n\ge N\;(0<x_n).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Convergence to a positive limit eventually traps the sequence inside the
positive half-line. This result is the sign-control counterpart of the
eventual nonzero theorem.
\end{remark*}

\begin{theorembox}{Theorem (Rational Sequence Limit)}
\begin{theorem}[Rational Sequence Limit]
\label{thm:rational-sequence-limit}
Let $(x_n)$ be a real sequence, let $L\in\mathbb{R}$, and let
$p,q\in\mathbb{R}[t]$ be polynomials. Suppose $q(L)\neq 0$ and
$q(x_n)\neq 0$ for every $n\in\mathbb{N}$. If $x_n\to L$, then
\[
\frac{p(x_n)}{q(x_n)}\to \frac{p(L)}{q(L)}.
\]

\smallskip
\noindent
\hyperref[prf:rational-sequence-limit]{\textit{Go to proof.}}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (x_n) \text{ be a real sequence, } L\in\mathbb{R},
  \text{ and } p,q\in\mathbb{R}[t].\\
&\bigl[
q(L)\neq 0
\land
\forall n\in\mathbb{N}\;(q(x_n)\neq 0)
\land
x_n\to L
\bigr]\\
&\qquad\Longrightarrow\qquad
\left[
\frac{p(x_n)}{q(x_n)}\to\frac{p(L)}{q(L)}
\right].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{ConvergesTo}(x_n,L,\mathbb{R})
\Longrightarrow
\operatorname{ConvergesTo}
\left(\frac{p(x_n)}{q(x_n)},\frac{p(L)}{q(L)},\mathbb{R}\right).
\]
\end{remark*}

\begin{remark*}[Interpretation]
A rational expression is a quotient of two polynomial expressions. The
condition $q(L)\neq 0$ keeps the limiting denominator legitimate, and the
pointwise condition $q(x_n)\neq 0$ keeps every term of the quotient sequence
defined.
\end{remark*}
