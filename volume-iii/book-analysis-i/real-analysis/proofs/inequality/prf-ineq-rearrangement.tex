\begin{theorem*}[Rearrangement Inequality]
Let $a_1\le\cdots\le a_n$ and $b_1\le\cdots\le b_n$. For every permutation
$(b'_1,\dots,b'_n)$ of $(b_1,\dots,b_n)$,
\[
\sum_{j=1}^{n}a_jb_{n+1-j}
\le
\sum_{j=1}^{n}a_jb'_j
\le
\sum_{j=1}^{n}a_jb_j.
\]
\end{theorem*}
