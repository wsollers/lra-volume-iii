\begin{lemma*}[Finite Sum of Scalar Multiples]
Let \(m \le n\) be integers, let \(a_i\) be a real number assigned to
each integer \(i\) satisfying \(m \le i \le n\), and let
\(c \in \mathbb{R}\). Then
\[
  \sum_{i=m}^{n} (c a_i)
  =
  c\left(\sum_{i=m}^{n} a_i\right).
\]
\end{lemma*}
