\begin{theorem*}[Constant Squeeze Theorem]
Let $(x_n)$ be a real sequence, and let $L\in\mathbb{R}$. If there
exists $N_0\in\mathbb{N}$ such that $L\le x_n\le L$ for every
$n\ge N_0$, then $x_n\to L$.

\noindent
\end{theorem*}
