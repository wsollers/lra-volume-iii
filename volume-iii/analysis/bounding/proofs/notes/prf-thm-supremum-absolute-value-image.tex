\begin{theorem*}[Supremum of the Absolute-Value Image]
Let $A\subseteq\mathbb{R}$ be nonempty and bounded. Then
\[
  \sup |A|=\max\{|\sup A|,|\inf A|\},
  \qquad
  |A|=\{|a|:a\in A\}.
\]
\end{theorem*}
