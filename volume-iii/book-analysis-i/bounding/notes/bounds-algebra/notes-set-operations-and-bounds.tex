\begin{remark*}[Orientation]
Set operations change the collection of possible counterexamples to a proposed
bound.  Each result below isolates one operation and one direction of bound
transport.
\end{remark*}

\begin{lemmabox}{Lemma (Union Preserves Upper Bounds)}
\begin{lemma}[Union Preserves Upper Bounds]
\label{lem:union-preserves-upper-bounds}
Let $A,B\subseteq\mathbb{R}$. For every $u\in\mathbb{R}$, if $u$ is an
upper bound of both $A$ and $B$, then $u$ is an upper bound of
$A\cup B$; that is,
\[
\left[
(\forall a\in A,\ a\leq u)
\land
(\forall b\in B,\ b\leq u)
\right]
\Longrightarrow
(\forall x\in A\cup B,\ x\leq u).
\]
\hyperref[prf:union-preserves-upper-bounds]{Go to proof.}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\bigl[(\forall a\in A)(a\leq u)\land(\forall b\in B)(b\leq u)\bigr]
\Longrightarrow
(\forall x\in A\cup B)(x\leq u).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&U=A\cup B,\\
&\operatorname{UpperBound}(u,A,P)\land\operatorname{UpperBound}(u,B,P)
\Longrightarrow \operatorname{UpperBound}(u,U,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
(\forall a\in A)(a\leq u),\quad
(\forall b\in B)(b\leq u),\quad
(\exists x\in A\cup B)(u<x).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&U=A\cup B,\\
&\operatorname{UpperBound}(u,A,P)\land\operatorname{UpperBound}(u,B,P)\land
\neg\operatorname{UpperBound}(u,U,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
A failure would require a union element larger than the proposed common
upper bound.


  \item[\operatorname{UpperBound}.]
  \[
(\forall a\in A)(a\leq u),\quad
(\forall b\in B)(b\leq u),\quad
(\exists x\in A\cup B)(u<x).
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&U=A\cup B,\\
&\operatorname{UpperBound}(u,A,P)\land\operatorname{UpperBound}(u,B,P)\land
\neg\operatorname{UpperBound}(u,U,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Every $x\in A\cup B$ lies in $A$ or in $B$.  In either case the common
upper-bound hypothesis gives $x\leq u$.
\end{remark*}

\begin{propositionbox}{Proposition (Union Is Bounded Above Exactly When Both Pieces Are)}
\begin{proposition}[Union Is Bounded Above Exactly When Both Pieces Are]
\label{prop:union-bounded-above-iff-pieces-bounded-above}
Let $A,B\subseteq\mathbb{R}$. The union $A\cup B$ is bounded above if and
only if both $A$ and $B$ are bounded above; equivalently,
\[
\left(\exists u\in\mathbb{R}\ \forall x\in A\cup B,\ x\leq u\right)
\Longleftrightarrow
\left[
\left(\exists u_A\in\mathbb{R}\ \forall a\in A,\ a\leq u_A\right)
\land
\left(\exists u_B\in\mathbb{R}\ \forall b\in B,\ b\leq u_B\right)
\right].
\]
\hyperref[prf:union-bounded-above-iff-pieces-bounded-above]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
\bigl[(\exists u\in\mathbb{R})(\forall x\in A\cup B)(x\leq u)\bigr]
\Longleftrightarrow
\bigl[(\exists u_A\in\mathbb{R})(\forall a\in A)(a\leq u_A)\bigr]
\land
\bigl[(\exists u_B\in\mathbb{R})(\forall b\in B)(b\leq u_B)\bigr].
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad U=A\cup B,\\
&\operatorname{BoundedAbove}(U,P)
\Longleftrightarrow
\operatorname{BoundedAbove}(A,P)\land\operatorname{BoundedAbove}(B,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
The equivalence fails precisely if one side holds and the other does not.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad U=A\cup B,\\
&\operatorname{BoundedAbove}(U,P)\oplus
\bigl(\operatorname{BoundedAbove}(A,P)\land\operatorname{BoundedAbove}(B,P)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
The forward direction could fail only if a bound for the union did not bound a
piece.  The reverse direction could fail only if bounds for both pieces did not
combine into a bound for the union.


  \item[\operatorname{BoundedAbove}.]

\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad U=A\cup B,\\
&\operatorname{BoundedAbove}(U,P)\oplus
\bigl(\operatorname{BoundedAbove}(A,P)\land\operatorname{BoundedAbove}(B,P)\bigr).
\end{aligned}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad U=A\cup B,\\
&\operatorname{BoundedAbove}(U,P)
\Longleftrightarrow
\operatorname{BoundedAbove}(A,P)\land\operatorname{BoundedAbove}(B,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Use subset inheritance for the forward direction and
\hyperref[lem:union-preserves-upper-bounds]{Union Preserves Upper Bounds} for
the reverse direction.
\end{remark*}

\begin{corollarybox}{Corollary (Union Is Bounded Exactly When Both Pieces Are)}
\begin{corollary}[Union Is Bounded Exactly When Both Pieces Are]
\label{cor:union-bounded-iff-pieces-bounded}
Let $A,B\subseteq\mathbb{R}$. The union $A\cup B$ is bounded if and only
if both $A$ and $B$ are bounded; that is,
\[
A\cup B\text{ is bounded}
\Longleftrightarrow
\bigl(A\text{ is bounded}\land B\text{ is bounded}\bigr).
\]
\hyperref[prf:union-bounded-iff-pieces-bounded]{Go to proof.}
\end{corollary}
\end{corollarybox}

\[
\operatorname{Bounded}(A\cup B)
\Longleftrightarrow
\operatorname{Bounded}(A)\land\operatorname{Bounded}(B).
\]
\end{remark*}

\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad U=A\cup B,\\
&\operatorname{Bounded}(U,P)
\Longleftrightarrow
\operatorname{Bounded}(A,P)\land\operatorname{Bounded}(B,P).
\end{aligned}
\]
\end{remark*}

\end{aligned}
\]
\end{remark*}

\begin{lemmabox}{Lemma (Subsets Preserve Upper Bounds)}
\begin{lemma}[Subsets Preserve Upper Bounds]
\label{lem:subsets-preserve-upper-bounds}
Let $C\subseteq A\subseteq\mathbb{R}$. Every upper bound of $A$ is an
upper bound of $C$; that is, for every $u\in\mathbb{R}$,
\[
\left(\forall a\in A,\ a\leq u\right)
\Longrightarrow
(\forall c\in C,\ c\leq u).
\]
Consequently, every subset of a set bounded above is bounded above.
\hyperref[prf:subsets-preserve-upper-bounds]{Go to proof.}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
C\subseteq A\land(\forall a\in A)(a\leq u)
\Longrightarrow
(\forall c\in C)(c\leq u).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad C\subseteq A\subseteq\mathbb{R},\\
&\operatorname{UpperBound}(u,A,P)
\Longrightarrow \operatorname{UpperBound}(u,C,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
C\subseteq A,\quad (\forall a\in A)(a\leq u),\quad
(\exists c\in C)(u<c).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad C\subseteq A\subseteq\mathbb{R},\\

\end{aligned}
\]
\end{remark*}

\begin{corollarybox}{Corollary (Intersections Inherit Bounds)}
\begin{corollary}[Intersections Inherit Bounds]
\label{cor:intersections-inherit-bounds}
Let $A,B\subseteq\mathbb{R}$. Every upper bound of $A$ or of $B$ is an
upper bound of $A\cap B$, and every lower bound of $A$ or of $B$ is a
lower bound of $A\cap B$; that is, for all $u,\ell\in\mathbb{R}$,
\[
\left[
(\forall a\in A,\ a\leq u)
\lor
(\forall b\in B,\ b\leq u)
\right]
\Longrightarrow
(\forall x\in A\cap B,\ x\leq u),
\]
and
\[
\left[
(\forall a\in A,\ \ell\leq a)
\lor
(\forall b\in B,\ \ell\leq b)
\right]
\Longrightarrow
(\forall x\in A\cap B,\ \ell\leq x).
\]
\hyperref[prf:intersections-inherit-bounds]{Go to proof.}
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\bigl[(\forall a\in A)(a\leq u)\lor(\forall b\in B)(b\leq u)\bigr]
\Longrightarrow(\forall x\in A\cap B)(x\leq u),\\
&\bigl[(\forall a\in A)(\ell\leq a)\lor(\forall b\in B)(\ell\leq b)\bigr]
\Longrightarrow(\forall x\in A\cap B)(\ell\leq x).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad C=A\cap B,\\
&\operatorname{UpperBound}(u,A,P)\Longrightarrow\operatorname{UpperBound}(u,C,P),\\
&\operatorname{UpperBound}(u,B,P)\Longrightarrow\operatorname{UpperBound}(u,C,P),\\
&\operatorname{LowerBound}(\ell,A,P)\Longrightarrow\operatorname{LowerBound}(\ell,C,P),\\
&\operatorname{LowerBound}(\ell,B,P)\Longrightarrow\operatorname{LowerBound}(\ell,C,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
A failure would be a failure of subset inheritance for $A\cap B\subseteq A$.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad C=A\cap B,\\
&\operatorname{UpperBound}(u,A,P)\land
\neg\operatorname{UpperBound}(u,C,P)
\end{aligned}
\]
is contradictory, and the same holds for lower bounds.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Intersection can lose elements, but it cannot introduce an element outside the
parent set $A$.


  \item[\operatorname{UpperBound} / \operatorname{LowerBound}.]

\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad C=A\cap B,\\
&\operatorname{UpperBound}(u,A,P)\land
\neg\operatorname{UpperBound}(u,C,P)
\end{aligned}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad C=A\cap B,\\
&\operatorname{UpperBound}(u,A,P)\Longrightarrow\operatorname{UpperBound}(u,C,P),\\
&\operatorname{LowerBound}(\ell,A,P)\Longrightarrow\operatorname{LowerBound}(\ell,C,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Apply subset inheritance to $C=A\cap B$.
\end{remark*}

\begin{corollarybox}{Corollary (Differences Inherit Bounds)}
\begin{corollary}[Differences Inherit Bounds]
\label{cor:differences-inherit-bounds}
Let $A,B\subseteq\mathbb{R}$. Every upper bound of $A$ is an upper bound
of $A\setminus B$, and every lower bound of $A$ is a lower bound of
$A\setminus B$; that is, for all $u,\ell\in\mathbb{R}$,
\[
\left(\forall a\in A,\ a\leq u\right)
\Longrightarrow
(\forall x\in A\setminus B,\ x\leq u),
\]
and
\[
\left(\forall a\in A,\ \ell\leq a\right)
\Longrightarrow
(\forall x\in A\setminus B,\ \ell\leq x).
\]
\hyperref[prf:differences-inherit-bounds]{Go to proof.}
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&(\forall a\in A)(a\leq u)
\Longrightarrow(\forall x\in A\setminus B)(x\leq u),\\
&(\forall a\in A)(\ell\leq a)
\Longrightarrow(\forall x\in A\setminus B)(\ell\leq x).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad D=A\setminus B,\\
&\operatorname{UpperBound}(u,A,P)\Longrightarrow\operatorname{UpperBound}(u,D,P),\\
&\operatorname{LowerBound}(\ell,A,P)\Longrightarrow\operatorname{LowerBound}(\ell,D,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
A failure would be a failure of subset inheritance for
$A\setminus B\subseteq A$.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad D=A\setminus B,\\
&\operatorname{UpperBound}(u,A,P)\land
\neg\operatorname{UpperBound}(u,D,P)
\end{aligned}
\]
is contradictory, and the same holds for lower bounds.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Difference can remove elements from $A$, but it cannot create a new element
outside $A$.


  \item[\operatorname{UpperBound} / \operatorname{LowerBound}.]

\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad D=A\setminus B,\\
&\operatorname{UpperBound}(u,A,P)\land
\neg\operatorname{UpperBound}(u,D,P)
\end{aligned}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad D=A\setminus B,\\
&\operatorname{UpperBound}(u,A,P)\Longrightarrow\operatorname{UpperBound}(u,D,P),\\
&\operatorname{LowerBound}(\ell,A,P)\Longrightarrow\operatorname{LowerBound}(\ell,D,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Apply subset inheritance to $C=A\setminus B$.
\end{remark*}

\begin{corollarybox}{Corollary (Complements Inherit Ambient Bounds)}
\begin{corollary}[Complements Inherit Ambient Bounds]
\label{cor:complements-inherit-ambient-bounds}
Let $A\subseteq S\subseteq\mathbb{R}$. Every upper bound of $S$ is an
upper bound of the relative complement $S\setminus A$, and every lower
bound of $S$ is a lower bound of $S\setminus A$; that is, for all
$u,\ell\in\mathbb{R}$,
\[
\left(\forall s\in S,\ s\leq u\right)
\Longrightarrow
(\forall x\in S\setminus A,\ x\leq u),
\]
and
\[
\left(\forall s\in S,\ \ell\leq s\right)
\Longrightarrow
(\forall x\in S\setminus A,\ \ell\leq x).
\]
\hyperref[prf:complements-inherit-ambient-bounds]{Go to proof.}
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&(\forall s\in S)(s\leq u)
\Longrightarrow(\forall x\in S\setminus A)(x\leq u),\\
&(\forall s\in S)(\ell\leq s)
\Longrightarrow(\forall x\in S\setminus A)(\ell\leq x).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq S\subseteq\mathbb{R},\qquad C=S\setminus A,\\
&\operatorname{UpperBound}(u,S,P)\Longrightarrow\operatorname{UpperBound}(u,C,P),\\
&\operatorname{LowerBound}(\ell,S,P)\Longrightarrow\operatorname{LowerBound}(\ell,C,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\operatorname{UpperBound}(u,S)\land\neg\operatorname{UpperBound}(u,S\setminus A)
\]
or
\[
\operatorname{LowerBound}(\ell,S)\land\neg\operatorname{LowerBound}(\ell,S\setminus A).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq S\subseteq\mathbb{R},\qquad C=S\setminus A,\\
&\bigl(\operatorname{UpperBound}(u,S,P)\land\neg\operatorname{UpperBound}(u,C,P)\bigr)\\
&\qquad{}\lor
\bigl(\operatorname{LowerBound}(\ell,S,P)\land\neg\operatorname{LowerBound}(\ell,C,P)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
The complement does not inherit bounds from $A$; it inherits them from the
ambient set $S$.


  \item[\operatorname{UpperBound} / \operatorname{LowerBound}.]
  \[
\operatorname{UpperBound}(u,S)\land\neg\operatorname{UpperBound}(u,S\setminus A)
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq S\subseteq\mathbb{R},\qquad C=S\setminus A,\\
&\bigl(\operatorname{UpperBound}(u,S,P)\land\neg\operatorname{UpperBound}(u,C,P)\bigr)\\
&\qquad{}\lor
\bigl(\operatorname{LowerBound}(\ell,S,P)\land\neg\operatorname{LowerBound}(\ell,C,P)\bigr).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Every point of $S\setminus A$ is a point of $S$, so every ambient bound of
$S$ applies to the complement.
\end{remark*}
