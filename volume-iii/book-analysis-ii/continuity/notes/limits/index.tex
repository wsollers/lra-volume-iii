\section{Limits}

\begin{toolkitbox}{One-Sided Limits --- Quick Reference}
\small
\begin{tabular}{l l}
\toprule
\textbf{Name} & \textbf{Meaning}\\
\midrule
\hyperref[def:right-hand-limit]{Right-hand limit} & Approach the point from inputs larger than $c$.\\
\hyperref[def:left-hand-limit]{Left-hand limit} & Approach the point from inputs smaller than $c$.\\
\hyperref[thm:sequential-criterion-right-hand-limit]{Right sequential criterion} & Test right limits by right-approaching sequences.\\
\hyperref[cor:sequential-criterion-left-hand-limit]{Left sequential criterion} & Test left limits by left-approaching sequences.\\
\hyperref[thm:two-sided-limit-iff-matching-one-sided-limits]{Matching criterion} & Two-sided limits exist exactly when side limits agree.\\
\bottomrule
\end{tabular}
\end{toolkitbox}


\begin{toolkitbox}{Limits of Functions --- Quick Reference}
\small
\begin{tabular}{l l}
\toprule
\textbf{Name} & \textbf{Meaning}\\
\midrule
\hyperref[def:limit-function]{Function limit} & Punctured input control near a cluster point.\\
\hyperref[def:limit-function-sequential]{Sequential limit} & All approaching sequences have the same output limit.\\
\hyperref[def:limit-function-neighbourhood]{Neighbourhood limit} & Neighbourhood formulation of the same condition.\\
\hyperref[thm:limit-unique]{Uniqueness} & A function has at most one limit at a point.\\
\hyperref[prop:sequential-criterion-limits]{Sequential criterion} & Sequence convergence is the main working test.\\
\hyperref[thm:limit-implies-local-bounding]{Local boundedness} & Finite limits bound the function near the point.\\
\hyperref[thm:squeeze-function-limits]{Squeeze theorem} & Ordered comparison transfers limits.\\
\hyperref[thm:limits-are-local]{Locality} & Only behaviour near the approach point matters.\\
\bottomrule
\end{tabular}
\end{toolkitbox}
