\begin{definitionbox}{Definition (Epsilon Characterization of Supremum)}
\begin{definition}[Epsilon Characterization of Supremum]
\label{def:epsilon-characterization-of-supremum}
Let $A\subseteq\mathbb{R}$ and let $s\in\mathbb{R}$. The element $s$ satisfies the \emph{epsilon characterization of the supremum} of $A$ if and only if $s$ is an upper bound of $A$ and for every $\varepsilon>0$ there exists $a\in A$ such that $s-\varepsilon<a$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } A\subseteq\mathbb{R} \text{ and } s\in\mathbb{R}.\\
&s \text{ satisfies the epsilon characterization of } \sup A\\
&\Longleftrightarrow
\Bigl[\forall x\in A\;(x\le s)\Bigr]
\land
\Bigl[\forall \varepsilon>0\;\exists a\in A\;(s-\varepsilon<a)\Bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{EpsilonCharacterizationOfSupremum}(s,A,\mathbb{R})
\Longleftrightarrow
\operatorname{LeastUpperBound}(s,A,\mathbb{R})
\Longleftrightarrow
\operatorname{UpperBound}(s,A)
\land
\forall \varepsilon>0\;\exists a\in A\;(s-\varepsilon<a).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let } A\subseteq\mathbb{R} \text{ and } s\in\mathbb{R}.\\
&s \text{ does not satisfy the epsilon characterization of } \sup A\\
&\Longleftrightarrow
\Bigl[\exists x\in A\;(s<x)\Bigr]
\lor
\Bigl[\exists \varepsilon_0>0\;\forall a\in A\;(a\le s-\varepsilon_0)\Bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{EpsilonCharacterizationOfSupremum}(s,A,\mathbb{R})\Longleftrightarrow
\Bigl[\exists x\in A\;(s<x)\Bigr]
\lor
\Bigl[\exists \varepsilon_0>0\;\forall a\in A\;(a\le s-\varepsilon_0)\Bigr].
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
The characterization fails either because $s$ is not an upper bound of $A$, or because $A$ remains at least some positive distance below $s$.
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
\neg\operatorname{EpsilonCharacterizationOfSupremum}(s,A,\mathbb{R})
=
\underbrace{\exists x\in A\;(s<x)}_{\text{upper-bound failure}}
\;\lor\;
\underbrace{\exists \varepsilon_0>0\;\forall a\in A\;(a\le s-\varepsilon_0)}_{\text{positive gap below }s}.
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
This condition says that $s$ caps the set from above while points of $A$ approach $s$ from below at every positive scale. The upper-bound clause prevents $A$ from crossing above $s$, and the epsilon clause prevents $s$ from floating strictly above the set. Geometrically, $s$ is a ceiling that the set presses against arbitrarily closely.
\end{remark*}

\begin{definitionbox}{Definition (Epsilon Characterization of Infimum)}
\begin{definition}[Epsilon Characterization of Infimum]
\label{def:epsilon-characterization-of-infimum}
Let $A\subseteq\mathbb{R}$ and let $t\in\mathbb{R}$. The element $t$ satisfies the \emph{epsilon characterization of the infimum} of $A$ if and only if $t$ is a lower bound of $A$ and for every $\varepsilon>0$ there exists $a\in A$ such that $a<t+\varepsilon$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } A\subseteq\mathbb{R} \text{ and } t\in\mathbb{R}.\\
&t \text{ satisfies the epsilon characterization of } \inf A\\
&\Longleftrightarrow
\Bigl[\forall x\in A\;(t\le x)\Bigr]
\land
\Bigl[\forall \varepsilon>0\;\exists a\in A\;(a<t+\varepsilon)\Bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\operatorname{EpsilonCharacterizationOfInfimum}(t,A,\mathbb{R})
\Longleftrightarrow
\operatorname{GreatestLowerBound}(t,A,\mathbb{R})
\Longleftrightarrow
\operatorname{LowerBound}(t,A)
\land
\forall \varepsilon>0\;\exists a\in A\;(a<t+\varepsilon).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let } A\subseteq\mathbb{R} \text{ and } t\in\mathbb{R}.\\
&t \text{ does not satisfy the epsilon characterization of } \inf A\\
&\Longleftrightarrow
\Bigl[\exists x\in A\;(x<t)\Bigr]
\lor
\Bigl[\exists \varepsilon_0>0\;\forall a\in A\;(t+\varepsilon_0\le a)\Bigr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg\operatorname{EpsilonCharacterizationOfInfimum}(t,A,\mathbb{R})\Longleftrightarrow
\Bigl[\exists x\in A\;(x<t)\Bigr]
\lor
\Bigl[\exists \varepsilon_0>0\;\forall a\in A\;(t+\varepsilon_0\le a)\Bigr].
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
The characterization fails either because $t$ is not a lower bound of $A$, or because $A$ remains at least some positive distance above $t$.
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
\[
\neg\operatorname{EpsilonCharacterizationOfInfimum}(t,A,\mathbb{R})
=
\underbrace{\exists x\in A\;(x<t)}_{\text{lower-bound failure}}
\;\lor\;
\underbrace{\exists \varepsilon_0>0\;\forall a\in A\;(t+\varepsilon_0\le a)}_{\text{positive gap above }t}.
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
This condition says that $t$ sits below the set while points of $A$ approach $t$ from above at every positive scale. The lower-bound clause prevents $A$ from crossing below $t$, and the epsilon clause prevents $t$ from sitting strictly below the set with a positive gap. Geometrically, $t$ is a floor that the set presses against arbitrarily closely from above.
\end{remark*}

\begin{remark*}[Interpretation]
The epsilon characterizations are one way to prove the sharpness part of a
supremum or infimum claim.  After proving the candidate is an upper or lower
bound, one can either compare it with arbitrary bounds directly or build
near-extremal points using an arbitrary $\varepsilon>0$.  The proof-techniques
chapter gives a fuller guide to choosing between these approaches:
\hyperref[subsec:eps-char]{The $\varepsilon$-Characterisation of Suprema and
Infima}.
\end{remark*}

\begin{lemmabox}{Lemma (Supremum Strict Upper Approximation)}
\begin{lemma}[Supremum Strict Upper Approximation]
\label{lem:supremum-strict-upper-approximation}
Let $A\subseteq\mathbb{R}$ be nonempty and bounded above, and let
$s=\sup A$. If $u<s$, then there exists $a\in A$ such that
\[
u<a\le s.
\]
\hyperref[prf:supremum-strict-upper-approximation]{Go to proof.}
\end{lemma}
\end{lemmabox}
\begin{remark*}[Standard quantified statement]
\[
\forall A\subseteq\mathbb{R}\;\forall u,s\in\mathbb{R}\;
\bigl((A\ne\varnothing\land s=\sup A\land u<s)\Rightarrow
\exists a\in A\;(u<a\le s)\bigr).
\]
\end{remark*}
\begin{remark*}[Predicate reading]
\[
u<\sup A\Rightarrow A\cap(u,\sup A]\ne\varnothing.
\]
\end{remark*}
\begin{remark*}[Negated quantified statement]
\[
\exists A\subseteq\mathbb{R}\;\exists u,s\in\mathbb{R}\;
\bigl(A\ne\varnothing\land s=\sup A\land u<s\land
\forall a\in A\;(a\le u\lor s<a)\bigr).
\]
\end{remark*}
\begin{remark*}[Negation predicate reading]
\[
u<\sup A\land A\cap(u,\sup A]=\varnothing.
\]
\end{remark*}
\begin{remark*}[Interpretation]
Every number strictly below a supremum is crossed by some element of the set.
\end{remark*}

\begin{lemmabox}{Lemma (Infimum Strict Lower Approximation)}
\begin{lemma}[Infimum Strict Lower Approximation]
\label{lem:infimum-strict-lower-approximation}
Let $A\subseteq\mathbb{R}$ be nonempty and bounded below, and let
$t=\inf A$. If $t<v$, then there exists $a\in A$ such that
\[
t\le a<v.
\]
\hyperref[prf:infimum-strict-lower-approximation]{Go to proof.}
\end{lemma}
\end{lemmabox}
\begin{remark*}[Standard quantified statement]
\[
\forall A\subseteq\mathbb{R}\;\forall t,v\in\mathbb{R}\;
\bigl((A\ne\varnothing\land t=\inf A\land t<v)\Rightarrow
\exists a\in A\;(t\le a<v)\bigr).
\]
\end{remark*}
\begin{remark*}[Predicate reading]
\[
\inf A<v\Rightarrow A\cap[\inf A,v)\ne\varnothing.
\]
\end{remark*}
\begin{remark*}[Negated quantified statement]
\[
\exists A\subseteq\mathbb{R}\;\exists t,v\in\mathbb{R}\;
\bigl(A\ne\varnothing\land t=\inf A\land t<v\land
\forall a\in A\;(a<t\lor v\le a)\bigr).
\]
\end{remark*}
\begin{remark*}[Negation predicate reading]
\[
\inf A<v\land A\cap[\inf A,v)=\varnothing.
\]
\end{remark*}
\begin{remark*}[Interpretation]
Every number strictly above an infimum is reached from below by some element
of the set.
\end{remark*}
