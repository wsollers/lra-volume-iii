\begin{proposition*}[Bounded Factor Rule for Little-o]
If $f(x)=o(g(x))$ as $x\to a$ and $m$ is bounded near $a$, then
\[
  m(x)f(x)=o(g(x))\qquad (x\to a).
\]
\end{proposition*}

\begin{proof}[Professional Standard Proof]
\LRAProofBodyStart
Let $\varepsilon>0$. Since $m$ is bounded near $a$, choose $M>0$ and a
punctured neighbourhood on which $|m(x)|\leq M$. Since $f=o(g)$, choose a
smaller punctured neighbourhood on which
$|f(x)|\leq(\varepsilon/M)|g(x)|$. Then
\[
  |m(x)f(x)|\leq M|f(x)|\leq \varepsilon |g(x)|.
\]
Thus $mf=o(g)$.
\end{proof}

\begin{proof}[Detailed Learning Proof]
\LRAProofBodyStart
The bounded factor can cost only a fixed multiplicative constant. If
$|m(x)|\leq M$ near $a$, ask the little-o estimate for $f$ to be
$\varepsilon/M$ instead of $\varepsilon$. Multiplying the two estimates gives
the final tolerance $\varepsilon$.
\end{proof}
