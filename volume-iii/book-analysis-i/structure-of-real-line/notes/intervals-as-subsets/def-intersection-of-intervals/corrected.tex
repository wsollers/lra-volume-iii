\begin{definitionbox}{Definition (Intersection of Intervals)}
\begin{definition}[Intersection of Intervals]
\label{def:intersection-of-intervals}
Let $\{I_\lambda\}_{\lambda\in\Lambda}$ be a family of intervals in
$\mathbb{R}$.  Their \emph{intersection} is the set
\[
  \bigcap_{\lambda\in\Lambda}I_\lambda
  :=
  \{x\in\mathbb{R}:(\forall\lambda\in\Lambda)(x\in I_\lambda)\}.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
I\cap J=\{x\in\mathbb{R}:x\in I\land x\in J\}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The intersection keeps exactly the points common to both intervals. On the
real line this common part is either empty or another interval.
\end{remark*}
