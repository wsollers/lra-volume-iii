\begin{theorem*}[Uniqueness of the Supremum]
If a set $A \subseteq \mathbb{R}$ has a supremum, then that supremum is unique.
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
Let $s_1 = \sup A$ and $s_2 = \sup A$. Since $s_1$ is a supremum, it is an upper bound of $A$, and because $s_2$ is the least upper bound, $s_2 \le s_1$. Conversely, $s_2$ is an upper bound of $A$, and because $s_1$ is the least upper bound, $s_1 \le s_2$. By the antisymmetry of the standard order on $\mathbb{R}$, $s_1 \le s_2$ and $s_2 \le s_1$ implies $s_1 = s_2$.
\end{proof}

\begin{proof}
\textbf{Detailed Learning Proof.}~\\

\textbf{Step 1.} Define two potential candidates for the supremum.
Let $A \subseteq \mathbb{R}$ be a non-empty subset with $s_1 = \sup A$ and suppose there exists a second supremum $s_2 = \sup A$.

\textbf{Step 2.} Apply the "Least" property to the first candidate.
By the definition of supremum, $s_2$ is an upper bound of $A$. Since $s_1$ is the \textit{least} upper bound, it must satisfy $s_1 \le s_2$.

\textbf{Step 3.} Apply the "Least" property to the second candidate.
Similarly, $s_1$ is an upper bound of $A$. Since $s_2$ is also a \textit{least} upper bound, it must satisfy $s_2 \le s_1$.

\textbf{Step 4.} Conclude using antisymmetry.
In the real number system, the relation $\le$ is antisymmetric. Because $s_1 \le s_2$ and $s_2 \le s_1$, the two values must be identical. Thus, $s_1 = s_2$.
\end{proof}

\begin{remark*}[Proof structure]
Direct proof utilizing the minimality of the supremum against itself to force equality via antisymmetry.
\end{remark*}
