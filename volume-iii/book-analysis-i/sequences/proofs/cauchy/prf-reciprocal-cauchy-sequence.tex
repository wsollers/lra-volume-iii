\begin{theorem*}[Reciprocals of Cauchy Sequences]
Let $(x_n)$ be a Cauchy real sequence. If there exist $c>0$ and
$N_0\in\mathbb{N}$ such that $|x_n|\ge c$ for every $n\ge N_0$, then
$(1/x_n)$ is Cauchy.

\smallskip
\noindent
\end{theorem*}
