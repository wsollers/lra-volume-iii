\begin{theorem}[Supremum of the Pairwise Minimum Set]
\label{thm:supremum-pairwise-minimum-set}
Let $A,B\subseteq\mathbb{R}$ be nonempty and bounded above. Then $A\wedge B$
is bounded above and
\[
\sup(A\wedge B)=\min\{\sup A,\sup B\}.
\]
\hyperref[prf:supremum-pairwise-minimum-set]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
A,B\neq\varnothing,\quad A,B\text{ bounded above}
\quad\Longrightarrow\quad
\sup(A\wedge B)=\min\{\sup A,\sup B\}.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&N=A\wedge B,\qquad U=\min\{s_A,s_B\},\\
&\operatorname{LeastUpperBound}(s_A,A,P)\land\operatorname{LeastUpperBound}(s_B,B,P)
\Longrightarrow
\operatorname{LeastUpperBound}(U,N,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\sup(A\wedge B)\neq\min\{\sup A,\sup B\}
\]
is impossible under the stated hypotheses.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&N=A\wedge B,\qquad U=\min\{s_A,s_B\},\\
&\operatorname{LeastUpperBound}(s_A,A,P)\land\operatorname{LeastUpperBound}(s_B,B,P)\land
\neg\operatorname{LeastUpperBound}(U,N,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
The formula could fail only if the smaller supremum failed to bound all
pairwise minima, or if that upper bound were not sharp.


  \item[\operatorname{LeastUpperBound}.]
  \[
\sup(A\wedge B)\neq\min\{\sup A,\sup B\}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&N=A\wedge B,\qquad U=\min\{s_A,s_B\},\\
&\operatorname{LeastUpperBound}(s_A,A,P)\land\operatorname{LeastUpperBound}(s_B,B,P)\land
\neg\operatorname{LeastUpperBound}(U,N,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Every pairwise minimum is below both suprema, hence below their minimum.
Sharpness comes from taking both inputs near their own suprema.


  \item[\operatorname{LeastUpperBound}.]
  \[
\sup(A\wedge B)\neq\min\{\sup A,\sup B\}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&N=A\wedge B,\qquad U=\min\{s_A,s_B\},\\
&\operatorname{LeastUpperBound}(s_A,A,P)\land\operatorname{LeastUpperBound}(s_B,B,P)\land
\neg\operatorname{LeastUpperBound}(U,N,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
For pairwise minima, the upper edge is limited by the smaller of the two
individual upper edges.
\end{remark*}
