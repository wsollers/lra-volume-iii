\begin{theorem*}[Derivative of the Identity]
Let $I \subseteq \mathbb{R}$ be an open interval, and let $\operatorname{id}_I : I \to \mathbb{R}$ be the identity function defined by $\operatorname{id}_I(x) := x$. Then $\operatorname{id}_I$ is differentiable at every $c \in I$, and $\operatorname{id}_I'(c) = 1$.
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
For any $c \in I$ and $x \in I$ with $x \neq c$, the difference quotient of $\operatorname{id}_I$ is
\[
\frac{\operatorname{id}_I(x) - \operatorname{id}_I(c)}{x - c} = \frac{x - c}{x - c} = 1.
\]
Since the difference quotient is identically $1$ on $I \setminus \{c\}$, its limit as $x \to c$ is $1$. Therefore $\operatorname{id}_I'(c) = 1$.
\end{proof}

\begin{proof}
\textbf{Detailed Learning Proof.}~\\

\textbf{Step 1.} Write out the difference quotient. Fix $c \in I$. For any $x \in I$ with $x \neq c$, substitute $\operatorname{id}_I(x) = x$ and $\operatorname{id}_I(c) = c$:
\[
\frac{\operatorname{id}_I(x) - \operatorname{id}_I(c)}{x - c} = \frac{x - c}{x - c}.
\]

\textbf{Step 2.} Simplify the difference quotient. Since $x \neq c$, the denominator $x - c \neq 0$, so cancellation is valid:
\[
\frac{x - c}{x - c} = 1 \qquad \text{for all } x \in I \setminus \{c\}.
\]
The difference quotient is the constant function $1$ on the entire punctured domain $I \setminus \{c\}$.

\textbf{Step 3.} Evaluate the limit. A constant function has that constant as its limit at every point. Therefore
\[
\operatorname{id}_I'(c) = \lim_{x \to c} \frac{\operatorname{id}_I(x) - \operatorname{id}_I(c)}{x - c} = \lim_{x \to c} 1 = 1.
\]

\textbf{Step 4.} Conclude. Since $c \in I$ was arbitrary, $\operatorname{id}_I$ is differentiable at every point of $I$ with derivative $1$ throughout.
\end{proof}

\begin{remark*}[Proof structure]
This is a direct computation from the definition of the derivative. The difference quotient of the identity function simplifies to $1$ identically, making the limit trivial. No limit law beyond the constant function limit is required.
\end{remark*}
