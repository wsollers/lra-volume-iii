\begin{definitionbox}{Bound}
\begin{definition}[Bound]
\label{def:bound}
Let $(S,\leq)$ be an ordered set, let $A\subseteq S$ be nonempty, and let
$b\in S$.  The element $b$ is a \emph{bound} of $A$ if and only if it
controls every element of $A$ from at least one side: either $x\leq b$
for every $x\in A$, or $b\leq x$ for every $x\in A$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  b\text{ is a bound of }A
  \quad\Longleftrightarrow\quad
  \bigl[(\forall x\in A)(x\leq b)
  \ \text{or}\ 
  (\forall x\in A)(b\leq x)\bigr].
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  P=\mathsf{OrderedSet}(S,\leq),\qquad
  \operatorname{UpperBound}(b,A,P)
  \lor
  \operatorname{LowerBound}(b,A,P).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  (\exists x\in A)(x\nleq b)
  \quad\text{and}\quad
  (\exists y\in A)(b\nleq y).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  P=\mathsf{OrderedSet}(S,\leq),\qquad
  \neg\operatorname{UpperBound}(b,A,P)
  \land
  \neg\operatorname{LowerBound}(b,A,P).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
  Under the standing nonemptiness assumption, a candidate fails to be a
  bound exactly when it fails both one-sided comparison conditions.  Empty-set
  input is instead a failure of applicability.

  \item[Applicability failure.]
  The definition is stated only for nonempty subsets.
  \[
    A=\varnothing.
  \]
  \[
    \operatorname{IsEmpty}(A).
  \]

  \item[Failure of the bound condition.]
  There is an element of $A$ that is not below $b$, and there is an element
  of $A$ that is not above $b$.
  \[
    \begin{aligned}
      &(\exists x\in A)(x\nleq b),\\
      &(\exists y\in A)(b\nleq y).
    \end{aligned}
  \]
  \[
    P=\mathsf{OrderedSet}(S,\leq),\qquad
    \neg\operatorname{UpperBound}(b,A,P)
    \land
    \neg\operatorname{LowerBound}(b,A,P).
  \]
  The non-comparisons are not rewritten as strict inequalities without an
  additional total-order law.
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
A bound controls the set from at least one side.  It need not be sharp and it
need not belong to the set being bounded.
\end{remark*}
