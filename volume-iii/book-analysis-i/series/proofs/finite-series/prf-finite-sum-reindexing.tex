\begin{lemma*}[Reindexing a Finite Sum]
Let \(m \le n\) be integers, let \(k\) be an integer, and let \(a_i\)
be a real number assigned to each integer \(i\) satisfying \(m \le i
\le n\). Then
\[
  \sum_{i=m}^{n} a_i
  =
  \sum_{j=m+k}^{n+k} a_{j-k}.
\]
\end{lemma*}
