\subsection*{Density}




\begin{definitionbox}{Definition (Dense Subset)}
\begin{definition}[Dense Subset]
\label{def:dense-subset}
Let $D\subseteq S\subseteq\mathbb{R}$. The set $D$ is \emph{dense in} $S$ if and only if for every $x\in S$ and every $\varepsilon>0$ there exists $d\in D$ such that $|x-d|<\varepsilon$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let }D\subseteq S\subseteq\mathbb{R}.\\
&D\text{ is dense in }S
\Longleftrightarrow
\forall x\in S\;\forall \varepsilon>0\;\exists d\in D\;\bigl(|x-d|<\varepsilon\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
D\subseteq S\subseteq\mathbb{R},\qquad
\operatorname{IsDense}(D,S)\coloneqq
\forall x\in S\;\forall \varepsilon>0\;\exists d\in D\;\bigl(|x-d|<\varepsilon\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let }D\subseteq S\subseteq\mathbb{R}.\\
&D\text{ is not dense in }S
\Longleftrightarrow
\exists x\in S\;\exists \varepsilon_0>0\;\forall d\in D\;\bigl(|x-d|\ge \varepsilon_0\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
D\subseteq S\subseteq\mathbb{R},\qquad
\neg\operatorname{IsDense}(D,S)\Longleftrightarrow
\exists x\in S\;\exists \varepsilon_0>0\;\forall d\in D\;\bigl(|x-d|\ge \varepsilon_0\bigr).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Density fails when some point of $S$ has a positive-radius neighborhood that contains no point of $D$.
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
\[
\neg\operatorname{IsDense}(D,S)
=
\underbrace{\exists x\in S}_{\text{point not approximated}}
\;\underbrace{\exists \varepsilon_0>0}_{\text{positive exclusion radius}}
\;\underbrace{\forall d\in D\;\bigl(|x-d|\ge \varepsilon_0\bigr)}_{\text{no point of }D\text{ enters the neighborhood}}.
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Density is an approximation property inside $S$. It does not say that every point of $S$ already belongs to $D$; it says that points of $D$ occur arbitrarily close to every point of $S$. Thus $D$ can be a proper subset of $S$ while still leaving no positive-width gap inside $S$.
\end{remark*}

\begin{definitionbox}{Definition (Order Dense Subset)}
\begin{definition}[Order Dense Subset]
\label{def:order-dense-subset}
Let $X$ be a linearly ordered set and let $A\subseteq X$. The subset $A$ is \emph{order dense in} $X$ if and only if for every $a,b\in X$ with $a<b$ there exists $c\in A$ such that $a<c<b$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let }X\text{ be a linearly ordered set and }A\subseteq X.\\
&A\text{ is order dense in }X
\Longleftrightarrow
\forall a,b\in X\;\bigl(a<b\rightarrow \exists c\in A\;(a<c<b)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
X\in\mathbf{Set},\qquad A\subseteq X,\qquad P_X=\mathsf{OrderedSet}(X,\leq),\qquad
D_A=\mathsf{OrderDenseSet}(A,P_X),\qquad
\operatorname{IsOrderDenseSubset}(A,P_X)\coloneqq
\forall a,b\in X\;\bigl(a<b\rightarrow \exists c\in A\;(a<c<b)\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let }X\text{ be a linearly ordered set and }A\subseteq X.\\
&A\text{ is not order dense in }X
\Longleftrightarrow
\exists a,b\in X\;\bigl(a<b\land \forall c\in A\;(c\le a\lor b\le c)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
X\in\mathbf{Set},\qquad A\subseteq X,\qquad P_X=\mathsf{OrderedSet}(X,\leq),\qquad
D_A=\mathsf{OrderDenseSet}(A,P_X),\qquad
\neg\operatorname{IsOrderDenseSubset}(A,P_X)\Longleftrightarrow
\exists a,b\in X\;\bigl(a<b\land \forall c\in A\;(c\le a\lor b\le c)\bigr).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Order density fails when there is a nonempty open interval of $X$ that contains no point of $A$.
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
\[
\neg\operatorname{IsOrderDenseSubset}(A,X)
=
\underbrace{\exists a,b\in X\;(a<b)}_{\text{nonempty interval}}
\;\land\;
\underbrace{\forall c\in A\;(c\le a\lor b\le c)}_{\text{no point of }A\text{ lies between }a\text{ and }b}.
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Order density is an interval-penetration property. It ignores metric distance and asks only whether every ordered gap contains a point of the subset. A subset can therefore be order dense in $X$ precisely when no nonempty open interval of $X$ is missed entirely by that subset.
\end{remark*}

\begin{theorembox}{Theorem (Density of the Rationals)}
\begin{theorem}[Density of the Rationals]
\label{thm:density-of-rationals-in-reals}
For every $a,b\in\mathbb{R}$ with $a<b$, there exists $q\in\mathbb{Q}$ such that $a<q<b$.
\hyperref[prf:density-of-rationals-in-reals]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{R}\;\bigl(a<b\rightarrow \exists q\in\mathbb{Q}\;(a<q<b)\bigr).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad \mathbb{Q}\subseteq\mathbb{R},\qquad
D_{\mathbb{Q}}=\mathsf{OrderDenseSet}(\mathbb{Q},P_{\mathbb{R}}),\qquad
\operatorname{IsOrderDenseSubset}(\mathbb{Q},P_{\mathbb{R}})\Longleftrightarrow
\forall a,b\in\mathbb{R}\;\bigl(a<b\rightarrow \exists q\in\mathbb{Q}\;(a<q<b)\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists a,b\in\mathbb{R}\;\bigl(a<b\land \forall q\in\mathbb{Q}\;(q\le a\lor b\le q)\bigr).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad \mathbb{Q}\subseteq\mathbb{R},\qquad
D_{\mathbb{Q}}=\mathsf{OrderDenseSet}(\mathbb{Q},P_{\mathbb{R}}),\qquad
\neg\operatorname{IsOrderDenseSubset}(\mathbb{Q},P_{\mathbb{R}})\Longleftrightarrow
\exists a,b\in\mathbb{R}\;\bigl(a<b\land \forall q\in\mathbb{Q}\;(q\le a\lor b\le q)\bigr).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
The theorem would fail precisely if some nonempty real interval contained no rational number.
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
\[
\neg\operatorname{IsOrderDenseSubset}(\mathbb{Q},\mathbb{R})
=
\underbrace{\exists a,b\in\mathbb{R}\;(a<b)}_{\text{nonempty real interval}}
\;\land\;
\underbrace{\forall q\in\mathbb{Q}\;(q\le a\lor b\le q)}_{\text{no rational lies inside}}.
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Rational numbers penetrate every real interval. The theorem does not say that every real number is rational; it says that rational numbers are never separated from a real interval by a positive order gap. This is the interval form of rational density in the real line.
\end{remark*}

\begin{lemma}[Rational Translation Preserves Rationality]
\label{lem:rational-translation-preserves-rationality}
If $r,q\in\mathbb{Q}$, then
\[
r+q\in\mathbb{Q}.
\]
\hyperref[prf:rational-translation-preserves-rationality]{\textit{Go to proof.}}
\end{lemma}

\begin{remark*}[Standard quantified statement]
\[
\forall r,q\in\mathbb{Q}\;(r+q\in\mathbb{Q}).

\end{remark*}

r\in\mathbb{Q}\land q\in\mathbb{Q}\Longrightarrow r+q\in\mathbb{Q}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The rational numbers are closed under addition, so translating a rational by a
rational keeps it rational.

\begin{corollary}[No Adjacent Real Numbers]

\begin{corollarybox}{Corollary (Rational and Irrational Points in Every Open Interval)}
\begin{corollary}[Rational and Irrational Points in Every Open Interval]
\label{cor:every-open-interval-contains-rational-and-irrational}
Every open interval in $\mathbb{R}$ contains both a rational number and an irrational number.
\hyperref[prf:every-open-interval-contains-rational-and-irrational]{Go to proof.}
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{R}\;
\Bigl[
a<b\rightarrow
\bigl(\exists q\in\mathbb{Q}\;(a<q<b)\land \exists s\in\mathbb{R}\setminus\mathbb{Q}\;(a<s<b)\bigr)
\Bigr].
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad I=\mathbb{R}\setminus\mathbb{Q},\qquad
D_{\mathbb{Q}}=\mathsf{OrderDenseSet}(\mathbb{Q},P_{\mathbb{R}}),\qquad
D_I=\mathsf{OrderDenseSet}(I,P_{\mathbb{R}}),\qquad
\operatorname{IsOrderDenseSubset}(\mathbb{Q},P_{\mathbb{R}})
\land
\operatorname{IsOrderDenseSubset}(I,P_{\mathbb{R}}).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists a,b\in\mathbb{R}\;
\Bigl[
a<b\land
\bigl(
\forall q\in\mathbb{Q}\;(q\le a\lor b\le q)
\lor
\forall s\in\mathbb{R}\setminus\mathbb{Q}\;(s\le a\lor b\le s)
\bigr)
\Bigr].
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
P_{\mathbb{R}}=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad I=\mathbb{R}\setminus\mathbb{Q},\qquad
D_{\mathbb{Q}}=\mathsf{OrderDenseSet}(\mathbb{Q},P_{\mathbb{R}}),\qquad
D_I=\mathsf{OrderDenseSet}(I,P_{\mathbb{R}}),\qquad
\neg\Bigl[
\operatorname{IsOrderDenseSubset}(\mathbb{Q},P_{\mathbb{R}})
\land
\operatorname{IsOrderDenseSubset}(I,P_{\mathbb{R}})
\Bigr].
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
The corollary fails if a nonempty real interval misses all rational points or misses all irrational points.
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
\[
\exists a,b\in\mathbb{R}\;(a<b)
\;\land\;
\Bigl[
\underbrace{\forall q\in\mathbb{Q}\;(q\le a\lor b\le q)}_{\text{no rational point}}
\;\lor\;
\underbrace{\forall s\in\mathbb{R}\setminus\mathbb{Q}\;(s\le a\lor b\le s)}_{\text{no irrational point}}
\Bigr].
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
The corollary combines the two density theorems. Every real interval, no matter how small, contains representatives of both arithmetic types. Thus neither the rational numbers nor the irrational numbers occupy isolated regions of the real line.
\end{remark*}

\begin{exposition}
Density in $\mathbb{R}$ has a scale-invariant consequence: no matter how small a nonempty
open interval becomes, it still contains both rational and irrational points. The interleaving
of $\mathbb{Q}$ and $\mathbb{R}\setminus\mathbb{Q}$ is therefore not a large-scale phenomenon.
It persists at every interval scale.

This has three important consequences. First, the continuum does not become locally discrete
under magnification. For every nonempty open interval $(a,b)$, both
\[
\mathbb{Q}\cap(a,b)
\qquad\text{and}\qquad
(\mathbb{R}\setminus\mathbb{Q})\cap(a,b)
\]
remain nonempty, and in fact infinite. Density is therefore stable under repeated restriction
to smaller intervals.

Second, this behavior sharply distinguishes the real line from discrete numerical models.
Floating-point systems form a discrete set of representable values, so repeated magnification
eventually reaches a scale at which adjacent representable numbers leave no further point
strictly between them. At that stage, interval penetration fails in the discrete model even
though it continues to hold in $\mathbb{R}$.

Third, the Archimedean Property supplies the scale mechanism behind this behavior. Given any
positive scale $\varepsilon$, there exists $n\in\mathbb{N}$ with
\[
\frac{1}{n}<\varepsilon.
\]
This makes it possible to construct rational approximations at arbitrarily small interval
scales, which is one of the basic engines behind density arguments.

The resulting contrast is structural. Smooth mathematical models are built in a setting where
interval penetration persists at every scale, whereas finite computational models eventually
encounter a smallest representable separation. The transition from continuum behavior to
discrete behavior occurs exactly where nonempty intervals in the numerical model cease to
contain further \\representable points.
\end{exposition}
