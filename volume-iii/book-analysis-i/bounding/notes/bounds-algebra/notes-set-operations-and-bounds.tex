\begin{remark*}[Orientation]
Set operations change the collection of possible counterexamples to a proposed
bound.  Each result below isolates one operation and one direction of bound
transport.
\end{remark*}

\begin{lemmabox}{Lemma (Union Preserves Upper Bounds)}
\begin{lemma}[Union Preserves Upper Bounds]
\label{lem:union-preserves-upper-bounds}
Let $A,B\subseteq\mathbb{R}$.  If $u_A$ is an upper bound of $A$ and
$u_B$ is an upper bound of $B$, then $\max\{u_A,u_B\}$ is an upper bound
of $A\cup B$.
\hyperref[prf:union-preserves-upper-bounds]{Go to proof.}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
(\forall a\in A)(a\leq u_A)\land(\forall b\in B)(b\leq u_B)
\Longrightarrow
(\forall x\in A\cup B)(x\leq \max\{u_A,u_B\}).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&U=A\cup B,\qquad m=\max\{u_A,u_B\},\\
&\operatorname{UpperBound}(u_A,A,P)\land\operatorname{UpperBound}(u_B,B,P)
\Longrightarrow \operatorname{UpperBound}(m,U,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
(\forall a\in A)(a\leq u_A),\quad
(\forall b\in B)(b\leq u_B),\quad
(\exists x\in A\cup B)(\max\{u_A,u_B\}<x).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&U=A\cup B,\qquad m=\max\{u_A,u_B\},\\
&\operatorname{UpperBound}(u_A,A,P)\land\operatorname{UpperBound}(u_B,B,P)\land
\neg\operatorname{UpperBound}(m,U,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
A failure would require a union element larger than both proposed upper
bounds.


  \item[\operatorname{UpperBound}.]
  \[
(\forall a\in A)(a\leq u_A),\quad
(\forall b\in B)(b\leq u_B),\quad
(\exists x\in A\cup B)(\max\{u_A,u_B\}<x).
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\\
&U=A\cup B,\qquad m=\max\{u_A,u_B\},\\
&\operatorname{UpperBound}(u_A,A,P)\land\operatorname{UpperBound}(u_B,B,P)\land
\neg\operatorname{UpperBound}(m,U,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Every $x\in A\cup B$ lies in $A$ or in $B$.  In the first case
$x\leq u_A\leq\max\{u_A,u_B\}$; in the second case
$x\leq u_B\leq\max\{u_A,u_B\}$.
\end{remark*}

\begin{propositionbox}{Proposition (Union Is Bounded Above Exactly When Both Pieces Are)}
\begin{proposition}[Union Is Bounded Above Exactly When Both Pieces Are]
\label{prop:union-bounded-above-iff-pieces-bounded-above}
Let $A,B\subseteq\mathbb{R}$ be nonempty.  Then $A\cup B$ is bounded above
if and only if $A$ is bounded above and $B$ is bounded above.
\hyperref[prf:union-bounded-above-iff-pieces-bounded-above]{Go to proof.}
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
A\neq\varnothing,\quad B\neq\varnothing
\quad\Longrightarrow\quad
\operatorname{BoundedAbove}(A\cup B)
\Longleftrightarrow
\operatorname{BoundedAbove}(A)\land\operatorname{BoundedAbove}(B).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad U=A\cup B,\\
&\operatorname{BoundedAbove}(U,P)
\Longleftrightarrow
\operatorname{BoundedAbove}(A,P)\land\operatorname{BoundedAbove}(B,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
The equivalence fails precisely if one side holds and the other does not.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad U=A\cup B,\\
&\operatorname{BoundedAbove}(U,P)\oplus
\bigl(\operatorname{BoundedAbove}(A,P)\land\operatorname{BoundedAbove}(B,P)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
The forward direction could fail only if a bound for the union did not bound a
piece.  The reverse direction could fail only if bounds for both pieces did not
combine into a bound for the union.


  \item[\operatorname{BoundedAbove}.]

\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad U=A\cup B,\\
&\operatorname{BoundedAbove}(U,P)\oplus
\bigl(\operatorname{BoundedAbove}(A,P)\land\operatorname{BoundedAbove}(B,P)\bigr).
\end{aligned}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad U=A\cup B,\\
&\operatorname{BoundedAbove}(U,P)
\Longleftrightarrow
\operatorname{BoundedAbove}(A,P)\land\operatorname{BoundedAbove}(B,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Use subset inheritance for the forward direction and
\hyperref[lem:union-preserves-upper-bounds]{Union Preserves Upper Bounds} for
the reverse direction.

\begin{lemmabox}{Lemma (Subsets Preserve Upper Bounds)}
\begin{lemma}[Subsets Preserve Upper Bounds]
\label{lem:subsets-preserve-upper-bounds}
Let $C\subseteq A\subseteq\mathbb{R}$.  If $u$ is an upper bound of $A$,
then $u$ is an upper bound of $C$.
\hyperref[prf:subsets-preserve-upper-bounds]{Go to proof.}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
C\subseteq A\land(\forall a\in A)(a\leq u)
\Longrightarrow
(\forall c\in C)(c\leq u).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad C\subseteq A\subseteq\mathbb{R},\\
&\operatorname{UpperBound}(u,A,P)
\Longrightarrow \operatorname{UpperBound}(u,C,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
C\subseteq A,\quad (\forall a\in A)(a\leq u),\quad
(\exists c\in C)(u<c).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad C\subseteq A\subseteq\mathbb{R},\\
&\operatorname{UpperBound}(u,A,P)\land\neg\operatorname{UpperBound}(u,C,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]

counterexample in $A$.

(\exists c\in C)(u<c).

\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad C\subseteq A\subseteq\mathbb{R},\\

\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Since $C\subseteq A$, every $c\in C$ is an element of $A$.  The upper-bound
condition on $A$ therefore applies directly to $c$.
\end{remark*}

\begin{corollarybox}{Corollary (Intersections Inherit Bounds)}
\begin{corollary}[Intersections Inherit Bounds]
\label{cor:intersections-inherit-bounds}
Let $A,B\subseteq\mathbb{R}$.  Every upper bound of $A$ is an upper bound
of $A\cap B$, and every lower bound of $A$ is a lower bound of $A\cap B$.
\hyperref[prf:intersections-inherit-bounds]{Go to proof.}
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
A\cap B\subseteq A.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad C=A\cap B,\\
&\operatorname{UpperBound}(u,A,P)\Longrightarrow\operatorname{UpperBound}(u,C,P),\\
&\operatorname{LowerBound}(\ell,A,P)\Longrightarrow\operatorname{LowerBound}(\ell,C,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
A failure would be a failure of subset inheritance for $A\cap B\subseteq A$.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad C=A\cap B,\\
&\operatorname{UpperBound}(u,A,P)\land
\neg\operatorname{UpperBound}(u,C,P)
\end{aligned}
\]
is contradictory, and the same holds for lower bounds.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Intersection can lose elements, but it cannot introduce an element outside the
parent set $A$.


  \item[\operatorname{UpperBound} / \operatorname{LowerBound}.]

\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad C=A\cap B,\\
&\operatorname{UpperBound}(u,A,P)\land
\neg\operatorname{UpperBound}(u,C,P)
\end{aligned}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad C=A\cap B,\\
&\operatorname{UpperBound}(u,A,P)\Longrightarrow\operatorname{UpperBound}(u,C,P),\\
&\operatorname{LowerBound}(\ell,A,P)\Longrightarrow\operatorname{LowerBound}(\ell,C,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Apply subset inheritance to $C=A\cap B$.
\end{remark*}

A failure would be a failure of subset inheritance for
$A\setminus B\subseteq A$.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad D=A\setminus B,\\
&\operatorname{UpperBound}(u,A,P)\land
\neg\operatorname{UpperBound}(u,D,P)
\end{aligned}
\]
is contradictory, and the same holds for lower bounds.
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Difference can remove elements from $A$, but it cannot create a new element
outside $A$.


  \item[\operatorname{UpperBound} / \operatorname{LowerBound}.]

\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad D=A\setminus B,\\
&\operatorname{UpperBound}(u,A,P)\land
\neg\operatorname{UpperBound}(u,D,P)
\end{aligned}
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A,B\subseteq\mathbb{R},\qquad D=A\setminus B,\\
&\operatorname{UpperBound}(u,A,P)\Longrightarrow\operatorname{UpperBound}(u,D,P),\\
&\operatorname{LowerBound}(\ell,A,P)\Longrightarrow\operatorname{LowerBound}(\ell,D,P).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Apply subset inheritance to $C=A\setminus B$.
\end{remark*}

\begin{corollarybox}{Corollary (Complements Inherit Ambient Bounds)}
\begin{corollary}[Complements Inherit Ambient Bounds]
\label{cor:complements-inherit-ambient-bounds}
Let $A\subseteq E\subseteq\mathbb{R}$, and take complements relative to
$E$.  Every upper bound of $E$ is an upper bound of $E\setminus A$, and
every lower bound of $E$ is a lower bound of $E\setminus A$.
\hyperref[prf:complements-inherit-ambient-bounds]{Go to proof.}
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
E\setminus A\subseteq E.
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq E\subseteq\mathbb{R},\qquad C=E\setminus A,\\
&\operatorname{UpperBound}(u,E,P)\Longrightarrow\operatorname{UpperBound}(u,C,P),\\
&\operatorname{LowerBound}(\ell,E,P)\Longrightarrow\operatorname{LowerBound}(\ell,C,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\operatorname{UpperBound}(u,E)\land\neg\operatorname{UpperBound}(u,E\setminus A)
\]
or
\[
\operatorname{LowerBound}(\ell,E)\land\neg\operatorname{LowerBound}(\ell,E\setminus A).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq E\subseteq\mathbb{R},\qquad C=E\setminus A,\\
&\bigl(\operatorname{UpperBound}(u,E,P)\land\neg\operatorname{UpperBound}(u,C,P)\bigr)\\
&\qquad{}\lor
\bigl(\operatorname{LowerBound}(\ell,E,P)\land\neg\operatorname{LowerBound}(\ell,C,P)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
The complement does not inherit bounds from $A$; it inherits them from the
ambient set $E$.


  \item[\operatorname{UpperBound} / \operatorname{LowerBound}.]
  \[
\operatorname{UpperBound}(u,E)\land\neg\operatorname{UpperBound}(u,E\setminus A)
\]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq E\subseteq\mathbb{R},\qquad C=E\setminus A,\\
&\bigl(\operatorname{UpperBound}(u,E,P)\land\neg\operatorname{UpperBound}(u,C,P)\bigr)\\
&\qquad{}\lor
\bigl(\operatorname{LowerBound}(\ell,E,P)\land\neg\operatorname{LowerBound}(\ell,C,P)\bigr).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Every point of $E\setminus A$ is a point of $E$, so every ambient bound of
$E$ applies to the complement.
\end{remark*}
