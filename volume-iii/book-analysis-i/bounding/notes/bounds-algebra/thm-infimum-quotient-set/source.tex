\begin{theorem}[Infimum of a Quotient Set]
\label{thm:infimum-quotient-set}
Let $A,B\subseteq\mathbb{R}$ be nonempty and bounded, and suppose
$0<\inf B$ or $\sup B<0$. Then
\[
  \inf(A/B)=
  \min\left\{\frac{\inf A}{\inf B},\frac{\inf A}{\sup B},
              \frac{\sup A}{\inf B},\frac{\sup A}{\sup B}\right\}.
\]
\hyperref[prf:infimum-quotient-set]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
A,B\neq\varnothing,\quad A,B\text{ bounded},\quad 0\notin[\inf B,\sup B]
\]
implies that the infimum of $\{a/b:a\in A,\ b\in B\}$ is the minimum of the
four endpoint quotients.
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
A,B\subseteq\mathbb{R},\qquad Q=A/B=\{a/b:a\in A,\ b\in B\},\\
&L=\min\{\alpha/\gamma,\alpha/\delta,\beta/\gamma,\beta/\delta\},\\
&\operatorname{GreatestLowerBound}(\alpha,A,P)\land
\operatorname{LeastUpperBound}(\beta,A,P)\land\\
&\operatorname{GreatestLowerBound}(\gamma,B,P)\land
\operatorname{LeastUpperBound}(\delta,B,P)\land
(0<\gamma\lor\delta<0)\\
&\Longrightarrow \operatorname{GreatestLowerBound}(L,Q,P).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
The displayed endpoint minimum is not the infimum of $A/B$.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}

\end{remark*}

\begin{remark*}[Interpretation]
The infimum of a quotient set is obtained by converting the quotient into
product with the reciprocal denominator set and taking the least endpoint
combination.
\end{remark*}
