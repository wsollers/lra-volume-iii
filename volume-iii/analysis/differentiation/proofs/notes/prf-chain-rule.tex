\begin{theorem*}[Chain Rule]
Let $J, I \subseteq \mathbb{R}$ be open intervals, let $f : J \to \mathbb{R}$ and $g : I \to \mathbb{R}$ be functions with $f(J) \subseteq I$, and let $c \in J$. If $f$ is differentiable at $c$ and $g$ is differentiable at $f(c)$, then $g \circ f$ is differentiable at $c$ and $(g \circ f)'(c) = g'(f(c)) \cdot f'(c)$.
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
Let $d = f(c)$. By Carathéodory's theorem, there exist continuous functions $\varphi : J \to \mathbb{R}$ and $\psi : I \to \mathbb{R}$ such that $f(x) - f(c) = \varphi(x)(x - c)$ with $\varphi(c) = f'(c)$ and $g(y) - g(d) = \psi(y)(y - d)$ with $\psi(d) = g'(d)$. Substituting $y = f(x)$ gives $g(f(x)) - g(f(c)) = \psi(f(x))[f(x) - f(c)] = \psi(f(x)) \cdot \varphi(x) \cdot (x - c)$. Define $\Phi(x) := \psi(f(x)) \cdot \varphi(x)$. Since $f$ is differentiable at $c$, it is continuous at $c$, so $\psi \circ f$ is continuous at $c$ by continuity of composition. Since $\varphi$ is continuous at $c$, their product $\Phi$ is continuous at $c$. By Carathéodory's theorem, $g \circ f$ is differentiable at $c$ with $(g \circ f)'(c) = \Phi(c) = \psi(f(c)) \cdot \varphi(c) = g'(f(c)) \cdot f'(c)$.
\end{proof}

\begin{proof}
\textbf{Detailed Learning Proof.}~\\

\textbf{Step 1.} Apply Carathéodory's theorem to linearize $f$ at $c$. Since $f$ is differentiable at $c$, there exists $\varphi : J \to \mathbb{R}$, continuous at $c$, such that $f(x) - f(c) = \varphi(x)(x - c)$ for all $x \in J$ and $\varphi(c) = f'(c)$. This factorization replaces the potentially problematic difference quotient with a continuous function.

\textbf{Step 2.} Apply Carathéodory's theorem to linearize $g$ at $d = f(c)$. Since $g$ is differentiable at $d$, there exists $\psi : I \to \mathbb{R}$, continuous at $d$, such that $g(y) - g(d) = \psi(y)(y - d)$ for all $y \in I$ and $\psi(d) = g'(d)$. This provides a factorization of increments in $g$ around the point $f(c)$.

\textbf{Step 3.} Compose the two factorizations to linearize $g \circ f$ at $c$. Substitute $y = f(x)$ into the identity for $g$: $g(f(x)) - g(f(c)) = \psi(f(x))[f(x) - f(c)]$. Substitute the identity for $f(x) - f(c)$ from Step 1: $g(f(x)) - g(f(c)) = \psi(f(x)) \cdot \varphi(x) \cdot (x - c)$.

\textbf{Step 4.} Identify the Carathéodory function for $g \circ f$. Define $\Phi : J \to \mathbb{R}$ by $\Phi(x) := \psi(f(x)) \cdot \varphi(x)$. The identity from Step 3 becomes $(g \circ f)(x) - (g \circ f)(c) = \Phi(x)(x - c)$ for all $x \in J$. By Carathéodory's theorem, $g \circ f$ is differentiable at $c$ if and only if $\Phi$ is continuous at $c$.

\textbf{Step 5.} Verify that $\Phi$ is continuous at $c$. Since $f$ is differentiable at $c$, it is continuous at $c$. Since $\psi$ is continuous at $d = f(c)$ and $f$ is continuous at $c$, the composition $\psi \circ f$ is continuous at $c$. Since $\varphi$ is continuous at $c$ from Step 1, the product $\Phi = (\psi \circ f) \cdot \varphi$ is continuous at $c$.

\textbf{Step 6.} Conclude differentiability and compute the derivative. By Carathéodory's theorem, $g \circ f$ is differentiable at $c$ and $(g \circ f)'(c) = \Phi(c)$. Evaluating: $(g \circ f)'(c) = \Phi(c) = \psi(f(c)) \cdot \varphi(c) = g'(f(c)) \cdot f'(c)$.
\end{proof}

\begin{remark*}[Proof structure]
The proof uses a constructive approach via Carathéodory's theorem. Both $f$ and $g$ are linearized using Carathéodory factorizations, yielding continuous functions $\varphi$ and $\psi$. The key insight is that composing these factorizations produces a single Carathéodory factorization for $g \circ f$ with Carathéodory function $\Phi = (\psi \circ f) \cdot \varphi$. The proof succeeds because this product function is continuous at $c$, which follows from combining the continuity properties of $f$, $\psi$, and $\varphi$ using standard continuity theorems. The derivative formula emerges naturally by evaluating $\Phi$ at $c$.
\end{remark*}
