\begin{theorem*}[\(\Delta = E-I\)]
The forward difference operator is the shift operator minus the identity
operator.
\end{theorem*}

\begin{proof}[Professional Standard Proof]
\LRAProofBodyStart
Applying the operator \(E-I\) to a function \(f\) gives
\[
(E-I)f(n) = Ef(n) - If(n) = f(n+1) - f(n) = \Delta f(n). \qedhere
\]
\end{proof}

\begin{proof}[Detailed Learning Proof]
\LRAProofBodyStart
Applying the operator \(E-I\) to a function \(f\) gives
\[
(E-I)f(n) = Ef(n) - If(n) = f(n+1) - f(n) = \Delta f(n). \qedhere
\]
\end{proof}
