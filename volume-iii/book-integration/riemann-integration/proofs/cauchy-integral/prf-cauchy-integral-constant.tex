\begin{proposition*}[Cauchy Integral of a Constant]
If \(f(x)=c\) on \([a,b]\), then \(\int_a^b c\,dx=c(b-a)\).
\end{proposition*}
\begin{proof}[Professional Standard Proof]
\LRAProofBodyStart
For every partition \(P\), \(C(f,P)=\sum_i c\Delta x_i=c(b-a)\). The net is
constant, so its limit is \(c(b-a)\).
\end{proof}
\begin{proof}[Detailed Learning Proof]
\LRAProofBodyStart
The rectangle heights are all \(c\), and the subinterval widths telescope to
the total length \(b-a\).
\end{proof}
\begin{remark*}[Proof structure]
Compute every finite sum and observe it is independent of the partition.
\end{remark*}
