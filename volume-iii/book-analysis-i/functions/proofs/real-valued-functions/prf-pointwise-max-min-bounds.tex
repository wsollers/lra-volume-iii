\begin{proposition*}[Pointwise Max/Min Bounds]
Let \(A\subseteq\mathbb{R}\) and let \(f,g:A\to\mathbb{R}\).  For every
\(x\in A\),
\[
  \min(f,g)(x)\leq f(x)\leq \max(f,g)(x)
  \quad\text{and}\quad
  \min(f,g)(x)\leq g(x)\leq \max(f,g)(x).
\]
\end{proposition*}
