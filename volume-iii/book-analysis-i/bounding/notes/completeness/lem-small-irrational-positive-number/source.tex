\begin{lemma}[Small Irrational Positive Number]
\label{lem:small-irrational-positive-number}
For every $\varepsilon\in\mathbb{R}$ with $\varepsilon>0$, there exists
$\eta\in\mathbb{R}\setminus\mathbb{Q}$ such that
\[
  0<\eta<\varepsilon.
\]
\hyperref[prf:small-irrational-positive-number]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\forall \varepsilon>0\;\exists \eta\in\mathbb{R}\setminus\mathbb{Q}\;
(0<\eta<\varepsilon).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The density of the irrationals applied to the interval $(0,\varepsilon)$ gives
an irrational positive number as small as desired.
\end{remark*}
