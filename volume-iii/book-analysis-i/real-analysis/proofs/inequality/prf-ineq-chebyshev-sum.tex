\begin{theorem*}[Chebyshev's Sum Inequality]
Let $a_1\le\cdots\le a_n$ and $b_1\le\cdots\le b_n$ be real sequences.
Then
\[
\left(\frac{a_1+\cdots+a_n}{n}\right)
\left(\frac{b_1+\cdots+b_n}{n}\right)
\le
\frac{a_1b_1+\cdots+a_nb_n}{n}.
\]
\end{theorem*}
