\begin{theorem*}[Supremum of a Sum Set]
Let $A, B \subseteq \mathbb{R}$ be non-empty sets bounded above. Define $A + B = \{a + b : a \in A, b \in B\}$. Then $\sup(A + B) = \sup A + \sup B$.
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
Let $s_a = \sup A$ and $s_b = \sup B$. For any $x \in A+B$, $x = a+b$ for some $a \in A, b \in B$. Since $a \le s_a$ and $b \le s_b$, we have $a+b \le s_a + s_b$, so $s_a + s_b$ is an upper bound of $A+B$. Thus $\sup(A+B) \le s_a + s_b$. Conversely, for any $\varepsilon > 0$, there exist $a \in A$ and $b \in B$ such that $a > s_a - \varepsilon/2$ and $b > s_b - \varepsilon/2$. Then $a+b > s_a + s_b - \varepsilon$. Since $a+b \in A+B$, this shows $\sup(A+B) \ge s_a + s_b$. Therefore, $\sup(A+B) = s_a + s_b$.
\end{proof}

\begin{proof}
\textbf{Detailed Learning Proof.}~\\

\textbf{Step 1. Show $s_a + s_b$ is an upper bound for $A+B$.}
Let $s_a = \sup A$ and $s_b = \sup B$. By definition of the supremum:
\[ \forall a \in A, a \le s_a \quad \text{and} \quad \forall b \in B, b \le s_b \]
Adding these inequalities, for any $a \in A, b \in B$:
\[ a + b \le s_a + s_b \]
Since every element of $A+B$ is of the form $a+b$, $s_a + s_b$ is an upper bound for $A+B$. Consequently:
\[ \sup(A+B) \le s_a + s_b \]

\textbf{Step 2. Use $\varepsilon$-characterization to show $s_a + s_b$ is the least upper bound.}
Let $\varepsilon > 0$ be arbitrary. By the $\varepsilon$-characterization of $\sup A$ and $\sup B$, there exist elements $a \in A$ and $b \in B$ such that:
\[ a > s_a - \frac{\varepsilon}{2} \quad \text{and} \quad b > s_b - \frac{\varepsilon}{2} \]
Summing these gives:
\[ a + b > (s_a - \frac{\varepsilon}{2}) + (s_b - \frac{\varepsilon}{2}) = s_a + s_b - \varepsilon \]
Since $a+b \in A+B$, we have found an element in the sum set strictly greater than $s_a + s_b - \varepsilon$.

\textbf{Step 3. Conclude the equality.}
Step 2 shows that no value smaller than $s_a + s_b$ can be an upper bound for $A+B$. Thus, the least upper bound must be $s_a + s_b$.
\end{proof}

\begin{remark*}[Proof structure]
The proof utilizes a double-inequality strategy: first proving $\sup(A+B) \le \sup A + \sup B$ via the upper bound property, and then $\sup(A+B) \ge \sup A + \sup B$ via the $\varepsilon$-characterization of the supremum.
\end{remark*}
