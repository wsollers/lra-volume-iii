\begin{proposition*}[Scalar Multiplication Rule for Finite Set Summation]
Let \(X\) be a finite set, let \(f:X\to\mathbb{R}\) be a function, and
let \(c\) be a real number. Then
\[
  \sum_{x\in X} c f(x)
  =
  c \sum_{x\in X} f(x).
\]
\end{proposition*}
