\begin{proposition*}[Substitution for Finite Set Summation]
If \(X\) is a finite set, \(f:X\to\mathbb{R}\) is a function, and
\(g:Y\to X\) is a bijection, then
\[
  \sum_{x\in X} f(x)
  =
  \sum_{y\in Y} f(g(y)).
\]
\end{proposition*}
