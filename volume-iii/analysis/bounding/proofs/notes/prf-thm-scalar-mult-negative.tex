\begin{theorem*}[Scalar Multiplication by a Negative Scalar]
Let $A\subseteq\mathbb{R}$ be nonempty and bounded below, and let $k<0$. Then
\[
  \sup(kA)=k\inf A,
  \qquad
  kA=\{ka:a\in A\}.
\]
\end{theorem*}
