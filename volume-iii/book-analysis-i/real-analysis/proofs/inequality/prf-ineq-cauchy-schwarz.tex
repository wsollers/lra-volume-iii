\begin{theorem*}[Cauchy-Schwarz Inequality]
For all real numbers $a_1,\dots,a_n$ and $b_1,\dots,b_n$,
\[
\left|\sum_{j=1}^{n}a_jb_j\right|
\le
\left(\sum_{j=1}^{n}a_j^2\right)^{1/2}
\left(\sum_{j=1}^{n}b_j^2\right)^{1/2}.
\]
\end{theorem*}
