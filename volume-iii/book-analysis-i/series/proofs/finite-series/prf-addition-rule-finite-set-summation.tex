\begin{proposition*}[Addition Rule for Finite Set Summation]
Let \(X\) be a finite set, and let \(f:X\to\mathbb{R}\) and
\(g:X\to\mathbb{R}\) be functions. Then
\[
  \sum_{x\in X} (f(x)+g(x))
  =
  \sum_{x\in X} f(x)
  +
  \sum_{x\in X} g(x).
\]
\end{proposition*}
