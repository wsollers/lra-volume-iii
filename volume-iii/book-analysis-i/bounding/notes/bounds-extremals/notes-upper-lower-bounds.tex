\subsection{Upper and Lower Bounds}

\begin{remark*}[Exposition]
Bounds are the first level of extremal information.  They compare one
candidate element with every element of a set, but they do not
require the candidate to be sharp or to belong to the set.
\end{remark*}

\begin{definitionbox}{Bound}
\begin{definition}[Bound]
\label{def:bound}
Let $(S,\leq)$ be an ordered set, let $A\subseteq S$, and let $b\in S$.
The element $b$ is a \emph{bound} of $A$ if $b$ is either an upper bound or
a lower bound of $A$; that is, if $a\leq b$ for every $a\in A$, or
$b\leq a$ for every $a\in A$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  \bigl[(\forall a\in A)(a\leq b)
  \ \text{or}\ 
  (\forall a\in A)(b\leq a)\bigr].
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  S\in\mathbf{Set},\qquad A\subseteq S,\qquad P=\mathsf{OrderedSet}(S,\leq),\qquad
  \operatorname{UpperBound}(b,A,P)
  \lor
  \operatorname{LowerBound}(b,A,P).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \bigl[(\exists x\in A)(b<x)
  \ \text{and}\ 
  (\exists y\in A)(y<b)\bigr].
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  S\in\mathbf{Set},\qquad A\subseteq S,\qquad P=\mathsf{OrderedSet}(S,\leq),\qquad
  \neg\operatorname{UpperBound}(b,A,P)
  \land
  \neg\operatorname{LowerBound}(b,A,P).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
  A candidate bound fails only when it does not control the set from either
  side.

  \item[\operatorname{UpperBound} / \operatorname{LowerBound}.]
  The candidate lies strictly between two elements of $A$.
  \[
    (\exists x,y\in A)(b<x\ \text{and}\ y<b).
  \]
  \[
    P=\mathsf{OrderedSet}(S,\leq),\qquad
    \neg\operatorname{UpperBound}(b,A,P)
    \land
    \neg\operatorname{LowerBound}(b,A,P).
  \]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
A bound is a one-sided comparison witness.  It need not be sharp and it
need not belong to the set being bounded.
\end{remark*}

\begin{remark*}[Empty-set convention]
If $A=\varnothing$, then every $b\in S$ is both an upper bound and a lower
bound of $A$, because the universal statements over $A$ are vacuously true.
This is intentional: nonemptiness is not part of being a bound.
\end{remark*}

\begin{definitionbox}{Upper Bound in the Real Line}
\begin{definition}[Upper Bound in the Real Line]
\label{def:real-upper-bound}
Let $A\subseteq\mathbb{R}$ and let $u\in\mathbb{R}$.  The element $u$ is an
\emph{upper bound} of $A$ in the real line if every element of $A$ is less
than or equal to $u$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  (\forall a\in A)(a\leq u).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad
  \operatorname{UpperBound}(u,A,P).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  (\exists a\in A)(u<a).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad
  \neg\operatorname{UpperBound}(u,A,P).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
  Upper-bound failure means the candidate does not sit above the whole
  set in the ambient order.

  \item[\operatorname{UpperBound}.]
  Some element of $A$ lies strictly above the proposed upper bound.
  \[
    (\exists a\in A)(u<a).
  \]
  \[
    P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
    \neg\operatorname{UpperBound}(u,A,P).
  \]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
An upper bound controls every element of the set from above, but it may
lie outside the set and need not be the least such control.
\end{remark*}

\begin{remark*}[Empty-set convention]
If $A=\varnothing$, then every $u\in\mathbb{R}$ is an upper bound of $A$.
The statement $(\forall a\in A)(a\leq u)$ is then vacuously true.
\end{remark*}

\begin{definitionbox}{Lower Bound in the Real Line}
\begin{definition}[Lower Bound in the Real Line]
\label{def:real-lower-bound}
Let $A\subseteq\mathbb{R}$ and let $\ell\in\mathbb{R}$.  The element
$\ell$ is a \emph{lower bound} of $A$ in the real line if every
element of $A$ is greater than or equal to $\ell$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[

\]
\end{remark*}

\begin{remark*}[Predicate reading]

  \operatorname{LowerBound}(\ell,A,P).

\end{remark*}

\begin{remark*}[Negated quantified statement]
\[

\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad

\]
\end{remark*}

\begin{remark*}[Failure modes]

\begin{definitionbox}{Bounded Above}
\begin{definition}[Bounded Above]
\label{def:bounded-above}
Let $A\subseteq\mathbb{R}$.  The set $A$ is \emph{bounded above} if it has
at least one upper bound in $\mathbb{R}$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  (\exists u\in\mathbb{R})(\forall a\in A)(a\leq u).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad
  \operatorname{BoundedAbove}(A,P)
  \Longleftrightarrow
  (\exists u\in\mathbb{R})\operatorname{UpperBound}(u,A,P).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  (\forall u\in\mathbb{R})(\exists a\in A)(u<a).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad
  \neg\operatorname{BoundedAbove}(A,P)
  \Longleftrightarrow
  (\forall u\in\mathbb{R})\neg\operatorname{UpperBound}(u,A,P).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
  Bounded-above failure means there is no single element of the ambient
  ordered set that lies above all elements of $A$.

  \item[\operatorname{BoundedAbove}.]
  Every proposed upper-bound candidate is defeated by a larger element of $A$.
  \[
    (\forall u\in\mathbb{R})(\exists a\in A)(u<a).
  \]
  \[
    P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
    \neg\operatorname{BoundedAbove}(A,P).
  \]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
A set is bounded above when at least one ambient element controls the set
from above.  No least upper bound is asserted here.
\end{remark*}

\begin{remark*}[Empty-set convention]
The empty set is bounded above in $\mathbb{R}$: every real number is an upper
bound of $\varnothing$, so an upper bound exists.
\end{remark*}

\begin{definitionbox}{Bounded Set}
\begin{definition}[Bounded Set]
\label{def:bounded}
Let $A\subseteq\mathbb{R}$.  The set $A$ is \emph{bounded} if it is bounded
above and bounded below.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
  (\exists \ell,u\in\mathbb{R})(\forall a\in A)(\ell\leq a\leq u).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
  P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad
  \operatorname{Bounded}(A,P)
  \Longleftrightarrow
  \operatorname{BoundedBelow}(A,P)\land\operatorname{BoundedAbove}(A,P).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \neg(\exists \ell,u\in\mathbb{R})(\forall a\in A)(\ell\leq a\leq u).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad A\subseteq\mathbb{R},\qquad
  \neg\operatorname{Bounded}(A,P)
  \Longleftrightarrow
  \neg\operatorname{BoundedBelow}(A,P)\lor\neg\operatorname{BoundedAbove}(A,P).
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
  \item[Exposition.]
  Boundedness fails when at least one of the two one-sided bounding
  requirements fails.

  \item[\operatorname{Bounded}.]
  The set is unbounded above or unbounded below.
  \[
    \neg(\exists \ell,u\in\mathbb{R})(\forall a\in A)(\ell\leq a\leq u).
  \]
  \[
    P=\mathsf{OrderedSet}(\mathbb{R},\leq),\qquad
    \neg\operatorname{Bounded}(A,P).
  \]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
Boundedness is the conjunction of boundedness below and boundedness above.
It records two-sided containment in the ambient order, not the existence of
best bounds.
\end{remark*}

\begin{remark*}[Empty-set convention]
The empty set is bounded in $\mathbb{R}$: it is both bounded above and
bounded below, vacuously.  Nonemptiness is a separate hypothesis and should be
stated explicitly when it is needed, for example in supremum and infimum
existence statements.
\end{remark*}
