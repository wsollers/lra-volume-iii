\label{sec:neighborhoods-and-balls}

\begin{topicbox}{Balls Are Open Intervals}
\begin{exposition}
On the line, the ball of radius $r$ about a point is an open interval centered at that point. Neighborhoods are sets that contain such a ball. These are the local objects from which open sets are assembled.
\end{exposition}
\end{topicbox}
\begin{definitionbox}{Definition (Open Ball on the Real Line)}
\begin{definition}[Open Ball on the Real Line]
\label{def:open-ball}
Let $x\in\mathbb{R}$ and $r>0$. The \emph{open ball} of radius $r$ about $x$ is
\[
B_{r}(x)=\{\,y\in\mathbb{R} : |y-x|<r\,\}=(x-r,\,x+r).
\]
\end{definition}
\end{definitionbox}
\begin{remark*}[Standard quantified statement]
\[
B_{r}(x)=\{\,y\in\mathbb{R} : d(y,x)<r\,\}=(x-r,x+r).
\]
\end{remark*}
\begin{remark*}[Interpretation]
A ball on the line is just a symmetric open interval: all points strictly within distance $r$ of $x$. Writing balls as intervals keeps the chapter inside the metric topology of $\mathbb{R}$.
\end{remark*}

\begin{definitionbox}{Definition (Neighborhood of a Point)}
\begin{definition}[Neighborhood of a Point]
\label{def:neighborhood-of-point}
A set $N\subseteq\mathbb{R}$ is a \emph{neighborhood} of $x\in\mathbb{R}$ exactly when there exists $r>0$ with $B_{r}(x)\subseteq N$.
\end{definition}
\end{definitionbox}
\begin{remark*}[Standard quantified statement]
\[
N \text{ is a neighborhood of } x \Longleftrightarrow \exists r>0\;(B_{r}(x)\subseteq N).
\]
\end{remark*}
\begin{remark*}[Interpretation]
A neighborhood of $x$ is any set with room to spare around $x$ --- it contains a whole ball, not just $x$ itself. Neighborhoods are the language of ``near $x$'' used to define interior points and limits.
\end{remark*}
