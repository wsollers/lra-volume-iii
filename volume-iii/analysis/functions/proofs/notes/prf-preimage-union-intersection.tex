\begin{proposition*}[Preimage Distributes over Union and Intersection]
Let $f:A\to B$, $S,T\subseteq B$. Then (i) $f^{-1}(S\cup T)=f^{-1}(S)\cup f^{-1}(T)$; (ii) $f^{-1}(S\cap T)=f^{-1}(S)\cap f^{-1}(T)$; (iii) $f^{-1}(S^c)=(f^{-1}(S))^c$.
\end{proposition*}
