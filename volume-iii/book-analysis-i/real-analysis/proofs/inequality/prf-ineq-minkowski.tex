\begin{theorem*}[Minkowski's Inequality]
Let $p\ge1$. For real numbers $a_1,\dots,a_n$ and $b_1,\dots,b_n$,
\[
\left(\sum_{j=1}^{n}|a_j+b_j|^p\right)^{1/p}
\le
\left(\sum_{j=1}^{n}|a_j|^p\right)^{1/p}
+
\left(\sum_{j=1}^{n}|b_j|^p\right)^{1/p}.
\]
\end{theorem*}
