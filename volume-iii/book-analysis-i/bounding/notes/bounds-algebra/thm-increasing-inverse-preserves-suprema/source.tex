\begin{theorem}[Increasing Inverse Preserves Suprema]
\label{thm:increasing-inverse-preserves-suprema}
Let $I,J\subseteq\mathbb{R}$ be intervals, and let $f\colon I\to J$ be a continuous
strictly increasing bijection.  Let $B\subseteq J$ be nonempty and bounded
above, and suppose $\sup B\in J$.  Then
$f^{-1}[B]:=\{f^{-1}(b):b\in B\}$ is bounded above, and
\[
  \sup f^{-1}[B]=f^{-1}(\sup B).
\]
\hyperref[prf:increasing-inverse-preserves-suprema]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
B\neq\varnothing,\quad \sup B\in J,\quad f^{-1}\text{ increasing and continuous at }\sup B
\Longrightarrow
\sup f^{-1}(B)=f^{-1}(\sup B).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\begin{aligned}
&B\subseteq J,\qquad g=f^{-1},\qquad g\colon J\to I,\qquad
C=g(B)=\{g(b):b\in B\},\\
&P_J=\mathsf{OrderedSet}(J,\leq),\qquad
P_I=\mathsf{OrderedSet}(I,\leq),\\
&\operatorname{LeastUpperBound}(u,B,P_J)\land
\operatorname{OrderPreserving}(g,P_J,P_I)\land
\operatorname{IsContinuous}(g,u,J,I)
\Longrightarrow
\operatorname{LeastUpperBound}(g(u),C,P_I).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\sup f^{-1}(B)\neq f^{-1}(\sup B)
\]
is impossible under the stated hypotheses.
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\begin{aligned}
&B\subseteq J,\qquad g=f^{-1},\qquad g\colon J\to I,\qquad
C=g(B)=\{g(b):b\in B\},\\
&P_J=\mathsf{OrderedSet}(J,\leq),\qquad
P_I=\mathsf{OrderedSet}(I,\leq),\\
&\operatorname{LeastUpperBound}(u,B,P_J)\land
\operatorname{OrderPreserving}(g,P_J,P_I)\land
\operatorname{IsContinuous}(g,u,J,I)\land
\neg\operatorname{LeastUpperBound}(g(u),C,P_I).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
The formula can fail if the inverse is not increasing, if $\sup B$ is outside
the codomain set $J$, or if the inverse is not continuous at $\sup B$.


  \item[\operatorname{LeastUpperBound} / \operatorname{OrderPreserving}.]
  \[
\sup f^{-1}(B)\neq f^{-1}(\sup B)
\]
\[
\begin{aligned}
&B\subseteq J,\qquad g=f^{-1},\qquad g\colon J\to I,\qquad
C=g(B)=\{g(b):b\in B\},\\
&P_J=\mathsf{OrderedSet}(J,\leq),\qquad
P_I=\mathsf{OrderedSet}(I,\leq),\\
&\operatorname{LeastUpperBound}(u,B,P_J)\land
\operatorname{OrderPreserving}(g,P_J,P_I)\land
\operatorname{IsContinuous}(g,u,J,I)\land
\neg\operatorname{LeastUpperBound}(g(u),C,P_I).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Failure modes]
\begin{description}
\item[Exposition.]
Apply the increasing image supremum theorem to $g=f^{-1}$ and the set $B$.


  \item[\operatorname{LeastUpperBound} / \operatorname{OrderPreserving}.]
  \[
\sup f^{-1}(B)\neq f^{-1}(\sup B)
\]
\[
\begin{aligned}
&B\subseteq J,\qquad g=f^{-1},\qquad g\colon J\to I,\qquad
C=g(B)=\{g(b):b\in B\},\\
&P_J=\mathsf{OrderedSet}(J,\leq),\qquad
P_I=\mathsf{OrderedSet}(I,\leq),\\
&\operatorname{LeastUpperBound}(u,B,P_J)\land
\operatorname{OrderPreserving}(g,P_J,P_I)\land
\operatorname{IsContinuous}(g,u,J,I)\land
\neg\operatorname{LeastUpperBound}(g(u),C,P_I).
\end{aligned}
\]
\end{description}
\end{remark*}

\begin{remark*}[Interpretation]
The inverse statement is the increasing-image supremum theorem applied to the
inverse map on the codomain side.
\end{remark*}
