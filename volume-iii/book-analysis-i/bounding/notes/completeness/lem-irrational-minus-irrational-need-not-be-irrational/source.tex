\begin{lemma}[Irrational Minus Irrational Need Not Be Irrational]
\label{lem:irrational-minus-irrational-need-not-be-irrational}
There exist irrational numbers $x,y\in\mathbb{R}\setminus\mathbb{Q}$ such
that
\[
  x-y\in\mathbb{Q}.
\]
For example, one may take $x=y=\sqrt{2}$.
\hyperref[prf:irrational-minus-irrational-need-not-be-irrational]{\textit{Go to proof.}}
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\exists x,y\in\mathbb{R}\setminus\mathbb{Q}\;(x-y\in\mathbb{Q}).
\]
\end{remark*}

\begin{remark*}[Predicate reading]
\[
\exists x,y\in\mathbb{R}\setminus\mathbb{Q}\;(x-y\in\mathbb{Q}).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Taking $x=y=\sqrt{2}$ gives $x-y=0$, so irrationality is not closed under
difference.
\end{remark*}
