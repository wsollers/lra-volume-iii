\begin{theorem*}[Quotients of Cauchy Sequences]
Let $(x_n)$ and $(y_n)$ be Cauchy real sequences. If there exist $c>0$
and $N_0\in\mathbb{N}$ such that $|y_n|\ge c$ for every $n\ge N_0$,
then $(x_n/y_n)$ is Cauchy.

\smallskip
\noindent
\end{theorem*}
