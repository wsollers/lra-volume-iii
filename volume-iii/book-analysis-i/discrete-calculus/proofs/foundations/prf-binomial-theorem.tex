\begin{theorem*}[Binomial Theorem]
For any integer \(n \ge 0\) and real numbers \(x,y\),
\[
(x+y)^n
=
\sum_{k=0}^{n} \binom{n}{k} x^{\,n-k} y^{\,k}.
\]
\end{theorem*}
